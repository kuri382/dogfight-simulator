12double calcRHead(const std::shared_ptr<PhysicalAssetAccessor>& parent,const MotionState& motion,const Track3D& track, bool shooter_head_on){
13 std::shared_ptr<FighterAccessor> f=std::dynamic_pointer_cast<FighterAccessor>(parent);
14 auto crs=motion.getCRS();
15 Eigen::Vector3d rt=track.pos(crs);
16 Eigen::Vector3d vt=track.vel(crs);
17 Eigen::Vector3d rs=motion.pos(crs);
18 Eigen::Vector3d vs=motion.vel(crs);
19 if(shooter_head_on){
20 double Vs=vs.norm();
21 vs=(rt-rs).normalized()*Vs;
22 }
23 return f->getRmax(rs,vs,rt,vt,M_PI,crs);
24}

38double calcRNorm(const std::shared_ptr<PhysicalAssetAccessor>& parent,const MotionState& motion,const Track3D& track, bool shooter_head_on){
39 double RHead=calcRHead(parent,motion,track,shooter_head_on);
40 double RTail=calcRTail(parent,motion,track,shooter_head_on);
41 auto crs=motion.getCRS();
42 Eigen::Vector3d rs=motion.pos(crs);
43 Eigen::Vector3d rt=track.pos(crs);
44 double r=(rs-rt).norm()-RTail;
45 double delta=RHead-RTail;
46 double outRangeScale=100000.0;
47 if(delta==0){
48 if(r<0){
49 r=r/outRangeScale;
50 }else if(r>0){
51 r=1+r/outRangeScale;
52 }else{
53 r=0;
54 }
55 }else{
56 if(r<0){
57 r=r/outRangeScale;
58 }else if(r>delta){
59 r=1+(r-delta)/outRangeScale;
60 }else{
61 r/=delta;
62 }
63 }
64 return r;
65}
