12double calcDistanceMargin(const std::string& team, const MotionState& parentMotion, const nl::json& parentObservables, const nl::json& rulerObservables){
13 //正規化した余剰燃料(距離換算)の計算
14 double optCruiseFuelFlowRatePerDistance=parentObservables.at("/spec/propulsion/optCruiseFuelFlowRatePerDistance"_json_pointer);
15 double fuelRemaining=parentObservables.at("/propulsion/fuelRemaining"_json_pointer);
16 double maxReachableRange=std::numeric_limits<double>::infinity();
17 if(optCruiseFuelFlowRatePerDistance>0.0){
18 maxReachableRange=fuelRemaining/optCruiseFuelFlowRatePerDistance;
19 }
20 Eigen::Vector2d forwardAx=rulerObservables.at("forwardAx").at(team);
21 double dLine=rulerObservables.at("dLine");
22 double distanceFromLine=forwardAx.dot(parentMotion.pos().block<2,1>(0,0,2,1))+dLine;
23 double distanceFromBase=rulerObservables.at("distanceFromBase").at(team);
24 double fuelMargin=rulerObservables.at("fuelMargin");
25 return std::clamp<double>((maxReachableRange/(1+fuelMargin)-(distanceFromLine+distanceFromBase))/(2*dLine),-1.0,1.0);
26}
