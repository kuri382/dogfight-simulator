13class RecurrentBlock(nn.Module):
14 """
15 HandyRL改において、リカレント部分の逐次処理を内部で実行し、
16 外部からは非リカレント部分と同様に全体をバッチとして一括で入力できるようにするためのラッパー
17 HandyRLが逐次処理を必要とする理由は、「推論を実行しない時刻」が存在したときにバッチから除外せずに、
18 学習時に該当時刻をマスクしてstateを更新せず次時刻に渡すことで対応されているためである。
19 """
20 def __init__(self, module):
21 super().__init__()
22 self.module=module
23 assert(isinstance(self.module,nn.RNNBase))
24 assert(self.module.batch_first)
25 dummy=torch.zeros([1,1,self.module.input_size])
26 with torch.no_grad():
27 _,hidden=self.module(dummy)
28 self.hidden_shape=map_r(hidden, lambda h: h.transpose(0,1))
29
30 def forward(self,inputs,states=None,seq_len=None,mask=None):
31 #通常のRNNに加えて、バッチ中の途中の時刻に「推論しない時刻」の存在を許容するもの
32 if(states is None):
33 states=map_r(self.init_hidden([inputs.shape[0]]), lambda h: h.to(inputs.device))
34 if(seq_len is None):
35 #単時刻の推論の場合、shape=[B,D]で時刻の次元が無い形で入力されることを前提とする
36 x,state_out=self.module(inputs.unsqueeze(1),map_r(states, lambda s: s.transpose(0,1).contiguous()))#[B×D_core]
37 x_out = x[:, 0, ...].squeeze(1)
38 state_out = map_r(state_out, lambda s: s.transpose(0,1).contiguous())
39 else:
40 #複数時刻にわたる推論の場合、shape=[B×T,D]の形で入力されることを前提とする
41 hs = states
42 if(mask is not None):
43 hs = bimap_r(hs, map_r(mask, lambda m:m[:,0]), lambda h, m: h * m)
44 x_out=[]
45 x_in=torch.reshape(inputs,(-1,seq_len)+inputs.shape[1:])
46 for t in range(seq_len):
47 x,state_out=self.module(x_in[:,t,...].unsqueeze(1),map_r(hs, lambda h: h.transpose(0,1).contiguous()))#[B,1,D]
48 x_out.append(x)
49 if(mask is not None):
50 state_out = map_r(state_out, lambda s: s.transpose(0,1).contiguous())
51 hs = trimap_r(hs, state_out, map_r(mask, lambda m:m[:,t]), lambda h, nh, m: h * (1 - m) + nh * m)
52 state_out=hs
53 x_out = torch.cat(x_out,dim=1).flatten(0,1)
54 return x_out,state_out
55
56 def init_hidden(self, batch_size=None):
57 if batch_size is None: # for inference
58 return map_r(self.hidden_shape, lambda h: np.zeros(h.shape[1:], dtype=np.float32))
59 else: # for training
60 return map_r(self.hidden_shape, lambda h: torch.zeros((*batch_size,*h.shape[1:]), dtype=torch.float32))
