9def map_r(x, callback_fn=None):
10 # recursive map function
11 if isinstance(x, (list, tuple, set)):
12 return type(x)(map_r(xx, callback_fn) for xx in x)
13 elif isinstance(x, dict):
14 return type(x)((key, map_r(xx, callback_fn)) for key, xx in x.items())
15 return callback_fn(x) if callback_fn is not None else None

34def rotate(x, max_depth=1024):
35 if max_depth == 0:
36 return x
37 if isinstance(x, (list, tuple)):
38 if isinstance(x[0], (list, tuple)):
39 return type(x[0])(
40 rotate(type(x)(xx[i] for xx in x), max_depth - 1)
41 for i, _ in enumerate(x[0])
42 )
43 elif isinstance(x[0], dict):
44 return type(x[0])(
45 (key, rotate(type(x)(xx[key] for xx in x), max_depth - 1))

48 elif isinstance(x, dict):
