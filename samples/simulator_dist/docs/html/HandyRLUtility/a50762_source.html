99class DiscreteActionDistribution(ActionDistributionBase):

102 """
103 def __init__(self, model, params, space=None, legal_actions=None, validate_args=None):
104 super().__init__(model, params, space, legal_actions, validate_args)
105 assert(isinstance(self.space,gym.spaces.Discrete))
106 self.ndim=self.space.n
107 self.action_mask=None
108 self.exploration_logits=None
109 if(self.asNumpy):
110 self.exploration_logits=np.zeros_like(self.params)
111 else:
112 self.exploration_logits=torch.zeros_like(self.params)
113 if(legal_actions is not None):
114 if(self.asNumpy):
115 self.action_mask = np.ones_like(self.params) * np.inf
116 else:
117 self.action_mask = torch.ones_like(self.params) * torch.inf
118 self.action_mask[legal_actions==1] = 0
119 self.params = self.params - self.action_mask
120 self.exploration_logits=-self.action_mask
121 if(self.asNumpy):
122 self.params = torch.tensor(self.params)
123 self.exploration_logits=torch.tensor(self.exploration_logits)
124 self.exploration_logits=torch.where(self.params==-torch.inf,-torch.inf,self.exploration_logits)
125 self.dist=dist.Categorical(logits=self.params, validate_args=validate_args)
126 def sample(self,deterministic=False,exploration=None):
127 if(exploration is None):
128 if(deterministic):
129 ret=self.dist.probs.argmax(dim=-1)
130 else:
131 ret=self.dist.sample()
132 else:
133 #ε-greedy
134 if(np.random.random()<exploration):
135 ret=dist.Categorical(logits=self.exploration_logits, validate_args=self.dist._validate_args).sample()
136 else:
137 if(deterministic):
138 ret=self.dist.probs.argmax(dim=-1)
139 else:
140 ret=self.dist.sample()
141 if(self.asNumpy):
142 ret=ret.numpy()
143 return ret
144 def unpack_policy(self,packed):
145 return [packed]
146 def unpack_scalar(self,packed,keepDim=True):
147 if(keepDim):
148 return [packed.unsqueeze(-1)]
149 else:
150 return [packed]

162class MultiDiscreteActionDistribution(ActionDistributionBase):
163 """MultiDiscreteな行動空間に対する分布

165 """
166 def __init__(self, model, params, space=None, legal_actions=None, validate_args=None):
167 super().__init__(model, params, space, legal_actions, validate_args)
168 assert(isinstance(self.space,gym.spaces.MultiDiscrete))
169 self.action_mask=None
170 splitted_params=self.unpack_policy(self.params)
171 self.exploration_logits=None
172 if(self.asNumpy):
173 self.exploration_logits=[np.zeros_like(p) for p in splitted_params]
174 else:
175 self.exploration_logits=[torch.zeros_like(p) for p in splitted_params]
176 if(legal_actions is not None):
177 self.action_mask=[]
178 for i in range(len(splitted_params)):
179 p=splitted_params[i]
180 if(self.asNumpy):
181 mask = np.ones_like(p) * np.inf
182 else:
183 mask = torch.ones_like(p) * torch.inf
184 mask[legal_actions[i]==1] = 0
185 splitted_params[i] = p - mask
186 self.action_mask.append(mask)
187 self.exploration_logits[i]=-mask
188 if(self.asNumpy):
189 splitted_params = [torch.tensor(p) for p in splitted_params]
190 if(self.action_mask is not None):
191 self.action_mask=np.concatenate(self.action_mask,axis=-1)
192 self.exploration_logits = [torch.tensor(l) for l in self.exploration_logits]
193 else:
194 if(self.action_mask is not None):
195 self.action_mask=torch.cat(self.action_mask,dim=-1)
196 self.exploration_logits=[torch.where(p==-torch.inf,-torch.inf,el) for p,el in zip(splitted_params,self.exploration_logits)]
197 self.dists=[dist.Categorical(logits=p, validate_args=validate_args) for p in splitted_params]
198 def log_prob(self,actions):
199 if(not isinstance(actions,torch.Tensor)):
200 actions=torch.tensor(actions,device=self.device)
201 splitted_actions=torch.split(actions, 1, -1)
202 ret=torch.stack([d.log_prob(a.squeeze(-1)) for d,a in zip(self.dists,splitted_actions)],dim=-1)
203 if(self.asNumpy):
204 ret=ret.numpy()
205 return ret
206 def sample(self,deterministic=False,exploration=None):
207 if(exploration is None):
208 if(deterministic):
209 ret=torch.stack([d.probs.argmax(dim=-1) for d in self.dists],dim=-1)
210 else:
211 ret=torch.stack([d.sample() for d in self.dists],dim=-1)
212 else:
213 #ε-greedy
214 if(np.random.random()<exploration):
215 ret=torch.stack([
216 dist.Categorical(logits=logits, validate_args=d._validate_args).sample()
217 for d,logits in zip(self.dists,self.exploration_logits)
218 ],dim=-1)
219 else:
220 if(deterministic):
221 ret=torch.stack([d.probs.argmax(dim=-1) for d in self.dists],dim=-1)
222 else:
223 ret=torch.stack([d.sample() for d in self.dists],dim=-1)
224 if(self.asNumpy):
225 ret=ret.numpy()
226 return ret
227 def entropy(self, actions=None):
228 ret=torch.stack([d.entropy() for d in self.dists],dim=-1)
229 if(self.asNumpy):
230 ret=ret.numpy()
231 return ret
232 def unpack_policy(self,packed):
233 if(isinstance(packed,torch.Tensor)):
234 return list(torch.split(packed, [s.n for s in self.space], -1))
235 else:
236 dims = [s.n for s in self.space]
237 return np.split(packed, [np.sum(dims[:i+1]) for i in range(len(dims)-1)], -1)
238 def unpack_scalar(self,packed,keepDim=True):
239 if(isinstance(packed,torch.Tensor)):
240 ret = list(torch.split(packed, 1, -1))
241 else:
242 dims = [s.n for s in self.space]
243 ret = np.split(packed, [i for i in range(len(dims)-1)], -1)
244 if(not keepDim):
245 ret = [v.squeeze(-1) for v in ret]
246 return ret

354class BoxActionDistribution(ActionDistributionBase):
355 """Boxな行動空間に対する分布
356 truncated gaussianとして表現する
357 mean=sigmoid(y1)*(high-low)+low
358 stddev=(sigmoid(y2)*upperLimit+lowerLimit)*(high-low)
359 a~truncnorm(mean,stddev)
360 Flattenされたparamsが入るのでspaceのshapeに復元が必要
361 """
362 stddev_min = 1e-2 #標準偏差の下限値(探索の余地を残すため)
363 stddev_scale = 3.0 #標準偏差のsigmoidにかかる係数≒上限値
364 def __init__(self, model, params, space=None, legal_actions=None, validate_args=None):
365 super().__init__(model, params, space, legal_actions, validate_args)
366 assert(isinstance(self.space,gym.spaces.Box))
367 self.low=torch.tensor(self.space.low)
368 self.high=torch.tensor(self.space.high)
369 self.ndim=np.product(self.space.shape)
370
371 self.action_mask=None
372 mean_, stddev_ = self.unpack_policy(self.params)
373 if(self.asNumpy):
374 mean_ = torch.tensor(mean_)
375 stddev_ = torch.tensor(stddev_)
376 self.action_mask = np.zeros_like(self.params)
377 else:
378 self.action_mask = torch.zeros_like(self.params)
379 if(legal_actions is not None):
380 self.low=legal_actions[0]
381 self.high=legal_actions[1]
382 if(isinstance(self.low,np.ndarray)):
383 self.low=torch.tensor(self.low)
384 if(isinstance(self.high,np.ndarray)):
385 self.high=torch.tensor(self.high)
386 mean_, stddev_, self.high, self.low = broadcast_all(mean_, stddev_, self.high, self.low)
387 self.mean=torch.sigmoid(mean_)*(self.high-self.low)+self.low
388 #self.stddev=torch.exp(stddev_)+1e-6
389 self.stddev=(torch.sigmoid(stddev_)*type(self).stddev_scale+type(self).stddev_min)*(self.high-self.low)
390 self.dist=TruncatedNormal(self.mean,self.stddev,self.low,self.high, validate_args=validate_args)
391 def sample(self,deterministic=False,exploration=None):
392 if(exploration is None):
393 if(deterministic):
394 ret=self.mean
395 else:
396 ret=self.dist.sample()
397 else:
398 #ε-greedy
399 if(np.random.random()<exploration):
400 ret=dist.Uniform(self.low,self.high,validate_args=self.dist._validate_args).sample()
401 #ret=TruncatedNormal(self.mean,self.stddev+exploration*(self.high-self.low),self.low,self.high, validate_args=self.dist._validate_args).sample()
402 else:
403 if(deterministic):
404 ret=self.mean
405 else:
406 ret=self.dist.sample()
407 ret=torch.clamp(ret,self.low,self.high) #just for numeric stability
408 if(self.asNumpy):
409 ret=ret.numpy()
410 return ret
411 def unpack_policy(self,packed):
412 dims = [self.ndim, self.ndim]
413 if(isinstance(packed,torch.Tensor)):
414 return [r.unflatten(-1,self.space.shape) for r in torch.split(packed, dims, -1)]
415 else:
416 return [np.reshape(r,packed.shape[:-1]+self.space.shape) for r in np.split(packed, [np.sum(dims[:i+1]) for i in range(len(dims)-1)], -1)]
417 def unpack_scalar(self,packed,keepDim=True):
418 if(keepDim):
419 return [packed]
420 else:
421 return [packed.squeeze(-1)]
