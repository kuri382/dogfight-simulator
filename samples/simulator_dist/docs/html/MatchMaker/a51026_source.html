68class MatchMakerOnRayBase(DefaultCallbacks):
69 """rayのCallbackとしてTrainer側でPolicyの重みを読み書きするクラス。
70 rayのtrainer_configにこれをそのまま登録することはせず、
71 createMatchMakerOnRayForEachTrainerによりTrainerの名称と紐付けたサブクラスを登録するものとする。
72 """
73 def __init__(self):
74 super().__init__()
75 self.currentWeights=defaultdict(lambda:defaultdict(lambda:defaultdict(lambda:-1)))
76 self.counter=0
77 self.trainerName="Learner"
78 self.resetRequested=False
79 self.resetWeights={}
80 self.numWeightsPerPolicy=100 #メモリに保持しておく重みの数(ポリシー毎)
81 self.weightBuffer=defaultdict(lambda:{})
82 self.trainingWeightCount=defaultdict(lambda:-1)
83 self.matchMaker=ray.get_actor("MatchMaker")
84 self.matchMonitors={}
85 self.weight_pool=ray.get(self.matchMaker.get_weight_pool.remote())
86 self.policy_config=ray.get(self.matchMaker.get_policy_config.remote())
87 def makeNextMatch(self,worker_index,vector_index):
88 matchType=self.trainerName
89 matchInfo=ray.get(self.matchMaker.makeNextMatch.remote(matchType,worker_index))
90 for team,info in matchInfo.get("teams",matchInfo).items():
91 if(info["Suffix"]!="" and vector_index > 0):
92 info["Suffix"]+="_{}".format(vector_index)
93 return matchInfo
94 def on_sub_environment_created(
95 self,
96 *,
97 worker: "RolloutWorker",
98 sub_environment: EnvType,
99 env_context: EnvContext,
100 env_index: Optional[int] = None,
101 **kwargs,) -> None:
102 """初回の対戦カードを設定する。
103 """
104 sub_environment.setMatch(self.makeNextMatch(sub_environment.worker_index,sub_environment.vector_index))
105 def on_episode_start(
106 self,
107 *,
108 worker: "RolloutWorker",
109 base_env: BaseEnv,
110 policies: Dict[PolicyID, Policy],
111 episode: Union[Episode, EpisodeV2],
112 env_index: Optional[int] = None,
113 **kwargs,
114 ) -> None:
115 env=base_env.get_sub_environments()[env_index]
116 self.matchMonitors[env_index]=self.matchMonitorClass(env)
117 self.matchMonitors[env_index].onEpisodeBegin()
118 self.counter+=1
119 for team,info in env.matchInfo.get("teams",env.matchInfo).items():
120 policyBase=info["Policy"]
121 policy=policyBase+info["Suffix"]
122 weight_id=info["Weight"]
123 if(weight_id>=0 and weight_id!=self.currentWeights[policy][env.worker_index][env.vector_index]):
124 self.loadWeights(policies[policy], policyBase, weight_id)
125 self.currentWeights[policy][env.worker_index][env.vector_index]=weight_id
126 def loadWeights(self, policy, policyBase, weight_id):
127 """重みを読み込む
128 """
129 if(self.policy_config[policyBase]["is_internal"]):
130 return
131 if(weight_id==0):
132 #weight_id==0 : その時点の学習中重みのコピー
133 count=ray.get(self.matchMaker.getTrainingWeightUpdateCount.remote(policyBase))
134 if(count>self.trainingWeightCount[policyBase]):
135 self.trainingWeightCount[policyBase]=count
136 latest=ray.get(self.matchMaker.getLatestTrainingWeight.remote(policyBase))
137 self.weightBuffer[policyBase][weight_id]=latest
138 policy.set_weights(self.weightBuffer[policyBase][weight_id])
139 elif(weight_id>0):
140 #weight_id>0 : 過去のある時点の重み
141 if(weight_id in self.weightBuffer[policyBase]):
142 weight=self.weightBuffer[policyBase].pop(weight_id)
143 else:
144 path=os.path.join(self.weight_pool,policyBase+"-"+str(weight_id)+".dat")
145 weight=loadWeights(path)
146 self.weightBuffer[policyBase][weight_id]=weight
147 policy.set_weights(weight)
148 if(len(self.weightBuffer[policyBase])>self.numWeightsPerPolicy):
149 #メモリに保持している重みが上限を超えたら直近の読み込み時刻が最も古いもの(ただし、0を除く)を削除する。
150 it=iter(self.weightBuffer[policyBase].keys())
151 removed=next(it)
152 if(removed==0):
153 removed=next(it)
154 self.weightBuffer[policyBase].pop(removed,None)
155 def on_episode_end(
156 self,
157 *,
158 worker: "RolloutWorker",
159 base_env: BaseEnv,
160 policies: Dict[PolicyID, Policy],
161 episode: Union[Episode, EpisodeV2, Exception],
162 env_index: Optional[int] = None,
163 **kwargs,
164 ) -> None:
165 env=base_env.get_sub_environments()[env_index]
166 matchType=(self.trainerName,'',env.worker_index)
167 result=self.matchMonitors[env_index].onEpisodeEnd(matchType)
168 populate_config=ray.get(self.matchMaker.onEpisodeEnd.remote(env.matchInfo,result))
169 for policyBase,conf in populate_config.items():
170 populate(self.weight_pool, policyBase, policies[policyBase].get_weights(), conf['weight_id'])
171 if(conf['reset']):
172 self.resetRequested=True
173 initial_weight=loadInitialWeight(policyBase,self.policy_config)
174 if(initial_weight is None):
175 raise ValueError("Weight reset is available only for pretrained policies.")
176 self.resetWeights[policyBase]=initial_weight
177 customMetrics=ray.get(self.matchMaker.get_metrics.remote(env.matchInfo,result))
178 for key,value in customMetrics.items():
179 episode.custom_metrics[key]=value
180 env.setMatch(self.makeNextMatch(env.worker_index,env.vector_index))
181 def on_train_result(
182 self,
183 *,
184 algorithm: "Algorithm",
185 result: dict,
186 **kwargs,
187 ) -> None:
188 if(self.resetRequested):
189 algorithm.set_weights(self.resetWeights)
190 algorithm.workers.sync_weights(list(self.resetWeights.keys()))
191 self.resetRequested=False
192 self.resetWeights={}
