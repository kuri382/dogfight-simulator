15void PlanarMissile::calcMotion(double tAftLaunch,double dt){
16 motion_prev=motion;
17 double V=motion.vel.norm();
18 auto propNav=getShared<PropNav>(controllers["Navigator"]);
19 Eigen::Vector3d accelCmd=propNav->commands["accel"];
20 Eigen::Vector3d omegaV=propNav->commands["omega"];
21 double omega=omegaV.norm();
22 double omegaMax=maxLoadG*gravity/V;
23 if(omega>omegaMax){
24 omegaV*=(omegaMax/omega);
25 omega=omegaMax;
26 accelCmd=omegaV.cross(motion.vel());
27 }
28 double alt=motion.getHeight();
29 bool pwroff=tAftLaunch>=tBurn;
30 double thrust_;
31 if(pwroff){
32 thrust_ = 0.0;
33 mass = massF;
34 lcg = lcgf;
35 }else{
36 thrust_ = thrust;
37 mass = massI+(massF-massI)*(tAftLaunch/tBurn-1);
38 lcg = lcgi+(lcgf-lcgi)*(tAftLaunch/tBurn-1);
39 }
40 std::map<std::string,double> atm=atmosphere(alt);
41 double M=V/atm["a"];
42 double qd=0.5*atm["rho"]*V*V;
43 Eigen::Vector3d ex=motion.vel()/V;
44 Eigen::Vector3d gravityAccelVector=motion.dirHtoP(Eigen::Vector3d(0,0,1),"NED")*gravity; //TODO 慣性力モデルに置換
45 Eigen::Vector3d ey=(accelCmd-gravityAccelVector).cross(ex)*mass;
46 double desiredSideForce=ey.norm();
47 Eigen::Vector3d sideAx;
48 double aoa,CL,CD;
49 if(desiredSideForce>0){
50 ey/=desiredSideForce;
51 sideAx=ex.cross(ey);
52 std::pair<double,double> tmp=EkkerMDT::cldcmd(d,l,ln,lcg,bt,St,M);
53 double cld=tmp.first;
54 double cmd=tmp.second;
55 auto func=[&](double aoa_){
56 std::pair<double,double> tmp=EkkerMDT::clacma(d,l,ln,lcg,bw,Sw,lw,bt,St,M,aoa_);
57 double cla=tmp.first;
58 double cma=tmp.second;
59 double delta=std::clamp<double>(-cma/cmd*aoa_,-maxD,maxD);
60 return (cla*aoa_+cld*delta)*(qd*Sref)+thrust_*sin(aoa_)-desiredSideForce;
61 };
62 if(func(0.0)>=0){
63 aoa=0.0;
64 }else if(func(maxA)<=0){
65 aoa=maxA;
66 }else{
67 try{
68 boost::math::tools::eps_tolerance<double> tol(16);
69 boost::uintmax_t max_iter=10;
70 auto result=boost::math::tools::toms748_solve(func,0.0,maxA,tol,max_iter);
71 aoa=(result.first+result.second)/2;
72 }catch(std::exception& e){
73 std::cout<<"exception at PlanarMissile::calcMotion"<<std::endl;
74 DEBUG_PRINT_EXPRESSION(func(0.0))
75 DEBUG_PRINT_EXPRESSION(func(maxA))
76 DEBUG_PRINT_EXPRESSION(alt)
77 DEBUG_PRINT_EXPRESSION(M)
78 DEBUG_PRINT_EXPRESSION(desiredSideForce)
79 DEBUG_PRINT_EXPRESSION(thrust_)
80 throw e;
81 }
82 }
83 tmp=EkkerMDT::clacma(d,l,ln,lcg,bw,Sw,lw,bt,St,M,aoa);
84 double cla=tmp.first;
85 double cma=tmp.second;
86 double delta=std::clamp<double>(-cma/cmd*aoa,-maxD,maxD);
87 //CL=(cla-cld*cma/cmd)*aoa;
88 CL=cla*aoa+cld*delta;
89 CD=EkkerMDT::cdtrim(d,l,ln,bw,Sw,thicknessRatio,bt,St,thicknessRatio,M,alt,aoa,delta,frictionFactor,pwroff);
90 }else{
91 sideAx<<0,0,0;
92 CL=0.0;
93 CD=EkkerMDT::cdtrim(d,l,ln,bw,Sw,thicknessRatio,bt,St,thicknessRatio,M,alt,0,0,frictionFactor,pwroff);
94 aoa=0.0;
95 }
96 Eigen::Vector3d sideAccel=sideAx*(CL*qd*Sref+thrust_*sin(aoa))/mass+(gravityAccelVector-ex*(gravityAccelVector.dot(ex)));
97 sideAccel=motion.relPtoH(sideAccel,motion.pos(),"NED",true);
98 sideAccel(2)=0;
99 sideAccel=motion.relHtoP(sideAccel,motion.pos(),"NED",true);
100 //attitude and deflection are assumed to be immidiately adjustable as desired.
101 accelScalar=(-CD*Sref*qd+thrust_*cos(aoa))/mass+gravityAccelVector.dot(ex);
102 Eigen::Vector3d accel=ex*accelScalar;
103 omegaV=motion.vel().cross(sideAccel)/(V*V);
104 omega=omegaV.norm();
105 double wt=omega*dt;
106 if(wt<1e-8){//ほぼ無回転
107 auto accelVector=motion.relBtoP(Eigen::Vector3d(accelScalar,0,0));
108 motion.setPos(motion.pos()+motion.vel()*dt+accelVector*dt*dt/2.);
109 auto posH=motion.absPtoH(motion.pos(),"NED",true);
110 posH(2)=-motion_prev.getHeight();
111 motion.setPos(motion.absHtoP(posH,"NED",true));
112 Eigen::Vector3d vn=motion.vel().normalized();
113 motion.setVel((V+accelScalar*dt)*vn);
114 auto velH=motion.velPtoH(motion.vel(),motion.pos(),"NED",true);
115 velH(2)=0;
116 motion.setVel(motion.velHtoP(velH,motion.pos(),"NED",true));
117 }else{
118 Eigen::Vector3d ex0=motion.vel()/V;
119 Eigen::Vector3d ey0=omegaV.cross(ex0).normalized();
120 Eigen::Vector3d dr=V*dt*(ex0*sinc(wt)+ey0*oneMinusCos_x2(wt)*wt)+accelScalar*dt*dt*(ex0*(sinc(wt)-oneMinusCos_x2(wt))+ey0*wt*(sincMinusOne_x2(wt)+oneMinusCos_x2(wt)));
121 motion.setPos(motion.pos()+dr);
122 auto posH=motion.absPtoH(motion.pos(),"NED",true);
123 posH(2)=-motion_prev.getHeight();
124 motion.setPos(motion.absHtoP(posH,"NED",true));
125 Eigen::Vector3d vn=(ex0*cos(wt)+ey0*sin(wt)).normalized();
126 motion.setVel((V+accelScalar*dt)*vn);
127 auto velH=motion.velPtoH(motion.vel(),motion.pos(),"NED",true);
128 velH(2)=0;
129 motion.setVel(motion.velHtoP(velH,motion.pos(),"NED",true));
130 }
131 motion.setOmega(omegaV);
132 calcQ();
133}

135bool PlanarMissile::calcRangeSub(double vs,double hs,double vt,double ht,double obs,double aa,double r){
136 Eigen::Vector3d posBef=motion.pos();
137 Eigen::Vector3d velBef=motion.vel();
138 Eigen::Vector3d omegaBef=motion.omega();
139 Quaternion qBef=motion.q;
140 Time timeBef=motion.time;
141 MotionState motion_prevBef=motion_prev;
142 bool isBoundToParentBef=isBoundToParent;
143 if(isBoundToParentBef){
144 unbindFromParent();
145 }
146 double dt=interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
147 motion.setPos(motion.absHtoP(Eigen::Vector3d(0,0,-hs),"NED",true));
148 motion.setVel(motion.velHtoP(Eigen::Vector3d(cos(obs),-sin(obs),0)*std::max(vs,1e-3),Eigen::Vector3d(-r,0,-hs),"NED",true));
149 motion.setOmega(Eigen::Vector3d(0,0,0));
150 calcQ();
151 Eigen::Vector3d tpos=motion.absHtoP(Eigen::Vector3d(r,0,-hs),"NED",true);
152 Eigen::Vector3d tvel=motion.velHtoP(Eigen::Vector3d(cos(aa),sin(aa),0)*vt,Eigen::Vector3d(0,0,-hs),"NED",true);
153 Eigen::Vector3d tpos_prev=tpos-dt*tvel;
154 motion_prev.setPos(motion.pos()-dt*motion.vel());
155 motion_prev.setVel(motion.vel());
156 double t=0;
157 bool finished=false;
158 bool hit=false;
159 auto propNav=getShared<PropNav>(controllers["Navigator"]);
160 bool hasLaunchedBef=hasLaunched;
161 hasLaunched=true;
162 while(!finished){
163 commands={
164 {"Navigator",{
165 {"rs",motion.pos()},
166 {"vs",motion.vel()},
167 {"rt",tpos},
168 {"vt",tvel}
169 }}
170 };
171 propNav->control();
172 calcMotion(t,dt);
173 tpos_prev=tpos;
174 tpos+=dt*tvel;
175 t+=dt;
176 if(hitCheck(tpos,tpos_prev)){
177 finished=true;
178 hit=true;
179 }else if(checkDeactivateCondition(t)){
180 finished=true;
181 hit=false;
182 }
183 }
184 if(isBoundToParentBef){
185 bindToParent();
186 }
187 motion.setPos(posBef);
188 motion.setVel(velBef);
189 motion.setOmega(omegaBef);
190 motion.setTime(timeBef);
191 motion.setQ(qBef);
192 motion_prev=motion_prevBef;
193 hasLaunched=hasLaunchedBef;
194 return hit;
195}

200 int nvs=7;
201 int nhs=6;
202 int nvt=7;
203 int nht=2;//
204 int nobs=7;
205 int naa=13;
206 Eigen::VectorXd vs=Eigen::VectorXd::LinSpaced(nvs,0.0,600.0);
207 Eigen::VectorXd hs=Eigen::VectorXd::LinSpaced(nhs,0.0,20000.0);

209 Eigen::VectorXd ht=Eigen::VectorXd::LinSpaced(nht,0.0,20000.0);

211 Eigen::VectorXd aa=Eigen::VectorXd::LinSpaced(naa,-M_PI,M_PI);
212 int numDataPoints=nvs*nhs*nht*nvt*nobs*naa;
213 Eigen::MatrixXd indices=Eigen::MatrixXd::Zero(6,numDataPoints);//ColMajor
214 Eigen::MatrixXd args=Eigen::MatrixXd::Zero(6,numDataPoints);//ColMajor
215 std::cout<<"create indices..."<<std::endl;
216 int idx=0;
217 for(int aa_i=0;aa_i<naa;aa_i++){
218 for(int obs_i=0;obs_i<nobs;obs_i++){
219 for(int ht_i=0;ht_i<nht;ht_i++){
220 for(int vt_i=0;vt_i<nvt;vt_i++){
221 for(int hs_i=0;hs_i<nhs;hs_i++){
222 for(int vs_i=0;vs_i<nvs;++vs_i){
223 args.block(0,idx,6,1)<<vs(vs_i),hs(hs_i),vt(vt_i),ht(ht_i),obs(obs_i),aa(aa_i);
224 idx++;
225 }
226 }
227 }
228 }
229 }
230 }
231 std::cout<<"create indices done."<<std::endl;
232 int numProcess=std::thread::hardware_concurrency();
233 std::vector<std::thread> th;
234 std::vector<std::future<Eigen::VectorXd>> f;
235 std::vector<int> begins;
236 int numPerProc=numDataPoints/numProcess;
237 std::cout<<"run in "<<numProcess<<" processes. total="<<numDataPoints<<", perProc="<<numPerProc<<std::endl;
238 std::mutex mtx;
239 std::vector<std::shared_ptr<Missile>> msls;
240 for(int i=0;i<numProcess;++i){
241 nl::json dummy_ic=instanceConfig;
242 dummy_ic["entityFullName"]=getFullName()+"/for_range_table_"+std::to_string(i+1);
243 msls.push_back(createUnmanagedEntityByClassName<Missile>(isEpisodic(),"PhysicalAsset",getFactoryClassName(),modelConfig,dummy_ic));
244 std::promise<Eigen::VectorXd> p;
245 f.emplace_back(p.get_future());
246 Eigen::MatrixXd subargs=args.block(0,numPerProc*i,6,(i!=numProcess-1 ? numPerProc : numDataPoints-numPerProc*i));
247 th.emplace_back(makeRangeTableSub,std::ref(mtx),std::move(p),msls[i],subargs);
248 }
249 std::cout<<"waiting calculation"<<std::endl;
250 std::vector<Eigen::VectorXd> rangeTablePoints_={vs,hs,vt,ht,obs,aa};
251 Eigen::VectorXd returned(numDataPoints);
252 for(int i=0;i<numProcess;++i){
253 try{
254 returned.block(numPerProc*i,0,(i!=numProcess-1 ? numPerProc : numDataPoints-numPerProc*i),1)=f[i].get();
255 }catch(std::exception& ex){
256 {
257 std::lock_guard<std::mutex> lock(mtx);
258 std::cout<<"exception in proc("<<i<<"): "<<ex.what()<<std::endl;
259 }
260 }
261 }
262 Eigen::Tensor<double,6> rangeTable_=Eigen::TensorMap<Eigen::Tensor<double,6>>(returned.data(),nvs,nhs,nvt,nht,nobs,naa);
263 for(int i=0;i<numProcess;++i){
264 th[i].join();
265 }
266 std::cout<<"exporting to npz file"<<std::endl;
267 auto np=py::module_::import("numpy");
268 np.attr("savez")(dstPath,py::arg("vs")=vs,py::arg("hs")=hs,py::arg("vt")=vt,py::arg("ht")=ht,py::arg("obs")=obs,py::arg("aa")=aa,py::arg("ranges")=rangeTable_);
269 std::cout<<"makeRangeTable done."<<std::endl;
270}
