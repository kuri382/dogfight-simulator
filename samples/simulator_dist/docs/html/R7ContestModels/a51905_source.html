110double R7ContestMassPointFighter::getThrust(){
111 auto [accelMin,accelMax] = getAccelRange(motion);
112 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
113 double pCmd=ctrl.at("pCmd");
114
115 // 重力の影響を反映
116 double V=motion.vel.norm();
117 Eigen::Vector3d vn=motion.vel()/V;
118 Eigen::Vector3d gravityAxis=motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false);
119 double gOfs=vn.dot(gravityAxis)*gravity;
120
121 return m*(accelMin+pCmd*(accelMax-accelMin)+gOfs);
122}

124void R7ContestMassPointFighter::calcMotion(double dt){
125 motion_prev=motion;
126 double V=motion.vel.norm();
127 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
128 Eigen::Vector3d omegaB=ctrl.at("omegaB");
129 auto extra=getExtraDynamicsState();
130 auto [vMin,vMax]=getVelocityRange(motion,extra);
131 auto [rollMin,rollMax]=getRollRateRange(motion,extra);
132 auto [pitchMin,pitchMax]=getPitchRateRange(motion,extra);
133 auto [yawMin,yawMax]=getYawRateRange(motion,extra);
134 omegaB(0)=std::clamp(omegaB(0),rollMin,rollMax);
135 omegaB(1)=std::clamp(omegaB(1),pitchMin,pitchMax);
136 omegaB(2)=std::clamp(omegaB(2),yawMin,yawMax);
137 double accel=getThrust()/m;
138 motion.setOmega(relBtoP(omegaB));
139 double theta=motion.omega.norm()*dt;
140 if(theta>1e-6){
141 Eigen::Vector3d ax=motion.omega.normalized()();
142 Quaternion dq=Quaternion::fromAngle(ax,theta);
143 motion.setQ(dq*motion.q);
144 }else{
145 Eigen::VectorXd dq=motion.q.dqdwi()*motion.omega()*dt;
146 motion.setQ((motion.q+Quaternion(dq)).normalized());
147 }
148 double Vaft=std::max(vMin,std::min(vMax,V+accel*dt));
149 motion.setVel(relBtoP(Eigen::Vector3d(Vaft,0,0)));
150 motion.setPos(motion.pos()+motion.vel()*dt);
151}

166double R7ContestMassPointFighter::getTwoDimTableValue(const std::string& key,const MotionState& m,const nl::json& extraArgs){
167 double alt=m.getHeight();
168 double V=m.vel.norm();
169 int alt_i=0;
170 Eigen::VectorXd vs;
171 Eigen::MatrixXd arg(1,1);
172 double vMin,vMax;
173 Eigen::DSizes<Eigen::Index, 2> startIndices{0,0};
174 Eigen::DSizes<Eigen::Index, 2> sliceSizes=twoDimTables[key].dimensions();
175 Eigen::DSizes<Eigen::Index, 1> squeezedSizes{twoDimTables[key].dimensions()[1]};
176 sliceSizes[0]=1;
177 if(alt<altTable(0) || alt>=altTable(altTable.size()-1)){
178 if(alt<altTable(0)){
179 alt_i=0;
180 }else{
181 alt_i=altTable.size()-1;
182 }
183 vMin=vMinTable(alt_i)*scaleFactor["vMin"];
184 vMax=vMaxTable(alt_i)*scaleFactor["vMax"];
185 if(vMin==vMax){
186 return twoDimTables[key](alt_i,0)*scaleFactor[key];
187 }
188 vs=Eigen::VectorXd::LinSpaced(twoDimTables[key].dimensions()[1],vMin,vMax);
189 arg<<std::clamp(V,vMin,vMax);
190 startIndices[0]=alt_i;
191 return interpn({vs},twoDimTables[key].slice(startIndices,sliceSizes).reshape(squeezedSizes),arg)(0);
192 }else{
193 for(alt_i=0;alt_i<altTable.size()-1;++alt_i){
194 if(altTable(alt_i)<=alt && alt<altTable(alt_i+1)){
195 double ret1,ret2;
196 vMin=vMinTable(alt_i)*scaleFactor["vMin"];
197 vMax=vMaxTable(alt_i)*scaleFactor["vMax"];
198 if(V<=vMin){
199 ret1=twoDimTables[key](alt_i,0);
200 }else if(V>=vMax){
201 ret1=twoDimTables[key](alt_i,altTable.size()-1);
202 }else{
203 vs=Eigen::VectorXd::LinSpaced(twoDimTables[key].dimensions()[1],vMin,vMax);
204 arg<<std::clamp(V,vMin,vMax);
205 startIndices[0]=alt_i;
206 ret1=interpn({vs},twoDimTables[key].slice(startIndices,sliceSizes).reshape(squeezedSizes),arg)(0);
207 }
208 vMin=vMinTable(alt_i+1)*scaleFactor["vMin"];
209 vMax=vMaxTable(alt_i+1)*scaleFactor["vMax"];
210 if(V<=vMin){
211 ret2=twoDimTables[key](alt_i+1,0);
212 }else if(V>=vMax){
213 ret2=twoDimTables[key](alt_i+1,altTable.size()-1);
214 }else{
215 vs=Eigen::VectorXd::LinSpaced(twoDimTables[key].dimensions()[1],vMin,vMax);
216 arg<<std::clamp(V,vMin,vMax);
217 startIndices[0]=alt_i+1;
218 ret2=interpn({vs},twoDimTables[key].slice(startIndices,sliceSizes).reshape(squeezedSizes),arg)(0);
219 }
220 return (ret1+(ret2-ret1)*(alt-altTable(alt_i))/(altTable(alt_i+1)-altTable(alt_i)))*scaleFactor[key];
221 }
222 }
223 throw std::runtime_error("Something went wrong.");
224 }

240void R7ContestMassPointFighter::loadTableFromJsonObject(const nl::json& j_table){
241 //
242 bool altitude_table;
243 if(j_table.contains("altitude")){
244 const nl::json& j=j_table.at("altitude");
245 if(j.is_array()){
246 j.get_to(altTable);
247 assert(altTable.size()>1);
248 }else{
249 throw std::runtime_error("The value of 'altitude' should be 1-dim array.");
250 }
251 }else{
252 altTable=Eigen::Vector2d(0.0,20000.0);
253 }
254 if(j_table.contains("vMin")){
255 const nl::json& j=j_table.at("vMin");
256 if(j.is_array()){
257 j.get_to(vMinTable);
258 if(vMinTable.size()==1){
259 vMinTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(vMinTable(0));
260 }
261 assert(vMinTable.size()==altTable.size());
262 }else if(j.is_number()){
263 vMinTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(j.get<double>());
264 }else{
265 throw std::runtime_error("The value of 'vMin' should be 1-dim array or number.");
266 }
267 }else{
268 vMinTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(150.0);
269 }
270 if(j_table.contains("vMax")){
271 const nl::json& j=j_table.at("vMax");
272 if(j.is_array()){
273 j.get_to(vMaxTable);
274 if(vMaxTable.size()==1){
275 vMaxTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(vMaxTable(0));
276 }
277 assert(vMaxTable.size()==altTable.size());
278 }else if(j.is_number()){
279 vMaxTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(j.get<double>());
280 }else{
281 throw std::runtime_error("The value of 'vMax' should be 1-dim array or number.");
282 }
283 }else{
284 vMaxTable=Eigen::Tensor<double,1>(altTable.size()).setConstant(450.0);
285 }
286 std::vector<std::string> keys{
287 "accelMin",
288 "accelMax",
289 "rollMin",
290 "rollMax",
291 "pitchMin",
292 "pitchMax",
293 "yawMin",
294 "yawMax"
295 };
296 std::map<std::string,double> defaultValues{
297 {"accelMin",-gravity},
298 {"accelMax",gravity},
299 {"rollMin",-deg2rad(180.0)},
300 {"rollMax",deg2rad(180.0)},
301 {"pitchMin",-deg2rad(30.0)},
302 {"pitchMax",deg2rad(30.0)},
303 {"yawMin",-deg2rad(30.0)},
304 {"yawMax",deg2rad(30.0)}
305 };
306 int nvs=-1;
307 for(auto&& key : keys){
308 if(j_table.contains(key)){
309 const nl::json& j=j_table.at(key);
310 if(j.is_array()){
311 assert(j.size()>0);
312 if(j[0].is_array()){
313 // 2-dim array
314 j.get_to(twoDimTables[key]);
315 }else if(j[0].is_number()){
316 // 1-dim array
317 Eigen::Tensor<double,1> tmp=j;
318 if(tmp.size()==1){
319 if(nvs<0){
320 nvs=2;
321 }
322 tmp=Eigen::Tensor<double,1>(nvs).setConstant(tmp(0));
323 }else{
324 if(nvs<0){
325 nvs=tmp.size();
326 }
327 }
328 twoDimTables[key]=tmp.reshape(Eigen::DSizes<Eigen::Index,2>{1,nvs});
329 }else{
330 throw std::runtime_error("The value of '"+key+"' should be 2-dim array, 1-dim array or number.");
331 }
332 if(twoDimTables[key].size()==1){
333 if(nvs<0){
334 nvs=2;
335 }
336 twoDimTables[key]=Eigen::Tensor<double,2>(altTable.size(),nvs).setConstant(defaultValues[key]);
337 }else if(twoDimTables[key].dimension(0)==1){
338 if(nvs<0){
339 nvs=twoDimTables[key].dimension(1);
340 }
341 Eigen::Tensor<double,2> tmp=twoDimTables[key];
342 twoDimTables[key]=Eigen::Tensor<double,2>(altTable.size(),nvs);
343 for(int i=0;i<altTable.size();++i){
344 twoDimTables[key].slice(Eigen::DSizes<Eigen::Index,2>{i,0},tmp.dimensions())=tmp;
345 }
346 }else if(twoDimTables[key].dimension(1)==1){
347 if(nvs<0){
348 nvs=2;
349 }
350 Eigen::Tensor<double,2> tmp=twoDimTables[key];
351 twoDimTables[key]=Eigen::Tensor<double,2>(tmp.dimension(0),nvs);
352 for(int i=0;i<nvs;++i){
353 twoDimTables[key].slice(Eigen::DSizes<Eigen::Index,2>{0,i},tmp.dimensions())=tmp;
354 }
355 }
356 assert(twoDimTables[key].dimension(0)==altTable.size() && twoDimTables[key].dimension(1)==nvs);
357 }else if(j.is_number()){
358 if(nvs<0){
359 nvs=2;
360 }
361 twoDimTables[key]=Eigen::Tensor<double,2>(altTable.size(),nvs).setConstant(j.get<double>());
362 }else{
363 throw std::runtime_error("The value of '"+key+"' should be 2-dim array, 1-dim array or number.");
364 }
365 }else{
366 if(nvs<0){
367 nvs=2;
368 }
369 twoDimTables[key]=Eigen::Tensor<double,2>(altTable.size(),nvs).setConstant(defaultValues[key]);
370 }
371 }

378nl::json R7ContestMassPointFlightController::calcDirect(const nl::json &cmd){
379 auto p=getShared<R7ContestMassPointFighter>(parent);
380 auto extra=p->getExtraDynamicsState();
381 auto [rollMin,rollMax]=p->getRollRateRange(p->motion,extra);
382 auto [pitchMin,pitchMax]=p->getPitchRateRange(p->motion,extra);
383 auto [yawMin,yawMax]=p->getYawRateRange(p->motion,extra);
384 auto roll=std::clamp(cmd.at("roll").get<double>(),-1.,1.);
385 if(roll>=0){
386 if(rollMax>0){
387 roll=std::max(rollMin,roll*abs(rollMax));
388 }else{
389 roll=rollMax;
390 }
391 }else{
392 if(rollMin<0){
393 roll=std::min(rollMax,roll*abs(rollMin));
394 }else{
395 roll=rollMin;
396 }
397 }
398 auto pitch=std::clamp(cmd.at("pitch").get<double>(),-1.,1.);
399 if(pitch>=0){
400 if(pitchMax>0){
401 pitch=std::max(pitchMin,pitch*abs(pitchMax));
402 }else{
403 pitch=pitchMax;
404 }
405 }else{
406 if(pitchMin<0){
407 pitch=std::min(pitchMax,pitch*abs(pitchMin));
408 }else{
409 pitch=pitchMin;
410 }
411 }
412 auto yaw=std::clamp(cmd.at("yaw").get<double>(),-1.,1.);
413 if(yaw>=0){
414 if(yawMax>0){
415 yaw=std::max(yawMin,yaw*abs(yawMax));
416 }else{
417 yaw=yawMax;
418 }
419 }else{
420 if(yawMin<0){
421 yaw=std::min(yawMax,yaw*abs(yawMin));
422 }else{
423 yaw=yawMin;
424 }
425 }
426 double pCmd;
427 if(cmd.contains("throttle")){//0〜1
428 pCmd=std::clamp(cmd.at("throttle").get<double>(),0.,1.);
429 }else if(cmd.contains("accel")){//-1〜+1
430 double accel=std::clamp(cmd.at("accel").get<double>(),-1.,1.);
431 auto [accelMin,accelMax]=p->getRollRateRange(p->motion,extra);
432 if(accel>=0){
433 if(accelMax>0){
434 accel=std::max(accelMin,accel*abs(accelMax));
435 }else{
436 accel=accelMax;
437 }
438 }else{
439 if(accelMin<0){
440 accel=std::min(accelMax,accel*abs(accelMin));
441 }else{
442 accel=accelMin;
443 }
444 }
445 pCmd=(accel-accelMin)/(accelMax-accelMin);
446 }else{
447 throw std::runtime_error("Either 'throttle' or 'accel' must be in cmd keys.");
448 }
449 return {
450 {"omegaB",Eigen::Vector3d(roll,pitch,yaw)},
451 {"pCmd",pCmd}
452 };

454nl::json R7ContestMassPointFlightController::calcFromDirAndVel(const nl::json &cmd){
455 auto p=getShared<R7ContestMassPointFighter>(parent);
456 auto extra=p->getExtraDynamicsState();
457 auto [accelMin,accelMax]=p->getAccelRange(p->motion,extra);
458 double pCmd;
459 if(cmd.contains("dstV")){
460 double V=p->motion.vel.norm();
461 double dstV=cmd.at("dstV");
462 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
463 double dstAccel=(dstV-V)/dt;
464 pCmd=std::clamp((dstAccel-accelMin)/(accelMax-accelMin),0.0,1.0);
465 }else if(cmd.contains("dstAccel")){
466 double dstAccel=cmd.at("dstAccel");
467 pCmd=std::clamp((dstAccel-accelMin)/(accelMax-accelMin),0.0,1.0);
468 }else if(cmd.contains("dstThrust")){
469 double dstT=cmd.at("dstThrust");
470 double Tmin=accelMin*p->m;
471 double Tmax=accelMax*p->m;
472 pCmd=std::clamp((dstT-Tmin)/(Tmax-Tmin),0.0,1.0);
473 }else if(cmd.contains("dstThrottle")){
474 double dstT=cmd.at("dstThrottle");
475 pCmd=std::clamp(dstT,0.0,1.0);
476 }else{
477 throw std::runtime_error("Only one of dstV, dstAccel, dstThrust or dstThrottle is acceptable.");
478 }
479
480 Eigen::Vector3d omegaB;
481 if(cmd.contains("dstDir")){
482 Eigen::Vector3d dstDir=cmd.at("dstDir");
483 if(cmd.contains("dstAlt")){
484 dstDir=altitudeKeeper(p->motion,dstDir,cmd.at("dstAlt").get<double>());
485 }
486 double V=p->motion.vel.norm();
487 Eigen::Vector3d currentDir=p->motion.vel()/V;
488 dstDir.normalize();
489 Eigen::Vector3d rotationAxis=currentDir.cross(dstDir);
490 auto ex=p->dirBtoP(Eigen::Vector3d(1,0,0));
491 auto ey=p->dirBtoP(Eigen::Vector3d(0,1,0));
492 auto ez=p->dirBtoP(Eigen::Vector3d(0,1,1));
493 auto down=p->motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false);
494 auto horizontalY=down.cross(ex);
495 double snTheta=std::min(1.0,rotationAxis.norm());
496 double csTheta=std::clamp(currentDir.dot(dstDir),-1.0,1.0);
497 double eTheta=atan2(snTheta,csTheta);
498 double eps=1e-4;
499 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
500 if(snTheta<eps){
501 //Almost no rotation or 180deg rotation
502 rotationAxis=currentDir.cross(down.cross(currentDir)).normalized();
503 }else{
504 rotationAxis/=snTheta;
505 }
506 Quaternion dq=Quaternion::fromAngle(rotationAxis,eTheta);
507 Quaternion q1=dq*p->motion.q;
508 ex=q1.transformVector(Eigen::Vector3d(1,0,0));// == dstDir
509 ey=q1.transformVector(Eigen::Vector3d(0,1,0));
510 double sinRoll=ex.dot(ey.cross(rotationAxis));
511 double cosRoll=ey.dot(rotationAxis);
512 double roll=atan2(sinRoll,cosRoll);
513 if(roll>M_PI_2){
514 roll=M_PI-roll;
515 }else if(roll<-M_PI_2){
516 roll=M_PI+roll;
517 }
518 auto [rollMin,rollMax]=p->getRollRateRange(p->motion,extra);
519 roll=std::clamp(roll,rollMin,rollMax);
520 dq=Quaternion::fromAngle(ex,roll)*dq;
521 if(dq.w()<0){
522 dq=-dq;
523 }
524 double cosRot=dq.w();
525 double sinRot=dq.vec().norm();
526 Eigen::Vector3d ax=dq.vec().normalized();
527 omegaB=p->relPtoB(ax*atan2(sinRot,cosRot)*2/dt);
528 auto [pitchMin,pitchMax]=p->getPitchRateRange(p->motion,extra);
529 auto [yawMin,yawMax]=p->getYawRateRange(p->motion,extra);
530 omegaB*=std::min<double>({
531 1.0,
532 omegaB(0)==0 ? 1.0 : abs((omegaB(0)>0 ? rollMax : rollMin)/omegaB(0)),
533 omegaB(1)==0 ? 1.0 : abs((omegaB(1)>0 ? pitchMax : pitchMin)/omegaB(1)),
534 omegaB(2)==0 ? 1.0 : abs((omegaB(2)>0 ? yawMax : yawMin)/omegaB(2)),
535 });
536 }else if(cmd.contains("dstTurnRate")){
537 omegaB=cmd.at("dstTurnRate");
538 throw std::runtime_error("Only one of dstDir, or dstTurnRate is acceptable.");
539 }
540
541 return {
542 {"omegaB",omegaB},
543 {"pCmd",pCmd}
544 };
