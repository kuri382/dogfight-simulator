72std::pair<Eigen::Vector3d,double> CapsuleCoursePatrolAgent::getCoursePoint(const double& phase) const{
73 Eigen::Vector3d dp=p2-p1;
74 double distance=sqrt(dp(0)*dp(0)+dp(1)*dp(1));
75 Eigen::Vector3d ex=dp.normalized();
76 Eigen::Vector3d ey=Eigen::Vector3d(0,0,1).cross(ex).normalized();
77 double phase2rad=course_length/radius;
78
79 std::vector<double> anchors = {
80 (radius*M_PI_2)/course_length,
81 (radius*M_PI_2+distance)/course_length,
82 (radius*M_PI_2*3+distance)/course_length,
83 (radius*M_PI_2*3+2*distance)/course_length
84 };
85 if(phase<anchors[0]){
86 double angle=phase*phase2rad;
87 return {
88 p1+radius*(-ex*cos(angle)-ey*sin(angle)),
89 rad2deg(angle)
90 };
91 }else if(phase<anchors[1]){
92 return {
93 p1+(ex*(phase-anchors[0])*course_length-ey*radius),
94 M_PI
95 };
96 }else if(phase<anchors[2]){
97 double angle=(phase-anchors[1])*phase2rad;
98 return {
99 p2+radius*(ex*sin(angle)-ey*cos(angle)),
100 M_PI_2+rad2deg(angle)
101 };
102 }else if(phase<anchors[3]){
103 return {
104 p2+(-ex*(phase-anchors[2])*course_length+ey*radius),
105 M_PI
106 };
107 }else{
108 double angle=(1.0-phase)*phase2rad;
109 return {
110 p1+radius*(-ex*cos(angle)+ey*sin(angle)),
111 -rad2deg(angle)
112 };
113 }

115double CapsuleCoursePatrolAgent::getCoursePhase(const Eigen::Vector3d& pos) const{
116 Eigen::Vector3d dp=p2-p1;
117 double distance=sqrt(dp(0)*dp(0)+dp(1)*dp(1));
118 Eigen::Vector3d ex=dp.normalized();
119 Eigen::Vector3d ey=Eigen::Vector3d(0,0,1).cross(ex).normalized();
120 double rad2phase=radius/course_length;
121
122 std::vector<double> anchors = {
123 (radius*M_PI_2)/course_length,
124 (radius*M_PI_2+distance)/course_length,
125 (radius*M_PI_2*3+distance)/course_length,
126 (radius*M_PI_2*3+2*distance)/course_length
127 };
128
129 double x=ex.dot(pos-p1);
130 double y=ey.dot(pos-p1);
131 if(x<0){
132 double angle=atan2(-y,-x);
133 return angle*rad2phase;
134 }else if(x>distance){
135 double angle=atan2(x-distance,-y);
136 return anchors[1]+angle*rad2phase;
137 }else{
138 if(y<0){
139 return anchors[0]+x/course_length;
140 }else{
141 return anchors[2]+(distance-x)/course_length;
142 }
143 }

146void CapsuleCoursePatrolAgent::control(){
147 MotionState originalMyMotion(parent->observables.at("motion"));
148 MotionState myMotion=originalMyMotion.transformTo(getLocalCRS());
149 Eigen::Vector3d pos=myMotion.pos();
150 Eigen::Vector3d vel=myMotion.vel();
151
152 Eigen::Vector3d dstDir;
153 if(chkMWS()){
154 std::vector<std::pair<double,Eigen::Vector3d>> tmp;
155 for(auto& m:parent->observables.at("/sensor/mws/track"_json_pointer)){
156 Eigen::Vector3d dir=m.get<Track2D>().dir(getLocalCRS());
157 tmp.push_back(std::make_pair(-dir.dot(myMotion.dirBtoP(Eigen::Vector3d(1,0,0))),-dir));
158 }
159 Eigen::Vector3d dr=(*std::min_element(tmp.begin(),tmp.end(),
160 [](const std::pair<double,Eigen::Vector3d> &lhs,const std::pair<double,Eigen::Vector3d> &rhs){
161 return lhs.first<rhs.first;
162 })).second;
163 dstDir=dr;
164 dstDir(2)=0;
165 dstDir.normalize();
166 double dAz=atan2(dstDir(1),dstDir(0))-myMotion.getAZ();
167 while(dAz>M_PI){dAz-=2*M_PI;}
168 while(dAz<=-M_PI){dAz+=2*M_PI;}
169 if(dAz > deltaAZLimit){
170 dstDir << cos(myMotion.getAZ()+deltaAZLimit), sin(myMotion.getAZ()+deltaAZLimit), 0;
171 }else if(dAz < -deltaAZLimit){
172 dstDir << cos(myMotion.getAZ()-deltaAZLimit), sin(myMotion.getAZ()-deltaAZLimit), 0;
173 }
174 altitudeKeeper.minPitch=minPitchForEvasion;
175 altitudeKeeper.maxPitch=maxPitchForEvasion;
176 dstDir=altitudeKeeper(myMotion,dstDir,minAltForEvasion);
177 }else{
178 double currentPhase=getCoursePhase(pos);
179 double dt=interval[SimPhase::CONTROL]*manager->getBaseTimeStep()*10;
180 double delta = velocity*dt/course_length;
181 auto [targetPoint, targetHeading] = getCoursePoint(currentPhase+delta);
182 dstDir=targetPoint-pos;
183 dstDir(2)=0;
184 dstDir.normalize();
185 double dAz=atan2(dstDir(1),dstDir(0))-myMotion.getAZ();
186 while(dAz>M_PI){dAz-=2*M_PI;}
187 while(dAz<=-M_PI){dAz+=2*M_PI;}
188 if(dAz > deltaAZLimit){
189 dstDir << cos(myMotion.getAZ()+deltaAZLimit), sin(myMotion.getAZ()+deltaAZLimit), 0;
190 }else if(dAz < -deltaAZLimit){
191 dstDir << cos(myMotion.getAZ()-deltaAZLimit), sin(myMotion.getAZ()-deltaAZLimit), 0;
192 }
193 altitudeKeeper.minPitch=minPitchForNormal;
194 altitudeKeeper.maxPitch=maxPitchForNormal;
195 dstDir=altitudeKeeper(myMotion,dstDir,-targetPoint(2));
196 }
197
198 commands[parent->getFullName()]={
199 {"motion",{
200 {"dstDir",originalMyMotion.dirAtoP(dstDir,myMotion.pos(),getLocalCRS())}//元のparent座標系に戻す
201 }},
202 {"weapon",{
203 {"launch",false},
204 {"target",Track3D()}
205 }}
206 };
207 commands[parent->getFullName()]["motion"]["dstV"]=velocity;
208}
