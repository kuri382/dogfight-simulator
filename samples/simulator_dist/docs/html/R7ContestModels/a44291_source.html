13void PlanarFighter::calcMotion(double dt){
14 motion_prev=motion;
15 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
16 double yaw=ctrl.at("yaw");
17 auto extra=getExtraDynamicsState();
18 auto [vMin,vMax]=getVelocityRange(motion,extra);
19 auto [pitchMin,pitchMax]=getPitchRateRange(motion,extra);
20 auto [yawMin,yawMax]=getYawRateRange(motion,extra);
21 yawMin=std::min(pitchMin,yawMin);
22 yawMax=std::max(pitchMax,yawMax);
23 yaw=std::clamp(yaw,yawMin,yawMax);
24 double accel=getThrust()/m;
25 Eigen::Vector3d omega=motion.relHtoP(Eigen::Vector3d(0,0,yaw),"NED",false);
26 auto qToParentFromNED=motion.getCRS()->getEquivalentQuaternionToTopocentricCRS(motion.pos(),motion.time,"NED").conjugate();
27 double newAZ=motion.getAZ()+yaw*dt;
28 double V=motion.vel.norm();
29 double wt=abs(yaw*dt);
30 if(wt<1e-8){//ほぼ無回転
31 auto accelVector=motion.relBtoP(Eigen::Vector3d(accel,0,0));
32 motion.setPos(motion.pos()+motion.vel()*dt+accelVector*dt*dt/2.);
33 auto posH=motion.absPtoH(motion.pos(),"NED",true);
34 posH(2)=-motion_prev.getHeight();
35 motion.setPos(motion.absHtoP(posH,"NED",true));
36 Eigen::Vector3d vn=motion.vel().normalized();
37 double Vaft=std::max(vMin,std::min(vMax,V+accel*dt));
38 motion.setVel(Vaft*vn);
39 auto velH=motion.velPtoH(motion.vel(),motion.pos(),"NED",true);
40 velH(2)=0;
41 motion.setVel(motion.velHtoP(velH,motion.pos(),"NED",true));
42 }else{
43 Eigen::Vector3d ex0=motion.vel()/V;
44 Eigen::Vector3d ey0=omega.cross(ex0).normalized();
45
46 Eigen::Vector3d dr=V*dt*(ex0*sinc(wt)+ey0*oneMinusCos_x2(wt)*wt)+accel*dt*dt*(ex0*(sinc(wt)-oneMinusCos_x2(wt))+ey0*wt*(sincMinusOne_x2(wt)+oneMinusCos_x2(wt)));
47 motion.setPos(motion.pos()+dr);
48 auto posH=motion.absPtoH(motion.pos(),"NED",true);
49 posH(2)=-motion_prev.getHeight();
50 motion.setPos(motion.absHtoP(posH,"NED",true));
51 Eigen::Vector3d vn=(ex0*cos(wt)+ey0*sin(wt)).normalized();
52 double Vaft=std::max(vMin,std::min(vMax,V+accel*dt));
53 motion.setVel(Vaft*vn);
54 auto velH=motion.velPtoH(motion.vel(),motion.pos(),"NED",true);
55 velH(2)=0;
56 motion.setVel(motion.velHtoP(velH,motion.pos(),"NED",true));
57 }
58 motion.setOmega(omega);
59 motion.setQ(
60 qToParentFromNED* //[parent <- NED]
61 Quaternion::fromAngle(Eigen::Vector3d(0,0,1),newAZ)* //[NED <- FSD(body)]
62 getAxisSwapQuaternion(bodyAxisOrder,"FSD") // [FSD(body) <- this]
63 );
64}

79nl::json PlanarFlightController::calcDirect(const nl::json &cmd){
80 auto p=getShared<PlanarFighter>(parent);
81 auto extra=p->getExtraDynamicsState();
82 auto [pitchMin,pitchMax]=p->getPitchRateRange(p->motion,extra);
83 auto [yawMin,yawMax]=p->getYawRateRange(p->motion,extra);
84 auto yaw=std::clamp(cmd.at("yaw").get<double>(),-1.,1.);
85 yawMin=std::min(pitchMin,yawMin);
86 yawMax=std::max(pitchMax,yawMax);
87 if(yaw>=0){
88 if(yawMax>0){
89 yaw=std::max(yawMin,yaw*abs(yawMax));
90 }else{
91 yaw=yawMax;
92 }
93 }else{
94 if(yawMin<0){
95 yaw=std::min(yawMax,yaw*abs(yawMin));
96 }else{
97 yaw=yawMin;
98 }
99 }
100 double pCmd;
101 if(cmd.contains("throttle")){//0〜1
102 pCmd=std::clamp(cmd.at("throttle").get<double>(),0.,1.);
103 }else if(cmd.contains("accel")){//-1〜+1
104 double accel=std::clamp(cmd.at("accel").get<double>(),-1.,1.);
105 auto [accelMin,accelMax]=p->getAccelRange(p->motion,extra);
106 if(accel>=0){
107 if(accelMax>0){
108 accel=std::max(accelMin,accel*abs(accelMax));
109 }else{
110 accel=accelMax;
111 }
112 }else{
113 if(accelMin<0){
114 accel=std::min(accelMax,accel*abs(accelMin));
115 }else{
116 accel=accelMin;
117 }
118 }
119 pCmd=(accel-accelMin)/(accelMax-accelMin);
120 }else{
121 throw std::runtime_error("Either 'throttle' or 'accel' must be in cmd keys.");
122 }
123 return {
124 {"yaw",0.0},
125 {"pCmd",pCmd}
126 };

128nl::json PlanarFlightController::calcFromDirAndVel(const nl::json &cmd){
129 auto p=getShared<PlanarFighter>(parent);
130 auto extra=p->getExtraDynamicsState();
131 auto [accelMin,accelMax]=p->getAccelRange(p->motion,extra);
132 double pCmd;
133 if(cmd.contains("dstV")){
134 double V=p->motion.vel.norm();
135 double dstV=cmd.at("dstV");
136 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
137 double dstAccel=(dstV-V)/dt;
138 pCmd=std::clamp((dstAccel-accelMin)/(accelMax-accelMin),0.0,1.0);
139 }else if(cmd.contains("dstAccel")){
140 double dstAccel=cmd.at("dstAccel");
141 pCmd=std::clamp((dstAccel-accelMin)/(accelMax-accelMin),0.0,1.0);
142 }else if(cmd.contains("dstThrust")){
143 double dstT=cmd.at("dstThrust");
144 double Tmin=accelMin*p->m;
145 double Tmax=accelMax*p->m;
146 pCmd=std::clamp((dstT-Tmin)/(Tmax-Tmin),0.0,1.0);
147 }else if(cmd.contains("dstThrottle")){
148 double dstT=cmd.at("dstThrottle");
149 pCmd=std::clamp(dstT,0.0,1.0);
150 }else{
151 throw std::runtime_error("Only one of dstV, dstAccel, dstThrust or dstThrottle is acceptable.");
152 }
153
154 double yaw;
155 if(cmd.contains("dstDir")){
156 Eigen::Vector3d dstDir=cmd.at("dstDir");
157 auto dstDirH=p->motion.dirPtoH(dstDir,p->motion.pos(),"NED",true);
158 dstDirH(2)=0;
159 dstDir=p->motion.dirHtoP(dstDirH,p->motion.pos(),"NED",true);
160 double V=p->motion.vel.norm();
161 Eigen::Vector3d currentDir=p->motion.vel()/V;
162 Eigen::Vector3d rotationAxis=currentDir.cross(dstDir);
163 double snTheta=std::min(1.0,rotationAxis.norm());
164 double csTheta=std::clamp(currentDir.dot(dstDir),-1.0,1.0);
165 double eTheta=atan2(snTheta,csTheta);
166 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
167 auto [pitchMin,pitchMax]=p->getPitchRateRange(p->motion,extra);
168 auto [yawMin,yawMax]=p->getYawRateRange(p->motion,extra);
169 yawMin=std::min(pitchMin,yawMin);
170 yawMax=std::max(pitchMax,yawMax);
171 if(rotationAxis(2)>=0){
172 yaw=std::clamp(eTheta/dt,yawMin,yawMax);
173 }else{
174 yaw=std::clamp(-eTheta/dt,yawMin,yawMax);
175 }
176 }else if(cmd.contains("dstTurnRate")){
177 if(cmd.at("dstTurnRate").is_number()){
178 yaw=cmd.at("dstTurnRate").get<double>();
179 }else{
180 yaw=cmd.at("dstTurnRate").get<Eigen::Vector3d>()(2);
181 }
182 }else{
183 throw std::runtime_error("Only one of dstDir, or dstTurnRate is acceptable.");
184 }
185 return {
186 {"yaw",yaw},
187 {"pCmd",pCmd}
188 };
