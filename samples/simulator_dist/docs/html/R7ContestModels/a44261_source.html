13def getForEscort(phase):
14 #護衛対象機の初期位置を計算する
15 p1=np.array([60000.0,100000.0,-10000.0])
16 p2=np.array([-60000.0,100000.0,-10000.0])
17 dp=p2-p1
18 radius=5000.0
19 distance=np.sqrt(dp[0]**2+dp[1]**2)
20 ex=dp/np.linalg.norm(dp)
21 ey=np.cross(np.array([0.,0.,1.]),ex)
22 ey/=np.linalg.norm(ey)
23
24 length=2*(radius*np.pi+distance)
25 phase2rad=length/radius
26 anchors=[
27 (radius*np.pi/2.)/length,
28 (radius*np.pi/2.+distance)/length,
29 (radius*np.pi*3/2.+distance)/length,
30 (radius*np.pi*3/2.+2*distance)/length,
31 ]
32 if phase < anchors[0]:
33 angle=phase*phase2rad
34 pos=p1+radius*(-ex*np.cos(angle)-ey*np.sin(angle))
35 heading=angle/np.pi*180
36 elif phase < anchors[1]:
37 pos=p1+(ex*(phase-anchors[0])*length-ey*radius)
38 heading=180.0
39 elif phase < anchors[2]:
40 angle=(phase-anchors[1])*phase2rad
41 pos=p2+radius*(ex*np.sin(angle)-ey*np.cos(angle))
42 heading=90.0+angle/np.pi*180
43 elif phase < anchors[3]:
44 pos=p2+(-ex*(phase-anchors[2])*length+ey*radius)
45 heading=0.0
46 else:
47 angle=(1.0-phase)*phase2rad
48 pos=p1+radius*(-ex*np.cos(angle)+ey*np.sin(angle))
49 heading=-angle/np.pi*180
50 return {
51 "pos":pos,
52 "vel":300.0,
53 "heading": heading,
54 }
