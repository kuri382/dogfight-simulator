23 def get_torch_exploration_action(
24 self, action_dist: ActionDistribution, explore: bool
25 ):
26 if explore:
27 req = force_tuple(
28 action_dist.required_model_output_shape(
29 self.action_space, getattr(self.model, "model_config", None)
30 )
31 )
32
33 # TupleとDictは中身を再帰で辿る
34 # Boxと(Multi)Discreteに
35 def add_batch_dim_and_to_torch(s):
36 if isinstance(s, gym.spaces.Tuple):
37 return tuple(add_batch_dim_and_to_torch(v) for v in s.spaces)
38 elif isinstance(s, gym.spaces.Dict):
39 return {k: add_batch_dim_and_to_torch(v) for k, v in s.spaces.items()}
40 else:
41 # Add a batch dimension?
42 if len(action_dist.inputs.shape) == len(req) + 1:
43 batch_size = action_dist.inputs.shape[0]
44 a = np.stack([s.sample() for _ in range(batch_size)])
45 else:
46 a = s.sample()
47 a = torch.from_numpy(a).to(self.device)
48 return a
49 action = add_batch_dim_and_to_torch(self.action_space)
50 else:
51 action = action_dist.deterministic_sample()
52
53 # logpはtf_utils.pyのzero_logps_from_actionsと同じ方法で計算
54 action_component = tree.flatten(action)[0]
55 logp = torch.zeros_like(action_component, dtype=torch.float32)
56 while len(logp.shape) > 1:
57 logp = logp[:, 0]
58
59 return action, logp
