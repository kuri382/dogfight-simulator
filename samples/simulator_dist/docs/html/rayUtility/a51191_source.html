38class PostprocessRNNStates(PostprocessAdvantages):
39 """Add seq_lens and states into the offline trajectory data without RNN states.
40 """
41
42 def postprocess_trajectory(
43 self,
44 sample_batch: SampleBatch,
45 other_agent_batches: Optional[Dict[Any, SampleBatch]] = None,
46 episode=None,
47 ):
48 if self.is_recurrent() and not "state_in_0" in sample_batch:
49 count = sample_batch.count
50 seq_lens = []
51 while count > 0:
52 seq_lens.append(min(count, self.max_seq_len))
53 count -= self.max_seq_len
54 sample_batch["seq_lens"] = np.array(seq_lens)
55 sample_batch.max_seq_len = self.max_seq_len
56 dummy=self._get_dummy_batch_from_view_requirements(len(seq_lens))
57 num_states=len([k for k in dummy.keys() if k.startswith("state_in")])
58 for i in range(num_states):
59 sample_batch["state_in_{}".format(i)]=dummy["state_in_{}".format(i)]
60
61 return super().postprocess_trajectory(
62 sample_batch, other_agent_batches, episode
63 )
