53 @override(TorchPolicyV2)
54 def loss(
55 self,
56 model: ModelV2,
57 dist_class: Type[TorchDistributionWrapper],
58 train_batch: SampleBatch,
59 ) -> Union[TensorType, List[TensorType]]:
60 model_out, _ = model(train_batch)
61 action_dist = dist_class(model_out, model)
62 actions = train_batch[SampleBatch.ACTIONS]
63 # log\pi_\theta(a|s)
64 logprobs = action_dist.logp(actions)
65
66 # Advantage estimation.
67 if self.config["beta"] != 0.0:
68 cumulative_rewards = train_batch[Postprocessing.ADVANTAGES]
69 state_values = model.value_function()
70 adv = cumulative_rewards - state_values
71 adv_squared_mean = torch.mean(torch.pow(adv, 2.0))
72
73 explained_var = explained_variance(cumulative_rewards, state_values)
74 ev = torch.mean(explained_var)
75 model.tower_stats["explained_variance"] = ev
76
77 # Policy loss.
78 # Update averaged advantage norm.
79 rate = self.config["moving_average_sqd_adv_norm_update_rate"]
80 self._moving_average_sqd_adv_norm = (
81 rate * (adv_squared_mean.detach() - self._moving_average_sqd_adv_norm)
82 + self._moving_average_sqd_adv_norm
83 )
84 model.tower_stats[
85 "_moving_average_sqd_adv_norm"
86 ] = self._moving_average_sqd_adv_norm
87 # Exponentially weighted advantages.
88 exp_advs = torch.exp(
89 self.config["beta"]
90 * (adv / (1e-8 + torch.pow(self._moving_average_sqd_adv_norm, 0.5)))
91 ).detach()
92 # Value loss.
93 v_loss = 0.5 * adv_squared_mean
94 else:
95 # Policy loss (simple BC loss term).
96 exp_advs = 1.0
97 # Value loss.
98 v_loss = 0.0
99 model.tower_stats["v_loss"] = v_loss
100 # logprob loss alone tends to push action distributions to
101 # have very low entropy, resulting in worse performance for
102 # unfamiliar situations.
103 # A scaled logstd loss term encourages stochasticity, thus
104 # alleviate the problem to some extent.
105 logstd_coeff = self.config["bc_logstd_coeff"]
106 if logstd_coeff > 0.0:
107 logstds = torch.mean(action_dist.log_std, dim=1)
108 else:
109 logstds = 0.0
110
111 p_loss = -torch.mean(exp_advs * (logprobs + logstd_coeff * logstds))
112 model.tower_stats["p_loss"] = p_loss
113 # Combine both losses.
114 self.v_loss = v_loss
115 self.p_loss = p_loss
116 total_loss = p_loss + self.config["vf_coeff"] * v_loss
117 model.tower_stats["total_loss"] = total_loss
118 return total_loss
