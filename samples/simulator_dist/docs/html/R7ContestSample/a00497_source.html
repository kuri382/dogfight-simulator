33void VirtualSimulatorSample::step(const std::shared_ptr<asrc::core::Agent>& agent, const std::vector<Track3D>& tracks){
34 auto now=agent->manager->getTickCount();
35 for(auto&& [port, parent] : agent->parents){
36 auto fullName=parent->getFullName();
37 if(parent->isAlive()){
38 MotionState motion(parent->observables.at("motion"));
39 auto crs=agent->manager->getRootCRS();
40 auto time=agent->manager->getTime();
41 Track3D selfTrack(
42 parent->getUUIDForTrack(),
43 crs,
44 time,
45 motion.pos(crs),
46 motion.vel(crs)
47 );
48 {
49 // 仮想誘導弾の諸元を更新
50 auto it=virtualMissiles[fullName].begin();
51 minimumDistances[fullName]=std::numeric_limits<double>::infinity();
52 bool hit=false;
53 while(it!=virtualMissiles[fullName].end()){
54 auto missile=*it;
55 assert(missile);
56 if(missile->isAlive()){
57 missile->communicationBuffers["MissileComm:"+missile->getFullName()].lock()->send({
58 {"target",selfTrack}
59 },CommunicationBuffer::UpdateType::MERGE);
60
61 double dt=agent->interval[SimPhase::AGENT_STEP]*agent->manager->getBaseTimeStep();
62 if(missile->hitCheck(motion.pos(missile->getParentCRS()),motion.pos(missile->getParentCRS())-motion.vel(missile->getParentCRS())*dt)){
63 // 命中
64 missile->manager->requestToKillAsset(missile);
65 hit=true;
66 }
67 minimumDistances[fullName]=std::min<double>(
68 minimumDistances[fullName],
69 (missile->pos()-motion.pos(missile->getParentCRS())).norm()
70 );
71 ++it;
72 }else{
73 // 飛翔終了していたら消す
74 it=virtualMissiles[fullName].erase(it);
75 }
76 }
77 //std::cout<<fullName<<" tick="<<agent->manager->getTickCount()<<": virtualMissiles size="<<virtualMissiles[fullName].size()<<", minimumDistance="<<minimumDistances[fullName]<<", hit="<<hit<<std::endl;
78 }
79 }else{
80 // parentが消えたら仮想誘導弾も消す
81 if(virtualMissiles.contains(fullName)){
82 for(auto&& missile : virtualMissiles[fullName]){
83 missile->manager->requestToKillAsset(missile);
84 }
85 virtualMissiles.erase(virtualMissiles.find(fullName));
86 minimumDistances.erase(minimumDistances.find(fullName));
87 }
88 }
89 }
90
91 // 仮想シミュレータの時間を進める
92 while(virtualSimulator->getTickCount()<now){
93 virtualSimulator->step(py::dict());
94 }
95
96 // 生存中のparentsに対して、新たな仮想誘導弾を生成する
97 for(auto&& [port, parent] : agent->parents){
98 auto fullName=parent->getFullName();
99 if(parent->isAlive()){
100 MotionState motion(parent->observables.at("motion"));
101 auto crs=agent->manager->getRootCRS();
102 auto time=agent->manager->getTime();
103 Track3D selfTrack(
104 parent->getUUIDForTrack(),
105 crs,
106 time,
107 motion.pos(crs),
108 motion.vel(crs)
109 );
110 for(auto&& t:tracks){
111 //std::cout<<"launch virtual missile at tick="<<manager->getTickCount()<<" (v "<<virtualSimulator->getTickCount()<<")"<<std::endl;
112 if(virtualMissiles[fullName].size()<maxNumPerParent){
113 MotionState shooterMotion(
114 crs,
115 time,
116 t.pos(crs),
117 t.vel(crs),
118 Eigen::Vector3d(0,0,0),
119 Quaternion(1,0,0,0),
120 "FSD"
121 );
122 if(launchCondition(parent,shooterMotion,selfTrack)){
123 nl::json launchCommand={
124 {"target",selfTrack},
125 {"motion",shooterMotion}
126 };
127 virtualMissiles[fullName].push_back(getShared<FighterAccessor>(parent)->launchInVirtual(virtualSimulator,launchCommand));
128 }
129 }
130 }
131 }
132 }
