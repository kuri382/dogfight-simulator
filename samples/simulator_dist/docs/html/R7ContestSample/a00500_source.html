101void R7ContestRewardSample01::onInnerStepEnd(){
102 nl::json track=nl::json::array();
103 auto ruler_=getShared<Ruler,Ruler>(manager->getRuler());
104 auto crs=ruler_->getLocalCRS();
105 for(auto&& team:target){
106 //(1)Biteへの加点、(2)メモリトラック落ちへの減点
107 int i=0;
108 for(auto&& m_:friendMsls[team]){
109 auto m=m_.lock();
110 if(m->hasLaunched && m->isAlive()){
111 if(m->mode==Missile::Mode::SELF && !biteFlag[team](i)){
112 reward[team]+=pBite;
113 biteFlag[team](i)=true;
114 }
115 if(m->mode==Missile::Mode::MEMORY && !memoryTrackFlag[team](i)){
116 reward[team]-=pMemT;
117 memoryTrackFlag[team](i)=true;
118 }
119 }
120 ++i;
121 }
122 //(3)敵探知への加点(生存中の敵の何%を探知できているか)(DL前提)
123 track=nl::json::array();
124 for(auto&& f_:friends[team]){
125 if(f_.lock()->isAlive()){
126 track=f_.lock()->observables.at("/sensor/track"_json_pointer);
127 break;
128 }
129 }
130 int numAlive=0;
131 int numTracked=0;
132 for(auto&& f_:enemies[team]){
133 auto f=f_.lock();
134 if(f->isAlive()){
135 numAlive+=1;
136 for(auto&& t_:track){
137 if(t_.get<Track3D>().isSame(f)){
138 numTracked+=1;
139 break;
140 }
141 }
142 }
143 }
144 if(numAlive>0){
145 reward[team]+=(1.0*numTracked/numAlive)*pDetect*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep();;
146 }
147 double ene=0;
148 std::vector<double> tmp;
149 tmp.push_back(-dLine);
150 for(auto&& f_:friends[team]){
151 auto f=f_.lock();
152 if(f->isAlive()){
153 //(4)過剰な機動への減点(角速度ノルムに対してL2、高度方向速度に対してL1正則化)
154 double vAlt=f->motion.relPtoH(f->vel(),"ENU")(2);
155 reward[team]+=(-pVel*abs(vAlt)-(f->omega().squaredNorm())*pOmega)*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep();
156 //(5)前進・後退への更なる加減点
157 tmp.push_back(forwardAx[team].dot(f->pos(crs).block<2,1>(0,0,2,1)));
158 }
159 //(6)保持している力学的エネルギー(回転を除く)の増減による加減点
160 //被撃墜・墜落後も状態量としては位置・速度等の値を保持しているためRewardからは参照可能(observableから消えるだけ)
161 ene+=f->vel().squaredNorm()/2+gravity*f->getHeight();
162 }
163 leadRange[team]=*std::max_element(tmp.begin(),tmp.end());
164 if(pLineAsPeak){
165 //最高到達点で前進の加点をする場合
166 if(leadRange[team]>leadRangePrev[team]){
167 reward[team]+=(leadRange[team]-leadRangePrev[team])*pLine;
168 leadRangePrev[team]=leadRange[team];
169 }
170 }else{
171 //都度前進・後退の加減点をする場合
172 reward[team]+=(leadRange[team]-leadRangePrev[team])*pLine;
173 leadRangePrev[team]=leadRange[team];
174 }
175 reward[team]+=(ene-totalEnergy[team])*pEnergy;
176 totalEnergy[team]=ene;
177 }
178}
