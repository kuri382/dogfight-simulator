11class GenericLayers(torch.nn.Module):
12 """jsonベースのconfigから様々なNNを生成するための基本クラス。
13 configは以下の形式とする。
14 {
15 "input_shape": list, #入力テンソルの形(バッチサイズを除く)
16 "layers": list[tuple[str,dict]], #各構成要素を生成するためのタプルのリスト。
17 #タプルの第1要素はクラス名。第2要素はコンストラクタ引数のうち、入力テンソルの形に関する引数を除いたもの。
18 }
19 途中の各構成要素の入出力テンソルの形状は自動計算される。
20 """
21 def __init__(self,config):
22 super().__init__()
23 self.config=copy.deepcopy(config)
24 self.input_shape=config['input_shape']
25 self.layerCount=defaultdict(lambda:0)
26 self.layers=[]
27 self.forwardConfigs=[]
28 dummy=torch.zeros([2,]+list(self.input_shape))
29 for layerType,layerConfig in config['layers']:
30 layer, dummy = self.makeLayer(layerType, layerConfig, dummy)
31 self.layerCount[layerType]+=1
32 layerName=layerType+str(self.layerCount[layerType])
33 setattr(self,layerName,layer)
34 self.layers.append(layer)
35 self.output_shape=dummy.shape
36
37 def makeLayer(self, layerType: str , layerConfig: dict, dummy):
38 if(layerType=='Conv1d'):
39 layer=torch.nn.Conv1d(in_channels=dummy.shape[1],**layerConfig)
40 elif(layerType=='BatchNorm1d'):
41 layer=torch.nn.BatchNorm1d(num_features=dummy.shape[1],**layerConfig)
42 elif(layerType=='transpose'):
43 def gen(dim0,dim1):
44 def func(x):
45 return torch.transpose(x,dim0,dim1)
46 return func
47 layer=gen(layerConfig[0],layerConfig[1])
48 elif(layerType=='mean'):
49 def gen(**layerConfig):
50 def func(x):
51 return torch.mean(x,**layerConfig)
52 return func
53 layer=gen(**layerConfig)
54 elif(layerType=='Conv2d'):
55 layer=torch.nn.Conv2d(in_channels=dummy.shape[1],**layerConfig)
56 elif(layerType=='AdaptiveAvgPool2d'):
57 layer=torch.nn.AdaptiveAvgPool2d(**layerConfig)
58 elif(layerType=='AdaptiveMaxPool2d'):
59 layer=torch.nn.AdaptiveMaxPool2d(**layerConfig)
60 elif(layerType=='AvgPool2d'):
61 layer=torch.nn.AvgPool2d(**layerConfig)
62 elif(layerType=='MaxPool2d'):
63 layer=torch.nn.MaxPool2d(**layerConfig)
64 elif(layerType=='BatchNorm2d'):
65 layer=torch.nn.BatchNorm2d(num_features=dummy.shape[1],**layerConfig)
66 elif(layerType=='Conv3d'):
67 layer=torch.nn.Conv3d(in_channels=dummy.shape[1],**layerConfig)
68 elif(layerType=='AvgPool3d'):
69 layer=torch.nn.AvgPool3d(**layerConfig)
70 elif(layerType=='MaxPool3d'):
71 layer=torch.nn.MaxPool3d(**layerConfig)
72 elif(layerType=='BatchNorm3d'):
73 layer=torch.nn.BatchNorm3d(num_features=dummy.shape[1],**layerConfig)
74 elif(layerType=='Flatten'):
75 layer=torch.nn.Flatten(**layerConfig)
76 elif(layerType=='Linear'):
77 layer=torch.nn.Linear(dummy.shape[-1],**layerConfig)
78 elif(layerType=='ReLU'):
79 layer=torch.nn.ReLU()
80 elif(layerType=='LeakyReLU'):
81 layer=torch.nn.LeakyReLU()
82 elif(layerType=='SiLU'):
83 layer=torch.nn.SiLU()
84 elif(layerType=='Tanh'):
85 layer=torch.nn.Tanh()
86 elif(layerType=='ResidualBlock'):
87 layer=ResidualBlock(dummy.shape[1:],layerConfig)
88 elif(layerType=='Concatenate'):
89 layer=Concatenate(dummy.shape[1:],layerConfig)
90 elif(layerType=='PositionalEncoding'):
91 layer=PositionalEncoding(dummy.shape[1:],layerConfig)
92 dummy=layer(dummy)
93 return layer, dummy
94
95 def forward(self,x):
96 self.batch_size=x.shape[0]
97 for layer in self.layers:
98 x=layer(x)
99 return x

101class ResidualBlock(torch.nn.Module):
102 """残差ブロック(y=x+h(x))。configは以下の形式とする。
103 {
104 "input_shape": list, #入力テンソルの形(バッチサイズを除く)
105 "block": dict, #残差を計算するためのGenericLayersを生成するためのconfig。
106 }
107 """
108 def __init__(self,input_shape,blockConfig):
109 super().__init__()
110 self.config=copy.deepcopy(blockConfig)
111 self.config["input_shape"]=input_shape
112 self.block=GenericLayers(self.config)
113 def forward(self,x):
114 return x+self.block(x)

116class Concatenate(torch.nn.Module):
117 """入力を複数のブロックに供給してその結果を結合するブロック。configは以下の形式とする。
118 {
119 "input_shape": list, #入力テンソルの形(バッチサイズを除く)
120 "blocks": list[dict], #分岐・結合対象の各ブロック(GenericLayers)を生成するためのconfig。
121 "dim": int, #結合対象の次元。省略時は-1(末尾)となる。
122
123 }
124 """
125 def __init__(self,input_shape,config):
126 super().__init__()
127 self.config=copy.deepcopy(config)
128 self.catDim=self.config.get("dim",-1)
129 self.blocks=[]
130 cnt=0
131 for c in self.config["blocks"]:
132 c["input_shape"]=input_shape
133 block=GenericLayers(c)
134 setattr(self,"Block"+str(cnt),block)
135 self.blocks.append(block)
136 cnt+=1
137 def forward(self,x):
138 return torch.cat([b(x) for b in self.blocks],dim=self.catDim)

140class PositionalEncoding(torch.nn.Module):
141 def __init__(self,dim,max_len=10000):
142 super().__init__()
143 pe=torch.zeros(max_len,dim)
144 position=(torch.arange(0,max_len,dtype=torch.float).float()/max_len).unsqueeze(1)
145 div_term=torch.exp(torch.arange(0,dim,2).float()/dim)
146 pe[:,0::2]=torch.sin(position*div_term)
147 pe[:,1::2]=torch.cos(position*div_term)
148 pe=pe.unsqueeze(0).transpose(0,1)
149 self.register_buffer('pe',pe)
150 def forward(self,x):
151 return x+self.pe[:x.size(0),:]
