43 def createVirtualSimulator(self, agent: Agent):
44 """agentの持つ情報に基づいて仮想シミュレータを起動する。agentのvalidate()で呼び出すことを想定。
45 """
46 option = nljson({
47 "patch": {
48 "Manager": {
49 "TimeStep": {
50 "defaultAgentStepInterval": agent.interval[SimPhase.AGENT_STEP]
51 },
52 "skipNoAgentStep": False
53 }
54 }
55 })
56 self.virtualSimulator = agent.manager.createVirtualSimulationManager(option)
57 self.virtualSimulator.reset()

62 def step(self, agent: Agent, tracks: list[Track3D]):
63 """agentの持つ情報に基づいて仮想シミュレータの時間を進め、情報を取得する。仮想敵の情報をtracksで与える。
64 agentのdeploy(action)で毎回呼び出すことを想定。
65 """
66 now = agent.manager.getTickCount()
67 for port, parent in agent.parents.items():
68 fullName = parent.getFullName()
69
70 if parent.isAlive():
71 motion = MotionState(parent.observables.at("motion"))
72 crs = agent.manager.getRootCRS()
73 time = agent.manager.getTime()
74 selfTrack = Track3D(
75 parent.getUUIDForTrack(),
76 crs,
77 time,
78 motion.pos(crs),
79 motion.vel(crs)
80 )
81
82 if not fullName in self.virtualMissiles:
83 self.virtualMissiles[fullName] = []
84 if not fullName in self.minimumDistances:
85 self.minimumDistances[fullName] = []
86
87 #仮想誘導弾の諸元を更新
88 self.minimumDistances[fullName]=np.inf
89 hit = False
90 i=0
91 while i < len(self.virtualMissiles[fullName]):
92 missile = self.virtualMissiles[fullName][i]
93 if missile.isAlive():
94 missile.communicationBuffers["MissileComm:"+missile.getFullName()]().send({
95 "target": selfTrack
96 },CommunicationBuffer.UpdateType.MERGE)
97
98 dt = agent.interval[SimPhase.AGENT_STEP]*agent.manager.getBaseTimeStep()
99
100 if missile.hitCheck(motion.pos(missile.getParentCRS()),motion.pos(missile.getParentCRS())-motion.vel(missile.getParentCRS())*dt):
101 # 命中
102 missile.manager.requestToKillAsset(missile)
103 hit=True
104
105 self.minimumDistances[fullName] = min(
106 self.minimumDistances[fullName],
107 np.linalg.norm(missile.pos()-motion.pos(missile.getParentCRS()))
108 )
109 i+=1
110 else:
111 # 飛翔終了していたら消す
112 self.virtualMissiles[fullName].pop(i)
113
114 #print(
115 # fullName," tick=",agent.manager.getTickCount(),
116 # ": virtualMissiles size=",len(self.virtualMissiles[fullName]),
117 # ", minimumDistance=",self.minimumDistances[fullName],
118 # ", hit=",hit
119 #)
120 else:
121 # parentが消えたら仮想誘導弾も消す
122 if fullName in self.virtualMissiles:
123 for missile in self.virtualMissiles[fullName]:
124 missile.manager.requestToKillAsset(missile)
125
126 self.virtualMissiles.pop(fullName)
127 self.minimumDistances.pop(fullName)
128
129 # 仮想シミュレータの時間を進める
130 while self.virtualSimulator.getTickCount() < now:
131 #print("call virtual simulator's step tick=",now," (v ",self.virtualSimulator.getTickCount(),")")
132 self.virtualSimulator.step({})
133
134 # 生存中のparentsに対して、新たな仮想誘導弾を生成する
135 for port, parent in agent.parents.items():
136 fullName = parent.getFullName()
137 if parent.isAlive():
138 motion = MotionState(parent.observables.at("motion"))
139 crs = agent.manager.getRootCRS()
140 time = agent.manager.getTime()
141 selfTrack = Track3D(
142 parent.getUUIDForTrack(),
143 crs,
144 time,
145 motion.pos(crs),
146 motion.vel(crs)
147 )
148 for t in tracks:
149 if len(self.virtualMissiles[fullName]) < self.maxNumPerParent:
150 shooterMotion = MotionState(
151 crs,
152 time,
153 t.pos(crs),
154 t.vel(crs),
155 np.array([0,0,0]),
156 Quaternion(1,0,0,0),
157 "FSD"
158 )
159 if self.launchCondition(parent, shooterMotion,selfTrack):
160 launchCommand=nljson({
161 "target": selfTrack,
162 "motion": shooterMotion
163 })
164 self.virtualMissiles[fullName].append(parent.launchInVirtual(self.virtualSimulator,launchCommand))
