112void R7ContestRewardSample02::onHit(const nl::json& args){//{"wpn":wpn,"tgt":tgt}
113 std::lock_guard<std::mutex> lock(mtx);
114 std::shared_ptr<PhysicalAsset> wpn=args.at("wpn");
115 std::shared_ptr<PhysicalAsset> tgt=args.at("tgt");
116 auto beg=deadFighters.begin();
117 auto end=deadFighters.end();
118 if(std::find(beg,end,tgt->getFullName())==end){
119 auto crs=manager->getRootCRS();//ここは直交座標系の方が適切。
120 Eigen::Vector3d vel=tgt->motion.vel(crs).normalized();
121 Eigen::Vector3d dir=(wpn->motion.pos(crs)-tgt->motion.pos(crs)).normalized();
122 double angle=acos(vel.dot(dir));
123 std::string name=tgt->getFullName();
124 if(assetToTargetAgent.find(name)!=assetToTargetAgent.end()){
125 reward[assetToTargetAgent[name]]+=((pHitE_head+pHitE_tail)+cos(angle)*(pHitE_head-pHitE_tail))/2;
126 }
127 name=wpn->parent.lock()->getFullName();
128 if(assetToTargetAgent.find(name)!=assetToTargetAgent.end()){
129 reward[assetToTargetAgent[name]]+=((pHit_head+pHit_tail)+cos(angle)*(pHit_head-pHit_tail))/2;
130 }
131 deadFighters.push_back(tgt->getFullName());
132 }
133}
134void R7ContestRewardSample02::onInnerStepEnd(){
135 auto ruler_=getShared<Ruler,Ruler>(manager->getRuler());
136 auto crs=ruler_->getLocalCRS();
137 for(auto&& e:manager->getAssets()){
138 if(isinstance<Missile>(e)){
139 auto m=getShared<Missile>(e);
140 if(!checked[m->getFullName()] && m->hasLaunched && !m->isAlive()){
141 Track3D tgt=m->target;
142 for(auto&& agentFullName:target){
143 for(auto& p:parents[agentFullName]){
144 if(tgt.isSame(p)){
145 reward[agentFullName]+=pAvoid;
146 }
147 }
148 }
149 checked[m->getFullName()]=true;
150 }
151 }else if(isinstance<Fighter>(e)){
152 auto f=getShared<Fighter>(e);
153 if(f->isAlive()){
154 std::string team=f->getTeam();
155 double outDist=std::max(0.0,abs(sideAx[team].dot(f->pos(crs).block<2,1>(0,0,2,1)))-dOut);
156 outDist=std::max(outDist,abs(forwardAx[team].dot(f->pos(crs).block<2,1>(0,0,2,1)))-dLine);
157 std::string name=f->getFullName();
158 if(assetToTargetAgent.find(name)!=assetToTargetAgent.end()){
159 reward[assetToTargetAgent[name]]+=(outDist/1000.)*pOut*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep();
160
161 if(f->optCruiseFuelFlowRatePerDistance>0.0){
162 double maxReachableRange=f->fuelRemaining/f->optCruiseFuelFlowRatePerDistance;
163 double distanceFromLine=forwardAx[team].dot(f->pos(crs).block<2,1>(0,0,2,1))+dLine;
164 double excess=maxReachableRange/(1+fuelMargin)-(distanceFromLine+distanceFromBase[team]);
165 if(excess<0){
166 reward[assetToTargetAgent[name]]+=pFuelShortage*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep();
167 }
168 }
169 }
170 }
171 }
172 }

174void R7ContestRewardSample02::onStepEnd(){
175 auto ruler=getShared<BasicAACRuler01>(manager->getRuler());
176 if(ruler->endReason!=BasicAACRuler01::EndReason::NOTYET){
177 for(auto&& e:manager->getAssets()){
178 if(isinstance<Fighter>(e)){
179 auto f=getShared<Fighter>(e);
180 if(f->isAlive()){
181 std::string name=f->getFullName();
182 if(assetToTargetAgent.find(name)!=assetToTargetAgent.end()){
183 if(ruler->isReturnableToBase(f)){
184 //帰還可能なら生存点
185 reward[assetToTargetAgent[name]]+=pAlive;
186 }else{
187 //帰還不可能なら墜落ペナルティ
188 reward[assetToTargetAgent[name]]+=pCrash;
189 }
190 }
191 }
192 }
193 }
194 }
