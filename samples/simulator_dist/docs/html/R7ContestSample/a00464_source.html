2"""
3順列不変性(permutation invariance)と順列同変性(permutation equivariance)をもったSet Transformer
4
5[Lee 2019] Lee, J., et al. "Set Transformer: A Framework for Attention-based Permutation-Invariant Neural Networks."
6
7著者による実装例（https://github.com/juho-lee/set_transformer）も提供されているが、
8maskに対応していないため、torch.nnのMultiHeadAttentionを用いて実装し直す。
9SAB,ISAB,PMAのforward()の引数key_padding_maskとして無効トークンのmaskを与えられるようにしている。
10
11なお、これらの性質は、torch.nnのTransformerEncoderLayerとTransformerDecoderLayerを用いても実現できる。
12
13"""
14import torch
15import torch.nn as nn

17class SAB(nn.Module):
18 def __init__(self, embed_dim, num_heads, **kwargs):
19 """
20 kwargのうち特にbatch_first,add_bias_kv,add_zero_attnは用途に応じて適切に選択すること。
21 特に、長さゼロ(全て無効トークン)のシーケンスが入力される可能性があるならば、add_zero_attnをTrueにする必要がある。
22 """
23 super(SAB, self).__init__()
24 self.batch_first = kwargs.get('batch_first', True)
25 kwargs['batch_first']=self.batch_first
26 self.mab = nn.MultiheadAttention(
27 embed_dim,
28 num_heads,
29 **kwargs
30 )
31 #self.mab = MAB(dim_in, dim_in, dim_out, num_heads, ln=ln)
32
33 def forward(self, X, key_padding_mask=None):
34 """
35 key_padding_maskは、サイズ(batch_size, embed_dim)のテンソルで与える。
36 bool型の場合はTrueのトークンが無視され、float型の場合はattention weightに値がそのまま加算される(-infを与えると無視)。
37 """
38 ret = self.mab(X, X, X,
39 key_padding_mask=key_padding_mask,
40 need_weights=False,

42 is_causal=False
43 )[0]

46class ISAB(nn.Module):
47 def __init__(self, embed_dim, num_heads, num_inds, **kwargs):
48 """
49 kwargのうち特にbatch_first,add_bias_kv,add_zero_attnは用途に応じて適切に選択すること。
50 特に、長さゼロ(全て無効トークン)のシーケンスが入力される可能性があるならば、add_zero_attnをTrueにする必要がある。
51 """
52 super(ISAB, self).__init__()
53 self.batch_first = kwargs.get('batch_first', True)
54 kwargs['batch_first']=self.batch_first
55 if self.batch_first:
56 self.I = nn.Parameter(torch.Tensor(1, num_inds, embed_dim))
57 else:
58 self.I = nn.Parameter(torch.Tensor(num_inds, 1, embed_dim))
59 nn.init.xavier_uniform_(self.I)
60 self.mab0 = nn.MultiheadAttention(
61 embed_dim,
62 num_heads,
63 **kwargs
64 )
65 self.mab1 = nn.MultiheadAttention(
66 embed_dim,
67 num_heads,
68 **kwargs
69 )
70
71 def forward(self, X, key_padding_mask=None):
72 """
73 key_padding_maskは、サイズ(batch_size, embed_dim)のテンソルで与える。
74 bool型の場合はTrueのトークンが無視され、float型の場合はattention weightに値がそのまま加算される(-infを与えると無視)。
75 """
76 if self.batch_first:
77 I = self.I.repeat(X.size(0), 1, 1)
78 else:
79 I = self.I.repeat(1, X.size(1), 1)
80 H = self.mab0(
81 I,
82 X,
83 X,
84 key_padding_mask=key_padding_mask,
85 need_weights=False,
86 attn_mask=None,
87 is_causal=False
88 )[0]
89 ret = self.mab1(
90 X,
91 H,
92 H,
93 key_padding_mask=None,
94 need_weights=False,
95 attn_mask=None,
96 is_causal=False
97 )[0]
