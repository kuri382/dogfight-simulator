2"""ray RLlibで学習するためのtorch.nn.Moduleのサンプル。
3R7ContestAgentSample01のobservation,actionに対応したものとなっている。
4
5* friend_enemy_relativeは使用していない。
6 このサンプルでは各Assetのembeddingを作成してpermutation equivariance/invarianceな操作を行う。
7 要素数Nfのシーケンスと要素数Neのシーケンスに対して[Nf,Ne,D]のshapeを持った付加情報を加えるためには、
8 様々な方式が考えられるが、シンプルに実装できるものは乏しい。
9 例えば、attention weightにバイアスを加える方法もあるが、この機能はPyTorchのMultiheadAttentionには存在しない。
10

94 #dummy entity (「射撃しない」等を表現するための空のエンティティ)
95 self.dummy_entity = torch.nn.Parameter(torch.empty((1,1,self.embed_dim), dtype=torch.float32))
96 bound = 1 / np.sqrt(self.embed_dim)
97 torch.nn.init.uniform_(self.dummy_entity,-bound,bound)

99 #permutation equivariant transformation
100 if 'entity_equivariant' in self.custom_model_config:
101 cfg=self.custom_model_config['entity_equivariant']
102 cfg['input_shape']=[1,self.embed_dim]
103 if cfg['type']=='SetTransformer':
104 self.entity_equivariant=SetTransformer(cfg)
105 elif cfg['type']=='TransformerEncoder':
106 self.entity_equivariant=TransformerEncoder(cfg)
107 else:
108 raise ValueError('type of entity_equivariant needs to be SetTransformer or TransformerEncoder.')
109
110 #permutation invariant transformation
111 cfg=self.custom_model_config['entity_invariant']
112 cfg['input_shape']=[1,self.embed_dim]
113 if cfg['type']=='SetTransformer':
114 self.entity_invariant=SetTransformer(cfg)
115 assert self.entity_invariant.hasPMA and self.entity_invariant.num_seeds==1
116 elif cfg['type']=='SetTransformerWithQuery':
117 self.entity_invariant=SetTransformerWithQuery(cfg)
118 self.entity_invariant_query = nn.Parameter(torch.Tensor(1, 1, self.embed_dim))
119 nn.init.xavier_uniform_(self.entity_invariant_query)
120 elif cfg['type']=='TransformerDecoder':
121 self.entity_invariant=TransformerDecoder(cfg)
122 self.entity_invariant_query = nn.Parameter(torch.Tensor(1, 1, self.embed_dim))
123 nn.init.xavier_uniform_(self.entity_invariant_query)
124 else:
125 raise ValueError('type of entity_invariant needs to be SetTransformer, SetTransformerWithQuery or TransformerDecoder.')
126
127 self.core_dim+=self.entity_invariant.output_shape[-1]
128
129 #merge branch
130 self.use_merger = 'merge' in self.custom_model_config
131 if self.use_merger:
132 cfg=self.custom_model_config['merge']
133 cfg['input_shape']=[self.core_dim]
134 self.merge=GenericLayers(cfg)
135 self.merged_dim=self.merge.output_shape[-1]
136 else:
137 self.merged_dim=self.core_dim
138
139 #recurrent block
140 self.use_lstm = self.custom_model_config.get("use_lstm",False)
141 if(self.use_lstm):
142 self.time_major = model_config.get("_time_major", False)
143 in_space =gym.spaces.Box(
144 float("-inf"),
145 float("inf"),
146 shape=(self.merged_dim, ),
147 dtype=np.float32)
148 self.lstm_cell_size = self.custom_model_config.get("lstm_cell_size",64)

150 pass
151 def dummy_forward(
152 self,
153 input_dict: Dict[str, TensorType],
154 state: List[TensorType],
155 seq_lens: TensorType,
156 ):
157 return input_dict["obs_flat"], state
158 PureLSTM._wrapped_forward = dummy_forward
159 self.core_recurrent_block=PureLSTM(
160 in_space,
161 action_space,
162 self.merged_dim,
163 {
164 "_time_major":self.time_major,
165 "lstm_cell_size":self.lstm_cell_size,
166 "lstm_use_prev_action":False,
167 "lstm_use_prev_reward":False
168 },
169 "core_recurrent_block"
170 )
171
172 #apply merged (recurrent) info to entities
173 #この例ではrecurrentの出力をKV(memory)、entityのembeddingをQ(tgt)としてTransformerDecoderに流す
174 #また、この例ではこれより後でparentとenemyしか使用しないため、いずれかが存在する場合のみ処理する
175 if 'parent' in self.entity_embeds or 'enemy' in self.entity_embeds:
176 if 'apply_merged_to_entity' in self.custom_model_config:
177 if(self.embed_dim!=self.merged_dim):
178 cfg={
179 'input_shape':[self.merged_dim],
180 'layers':[
181 ['Linear',{'out_features':self.embed_dim}]
182 ]
183 }
184 self.apply_merged_to_entity_pre=GenericLayers(cfg)
185
186 cfg=self.custom_model_config['apply_merged_to_entity']
187 cfg['input_shape']=[1,self.embed_dim]
188 if cfg['type']=='SetTransformerWithQuery':
189 self.apply_merged_to_entity=SetTransformerWithQuery(cfg)
190 elif cfg['type']=='TransformerDecoder':
191 self.apply_merged_to_entity=TransformerDecoder(cfg)
192 else:
193 raise ValueError('type of apply_merged_to_entity needs to be SetTransformerWithQuery or TransformerDecoder.')
194
195 #critic branch
196 self.critic_heads={}
197 for key in self.custom_model_config['critic']:
198 cfg=self.custom_model_config['critic'][key]
199 cfg['input_shape']=[self.merged_dim]
200 cfg['layers'].append(['Linear',{'out_features':1}])
201 head=GenericLayers(cfg)
202 setattr(self,'critic_'+key+'_head',head)
203 self.critic_heads[key]=head
204
205 #actor branch
206 if 'parent' in self.entity_embeds:
207 #このサンプルでは、parentのembeddingを作る場合、全parentに対して同一の重みで行動を計算する。
208 self.action_heads={}
209 for key in action_space[0].spaces:
210 if key=='target' and 'enemy' in self.entity_embeds:
211 # enemyのembeddingを作る場合の目標選択
212 # (1) 各parentを長さ1のKVとして、dummy+enemyをQとしたTransformerDecoderを流す。
213 # [B*N_parent,1,self.embed_dim], [B*N_parent,1+N_enemy,self.embed_dim] -> [B*N_parent,1+N_enemy,self.embed_dim]
214 # (2) Linearで末尾の次元を1にして、squeeze(-1)とunflatten(0,[B,N_parent])によって、
215 # [B,N_parent,1+N_enemy]の出力を得る。
216 cfg=self.custom_model_config['actor'][key]
217 cfg['input_shape']=[1,self.embed_dim]
218 cfg['layers'].append(['Linear',{'out_features':1}])
219 if cfg['type']=='SetTransformerWithQuery':
220 head=SetTransformerWithQuery(cfg)
221 elif cfg['type']=='TransformerDecoder':
222 head=TransformerDecoder(cfg)
223 else:
224 raise ValueError('type of actor["'+key+'"] needs to be SetTransformerWithQuery or TransformerDecoder.')
225 setattr(self,'actor_'+key+'_head',head)
226 self.action_heads[key]=head
227 else:
228 # それ以外はshape=[B,N_parent,self.embed_dim]のembed_dimの次元をデータ次元としてGenericLayersを用いる。
229 # 出力のshapeは[B,N_parent,sub_action_dim]となる。
230 cfg=self.custom_model_config['actor'][key]
231 cfg['input_shape']=[self.embed_dim]
232 _, sub_action_dim = ModelCatalog.get_action_dist(action_space[0][key], model_config, framework='torch')
233 cfg['layers'].append(['Linear',{'out_features':sub_action_dim}])
234 head=GenericLayers(cfg)
235 setattr(self,'actor_'+key+'_head',head)
236 self.action_heads[key]=head
237 else:
238 #もしparentのembeddingを作らない場合はmergedから単純な全結合で、actionに必要なパラメータ数が出力されるように計算する。
239 cfg=self.custom_model_config['actor']
240 cfg['input_shape']=[self.merged_dim]
241 cfg['layers'].append(['Linear',{'out_features':self.num_outputs}])
242 self.action_head=GenericLayers(cfg)
243
244 if(self.use_lstm):
245 self.view_requirements = self.core_recurrent_block.view_requirements
246 self.view_requirements["obs"].space = self.obs_space

248 @override(TorchModelV2)
249 def forward(self, input_dict: Dict[str, TensorType],
250 state: List[TensorType],
251 seq_lens: TensorType) -> (TensorType, List[TensorType]):
252 obs=input_dict[SampleBatch.OBS]
253
254 core=[]
255 B=None #バッチサイズ
256
257 if 'common' in obs and hasattr(self,'common'):
258 B=obs['common'].shape[0]
259 core.append(self.common(obs['common'])) #[B,Dc]→[B,D_common]
260
261 if 'image' in obs and hasattr(self,'image'):
262 B=obs['image'].shape[0]
263 core.append(self.image(obs['image'])) #[B,Ch,Lon,Lat]→[B,D_image]
264
265 embedded={} #{key:[B,N_key,self.embed_dim]}
266 entity_mask={} #{key:[B,N_key,1]}
267 if len(self.entity_embeds)>0:
268 for key in ['parent','friend','enemy','friend_missile','enemy_missile']:
269 if key in obs and key in self.entity_embeds:
270 B=obs[key].shape[0]
271 embedded[key]=self.entity_embeds[key](obs[key])
272
273 if 'observation_mask' in obs:
274 if key in obs['observation_mask']:
275 #PytorchではTrueのときに無効化される。obsは有効なものを1としているので反転する。
276 entity_mask[key]=obs['observation_mask'][key]!=1 #[B,N_key]
277 else:
278 #observation maskが与えられていなければ全て有効(False)とする
279 N=embedded[key].shape[-2]
280 entity_mask[key]=torch.full((B,N),False,device=self.dummy_entity.device)
281 embedded['dummy']=self.dummy_entity.expand(B,1,-1)
282 entity_mask['dummy']=torch.full((B,1),False,device=self.dummy_entity.device)
283
284 cat_embedded=torch.cat(list(embedded.values()),dim=-2) #[B,N,self.embed_dim]
285 cat_entity_mask=torch.cat(list(entity_mask.values()),dim=-1) #[B,N]
286 if 'entity_equivariant' in obs and hasattr(self,'entity_equivariant'):
287 cat_embedded=self.entity_equivariant(
288 src=cat_embedded,
289 src_key_padding_mask=cat_entity_mask
290 ) #[B,N,self.embed_dim]
291
292 if isinstance(self.entity_invariant,SetTransformer):
293 core.append(torch.reshape(self.entity_invariant(
294 src=cat_embedded,
295 src_key_padding_mask=cat_entity_mask
296 ),[B,self.embed_dim])) #[B,self.embed_dim]
297 else:
298 # Queryが必要
299 query = self.entity_invariant_query.repeat(B, 1, 1)
300 core.append(torch.reshape(self.entity_invariant(
301 tgt=query,
302 memory=cat_embedded,
303 memory_key_padding_mask=cat_entity_mask
304 ),[B,self.embed_dim])) #[B,self.embed_dim]
305
306 #merge
307 merged=torch.cat(core,dim=-1)+0 #[B,self.core_dim]
308 if self.use_merger:
309 merged=self.merge(merged)
310
311 #recurrent
312 if(self.use_lstm):
313 input_dict["obs_flat"]=input_dict["obs"]=merged
314 merged,state_out=self.core_recurrent_block(input_dict,state,seq_lens)#[B×D_core]
315 else:
316 state_out=state
317
318 #apply merged (recurrent) info to entities
319 if 'parent' in embedded or 'enemy' in embedded:
320 if hasattr(self,'apply_merged_to_entity'):
321 if hasattr(self,'apply_merged_to_entity_pre'):
322 kv=torch.reshape(self.apply_merged_to_entity_pre(merged),[B,1,self.embed_dim])
323 else:
324 kv=torch.reshape(merged,[B,1,self.embed_dim])
325 for key in ['parent','enemy']:
326 if key in embedded:
327 embedded[key]=self.apply_merged_to_entity(
328 tgt=embedded[key],
329 memory=kv
330 ) #[B,N_key,self.embed_dim]
331
332 #actor
333 if 'parent' in self.entity_embeds:
334 #parentのembeddingを作る場合
335 parent_action_list=[]
336 for key in self.action_space[0].spaces:
337 if key=='target' and 'enemy' in self.entity_embeds:
338 # (1) 各parentを長さ1のKVとして、dummy+enemyをQとしたTransformerDecoderを流す。
339 # [B*N_parent,1,self.embed_dim], [B*N_parent,1+N_enemy,self.embed_dim] -> [B*N_parent,1+N_enemy,self.embed_dim]
340 # (2) Linearで末尾の次元を1にして、squeeze(-1)とunflatten(0,[B,N_parent])によって、
341 # [B,N_parent,1+N_enemy]の出力を得る。
342 num_parents=embedded['parent'].shape[-2]
343 tgt=torch.cat([embedded['dummy'],embedded['enemy']],dim=-2)
344 tgt=tgt.unsqueeze(1).repeat(1, num_parents, 1, 1).flatten(0,1)
345 memory=embedded['parent'].flatten(0,1).unsqueeze(1)
346
347 logits=self.action_heads[key](
348 tgt=tgt,
349 memory=memory
350 ).squeeze(-1).unflatten(0,[B,num_parents]) #[B,N_parent,1+N_enemy]
351
352 # action maskを適用し、無効な対象が選ばれないようにする(-infにする)
353 # サンプラーが-infになっている選択肢を選ばない仕様になっていることを確認すること！
354 if self.custom_model_config['actor']['target'].get('apply_mask',True):
355 target_mask=torch.cat([entity_mask['dummy'],entity_mask['enemy']],dim=-1) #[B,1+N_enemy]
356 target_mask=target_mask.unsqueeze(1).repeat(1, num_parents, 1) #[B,N_parent,1+N_enemy]
357 logits=torch.where(target_mask,-torch.inf,logits)
358
359 parent_action_list.append(logits)
360
361 else:
362 logits=self.action_heads[key](embedded['parent'])
363
364 # 要すればaction maskを適用し、無効な対象が選ばれないようにする(-infにする)
365 # サンプラーが-infになっている選択肢を選ばない仕様になっていることを確認すること！
366 if (
367 'action_mask' in obs and key in obs['action_mask'][0]
368 and self.custom_model_config['actor'][key].get('apply_mask',True)
369 ):
370 mask=torch.stack([m[key] for m in obs['action_mask']],dim=1)
371
372 # all zeroのdummy dataが流れてきたときは全て1(有効にする)
373 mask=torch.where(torch.sum(mask,dim=-1,keepdim=True)==0,1,mask)
374
375 logits=torch.where(mask!=1,-torch.inf,logits)
376 parent_action_list.append(logits) #[B,N_parent,sub_action_dim]
377 parent_action=torch.cat(parent_action_list,dim=-1) #[B,N_parent,parent_action_dim]
378
379 #parentのentity_maskを反映し、生存していないparentのactionから勾配が流れないようにする
380 parent_action=torch.where(
381 entity_mask['parent'].unsqueeze(-1),
382 parent_action.detach(),
383 parent_action
384 )
385
386 #元のaction spaceの形(Tuple[Dict])に合わせて並び替える
387 action_logits=torch.cat(
388 [parent_action[:,i,...] for i in range(len(self.action_space))],
389 dim=-1) #[B,total_action_dim]
390 else:
391 #parentのembeddingを作らない場合
392 action_logits=self.action_head(merged)
393
394 #critic branch
395 self._value_base_features=merged+0
396
397 return action_logits,state_out
