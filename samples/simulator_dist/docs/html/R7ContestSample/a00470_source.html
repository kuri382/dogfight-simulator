106 def onInnerStepEnd(self):
107 crs=self.manager.getRuler()().getLocalCRS()
108 for team in self.target:
109 #(1)Biteへの加点、(2)メモリトラック落ちへの減点
110 for i,m in enumerate(self.friendMsls[team]):
111 if(m.hasLaunched and m.isAlive):
112 if(m.mode==Missile.Mode.SELF and not self.biteFlag[team][i]):
113 self.reward[team]+=self.pBite
114 self.biteFlag[team][i]=True
115 if(m.mode==Missile.Mode.MEMORY and not self.memoryTrackFlag[team][i]):
116 self.reward[team]-=self.pMemT
117 self.memoryTrackFlag[team][i]=True
118 #(3)敵探知への加点(生存中の敵の何%を探知できているか)(データリンク前提)
119 track=[]
120 for f in self.friends[team]:
121 if(f.isAlive()):
122 track=[Track3D(t) for t in f.observables["sensor"]["track"]]
123 break
124 numAlive=0
125 numTracked=0
126 for f in self.enemies[team]:
127 if(f.isAlive()):
128 numAlive+=1
129 for t in track:
130 if(t.isSame(f)):
131 numTracked+=1
132 break
133 if(numAlive>0):
134 self.reward[team]+=(1.0*numTracked/numAlive)*self.pDetect*self.interval[SimPhase.ON_INNERSTEP_END]*self.manager.getBaseTimeStep()
135 ene=0.0
136 tmp=-self.dLine
137 for f in self.friends[team]:
138 if(f.isAlive()):
139 #(4)過剰な機動への減点(角速度ノルムに対してL2、高度方向速度に対してL1正則化)
140 vAlt=f.motion.relPtoH(f.vel(),"ENU")[2]
141 self.reward[team]+=(-self.pVel*abs(vAlt)-(np.linalg.norm(f.omega())**2)*self.pOmega)*self.interval[SimPhase.ON_INNERSTEP_END]*self.manager.getBaseTimeStep()
142 #(5)前進・後退への更なる加減点
143 tmp=max(tmp,np.dot(self.forwardAx[team],f.pos(crs)[0:2]))

145 ene+=np.linalg.norm(f.vel())**2/2+gravity*f.getHeight()
