2"""HandyRLで学習するためのtorch.nn.Moduleのサンプル。
3R7ContestAgentSample01のobservation,actionに対応したものとなっている。
4
5* friend_enemy_relativeは使用していない。
6 このサンプルでは各Assetのembeddingを作成してpermutation equivariance/invarianceな操作を行う。
7 要素数Nfのシーケンスと要素数Neのシーケンスに対して[Nf,Ne,D]のshapeを持った付加情報を加えるためには、
8 様々な方式が考えられるが、シンプルに実装できるものは乏しい。
9 例えば、attention weightにバイアスを加える方法もあるが、この機能はPyTorchのMultiheadAttentionには存在しない。
10
11"""

65
66 # friend_enemy_relative (shape=[Np+Nf,Ne,Dfe])はこのサンプルでは使用しない。
67 # 例えばAttention weightに加算する等の用途が考えられる。

70 # embed_dimの整合確認
71 self.embed_dim=next(iter(self.entity_embeds.values())).output_shape[-1]

74
75 #dummy entity (「射撃しない」等を表現するための空のエンティティ)
76 self.dummy_entity = torch.nn.Parameter(torch.empty((1,1,self.embed_dim), dtype=torch.float32))
77 bound = 1 / np.sqrt(self.embed_dim)
78 torch.nn.init.uniform_(self.dummy_entity,-bound,bound)
79
80 #permutation equivariant transformation
81 if 'entity_equivariant' in self.model_config:
82 cfg=self.model_config['entity_equivariant']
83 cfg['input_shape']=[1,self.embed_dim]
84 if cfg['type']=='SetTransformer':
85 self.entity_equivariant=SetTransformer(cfg)
86 elif cfg['type']=='TransformerEncoder':
87 self.entity_equivariant=TransformerEncoder(cfg)
88 else:
89 raise ValueError('type of entity_equivariant needs to be SetTransformer or TransformerEncoder.')
90
91 #permutation invariant transformation
92 cfg=self.model_config['entity_invariant']
93 cfg['input_shape']=[1,self.embed_dim]
94 if cfg['type']=='SetTransformer':
95 self.entity_invariant=SetTransformer(cfg)
96 assert self.entity_invariant.hasPMA and self.entity_invariant.num_seeds==1
97 elif cfg['type']=='SetTransformerWithQuery':
98 self.entity_invariant=SetTransformerWithQuery(cfg)
99 self.entity_invariant_query = nn.Parameter(torch.Tensor(1, 1, self.embed_dim))
100 nn.init.xavier_uniform_(self.entity_invariant_query)
101 elif cfg['type']=='TransformerDecoder':
102 self.entity_invariant=TransformerDecoder(cfg)
103 self.entity_invariant_query = nn.Parameter(torch.Tensor(1, 1, self.embed_dim))
104 nn.init.xavier_uniform_(self.entity_invariant_query)
105 else:
106 raise ValueError('type of entity_invariant needs to be SetTransformer, SetTransformerWithQuery or TransformerDecoder.')
107
108 self.core_dim+=self.entity_invariant.output_shape[-1]
109
110 #merge branch
111 self.use_merger = 'merge' in self.model_config
112 if self.use_merger:
113 cfg=self.model_config['merge']
114 cfg['input_shape']=[self.core_dim]
115 self.merge=GenericLayers(cfg)
116 self.merged_dim=self.merge.output_shape[-1]
117 else:
118 self.merged_dim=self.core_dim
119
120 #recurrent block
121 self.use_lstm = self.model_config.get("use_lstm",False)
122 if(self.use_lstm):
123 self.lstm_cell_size = self.model_config.get("lstm_cell_size",64)
124 self.lstm_num_layers = self.model_config.get("lstm_num_layers",1)
125 self.lstm_dropout = self.model_config.get("lstm_dropout",0.1)
126 self.core_recurrent_block=RecurrentBlock(nn.LSTM(
127 self.merged_dim,
128 self.lstm_cell_size,
129 num_layers=self.lstm_num_layers,
130 batch_first=True,
131 dropout=self.lstm_dropout
132 ))
133 self.merged_dim=self.lstm_cell_size
134
135 #apply merged (recurrent) info to entities
136 #この例ではrecurrentの出力をKV(memory)、entityのembeddingをQ(tgt)としてTransformerDecoderに流す
137 #また、この例ではこれより後でparentとenemyしか使用しないため、いずれかが存在する場合のみ処理する
138 if 'parent' in self.entity_embeds or 'enemy' in self.entity_embeds:
139 if 'apply_merged_to_entity' in self.model_config:
140 if(self.embed_dim!=self.merged_dim):
141 cfg={
142 'input_shape':[self.merged_dim],
143 'layers':[
144 ['Linear',{'out_features':self.embed_dim}]
145 ]
146 }
147 self.apply_merged_to_entity_pre=GenericLayers(cfg)
148
149 cfg=self.model_config['apply_merged_to_entity']
150 cfg['input_shape']=[1,self.embed_dim]
151 if cfg['type']=='SetTransformerWithQuery':
152 self.apply_merged_to_entity=SetTransformerWithQuery(cfg)
153 elif cfg['type']=='TransformerDecoder':
154 self.apply_merged_to_entity=TransformerDecoder(cfg)
155 else:
156 raise ValueError('type of apply_merged_to_entity needs to be SetTransformerWithQuery or TransformerDecoder.')
157
158 #critic branch
159 self.critic_heads={}
160 for key in self.model_config['critic']:
161 cfg=self.model_config['critic'][key]
162 cfg['input_shape']=[self.merged_dim]
163 cfg['layers'].append(['Linear',{'out_features':1}])
164 head=GenericLayers(cfg)
165 setattr(self,'critic_'+key+'_head',head)
166 self.critic_heads[key]=head
167
168 #actor branch
169 if 'parent' in self.entity_embeds:
170 #このサンプルでは、parentのembeddingを作る場合、全parentに対して同一の重みで行動を計算する。
171 self.action_heads={}
172 sub=self.action_dist_class.getActionDistributionClass(ac_space[0])
173 for key in ac_space[0].spaces:
174 if key=='target' and 'enemy' in self.entity_embeds:
175 # enemyのembeddingを作る場合の目標選択
176 # (1) 各parentを長さ1のKVとして、dummy+enemyをQとしたTransformerDecoderを流す。
177 # [B*N_parent,1,self.embed_dim], [B*N_parent,1+N_enemy,self.embed_dim] -> [B*N_parent,1+N_enemy,self.embed_dim]
178 # (2) Linearで末尾の次元を1にして、squeeze(-1)とunflatten(0,[B,N_parent])によって、
179 # [B,N_parent,1+N_enemy]の出力を得る。
180 cfg=self.model_config['actor'][key]
181 cfg['input_shape']=[1,self.embed_dim]
182 cfg['layers'].append(['Linear',{'out_features':1}])
183 if cfg['type']=='SetTransformerWithQuery':
184 head=SetTransformerWithQuery(cfg)
185 elif cfg['type']=='TransformerDecoder':
186 head=TransformerDecoder(cfg)
187 else:
188 raise ValueError('type of actor["'+key+'"] needs to be SetTransformerWithQuery or TransformerDecoder.')
189 setattr(self,'actor_'+key+'_head',head)
190 self.action_heads[key]=head
191 else:
192 # それ以外はshape=[B,N_parent,self.embed_dim]のembed_dimの次元をデータ次元としてGenericLayersを用いる。
193 # 出力のshapeは[B,N_parent,sub_action_dim]となる。
194 cfg=self.model_config['actor'][key]
195 cfg['input_shape']=[self.embed_dim]
196 sub_action_dim=sub.getActionDistributionClass(ac_space[0][key]).get_param_dim(ac_space[0][key])
197 cfg['layers'].append(['Linear',{'out_features':sub_action_dim}])
198 head=GenericLayers(cfg)
199 setattr(self,'actor_'+key+'_head',head)
200 self.action_heads[key]=head
201 else:
202 #もしparentのembeddingを作らない場合はmergedから単純な全結合で、actionに必要なパラメータ数が出力されるように計算する。
203 cfg=self.model_config['actor']
204 cfg['input_shape']=[self.merged_dim]
205 cfg['layers'].append(['Linear',{'out_features':self.action_dim}])
206 self.action_head=GenericLayers(cfg)

208 def forward(self,obs,state,seq_len=None,mask=None):
209 core=[]
210 B=None #バッチサイズ
211
212 if 'common' in obs and hasattr(self,'common'):
213 B=obs['common'].shape[0]
214 core.append(self.common(obs['common'])) #[B,Dc]→[B,D_common]
215
216 if 'image' in obs and hasattr(self,'image'):
217 B=obs['image'].shape[0]
218 core.append(self.image(obs['image'])) #[B,Ch,Lon,Lat]→[B,D_image]
219
220 embedded={} #{key:[B,N_key,self.embed_dim]}
221 entity_mask={} #{key:[B,N_key,1]}
222 if len(self.entity_embeds)>0:
223 for key in ['parent','friend','enemy','friend_missile','enemy_missile']:
224 if key in obs and key in self.entity_embeds:
225 B=obs[key].shape[0]
226 embedded[key]=self.entity_embeds[key](obs[key])
227
228 if 'observation_mask' in obs:
229 if key in obs['observation_mask']:
230 #PytorchではTrueのときに無効化される。obsは有効なものを1としているので反転する。
231 entity_mask[key]=obs['observation_mask'][key]!=1 #[B,N_key]
232 else:
233 #observation maskが与えられていなければ全て有効(False)とする
234 N=embedded[key].shape[-2]
235 entity_mask[key]=torch.full((B,N),False,device=self.dummy_entity.device)
236 embedded['dummy']=self.dummy_entity.expand(B,1,-1)
237 entity_mask['dummy']=torch.full((B,1),False,device=self.dummy_entity.device)
238
239 cat_embedded=torch.cat(list(embedded.values()),dim=-2) #[B,N,self.embed_dim]
240 cat_entity_mask=torch.cat(list(entity_mask.values()),dim=-1) #[B,N]
241 if 'entity_equivariant' in obs and hasattr(self,'entity_equivariant'):
242 cat_embedded=self.entity_equivariant(
243 src=cat_embedded,
244 src_key_padding_mask=cat_entity_mask
245 ) #[B,N,self.embed_dim]
246
247 if isinstance(self.entity_invariant,SetTransformer):
248 core.append(torch.reshape(self.entity_invariant(
249 src=cat_embedded,
250 src_key_padding_mask=cat_entity_mask
251 ),[B,self.embed_dim])) #[B,self.embed_dim]
252 else:
253 # Queryが必要
254 query = self.entity_invariant_query.repeat(B, 1, 1)
255 core.append(torch.reshape(self.entity_invariant(
256 tgt=query,
257 memory=cat_embedded,
258 memory_key_padding_mask=cat_entity_mask
259 ),[B,self.embed_dim])) #[B,self.embed_dim]
260
261 #merge
262 merged=torch.cat(core,dim=-1) #[B,self.core_dim]
263 if self.use_merger:
264 merged=self.merge(merged)
265
266 #recurrent
267 if(self.use_lstm):
268 merged,state_out=self.core_recurrent_block(merged,state,seq_len,mask)
269
270 #apply merged (recurrent) info to entities
271 if 'parent' in embedded or 'enemy' in embedded:
272 if hasattr(self,'apply_merged_to_entity'):
273 if hasattr(self,'apply_merged_to_entity_pre'):
274 kv=torch.reshape(self.apply_merged_to_entity_pre(merged),[B,1,self.embed_dim])
275 else:
276 kv=torch.reshape(merged,[B,1,self.embed_dim])
277 for key in ['parent','enemy']:
278 if key in embedded:
279 embedded[key]=self.apply_merged_to_entity(
280 tgt=embedded[key],
281 memory=kv
282 ) #[B,N_key,self.embed_dim]
283
284 ret={}
285 #critic
286 for key,head in self.critic_heads.items():
287 ret[key]=head(merged)
288
289 #actor
290 if 'parent' in self.entity_embeds:
291 #parentのembeddingを作る場合
292 parent_action_list=[]
293 for key in self.action_space[0].spaces:
294 if key=='target' and 'enemy' in self.entity_embeds:
295 # (1) 各parentを長さ1のKVとして、dummy+enemyをQとしたTransformerDecoderを流す。
296 # [B*N_parent,1,self.embed_dim], [B*N_parent,1+N_enemy,self.embed_dim] -> [B*N_parent,1+N_enemy,self.embed_dim]
297 # (2) Linearで末尾の次元を1にして、squeeze(-1)とunflatten(0,[B,N_parent])によって、
298 # [B,N_parent,1+N_enemy]の出力を得る。
299 num_parents=embedded['parent'].shape[-2]
300 tgt=torch.cat([embedded['dummy'],embedded['enemy']],dim=-2)
301 tgt=tgt.unsqueeze(1).repeat(1, num_parents, 1, 1).flatten(0,1)
302 memory=embedded['parent'].flatten(0,1).unsqueeze(1)
303
304 logits=self.action_heads[key](
305 tgt=tgt,
306 memory=memory
307 ).squeeze(-1).unflatten(0,[B,num_parents]) #[B,N_parent,1+N_enemy]
308
309 # action maskを適用し、無効な対象が選ばれないようにする(-infにする)
310 # サンプラーが-infになっている選択肢を選ばない仕様になっていることを確認すること！
311 if self.model_config['actor']['target'].get('apply_mask',True):
312 target_mask=torch.cat([entity_mask['dummy'],entity_mask['enemy']],dim=-1) #[B,1+N_enemy]
313 target_mask=target_mask.unsqueeze(1).repeat(1, num_parents, 1) #[B,N_parent,1+N_enemy]
314 logits=torch.where(target_mask,-torch.inf,logits)
315
316 parent_action_list.append(logits)
317
318 else:
319 logits=self.action_heads[key](embedded['parent'])
320
321 # 要すればaction maskを適用し、無効な対象が選ばれないようにする(-infにする)
322 # サンプラーが-infになっている選択肢を選ばない仕様になっていることを確認すること！
323 if (
324 'action_mask' in obs and key in obs['action_mask'][0]
325 and self.model_config['actor'][key].get('apply_mask',True)
326 ):
327 mask=torch.stack([m[key] for m in obs['action_mask']],dim=1)
328
329 # all zeroのdummy dataが流れてきたときは全て1(有効にする)
330 mask=torch.where(torch.sum(mask,dim=-1,keepdim=True)==0,1,mask)
331
332 logits=torch.where(mask!=1,-torch.inf,logits)
333 parent_action_list.append(logits) #[B,N_parent,sub_action_dim]
334 parent_action=torch.cat(parent_action_list,dim=-1) #[B,N_parent,parent_action_dim]
335
336 #parentのentity_maskを反映し、生存していないparentのactionから勾配が流れないようにする
337 parent_action=torch.where(
338 entity_mask['parent'].unsqueeze(-1),
339 parent_action.detach(),
340 parent_action
341 )
342
343 #元のaction spaceの形(Tuple[Dict])に合わせて並び替える
344 ret['policy']=torch.cat(
345 [parent_action[:,i,...] for i in range(len(self.action_space))],
346 dim=-1) #[B,total_action_dim]
347 else:
348 #parentのembeddingを作らない場合
349 ret['policy']=self.action_head(merged)
350
351 if(self.use_lstm):
352 ret['hidden']=state_out
353 return ret
