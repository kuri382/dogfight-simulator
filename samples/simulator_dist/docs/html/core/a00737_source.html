3// Pythonオブジェクトをcerealでシリアライズする機能と、それを用いてnlohmann::basic_jsonを相互変換する機能
4//
5// pybind11::object等への代入は参照先を変える形となるため、基本的には既存のPythonオブジェクトに対するin-placeなloadができない。
6// in-placeなloadに相当する処理として、以下の2種類を提供する。Python側からは(2)の利用を推奨する。なお、C++側で保持するpybind11::object型変数に対してはin-placeなloadによる参照先の変更で事足りることが多いと思われる。
7// (1)pybind11::objectインスタンスを返す関数(load_python_object)
8// (2)setattrを用いて他のオブジェクトの属性としてloadする関数(load_attr)
9// また、load時には型情報を与えないと復元できないため、型情報の与え方として以下の2種類を提供する。(1)の利用を推奨する。
10// (1)saveの代わりにsave_with_type_infoを呼ぶことで型情報と一緒に保存しておき、loadの代わりにload_with_type_infoで同時に読み込む。
11// (2)save_type_info_ofを呼んで型情報を別途保存しておき、loadの代わりにload_with_external_type_infoでその型情報を引数に与えて読み込む。
12//
13// 上記の推奨する方式を踏まえて、Python側からは読み・書き共通のインターフェースとして以下の関数を用いることを推奨する。
14// (1) serialize_attr_with_type_info(archive: AvailableArchiveTypes ,obj: Any, *args: str)
15// objが持つargsで与えた属性について型情報とともに読み書きする。
16// (2) serialize_by_func(archive: AvailableArchiveTypes, func: Callable[[AvailableArchiveTypes],None])
17// serialize関数としてfunc(archive)を持つような仮想オブジェクトとして読み書きする。
18// (3) serialize_by_func(archive: AvailableArchiveTypes, attr_name: str, func: Callable[[AvailableArchiveTypes],None])
19// serialize関数としてfunc(archive)を持つような仮想オブジェクトを、attr_nameという名称を持つNameValuePairとして読み書きする。
20// (4) serialize_internal_state_by_func(archive: AvailableArchiveTypes, bool full, func: Callable[[AvailableArchiveTypes, bool],None])
21// serializeInternalState関数としてfunc(archive,full)を持つような仮想オブジェクトとして読み書きする。
22// (5) serialize_internal_state_by_func(archive: AvailableArchiveTypes, bool full, attr_name: str, func: Callable[[AvailableArchiveTypes, bool],None])
23// serializeInternalState関数としてfunc(archive,full)を持つような仮想オブジェクトを、attr_nameという名称を持つNameValuePairとして読み書きする。
24// (6) save_with_type_info(archive: AvailableOutputArchiveTypes, obj: Any)
25// objを型情報とともに書き込む。名前を指定しないのでarchiveによりデフォルトの名前が付与される。
26// (7) save_with_type_info(archive: AvailableOutputArchiveTypes, name: str, obj: Any)
27// nameという名前でcereal::NameValuePairを生成し、objを型情報とともに書き込む。
28// (8) load_with_type_info(archive: AvailableInputArchiveTypes) -> Any
29// 型情報とともにobjectを読み込んで返す。名前を指定しないのでarchiveによりデフォルトの名前が付与される。
30// (9) load_with_type_info(archive: AvailableInputArchiveTypes name: str) -> Any
31// nameという名前でcereal::NameValuePairを生成し、型情報とともにobjectを読み込んで返す。
32//

446 if constexpr (traits::same_archive_as<Archive,::asrc::core::util::NLJSONInputArchive>){
447 const nlohmann::ordered_json& j=archive.getCurrentNode();
448 size=j.size();
449 }else{
450 archive( ::cereal::make_size_tag( size ));
451 }
452 if(withHint){
453 assert(
454 (
455 hint.kind==PyObjectTypeKind::mutable_mapping
456 || hint.kind==PyObjectTypeKind::mapping
457 )
458 && (
459 hint.str_key_only ?
460 hint.mapping_str_key_only.size()==size :
461 hint.mapping_generic.size()==size
462 )
463 );
464 }
465 obj=::pybind11::dict();
466 if(withHint){
467 if(hint.str_key_only){
468 for(auto&& [h_key,h_value] : hint.mapping_str_key_only){
469 ::pybind11::object key,value;
470 if constexpr (traits::same_archive_as<Archive,::asrc::core::util::NLJSONInputArchive>){
471 // object
472 archive(::cereal::make_nvp(
473 h_key.c_str(),
474 makePyObjectWrapper(value,h_value)
475 ));
476 key=::pybind11::str(h_key);
477 }else{
478 auto strTypeInfo=PyTypeInfo(::pybind11::str());
479 archive(makeMappingItem(key,value,strTypeInfo,h_value));
480 }
481 obj[key]=value;
482 }
483 }else{
484 std::size_t i=0;
485 for(auto&& [h_key,h_value] : hint.mapping_generic){
486 ::pybind11::object key,value;
487 if constexpr (traits::same_archive_as<Archive,::asrc::core::util::NLJSONInputArchive>){
488 // array
489 const nlohmann::ordered_json& j=archive.getCurrentNode();
490 assert(j.at(i).is_array() && j.at(i).size()==2);
491 }
492 archive(makeMappingItem(key,value,h_key,h_value));
493 obj[key]=value;
494 ++i;
495 }
496 }
497 }else{
498 if constexpr (traits::same_archive_as<Archive,::asrc::core::util::NLJSONInputArchive>){
499 // object
500 const nlohmann::ordered_json& j=archive.getCurrentNode();
501 for(auto&& [j_key, j_value] : j.items()){
502 ::pybind11::object key,value;
503 archive(::cereal::make_nvp(
504 j_key.c_str(),
505 makePyObjectWrapper(value)
506 ));
507 obj[::pybind11::str(j_key)]=value;
508 }
509 }else{
510 for(std::size_t i=0;i<size;++i){
511 ::pybind11::object key,value;
512 archive(makeMappingItem(key,value));
513 obj[key]=value;
514 }
515 }
516 }
517 }
518};
