25Eigen::Vector3d pitchLimitter(const MotionState& motion, const Eigen::Vector3d& dstDir,const double& pitchLimit){
26 auto crs=motion.getCRS();
27 Eigen::Vector3d velInNED=motion.velPtoH(motion.vel(),motion.pos(),"NED");
28 Eigen::Vector3d t=velInNED.normalized();//現在の進行方向
29 Eigen::Vector3d dd=crs->transformToTopocentricCRS(
30 dstDir,
31 motion.pos(),
32 motion.time,
33 CoordinateType::DIRECTION,
34 motion.pos(),
35 "NED",
36 false
37 );//目標方向
38 Eigen::Vector3d b=t.cross(dd);//最小回転量で到達できる回転軸(=「当初回転軸」)
39 double snTheta=std::min(1.0,b.norm());
40 double csTheta=std::clamp(t.dot(dd),-1.0,1.0);
41 double eTheta=atan2(snTheta,csTheta);
42 b/=snTheta;
43 Eigen::Vector3d pz(0,0,1);//+z方向
44 double sint=abs(t.dot(pz));//現在のピッチ角の正弦
45 double sindd=abs(dd.dot(pz));//目標方向のピッチ角の正弦
46 double sing=std::max(sin(pitchLimit),std::max(sint,sindd));//制限ピッチ角の正弦
47 double sinu=pz.cross(b).norm();//b軸まわりの回転面上におけるピッチ角(の絶対値)の最大値
48 if(sinu>sing){
49 //回転面が制限範囲を通過する場合
50 Eigen::Vector3d vert=b.cross(pz.cross(b)).normalized();//ピッチ角(の絶対値)が最大となる方向
51 if(vert.dot(dd+t)<0){
52 vert=-vert;
53 }
54 if(b.dot(t.cross(vert))>=0 && b.dot(vert.cross(dd))>=0){
55 //最短経路が垂直面を通過する場合、ピッチ角が制限値以内に収まるように回転軸を補正する。
56 //補正後回転軸の候補は4種類あるため、もとの軸に最も近いものを選択する。
57 Eigen::Vector3d hh=t.cross(pz);
58 if(hh.norm()<1e-6){
59 hh=(b.cross(pz)).cross(pz).normalized();
60 }else{
61 hh.normalize();
62 }
63 Eigen::Vector3d vv=hh.cross(t).normalized();
64 double sinRot=1;
65 if(sing>sint){
66 double cosg=sqrt(std::max(0.0,1-sing*sing));
67 sinRot=cosg/sqrt(std::max(0.0,1-sint+sint));
68 }
69 double cosRot=sqrt(std::max(0.0,1-sinRot*sinRot));
70 Eigen::Vector3d bb1=hh*cosRot+vv*sinRot;
71 Eigen::Vector3d bb2=hh*cosRot-vv*sinRot;
72 Eigen::Vector3d bb3=-hh*cosRot+vv*sinRot;
73 Eigen::Vector3d bb4=-hh*cosRot-vv*sinRot;
74 Eigen::Vector3d bb12,bb34;
75 if(b.dot(bb1)>b.dot(bb2)){
76 bb12=bb1.normalized();
77 }else{
78 bb12=bb2.normalized();
79 }
80 if(b.dot(bb3)>b.dot(bb4)){
81 bb34=bb3.normalized();
82 }else{
83 bb34=bb4.normalized();
84 }
85 if(b.dot(bb12)>b.dot(bb34)){
86 b=bb12.normalized();
87 }else{
88 b=bb34.normalized();
89 }
90 }
91 }
92 return crs->transformFromTopocentricCRS(
93 b,
94 Eigen::Vector3d::Zero(),
95 motion.time,
96 CoordinateType::DIRECTION,
97 motion.pos(),
98 "NED",
99 false
100 );
101}

123Eigen::Vector3d AltitudeKeeper::operator()(const MotionState& motion, const Eigen::Vector3d& dstDir, const double& dstAlt){
124 auto crs=motion.getCRS();
125 auto dstDirInNED=crs->transformToTopocentricCRS(
126 dstDir,
127 motion.pos(),
128 motion.time,
129 CoordinateType::DIRECTION,
130 motion.pos(),
131 "NED",
132 false
133 );//目標方向
134 double eAlt=motion.getHeight() - dstAlt;
135 Eigen::Vector3d velInENU=motion.velPtoH(motion.vel(),motion.pos(),"ENU");
136 double vAlt=velInENU(2);
137 double V=velInENU.norm();
138 Eigen::Vector3d ret(dstDirInNED(0),dstDirInNED(1),0);
139 ret.normalize();
140 double dstPitch=asin(std::clamp<double>((pGain*eAlt+dGain*vAlt)/V,-1,1));//上向き正の目標ピッチ角
141 dstPitch=std::clamp<double>(dstPitch,minPitch,maxPitch);
142 double cs=cos(dstPitch);
143 double sn=sin(dstPitch);
144 return crs->transformFromTopocentricCRS(
145 Eigen::Vector3d(cs*ret(0),cs*ret(1),-sn),//NED
146 Eigen::Vector3d::Zero(),
147 motion.time,
148 CoordinateType::DIRECTION,
149 motion.pos(),
150 "NED",
151 false
152 );
153}
154double AltitudeKeeper::getDstPitch(const MotionState& motion, const double& dstAlt){
155 double eAlt=motion.getHeight() - dstAlt;
156 Eigen::Vector3d velInENU=motion.velPtoH(motion.vel(),motion.pos(),"ENU");
157 double vAlt=velInENU(2);
158 double V=velInENU.norm();
159 double dstPitch=asin(std::clamp<double>((pGain*eAlt+dGain*vAlt)/V,-1,1));//上向き正の目標ピッチ角
160 return std::clamp<double>(dstPitch,minPitch,maxPitch);

162double AltitudeKeeper::inverse(const MotionState& motion, const double& dstPitch){
163 double alt=motion.getHeight();
164 Eigen::Vector3d velInENU=motion.velPtoH(motion.vel(),motion.pos(),"ENU");
165 double vAlt=velInENU(2);
166 double V=velInENU.norm();
167 double eAlt=(V*sin(dstPitch)-dGain*vAlt)/pGain;
168 return alt-eAlt;
169}
