2 * Copyright (c) 2024-2025 Air Systems Research Center, Acquisition, Technology & Logistics Agency(ATLA)
3 * @file
4 *
5 * @brief Python側に公開可能とするクラスの基底クラスを定義する。
6 *
7 * @details このクラスを継承すると以下のような特性を有する。
8 *
9 * @par protectedなコンストラクタを使用可能
10 *
11 * あるクラスTのコンストラクタがpublicでない場合、make_sharedを使用できない。
12 * Tを継承したクラス内クラス(make_shared_enabler)をprotectedで定義し、そのコンストラクタをpublicとすることで、
13 * make_shared_enablerに対してはmake_sharedを使用できるようになる。
14 * これを用いると、shared_ptrを返すstaticメンバ関数(create)のみをpublicメンバとして、
15 * shared_ptrとしてのインスタンス化をユーザに強制することも可能となる。
16 *
17 * @par インターフェースクラスを多重継承した派生クラスをC++側、Python側の両方で定義することが可能
18 *
19 * Python側での仮想関数オーバーライドを簡便に実現するためには"trampoline"クラスを用いることになるが、
20 * trampolineクラスは元のC++クラスをテンプレートパラメータにとってこれを継承する形で実装される。
21 * 多重継承に対応するために各trampolineクラスは元のC++クラスをvirtual継承する必要があったが、
22 * virtual継承された派生クラスはstatic_castで基底クラスからダウンキャスト不可能であるため、
23 * pybind11のcastに失敗してしまう。
24 * この対策として、trampolineクラスが使用されるのはPython側でインスタンスが生成されたときのみであることを利用して、
25 * __init__で渡されるselfに相当するpybind11::handleをC++側のprivateメンバとして保持しておきpybind11::castではそれを返すようにした。
26 *
27 * @par 他のpybind11モジュール内でpybind11::module_local()とともに公開された派生クラスへのダウンキャストが可能
28 *
29 * pybind11::castによる自動ダウンキャストは、それを呼び出したモジュール上で見えているクラスにしか行われない。
30 * そのため、他のモジュールでpybind11::module_local()とされた派生クラスのインスタンスが入っている基底クラスポインタを
31 * 適切にダウンキャストしてpybind11::objectを得るためには、派生クラスが公開されたモジュール上で実際のダウンキャスト処理を行わせる必要がある。
32 * これを実現するために、各モジュール上に実体が生成されるような仮想メンバ関数を介してcastを行うような実装とした。
33 */

82 /**
83 * @internal
84 * @brief [For internal use] 派生クラスTのコンストラクタがpublicでない場合にmake_shared<T>を呼び出し可能とするためのenabler。
85 */
86 template<std::derived_from<PtrBaseType> T>
87 struct make_shared_enabler: T{
88 template<typename... Args>
89 explicit make_shared_enabler(Args&&... args): T(std::forward<Args>(args)...){}
90
91 virtual const std::type_info* _get_type_info() const{
92 return &typeid(T);
93 }
94 virtual std::string getDemangledName() const{
95 return cereal::util::demangle(typeid(T).name());

98 /**
99 * @internal
100 * @brief [For internal use] 派生クラスTのコンストラクタがpublicでない場合にmake_shared<T::TrampolineType>を呼び出し可能とするためのenabler。
101 */
102 template<std::derived_from<PtrBaseType> T>
103 struct make_shared_enabler_for_trampoline: T::TrampolineType<>{
104 template<typename... Args>
105 explicit make_shared_enabler_for_trampoline(Args&&... args): T(std::forward<Args>(args)...), T::TrampolineType<>(std::forward<Args>(args)...){}
106
107 virtual const std::type_info* _get_type_info() const{
108 return &typeid(typename T::TrampolineType<>);
109 }
110 virtual std::string getDemangledName() const{
111 return cereal::util::demangle(typeid(T).name());
112 }
113 };
114
115 public:
116 /**
117 * @brief インスタンス生成用のstatic関数。 asrc::core::py_init で使用される。
118 *
119 * @details
120 * 引数はコンストラクタが受け入れる任意の組み合わせを与えることができる。
121 */
122 template<std::derived_from<PtrBaseType> T=PtrBaseType, typename... Args>
123 requires (
124 std::constructible_from<typename T::TrampolineType<>::make_shared_enabler_for_trampoline<T>,Args...>
125 )
126 static std::shared_ptr<T> create(Args&&... args){
127 if constexpr (std::constructible_from<T,Args...>){
128 // public constructor
129 return std::make_shared<T>(std::forward<Args>(args)...);
130 }else if constexpr (std::constructible_from<typename T::make_shared_enabler<T>,Args...>){
131 // non-public constructor
132 return std::make_shared<typename T::make_shared_enabler<T>>(std::forward<Args>(args)...);
133 }else{
134 // pure virtual class
135 std::string demangledName = cereal::util::demangle(typeid(T).name());
136 throw std::runtime_error("You tried to call constructor of abstract class '"+demangledName+"'. from Python side.");
137 }
138 }
139 /**
140 * @brief Trampolineクラスのインスタンス生成用のstatic関数。 asrc::core::py_init で使用される。
141 *
142 * @details
143 * 引数はコンストラクタが受け入れる任意の組み合わせを与えることができる。
144 */
145 template<std::derived_from<PtrBaseType> T=PtrBaseType, typename... Args>
146 requires (
147 std::constructible_from<typename T::TrampolineType<>::make_shared_enabler_for_trampoline<T>,Args...>
148 )
149 static std::shared_ptr<typename T::TrampolineType<>> create_for_trampoline(Args&&... args){
150 if constexpr (std::constructible_from<T,Args...>){
151 // public constructor
152 return std::make_shared<typename T::TrampolineType<>>(std::forward<Args>(args)...);
153 }else{
154 // non-public constructor
155 return std::make_shared<typename T::TrampolineType<>::make_shared_enabler_for_trampoline<T>>(std::forward<Args>(args)...);
156 }
157 }

187 * @internal
188 * @brief [For internal use]
189 */
190 virtual pybind11::handle _call_cast(pybind11::return_value_policy policy, pybind11::handle parent) const{
191 if(
192 policy == pybind11::return_value_policy::copy
193 || policy == pybind11::return_value_policy::move
194 ){
195 // new instance
196 _internal_enforce_type_caster_base=true;
197 auto ret=pybind11::cast(this,policy,parent).release();
198 _internal_enforce_type_caster_base=false;
199 return std::move(ret);
200 }else{
201 // existing instance
202 if(_internal_handle){
203 return pybind11::reinterpret_borrow<pybind11::object>(_internal_handle).release();
204 }else{
205 _internal_enforce_type_caster_base=true;
206 auto ret=pybind11::cast(this,policy,parent).release();
207 _internal_enforce_type_caster_base=false;
208 return std::move(ret);
209 }
210 }
211 }

213 * @internal
214 * @brief [For internal use]
215 */
216 virtual pybind11::handle _call_cast_holder(const std::shared_ptr<const PtrBaseType>& holder) const{
217 // policy = return_value_policy::take_ownership;
218 if(_internal_handle){
219 return pybind11::reinterpret_borrow<pybind11::object>(_internal_handle).release();
220 }else{
221 _internal_enforce_type_caster_base=true;
222 auto ret=pybind11::cast(holder).release();
223 _internal_enforce_type_caster_base=false;
224 return std::move(ret);
225 }
226 }

228 * @internal
229 * @brief [For internal use]
230 */

232 return _internal_enforce_type_caster_base;
233 }

236 mutable bool _internal_enforce_type_caster_base; //!<
237};

244/**
245 * @brief PolymorphicPythonableBaseを継承したクラスTに対して任意のArgs...を引数にとる__init__を登録する。
246 *
247 * @details
248 * コンストラクタがpublicでない場合や、抽象クラスである場合にも同じ形で呼び出し可能。
249 * trampolineクラスとの呼び分けも対応している。
250 *
251 * execute 中の__init__は
252 * pybind11::detail::initimpl::factory<CFunc, AFunc, CReturn(CArgs...), AReturn(AArgs...)>
253 * が行う処理に相当する。
254 *
255 * 呼び出すコンストラクタは、Python側での継承の有無によって以下の2種類が使用される。
256 * 無-> T::template create<T,Args...>
257 * 有-> T::template create_for_trampoline<T,Args...>
258 *
259 * コンストラクタの呼び出し後、selfを表すpybind11::handleをC++側に伝える処理を追加している。
260 * この1行のためにpybind11::initのカスタマイズ用フックを用意した。
261*/
262template<polymorphic_pythonable T, typename... Args>
263struct py_init_constructor_impl<T,Args...> {
264 template <typename Class, typename... Extra>
265 requires ( std::same_as<T,typename Class::type> )
266 void execute(Class &cl, const Extra &...extra) const{
267 cl.def(
268 "__init__",
269 [](pybind11::detail::value_and_holder &v_h, Args... args) {
270 if (Py_TYPE(v_h.inst) == v_h.type->type) {
271 pybind11::detail::initimpl::construct<Class>(v_h, T::template create<T,Args...>(std::forward<Args>(args)...), false);
272 } else {
273 pybind11::detail::initimpl::construct<Class>(v_h, T::template create_for_trampoline<T,Args...>(std::forward<Args>(args)...), true);
274 }
275
276 // selfをsetPyHandleでC++インスタンス側に渡す
277 v_h.value_ptr<T>()->setPyHandle((PyObject*)(v_h.inst)); // この行を追加するためにこの構造体を定義した
278 },
279 pybind11::detail::is_new_style_constructor(),
280 extra...);
281 }
282};
