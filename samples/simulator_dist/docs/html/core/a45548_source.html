158Eigen::VectorXd SixDoFFighter::trim6(double alt,double V,double tgtBank){
159 auto eng=getShared<SimpleFighterJetEngine>(engine);
160 std::function<double(unsigned int,const double*,double*,void*)> obj=[&](unsigned int n,const double* x,double* grad,void* data){
161 //x=[alpha,beta,de,da,dr,P]
162 double alpha_=x[0];
163 double beta_=x[1];
164 double de_=x[2];
165 double da_=x[3];
166 double dr_=x[4];
167 if(tgtBank==0){
168 beta_=0;
169 da_=0;
170 dr_=0;
171 }
172 std::map<std::string,double> atm=atmosphere(alt);
173 double Mach=V/atm["a"];
174 double dp=atm["rho"]*V*V/2;
175 double thrust_=eng->calcThrust(x[5],alt,Mach);
176 Eigen::Vector3d vi(V,0,0);
177 Eigen::Vector3d omegai(0,0,0);
178 Quaternion q1=Quaternion::fromAngle(Eigen::Vector3d(0,0,1),0);
179 Quaternion q2=Quaternion::fromAngle(Eigen::Vector3d(1,0,0),tgtBank);
180 Quaternion q3=Quaternion::fromAngle(Eigen::Vector3d(0,0,1),-beta_);
181 Quaternion q4=Quaternion::fromAngle(Eigen::Vector3d(0,1,0),alpha_);
182 Quaternion qToNEDFromBody = q1*q2*q3*q4; // [NED <- Body]
183 Quaternion qToParentFromNED = motion.getCRS()->getEquivalentQuaternionToTopocentricCRS(pos,"NED").conjugate(); // [Parent <- NED]
184 Quaternion q=qToParentFromNED*qToNEDFromBody; // [Parent <- Body]
185 Eigen::Vector3d ex=q.transformVector(Eigen::Vector3d(1,0,0));
186 Eigen::Vector3d ey=q.transformVector(Eigen::Vector3d(0,1,0));
187 Eigen::Vector3d ez=q.transformVector(Eigen::Vector3d(0,0,1));
188 Eigen::Vector3d vb=q.conjugate().transformVector(vi);
189 Eigen::Vector3d omegab=q.conjugate().transformVector(omegai);
190 std::map<std::string,double> aero=calcAeroV(alpha_,beta_,de_,da_,dr_,Mach,false);
191 double pd=omegab(0)*b/(2*V);
192 double qd=omegab(1)*mac/(2*V);
193 double rd=omegab(2)*b/(2*V);
194 double CX=aero["Cx"]+aero["Cxq"]*qd;
195 double CY=aero["Cy"]+aero["Cyp"]*pd+aero["Cyr"]*rd;
196 double CZ=aero["Cz"]+aero["Czq"]*qd;
197 double CL=aero["Cl"]+aero["Clp"]*pd+aero["Clr"]*rd+aero["Clda"]*da_+aero["Cldr"]*dr_;
198 double CM=aero["Cm"]+aero["Cmq"]*qd+CZ*(XcgR-Xcg);
199 double CN=aero["Cn"]+aero["Cnp"]*pd+aero["Cnr"]*rd+aero["Cnda"]*da_+aero["Cndr"]*dr_-CY*(XcgR-Xcg)*(mac/b);
200 Eigen::Vector3d Fab=(Eigen::Vector3d(CX,CY,CZ)-aero["Cdw"]*vb/V)*dp*S;
201 Eigen::Vector3d Na=Eigen::Vector3d(CL*b,CM*mac,CN*b)*dp*S;
202 Eigen::Vector3d Ftb=Eigen::Vector3d(thrust_,0,0);
203 Eigen::Vector3d gravityVector=motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false)*m*gravity; //TODO 慣性力モデルに置換
204 Eigen::Vector3d Fb=Fab+Ftb+q.conjugate().transformVector(gravityVector);
205 Eigen::Vector3d omegaDotB=Iinv*(Na-omegab.cross(I*omegab+Eigen::Vector3d(Le,0.,0.)));
206 Eigen::Vector3d vbDot=Fb/m-omegab.cross(vb);
207 double VDot=vb.dot(vbDot)/V;
208 double Vxz=sqrt(vb(0)*vb(0)+vb(2)*vb(2));
209 double alphaDot=(vb(0)*vbDot(2)-vb(2)*vbDot(0))/(Vxz*Vxz);
210 double betaDot=(V*vbDot(1)-vb(1)*VDot)/(V*Vxz);
211 return (VDot*VDot)/(V*V)+alphaDot*alphaDot+betaDot*betaDot+omegaDotB.squaredNorm();
212 };
213 nlopt_delegator obj_d(&obj,nullptr);
214 std::vector<double> trimmed={0,0,0,0,0,0.5};
215 std::vector<double> lb={-deg2rad(10),-deg2rad(30),-deLimit,-daLimit,-drLimit,0};
216 std::vector<double> ub={deg2rad(45),deg2rad(30),deLimit,daLimit,drLimit,1};
217 nlopt::opt opt=nlopt::opt(nlopt::algorithm::LN_BOBYQA,6);
218 opt.set_min_objective(nlopt_delegatee,&obj_d);
219 opt.set_stopval(1e-10);
220 opt.set_maxeval(1e4);
221 double minf;
222 try{
223 opt.set_lower_bounds(lb);
224 opt.set_upper_bounds(ub);
225 auto result=opt.optimize(trimmed,minf);
226 }catch(...){
227 if(std::isnan(trimmed[0])){trimmed[0]=0;}
228 if(std::isnan(trimmed[1])){trimmed[1]=0;}
229 if(std::isnan(trimmed[2])){trimmed[2]=0;}
230 if(std::isnan(trimmed[3])){trimmed[3]=0;}
231 if(std::isnan(trimmed[4])){trimmed[4]=0;}
232 if(std::isnan(trimmed[5])){trimmed[5]=0.5;}
233 minf=std::numeric_limits<double>::infinity();
234 }
235 Eigen::VectorXd ret(7);
236 if(tgtBank==0){
237 trimmed[1]=0;//beta
238 trimmed[3]=0;//da
239 trimmed[4]=0;//dr
240 }
241 ret<<trimmed[0],trimmed[1],trimmed[2],trimmed[3],trimmed[4],trimmed[5],(minf<1e-10 ? 1.0 : 0.0);
242 return ret;
243}

249Eigen::VectorXd SixDoFFighter::calcDerivative(const Eigen::VectorXd& x,const Time& t,const Eigen::VectorXd& u){
250 //x,uの読み込み
251 MotionState tmp_motion(
252 motion.getCRS(),
253 t,
254 x.block(0,0,3,1),
255 x.block(3,0,3,1),
256 x.block(10,0,3,1),
257 Quaternion(x(6),x(7),x(8),x(9)).normalized()
258 );
259 double tmp_de=std::clamp<double>(x(13),-deLimit,deLimit);
260 double tmp_da=std::clamp<double>(x(14),-daLimit,daLimit);
261 double tmp_dr=std::clamp<double>(x(15),-drLimit,drLimit);
262 double tmp_power=std::clamp<double>(x(16),0,1);
263 double ue=u(0);
264 double ua=u(1);
265 double ur=u(2);
266 double pCmd=u(3);
267
268 //基本の状態量に関する値
269 Eigen::Vector3d ri=tmp_motion.pos();
270 Eigen::Vector3d vi=tmp_motion.vel();
271 Eigen::Vector3d vb=tmp_motion.relPtoB(vi);
272 double V=tmp_motion.vel.norm();
273 Eigen::Vector3d vni=vi/V;
274 Eigen::Vector3d omegai=tmp_motion.omega();
275 Eigen::Matrix<double,4,3> Jq=tmp_motion.q.dqdwi();
276 //大気,動圧に関する値
277 double alt=tmp_motion.getHeight();
278 std::map<std::string,double> atm=atmosphere(alt);
279 double Mach=V/atm["a"];
280 double dp=atm["rho"]*V*V/2;
281 double K=dp*S;
282 //α,βに関する値
283 double alpha=atan2(vb(2),vb(0));
284 double beta=asin(vb(1)/V);
285 //力、モーメントに関する値
286 //空気力
287 double alphaClamp=std::clamp<double>(alpha,-deg2rad(10),deg2rad(45));
288 double betaClamp=std::clamp<double>(beta,-deg2rad(30),deg2rad(30));
289 std::map<std::string,double> aero=calcAeroV(alphaClamp,betaClamp,tmp_de,tmp_da,tmp_dr,Mach,false);
290 Eigen::Vector3d omegaB=tmp_motion.relPtoB(omegai);
291 double pd=omegaB(0)*b/(2*V);
292 double qd=omegaB(1)*mac/(2*V);
293 double rd=omegaB(2)*b/(2*V);
294 double CX=aero["Cx"]+aero["Cxq"]*qd;
295 double CY=aero["Cy"]+aero["Cyp"]*pd+aero["Cyr"]*rd;
296 double CZ=aero["Cz"]+aero["Czq"]*qd;
297 double CL=aero["Cl"]+aero["Clp"]*pd+aero["Clr"]*rd+aero["Clda"]*tmp_da+aero["Cldr"]*tmp_dr;
298 double CM=aero["Cm"]+aero["Cmq"]*qd+CZ*(XcgR-Xcg);
299 double CN=aero["Cn"]+aero["Cnp"]*pd+aero["Cnr"]*rd+aero["Cnda"]*tmp_da+aero["Cndr"]*tmp_dr-CY*(XcgR-Xcg)*(mac/b);
300 Eigen::Vector3d CFa(CX,CY,CZ);
301 Eigen::Vector3d CNa(CL,CM,CN);
302 Eigen::Matrix<double,3,3> bcbMat=Eigen::Vector3d(b,mac,b).asDiagonal().toDenseMatrix();
303 //推力
304 auto eng=getShared<SimpleFighterJetEngine>(engine);
305 double thrust=calcThrust({{"power",tmp_power},{"alt",alt},{"rmach",Mach}});
306 //状態方程式
307 Eigen::Vector3d gravityVector=tmp_motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false)*m*gravity; //TODO 慣性力モデルに置換
308 Eigen::Vector3d Fi=(tmp_motion.relBtoP(CFa)-aero["Cdw"]*vni)*K+tmp_motion.relBtoP(Eigen::Vector3d(thrust,0,0))+gravityVector;
309 Eigen::Vector3d Nb=bcbMat*CNa*K;
310 Eigen::Vector3d Lb=I*omegaB+Eigen::Vector3d(Le,0.,0.);
311 Eigen::Vector3d omegaDotB=Iinv*(Nb-omegaB.cross(Lb));
312 Eigen::Vector3d omegaDotI=tmp_motion.relBtoP(omegaDotB);
313 Eigen::VectorXd f=Eigen::VectorXd::Zero(17);
314 f.block(0,0,3,1)=vi;
315 f.block(3,0,3,1)=Fi/m;
316 f.block(6,0,4,1)=Jq*omegai;
317 f.block(10,0,3,1)=omegaDotI;
318 f(13,0)=std::clamp<double>((ue*deLimit-tmp_de)/deTimeConstant,-deMaxRate,deMaxRate);
319 f(14,0)=std::clamp<double>((ua*daLimit-tmp_da)/daTimeConstant,-daMaxRate,daMaxRate);
320 f(15,0)=std::clamp<double>((ur*drLimit-tmp_dr)/drTimeConstant,-drMaxRate,drMaxRate);
321 if(eng->enableInstantResponse){
322 f(16,0)=0;
323 }else{
324 f(16,0)=eng->pdot(tmp_power,pCmd);
325 }
326 return f;

328void SixDoFFighter::calcMotion(double dt){
329 motion_prev=motion;
330 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
331 auto eng=getShared<SimpleFighterJetEngine>(engine);
332 Eigen::VectorXd x0=Eigen::VectorXd::Zero(17);
333 x0.block(0,0,3,1)=motion.pos();
334 x0.block(3,0,3,1)=motion.vel();
335 x0.block(6,0,4,1)=motion.q.toArray();
336 x0.block(10,0,3,1)=motion.omega();
337 x0(13)=de;
338 x0(14)=da;
339 x0(15)=dr;
340 x0(16)=eng->power;
341 Eigen::VectorXd u=Eigen::VectorXd::Zero(4);
342 u(0)=ctrl.at("ue").get<double>();
343 u(1)=ctrl.at("ua").get<double>();
344 u(2)=ctrl.at("ur").get<double>();
345 u(3)=ctrl.at("pCmd").get<double>();
346 std::function<Eigen::VectorXd (const Eigen::VectorXd& x,const Time& t)> deriv=[&](const Eigen::VectorXd& x,const Time& t){
347 return calcDerivative(x,t,u);
348 };
349 Eigen::VectorXd nextX=RK4(x0,manager->getTime(),dt,deriv);
350 motion.setPos(nextX.block(0,0,3,1));
351 motion.setVel(nextX.block(3,0,3,1));
352 motion.setOmega(nextX.block(10,0,3,1));
353 motion.setQ(Quaternion(nextX(6),nextX(7),nextX(8),nextX(9)).normalized());
354 de=std::clamp<double>(nextX(13),-deLimit,deLimit);
355 da=std::clamp<double>(nextX(14),-daLimit,daLimit);
356 dr=std::clamp<double>(nextX(15),-drLimit,drLimit);
357 //α、βが許容範囲外に出た場合も墜落として扱う
358 Eigen::Vector3d vb=motion.relPtoB(motion.vel());
359 double V=vb.norm();
360 double viex=vb(0);
361 double viey=vb(1);
362 double viez=vb(2);
363 double alpha=atan2(viez,viex);
364 double beta=asin(viey/V);
365 if(alpha<deg2rad(-10.0) ||
366 alpha>deg2rad(45.0) ||
367 beta<deg2rad(-30.0) ||
368 beta>deg2rad(30.0)
369 ){
370 if(motion.getHeight()>=0){
371 //高度が0以上の場合のみここで通知(高度0未満の場合は親クラスで通知されるため)
372 manager->triggerEvent("Crash",this->weak_from_this());
373 manager->requestToKillAsset(getShared<Asset>(this->shared_from_this()));
374 //std::cout<<"Crashed! alpha="<<rad2deg(alpha)<<", beta="<<rad2deg(beta)<<std::endl;
375 }
376 }

434nl::json SixDoFFlightController::calcDirect(const nl::json &cmd){
435 //操舵量を直接指定する場合
436 auto p=getShared<SixDoFFighter>(parent);
437 double alt=p->motion.getHeight();
438 double V=p->motion.vel.norm();
439 double tgtBank=0;
440 Eigen::VectorXd u;
441 Eigen::VectorXd trimmed=p->trim6(alt,V,tgtBank);
442 u=Eigen::VectorXd::Zero(4);
443 double roll_=0.0;
444 if(cmd.contains("roll")){
445 roll_=std::clamp(cmd.at("roll").get<double>(),-1.,1.);
446 }
447 double pitch_=0.0;
448 if(cmd.contains("pitch")){
449 pitch_=std::clamp(cmd.at("pitch").get<double>(),-1.,1.);
450 }

793Eigen::VectorXd SixDoFFlightController::calcU(const nl::json &cmd,const Eigen::VectorXd& x){

795 auto p=getShared<SixDoFFighter>(parent);
796 //xの読み込み
797 MotionState motion(
798 p->motion.getCRS(),
799 p->manager->getTime(),
800 x.block(0,0,3,1),
801 x.block(3,0,3,1),
802 x.block(10,0,3,1),
803 Quaternion(x(6),x(7),x(8),x(9))
804 );
805 double de=x(13);
806 double da=x(14);
807 double dr=x(15);
808 double power=x(16);

810 Eigen::Vector3d ri=motion.pos();
811 Eigen::Vector3d vi=motion.vel();
812 Eigen::Vector3d vb=motion.relPtoB(vi);
813 double V=motion.vel.norm();
814 double Vxz=sqrt(vb(0)*vb(0)+vb(2)*vb(2));

816 Eigen::Matrix<double,3,3> dvnidvi=(Eigen::Matrix3d::Identity()-vni*vni.transpose())/V;
817 Eigen::Vector3d omegai=motion.omega();

829 Eigen::Vector3d eyd=Rbi.block(1,0,1,3).transpose();

831 Eigen::Vector3d ezd=Rbi.block(2,0,1,3).transpose();
832 Eigen::Matrix<double,3,4> dezddq=motion.q.dR3dq();

837 domegabdq.block(2,0,1,4)=omegai.transpose()*dezdq;

866 dBdx.block(0,6,1,4)=vni.transpose()*deydq;
867 dBdx/=(Vxz/V);//sqrt(1-pow(viey/V,2));

869 //空気力
870 double alphaClamp=std::clamp<double>(alpha,-deg2rad(10),deg2rad(45));
871 double betaClamp=std::clamp<double>(beta,-deg2rad(30),deg2rad(30));
872 std::map<std::string,double> aeroV=p->calcAeroV(alphaClamp,betaClamp,de,da,dr,Mach,true);
873 Eigen::Vector3d omegaB=motion.relPtoB(omegai);

875 double qd=omegaB(1)*p->mac/(2*V);//omegai.dot(ey)
876 double rd=omegaB(2)*p->b/(2*V);//omegai.dot(ez)
877 double CX=aeroV["Cx"]+aeroV["Cxq"]*qd;

879 double dCXdB=aeroV["dCxdB"]+aeroV["dCxqdB"]*qd;

881 double dCYdA=aeroV["dCydA"]+aeroV["dCypdA"]*pd+aeroV["dCyrdA"]*rd;
882 double dCYdB=aeroV["dCydB"]+aeroV["dCypdB"]*pd+aeroV["dCyrdB"]*rd;
883 double CZ=aeroV["Cz"]+aeroV["Czq"]*qd;
884 double dCZdA=aeroV["dCzdA"]+aeroV["dCzqdA"]*qd;
885 double dCZdB=aeroV["dCzdB"]+aeroV["dCzqdB"]*qd;
886 double CL=aeroV["Cl"]+aeroV["Clp"]*pd+aeroV["Clr"]*rd+aeroV["Clda"]*da+aeroV["Cldr"]*dr;
887 double dCLdA=aeroV["dCldA"]+aeroV["dClpdA"]*pd+aeroV["dClrdA"]*rd+aeroV["dCldadA"]*da+aeroV["dCldrdA"]*dr;
888 double dCLdB=aeroV["dCldB"]+aeroV["dClpdB"]*pd+aeroV["dClrdB"]*rd+aeroV["dCldadB"]*da+aeroV["dCldrdA"]*dr;
889 double CM=aeroV["Cm"]+aeroV["Cmq"]*qd+CZ*(p->XcgR-p->Xcg);
890 double dCMdA=aeroV["dCmdA"]+aeroV["dCmqdA"]*qd+dCZdA*(p->XcgR-p->Xcg);
