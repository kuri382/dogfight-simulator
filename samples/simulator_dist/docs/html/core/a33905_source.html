56double Time::parseDatetime(const std::string& datetime){
57 // ISO 8601相当のタイムゾーン無し文字列で与えること。
58 // YYYYMMDDThhmmss.SSS 又は YYYY-MM-DDThh:mm:ss.SSS
59 int year, month, day, hour, minute;
60 double second;
61 std::size_t pos=0;
62 try{
63 year=std::stoi(datetime.substr(pos,4));
64 pos+=4;
65 if(!std::isdigit(datetime.at(pos))){pos++;}
66 month=std::stoi(datetime.substr(pos,2));
67 pos+=2;
68 if(!std::isdigit(datetime.at(pos))){pos++;}
69 day=std::stoi(datetime.substr(pos,2));
70 pos+=2;
71 if(!std::isdigit(datetime.at(pos))){pos++;}
72 hour=std::stoi(datetime.substr(pos,2));
73 pos+=2;
74 if(!std::isdigit(datetime.at(pos))){pos++;}
75 minute=std::stoi(datetime.substr(pos,2));
76 pos+=2;
77 if(!std::isdigit(datetime.at(pos))){pos++;}
78 std::istringstream iss(datetime.substr(pos)); // 最後は末尾に余計なものが付いていても良いようにstringstreamを使う。
79 iss >> second;
80 }catch(std::exception& ex){
81 std::cout<<"Failed to parse datetime '"<<datetime<<"'"<<std::endl;
82 std::cout<<ex.what()<<std::endl;
83 throw ex;
84 }
85 return (
86 (days_from_civil(year,month,day) - days_from_civil(2000,1,1)) * 86400 +
87 (hour - 12) * 3600 +
88 minute * 60 +
89 second
90 );
91}
92int Time::days_from_civil(int y,int m, int d){
93 //日付(y,m,d)を与え、1970-01-01からの経過日数を返す。 https://howardhinnant.github.io/date_algorithm.html より。
94 y -= m <= 2;
95 const int era = (y >= 0 ? y : y-399) /400;
96 const unsigned int yoe = y - era * 400;
97 const unsigned int doy = (153*(m > 2 ? m-3 : m+9) + 2)/5 + d-1;
98 const unsigned int doe = yoe * 365 + yoe /4 - yoe/100 + doy;
99 return era * 146097 + doe - 719468;
100}
101std::tuple<int,int,int> Time::civil_from_days(int z){
102 //1970-01-01からの経過日数を与え、日付(y,m,d)を返す。 https://howardhinnant.github.io/date_algorithm.html より。
103 z += 719468;
104 const int era = (z >= 0 ? z : z -146096) / 146097;
105 const unsigned int doe = z - era * 146097;
106 const unsigned int yoe = (doe - doe/1460 + doe/36524 - doe/146096) / 365;
107 const int y = yoe + era * 400;
108 const unsigned int doy = doe - (365*yoe + yoe/4 - yoe/100);
109 const unsigned int mp = (5*doy + 2)/153;
110 const unsigned int d = doy - (153*mp+2)/5 + 1;
111 const unsigned int m = mp < 10 ? mp+3 : mp-9;
112 return {y + (m <= 2), m, d};
113}

157Time Epoch::toTT(const Time& t){
158 if(t.timeSystem==TimeSystem::TT){
159 return t;
160 }else if(t.timeSystem==TimeSystem::TAI){
161 return Time(t.value+delta_tt_tai,TimeSystem::TT);
162 }else if(t.timeSystem==TimeSystem::UT1){
163 return Time(t.value+delta_tt_ut1,TimeSystem::TT);
164 }else if(t.timeSystem==TimeSystem::UTC){
165 return Time(t.value+delta_tt_ut1+delta_ut1_utc,TimeSystem::TT);
166 }else if(t.timeSystem==TimeSystem::GPS){
167 return Time(t.value+delta_tt_tai+delta_tai_gps,TimeSystem::TT);
168 }else if(t.timeSystem==TimeSystem::JY){
169 return Time(t.value*86400*365.25,TimeSystem::TT);
170 }else if(t.timeSystem==TimeSystem::JC){
171 return Time(t.value*86400*36525,TimeSystem::TT);
172 }else{
173 throw std::runtime_error("Invalid Time.");
174 }
