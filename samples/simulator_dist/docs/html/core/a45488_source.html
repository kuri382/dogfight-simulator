72void AircraftRadar::perceive(bool inReset){
73 //PhysicalAsset::perceive(inReset);//Sensors don't need motion and isAlive as observable.
74 track.clear();
75 auto crs=getParentCRS();// equals to parent.lock()->getBodyCRS() because this is bound to parent.
76 if(!crs){
77 throw std::runtime_error("crs is nullptr!");
78 }
79 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
80 return asset->getTeam()!=getTeam() && isinstance<Fighter>(asset);
81 })){
82 auto f=getShared<Fighter>(e);
83 if(f->isAlive()){
84 Eigen::Vector3d fpos=f->pos(crs);
85 Eigen::Vector3d rpos=absPtoB(fpos);
86 double L=rpos.norm();
87 if(rpos(0)>L*cos(thetaFOR) && (Lref<0 || L<=Lref*pow(f->rcsScale,0.25))){
88 track.push_back(Track3D(f,crs,manager->getTime(),fpos,f->vel(crs)));
89 }
90 }
91 }
92 observables["track"]=track;
93}

172void MWS::perceive(bool inReset){
173 //PhysicalAsset::perceive(inReset);//Sensors don't need motion and isAlive as observable.
174 track.clear();
175 auto crs=getParentCRS();
176 Eigen::Vector3d myPos=pos(crs);
177 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
178 return asset->getTeam()!=getTeam() && isinstance<Missile>(asset);
179 })){
180 auto m=getShared<Missile>(e);
181 Eigen::Vector3d rpos=absPtoB(m->pos(crs));
182 double L=rpos.norm();
183 if(isEsmIsh){
184 if(m->hasLaunched && m->isAlive() && m->sensor.lock()->isActive){
185 std::pair<bool,Track3D> r=m->sensor.lock()->isTracking(Track3D(parent,crs));
186 if(r.first && rpos(0)>L*cos(thetaFOR) && (Lref<0 || L<=Lref)){
187 track.push_back(Track2D(m,crs,myPos));//真値を入れる
188 }
189 }
190 }else{
191 if(m->hasLaunched && m->isAlive() && rpos(0)>L*cos(thetaFOR) && (Lref<0 || L<=Lref)){
192 track.push_back(Track2D(m,crs,myPos));//真値を入れる
193 }
194 }
195 }
196 observables["track"]=track;

271 //PhysicalAsset::perceive(inReset);//Sensors don't need motion and isAlive as observable.
272 track.clear();
273 if(isActive){
274 auto crs=getParentCRS();
275 Eigen::Vector3d steer=absPtoB(estTPos(crs)).normalized();
276 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
277 return asset->getTeam()!=getTeam() && isinstance<Fighter>(asset);
278 })){
279 auto f=getShared<Fighter>(e);
280 if(f->isAlive()){
281 Eigen::Vector3d fpos=f->pos(crs);
282 Eigen::Vector3d rpos=absPtoB(fpos);
283 double L=rpos.norm();
284 if(rpos(0)>L*cos(thetaFOR) && (Lref<0 || L<=Lref) && rpos.dot(steer)>L*cos(thetaFOV)){
285 track.push_back(Track3D(f,crs,manager->getTime(),fpos,f->vel(crs)));
286 }else{
287 }

293void MissileSensor::control(){
294 nl::json cmd=parent.lock()->commands["Sensor"];
295 for(auto&& elem:cmd){
296 std::string cmdName=elem.at("name");
297 if(cmdName=="steering"){
298 auto crs=getParentCRS();
299 Coordinate p(elem.at("estTPos"));
300 Coordinate v(elem.at("estTVel"));
301 //次のperceiveまでの時間を計算して速度を加算
302 double dt=(getNextTick(SimPhase::PERCEIVE,manager->getTickCount())-manager->getTickCountAt(p.getTime()))*manager->getBaseTimeStep();
303 p.setValue(p()+v(p.getCRS())*dt);
304 estTPos=p.transformTo(getParentCRS());
305 }else if(cmdName=="activate"){
306 if(!isActive){
307 target=elem.at("target");
308 isActive=true;
309 }
310 }
311 }
