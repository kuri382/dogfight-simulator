53/**
54 * @brief pを係数とするxの多項式の値を返す。
55 *
56 * @param [in] p 係数のリスト(先頭が最高次係数、末尾が定数項)
57 * @param [in] x 入力変数の値
58 */
59template<typename T, int _Rows>
60T polyval(const Eigen::Matrix<T, _Rows, 1>& p,const T &x){
61 Eigen::Index n=p.size();
62 T xx=1.0;
63 T ret=0;
64 for(int i=0;i<n;++i){
65 ret+=xx*p(n-1-i);
66 xx*=x;
67 }
68 return ret;
69}
70/**
71 * @brief pを係数とするxの多項式の導関数値を返す。
72 *
73 * @param [in] p 係数のリスト(先頭が最高次係数、末尾が定数項)
74 * @param [in] x 入力変数の値
75 */
76template<typename T, int _Rows>
77Eigen::Matrix<T,_Rows,1> polyderiv(const Eigen::Matrix<T,_Rows,1>& p,const T &x){
78 Eigen::Index n=p.size()-1;
79 Eigen::Matrix<T,_Rows,1> ret;
80 for(int i=0;i<n;++i){
81 ret(i)=p(i)*(n-1-i);
82 }
83 return ret;
84}

123/**
124 * @brief 1次元テーブルデータを内挿する際に参照すべきインデックスと重み付け係数を返す。
125 *
126 * @param [in] x テーブルのデータ点のリスト(ソート済)
127 * @param [in] p 内挿したい点の位置
128 * @param [in] clamp trueにすると、p が x の範囲外にある場合、xの端点にclampされる。falseの場合、線形外挿となる。
129 *
130 * @returns \f$ p = x(i) + a (x(i+1)-x(i)) \f$ を満たす\f$(i,a)\f$
131 */
132template<typename VectorType>
133requires (
134 ( traits::matrix<VectorType> )
135 || ( traits::tensor<VectorType>)
136)
137std::pair<Eigen::Index,typename Eigen::internal::traits<VectorType>::Scalar> _search1d(const VectorType &x,const typename Eigen::internal::traits<VectorType>::Scalar &p, bool clamp=false){
138 Eigen::Index lb=0;
139 Eigen::Index ub;
140 if constexpr (traits::matrix<VectorType>){
141 assert(x.rows()==1 || x.cols()==1);
142 ub=x.size()-2;
143 }else if constexpr (traits::tensor<VectorType>){
144 if constexpr (Eigen::internal::traits<VectorType>::NumDimensions>1){
145 auto dims=getTensorDimensions(x);
146 bool found_more_than_one=false;
147 for(auto d: dims){
148 if(d>1){
149 if(found_more_than_one){
150 throw std::runtime_error("asrc::core::_search1d failed. x is not 1-dim tensor.");
151 }else{
152 found_more_than_one=true;
153 }
154 }
155 }
156 return _search1d(x.reshape(Eigen::DSizes<Eigen::Index,1>{getTensorSize(x)}),p);
157 }
158 ub=getTensorSize(x)-2;
159 }else{
160 ub=x.size()-2;
161 }
162 if(ub<0){
163 // x has only one element
164 return {lb,x(lb)};
165 }
166 if(p<x(lb)){
167 if(clamp){
168 return {lb,0};
169 }else{
170 return {lb,(p-x(lb))/(x(lb+1)-x(lb))};
171 }
172 }
173 if(x(ub)<=p){
174 if(clamp){
175 return {ub,1};
176 }else{
177 return {ub,(p-x(ub))/(x(ub+1)-x(ub))};
178 }
179 }
180 Eigen::Index i;
181 while(ub-lb>1){
182 i=(lb+ub)/2;
183 if(x(i)<=p){
184 lb=i;
185 }else{// if(p<x(i)){
186 ub=i;
187 }
188 }
189 return {lb,(p-x(lb))/(x(lb+1)-x(lb))};
190}

192/**
193 * @brief N次元テーブルデータをM個の点において線形内挿する。
194 *
195 * @param [in] point テーブルの各次元のデータ点のリスト(ソート済)
196 * @param [in] table テーブルの値リスト(N次元のテンソル)
197 * @param [in] x 内挿したい点の位置(M×N行列)
198 * @param [in] clamp trueにすると、x が point の範囲外にある場合、 point の端点にclampされる。falseの場合、線形外挿となる。
199 *
200 * @returns 内挿された値を表すM次元ベクトル
201 */
202template<traits::tensor TensorType, traits::matrix MatrixType, typename Scalar=typename Eigen::internal::traits<TensorType>::Scalar>
203requires (
204 std::same_as<Scalar,typename Eigen::internal::traits<MatrixType>::Scalar>
205)
206Eigen::VectorX<Scalar> interpn(const std::vector<Eigen::VectorX<Scalar>>& point,const TensorType &table,const MatrixType &x, bool clamp=false){
207 using TensorTraits=Eigen::internal::traits<TensorType>;
208 using MatrixTraits=Eigen::internal::traits<MatrixType>;
209 constexpr int nDim=TensorTraits::NumDimensions;
210 Eigen::Index nx=x.rows();
211 Eigen::Index nt=getTensorDimensions(table)[nDim-1];
212 Eigen::Index m=getTensorSize(table);
213 Eigen::Tensor<Scalar,3,TensorTraits::Layout,typename TensorTraits::Index> store;
214 for(int d=nDim-1;d>=0;--d){
215 m/=nt;
216 Eigen::Tensor<Scalar,2> tmp2(nx,m);
217 if(d==nDim-1){
218 Eigen::Tensor<Scalar,2,TensorTraits::Layout,typename TensorTraits::Index> first=table.reshape(Eigen::array<Eigen::Index,2>{{m,nt}});
219 for(Eigen::Index i=0;i<nx;++i){
220 std::pair<Eigen::Index,Scalar> idx=_search1d(point[d],x(i,d),clamp);
221 for(Eigen::Index j=0;j<m;++j){
222 tmp2(i,j)=idx.second*(first(j,idx.first+1)-first(j,idx.first))+first(j,idx.first);
223 }
224 }
225 }else{
226 for(Eigen::Index i=0;i<nx;++i){
227 std::pair<Eigen::Index,Scalar> idx=_search1d(point[d],x(i,d),clamp);
228 for(Eigen::Index j=0;j<m;++j){
229 tmp2(i,j)=idx.second*(store(i,j,idx.first+1)-store(i,j,idx.first))+store(i,j,idx.first);
230 }
231 }
232 }
233 nt=(d>=1)?getTensorDimensions(table)[d-1]:1;
234 store=tmp2.reshape(Eigen::array<Eigen::Index,3>{{nx,m/nt,nt}});
235 }
236 typedef Eigen::Map<Eigen::VectorX<Scalar>> maptype;
237 return maptype(store.data(),nx);
238}

240/**
241 * @brief N次元テーブルデータをM個の点において線形内挿した際の偏導関数値を返す。
242 *
243 * @param [in] point テーブルの各次元のデータ点のリスト(ソート済)
244 * @param [in] table テーブルの値リスト(N次元のテンソル)
245 * @param [in] x 内挿したい点の位置(M×N行列)
246 *
247 * @returns 各点の偏導関数値を表すN×M行列(i行目がi次元目の変数による偏導関数値)
248 */
249template<traits::tensor TensorType, traits::matrix MatrixType, typename Scalar=typename Eigen::internal::traits<TensorType>::Scalar>
250requires (
251 std::same_as<Scalar,typename Eigen::internal::traits<MatrixType>::Scalar>
252)
253Eigen::Matrix<Scalar,-1,-1,Eigen::internal::traits<TensorType>::Layout> interpgradn(const std::vector<Eigen::VectorX<Scalar>>& point,const TensorType &table,const MatrixType &x){
254 using TensorTraits=Eigen::internal::traits<TensorType>;
255 using MatrixTraits=Eigen::internal::traits<MatrixType>;
256 constexpr int nDim=TensorTraits::NumDimensions;
257 Eigen::Index nx=x.rows();
258 Eigen::Index nt=getTensorDimensions(table)[nDim-1];
259 Eigen::Index m=getTensorSize(table);
260 Eigen::Tensor<Scalar,4,TensorTraits::Layout,typename TensorTraits::Index> store;
261 for(int d=nDim-1;d>=0;--d){
262 m/=nt;
263 Eigen::Tensor<Scalar,3,TensorTraits::Layout,typename TensorTraits::Index> tmp2(nDim,nx,m);
264 if(d==nDim-1){
265 Eigen::Tensor<Scalar,2,TensorTraits::Layout,typename TensorTraits::Index> first=table.reshape(Eigen::array<Eigen::Index,2>{{m,nt}});
266 for(Eigen::Index i=0;i<nx;++i){
267 std::pair<Eigen::Index,Scalar> idx=_search1d(point[d],x(i,d));
268 for(Eigen::Index j=0;j<m;++j){
269 for(Eigen::Index g=0;g<nDim;++g){
270 if(d==g){
271 tmp2(g,i,j)=(first(j,idx.first+1)-first(j,idx.first))/(point[d](idx.first+1)-point[d](idx.first));
272 }else{
273 tmp2(g,i,j)=idx.second*(first(j,idx.first+1)-first(j,idx.first))+first(j,idx.first);
274 }
275 }
276 }
277 }
278 }else{
279 for(Eigen::Index i=0;i<nx;++i){
280 std::pair<Eigen::Index,Scalar> idx=_search1d(point[d],x(i,d));
281 for(Eigen::Index j=0;j<m;++j){
282 for(Eigen::Index g=0;g<nDim;++g){
283 if(d==g){
284 tmp2(g,i,j)=(store(g,i,j,idx.first+1)-store(g,i,j,idx.first))/(point[d](idx.first+1)-point[d](idx.first));
285 }else{
286 tmp2(g,i,j)=idx.second*(store(g,i,j,idx.first+1)-store(g,i,j,idx.first))+store(g,i,j,idx.first);
287 }
288 }
289 }
290 }
291 }
292 nt=(d>=1)?getTensorDimensions(table)[d-1]:1;
293 store=tmp2.reshape(Eigen::array<Eigen::Index,4>{{nDim,nx,m/nt,nt}});
294 }
295 typedef Eigen::Map<Eigen::Matrix<Scalar,-1,-1,TensorTraits::Layout>> maptype;
296 return maptype(store.data(),nDim,nx);
297}
