553 ret.insert(key);

569bool Factory::checkPermissionImpl(const std::string& callerBaseName, const std::string& targetBaseName,const std::map<std::string, std::set<std::string>>& permissions){
570 std::shared_lock<std::shared_mutex> lock(static_mtx);
571 auto constBaseNameTree=std::as_const(baseNameTree);
572 if(permissions.find(targetBaseName) != permissions.end()){
573 auto&& permission=permissions.at(targetBaseName);
574 if(permission.find(callerBaseName)!=permission.end()){
575 //明示的許可
576 return true;
577 }else if(permission.find("!"+callerBaseName)!=permission.end()){
578 //明示的禁止
579 return false;
580 }else if(permission.find("AnyOthers")!=permission.end()){
581 //暗黙的許可の場合は親を辿って明示的に禁止されていなければ許可
582 std::string nextBaseName=constBaseNameTree.at(callerBaseName);
583 while(nextBaseName!=""){
584 if(!checkPermissionImpl(nextBaseName,targetBaseName,permissions)){
585 //親が禁止されていたら禁止
586 return false;
587 }
588 nextBaseName=constBaseNameTree.at(nextBaseName);
589 }
590 return true;
591 }else if(permission.find("!AnyOthers")!=permission.end()){
592 //暗黙的禁止の場合は親を辿って明示的に許可されていなければ禁止
593 std::string nextBaseName=constBaseNameTree.at(callerBaseName);
594 while(nextBaseName!=""){
595 if(checkPermissionImpl(nextBaseName,targetBaseName,permissions)){
596 //親が許可されていたら許可
597 return true;
598 }
599 nextBaseName=constBaseNameTree.at(nextBaseName);
600 }
601 return false;
602 }else{
603 //指定なしは許可とする
604 return true;
605 }
606 }else{
607 return false;
608 }

611 std::lock_guard<std::shared_mutex> lock(static_mtx);

624 permissionsForGet[baseName]=permissionForGet;
625 }else{
626 permissionsForCreate[baseName]=mergeParentPermission(permissionForCreate,permissionsForCreate.at(parentBaseName));
627 permissionsForGet[baseName]=mergeParentPermission(permissionForGet,permissionsForGet.at(parentBaseName));
628 }
