113Eigen::Matrix<double,3,4> Quaternion::dC1dq() const{//回転行列の1列目の自分自身による微分(3×4行列)

115 ret<<q.w(),q.x(),-q.y(),-q.z(),

155Eigen::Matrix<double,4,3> Quaternion::dqdwi() const{
156 double hw=0.5*q.w();
157 double hx=0.5*q.x();
158 double hy=0.5*q.y();
159 double hz=0.5*q.z();
160 Eigen::Matrix<double,4,3> ret;
161 ret<<-hx,-hy,-hz,
162 hw,hz,-hy,
163 -hz,hw,hx,
164 hy,-hx,hw;
165 return ret;
166}
167Quaternion Quaternion::fromAngle(const Eigen::Vector3d &ax,const double &theta){
168 //EigenではAngleAxisを経由して生成できるが、中身は以下の通り普通の演算。
169 double ha=0.5*theta;
170 Eigen::Vector3d v=sin(ha)*ax;
171 return Quaternion(cos(ha),v[0],v[1],v[2]);
172}
173Quaternion Quaternion::fromBasis(const Eigen::Vector3d &ex,const Eigen::Vector3d &ey,const Eigen::Vector3d &ez){
174 //基底が基準座標系iでの成分表示でex,ey,ezと表されるような座標系bの変換クォータニオンqを計算する。
175 //回転行列Rとしては、ex,ey,ezを列ベクトルとして並べたものになる。
176 Eigen::Matrix3d R;
177 R.block(0,0,3,1)=ex;R.block(0,1,3,1)=ey;R.block(0,2,3,1)=ez;
178 return fromRotationMatrix(R);
179}
180Quaternion Quaternion::fromRotationMatrix(const Eigen::Matrix3d& R){
181 //回転行列から対応するクォータニオンを生成する。
182 //Eigenでは、Quaternionのコンストラクタに3×3行列を入れると得られる。
183 //ベクトル量の成分表示間の変換としては、
184 //ri=R*rb
185 //ri=q*(rb,0)*q~
186 //となる。
187 return Quaternion(Eigen::Quaterniond(R).normalized());
188}
