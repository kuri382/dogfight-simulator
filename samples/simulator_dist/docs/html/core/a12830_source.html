175class SinglizedEnv(GymManager):
176 """! @brief 一つを除いたAgentのactionを内部で処理することによって、シングルエージェント環境として扱うためのラッパークラスの実装例。
177 """
178
179 def __init__(self,context: Dict[str,Any]):
180 """! @brief コンストラクタ
181
182 @details
183 contextの追加要素={
184 "target": Union[Callable[[str],bool],str],
185 "policies": Dict[str,StandalonePolicy],
186 "policyMapper": Optional[Callable[[str],str]],
187 "runUntilAllDone": bool (True if omitted)
188 }
189
190 Args:
191 target (Union[Callable[[str],bool],str]): observation,actionの入出力対象のAgentを特定するための関数。AgentのfullNameを引数にとってboolを返す関数を与える。最初にTrueとなったAgentを対象とみなす。
192 Callableの代わりに文字列を指定した場合、その文字列で始まるかどうか(startswith)を判定条件にすることが可能。
193 policies (Dict[str,StandalonePolicy]): 内部でactionを計算するためのStandalonePolicyオブジェクトのdict。キーはpolicyの名称となる。
194 policyMapper (Callable[[str],str]): AgentのfullNameから使用するpolicyの名称を計算する。省略した場合はfullNameの末尾にあるpolicyName部分が使われる。
195 runUntilAllDone (bool): 対象AgentのdoneがTrueとなっても、done["__all__"]がTrueとなるまで内部でエピソードの計算を継続するか否か。デフォルトはTrue。
196 """
197 target_=context["target"]
198 if(isinstance(target_,str)):
199 self.target=lambda fullName:fullName.startswith(target_)
200 elif(isinstance(target_,Callable)):
201 self.target=target_
202 else:
203 raise ValueError("SinglizedEnv requires Callable[[str],bool] or str for 'target' in context.")
204 self.targetAgent=None
205 self.policies=context.get("policies",{})
206 self.originalPolicyMapper=context.get("policyMapper",getDefaultPolicyMapper())
207 self.policyMapper=self.originalPolicyMapper
208 self.runUntilAllDone=context.get("runUntilAllDone",True)
209 super().__init__(context)
210 def reset(
211 self,
212 *,
213 seed: Optional[int] = None,
214 options: Optional[dict] = None,
215 ) -> Tuple[Any, dict]:
216 self.targetAgent=None
217 self.obs, infos=super().reset(seed=seed,options=options)
218 self.rewards={}
219 self.rewards={k:0.0 for k in self.obs.keys()}
220 self.terminateds={k:False for k in self.obs.keys()}
221 self.truncateds={k:False for k in self.obs.keys()}
222 self.infos={k:None for k in self.obs.keys()}
223 self.terminateds["__all__"]=False
224 self.truncateds["__all__"]=False
225 for p in self.policies.values():
226 p.reset()
227 if(self.isExpertWrapper):
228 self.targetObs=self.targetAgent.imitatorObs

232 def step(self,action) -> Tuple[Any, float, bool, bool, dict]:
233 observation_space=self.get_observation_space()
234 action_space=self.get_action_space()
235 actions={k:self.policies[self.policyMapper(k)].step(

238 self.terminateds[k] or self.truncateds[k],
239 self.infos[k],
240 k,
241 observation_space[k],
242 action_space[k]
243 ) for k,o in self.obs.items() if self.policyMapper(k) in self.policies}
244 actions[self.targetIdentifier]=action

246 if(self.isExpertWrapper):
247 self.targetObs=self.targetAgent.imitatorObs
248 else:
249 self.targetObs=self.obs[self.targetIdentifier]
250 ret=self.targetObs,self.rewards[self.targetIdentifier],self.terminateds[self.targetIdentifier],self.truncateds[self.targetIdentifier],self.infos[self.targetIdentifier]
251 dones={k:self.terminateds[k] or self.truncateds[k] for k in self.terminateds}
252 if(dones[self.targetIdentifier] and not dones["__all__"]):
253 if(self.runUntilAllDone):
254 while(not dones["__all__"]):
255 observation_space=self.get_observation_space()
256 action_space=self.get_action_space()

259 actions.pop(self.targetIdentifier)
260 self.obs,self.rewards,self.terminateds,self.truncateds,self.infos=super().step(actions)
261 dones={k:self.terminateds[k] or self.truncateds[k] for k in self.terminateds}
262 else:
263 self.manager.stopEpisodeExternally()
264 return ret
