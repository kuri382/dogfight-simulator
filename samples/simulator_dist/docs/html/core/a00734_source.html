226 // nlohmann's jsonの標準の変換機能を使用してシリアライズを行う型(*1)以外の型のうち、
227 // NLJSONOutputArchive,NLJSONInputArchiveで読み書きできる型について、
228 // これらのArchiveを介してnlohmann::basic_jsonとの相互変換を行うものとする。
229 //
230 // *1 数値型、文字列型、basic_json、nlohmann::detail::value_t及びそれらを要素とする以下のSTLコンテナ(*2)(コンテナのコンテナ及びコンテナの派生クラスを含む)が除外される。
231 // *2 set,multiset,unordered_set,unordered_multiset,
232 // array,deque,forward_list,list,vector,valarray,

242 >

246 template<typename BasicJsonType, typename TargetType=T>
247 static void to_json(BasicJsonType& j, TargetType&& m) {
248 if constexpr (std::same_as<std::remove_reference_t<BasicJsonType>,ordered_json>){
249 ::asrc::core::util::NLJSONOutputArchive ar(j,true);
250 ar(std::forward<TargetType>(m));
251 }else{
252 ordered_json oj;
253 ::asrc::core::util::NLJSONOutputArchive ar(oj,true);
254 ar(std::forward<TargetType>(m));
255 j=std::move(oj);
256 }
257 }
258 template<typename BasicJsonType, typename TargetType=T>
259 static void from_json(BasicJsonType&& j, TargetType& m) {
260 using Unref=std::remove_reference_t<BasicJsonType>;
261 if constexpr (std::same_as<Unref,ordered_json>){
262 ::asrc::core::util::NLJSONInputArchive ar(std::forward<BasicJsonType>(j),true);
263 ar(std::forward<TargetType>(m));
264 }else{
265 ordered_json oj=std::forward<BasicJsonType>(j);
266 ::asrc::core::util::NLJSONInputArchive ar(oj,true);
267 ar(std::forward<TargetType>(m));
268 }
269 }
