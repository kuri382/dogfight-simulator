91template<class Scalar>
92nl::json getValueFromDistributionImpl(const std::function<Scalar(Scalar,Scalar)>& sampler, const nl::json& first,const nl::json& second){
93 if(first.is_number() && second.is_number()){
94 return sampler(first.get<Scalar>(),second.get<Scalar>());
95 }else if(first.is_array() && second.is_array()){
96 if(first.size()==second.size()){
97 std::size_t size=first.size();
98 nl::json::array_t ret(size);
99 for(std::size_t i=0;i<size;++i){
100 ret[i]=getValueFromDistributionImpl<Scalar>(sampler,first.at(i),second.at(i));
101 }
102 return std::move(ret);
103 }else{
104 throw std::runtime_error("getValueFromJsonR failed. Distribution parameters need to have the same size.");
105 }
106 }else if(first.is_object() && second.is_object()){
107 nl::json::object_t ret;
108 for(auto && [key, first_val] : first.items()){
109 if(second.contains(key)){
110 ret[key]=getValueFromDistributionImpl<Scalar>(sampler,first_val,second.at(key));
111 }else{
112 throw std::runtime_error("getValueFromJsonR failed. Distribution parameters need to have the same keys when they are json object.");
113 }
114 }
115 return std::move(ret);
116 }else{
117 throw std::runtime_error("getValueFromJsonR failed. Distribution parameters need to be number, array, or object to use distribution.");
118 }
119}

121/**
122 * @brief 引数として与えたjsonから値をサンプリングして抽出する。
123 *
124 * @param [in] j 抽出元となるjson
125 * @param [in] gen 乱数生成器
126 *
127 * @details
128 * - j が object(dict)の場合
129 *
130 * "type"キーの値によって分岐する。省略時は"direct"とみなす。
131 *
132 * - "direct"の場合
133 * - j["type"]=="direct" かつ j が "value" キーを持っている場合
134 *
135 * j["value"]を返す。
136 * - それ以外の場合
137 *
138 * objectの各キーの値を再帰的に getValueFromJsonR を呼び出した結果に置き換えて返す。
139 *
140 * - "normal"の場合
141 *
142 * j["mean"]を平均、j["stddev"]を標準偏差とした正規分布からサンプリングして返す。
143 * 平均と標準偏差はスカラーに限らず、形が一致していれば任意のテンソルを与えることができる。
144 *
145 * - "uniform"の場合
146 *
147 * j["low"]を下限、j["high"]を上限とした一様分布でサンプリングして返す。
148 *
149 * j["dtype"]を"float"か"int"で指定(省略時は"float")し、実数と整数のどちらでサンプリングするかを選択可能。
150 *
151 * 上限と下限はスカラーに限らず、形が一致していれば任意のテンソルを与えることができる。
152 *
153 * - "choice"の場合
154 *
155 * j["weights"]で与えた重みにしたがって、j["candidates"]で与えた選択肢から一つを選んで返す。
156 *
157 * - j が array(list)の場合
158 *
159 * arrayの各要素を再帰的に getValueFromJsonR を呼び出した結果に置き換えて返す。
160 *
161 * - j がそれ以外(数値や文字列)の場合
162 *
163 * j をそのまま返す。
164 */
165template<class URBG>
166nl::json getValueFromJsonR(const nl::json& j, URBG& gen){
167 if(j.is_object()){
168 std::string type;
169 if(j.contains("type")){
170 type=j.at("type");
171 }else{
172 type="direct";
173 }
174 if(type=="normal"){
175 return getValueFromDistributionImpl<double>(
176 [&gen](double mean,double stddev){
177 std::normal_distribution<double> dist(mean,stddev);
178 return dist(gen);
179 },
180 j.at("mean"),j.at("stddev")
181 );
182 }else if(type=="uniform"){
183 std::string dtype;
184 if(j.contains("dtype")){
185 dtype=j.at("dtype");
186 }else{
187 dtype="float";
188 }
189 if(dtype=="int"){
190 return getValueFromDistributionImpl<std::int64_t>(
191 [&gen](std::int64_t low,std::int64_t high){
192 std::uniform_int_distribution<std::int64_t> dist(low,high);
193 return dist(gen);
194 },
195 j.at("low"),j.at("high")
196 );
197 }else{
198 return getValueFromDistributionImpl<double>(
199 [&gen](double low,double high){
200 std::uniform_real_distribution<double> dist(low,high);
201 return dist(gen);
202 },
203 j.at("low"),j.at("high")
204 );
205 }
206 }else if(type=="choice"){
207 nl::json weights=j.at("weights");
208 nl::json candidates=j.at("candidates");
209 std::discrete_distribution<std::size_t> dist(weights.begin(),weights.end());
210 return getValueFromJsonR(candidates[dist(gen)],gen);
211 }else{//direct
212 if(j.contains("type") && j.contains("value")){
213 return j.at("value");
214 }else{
215 nl::json ret=nl::json::object();
216 for(auto& e:j.items()){
217 ret[e.key()]=getValueFromJsonR(e.value(),gen);
218 }
219 return std::move(ret);
220 }
221 }
222 }else if(j.is_array()){
223 nl::json ret=nl::json::array();
224 for(auto& e:j){
225 ret.push_back(getValueFromJsonR(e,gen));
226 }
227 return std::move(ret);
228 }else{//direct
229 return j;
230 }
231}
