60struct PYBIND11_EXPORT PyTypeInfo{
61 PyObjectTypeKind kind;
62 bool is_basic_type;
63 pybind11::type self;
64 bool str_key_only;
65 std::vector<PyTypeInfo> sequence; // setもこれを使う
66 std::vector<std::pair<std::string,PyTypeInfo>> mapping_str_key_only;
67 std::vector<std::pair<PyTypeInfo,PyTypeInfo>> mapping_generic;
68 asrc::core::util::DTypes dtype;
69 bool is_col_major;
70 PyTypeInfo();
71 template<traits::pyobject Object>
72 PyTypeInfo(const Object& obj)
73 :kind(get_type_kind_of_object(obj)),
74 is_basic_type(is_basic_type_object(obj)),
75 self(pybind11::type::of(obj))
76 {
77 if constexpr (
78 traits::sequence_like_pyobject<Object>
79 || traits::set_like_pyobject<Object>
80 ){
81 for(auto&& v : obj){
82 sequence.emplace_back(PyTypeInfo(v));
83 }
84 assert(pybind11::len(obj)==(sequence.size()));
85 }else if constexpr (traits::mapping_like_pyobject<Object>){
86 auto asdict=pybind11::reinterpret_borrow<pybind11::dict>(obj);
87 str_key_only=is_all_keys_string_compatible(asdict);
88 if(str_key_only){
89 for(auto&& [k,v] : asdict){
90 mapping_str_key_only.emplace_back(k.template cast<std::string>(),PyTypeInfo(v));
91 }
92 }else{
93 for(auto&& [k,v] : asdict){
94 mapping_generic.emplace_back(PyTypeInfo(k),PyTypeInfo(v));
95 }
96 }
97 assert(pybind11::len(asdict)==(mapping_str_key_only.size()+mapping_generic.size()));
98 }else if constexpr (std::derived_from<Object,pybind11::array>){
99 dtype=asrc::core::util::get_dtype_enum(obj);
100 is_col_major=asrc::core::util::is_col_major(obj);
101 }else{
102 if(
103 kind==PyObjectTypeKind::set
104 || kind==PyObjectTypeKind::mutable_set
105 || kind==PyObjectTypeKind::sequence
106 || kind==PyObjectTypeKind::mutable_sequence
107 ){
108 for(auto&& v : obj){
109 sequence.emplace_back(PyTypeInfo(v));
110 }
111 assert(pybind11::len(obj)==(sequence.size()));
112 }
113 if(
114 kind==PyObjectTypeKind::mapping
115 || kind==PyObjectTypeKind::mutable_mapping
116 ){
117 auto asdict=pybind11::reinterpret_borrow<pybind11::dict>(obj);
118 str_key_only=is_all_keys_string_compatible(asdict);
119 if(str_key_only){
120 for(auto&& [k,v] : asdict){
121 mapping_str_key_only.emplace_back(k.template cast<std::string>(),PyTypeInfo(v));
122 }
123 }else{
124 for(auto&& [k,v] : asdict){
125 mapping_generic.emplace_back(PyTypeInfo(k),PyTypeInfo(v));
126 }
127 }
128 assert(pybind11::len(asdict)==(mapping_str_key_only.size()+mapping_generic.size()));
129 }
130 if(kind==PyObjectTypeKind::numpy_array){
131 auto arr=pybind11::reinterpret_borrow<pybind11::array>(obj);
132 dtype=asrc::core::util::get_dtype_enum(arr);
133 is_col_major=asrc::core::util::is_col_major(arr);
134 }
135 }
136 }
137 template<class Archive>
138 inline void serialize(Archive& archive){
139 archive(cereal::make_nvp("kind",kind));
140 archive(cereal::make_nvp("is_basic_type",is_basic_type));
141 if(!is_basic_type){
142 if constexpr (traits::output_archive<Archive>){
143 auto cloudpickle=pybind11::module::import("cloudpickle");
144 auto type_string=cloudpickle.attr("dumps")(self).cast<std::string>();
145
146 std::uint32_t nameid;
147 if constexpr (traits::text_archive<Archive>){
148 type_string=cereal::base64::encode(
149 reinterpret_cast<const unsigned char *>(type_string.c_str()),
150 type_string.size()
151 );
152 }
153 // output archiveではpolymorphic_idがconst char*をキーとしたmapで管理されているため、
154 // registerPolymorphicTypeに一時変数のtype_string.c_str()を渡すと上手く動かない(同じ文字列でもアドレスが異なるため。)
155 // この対策として、PyTypeInfoのstaticメンバとしてtype_stringを保持しておくことでポインタを固定する。
156 auto it=PyTypeInfo::typeStringCache.find(type_string);
157 if(it==PyTypeInfo::typeStringCache.end()){
158 it=PyTypeInfo::typeStringCache.insert(type_string).first;
159 }
160 nameid = archive.registerPolymorphicType((*it).c_str());
161 archive( CEREAL_NVP_("polymorphic_id", nameid) );
162 if( nameid & cereal::detail::msb_32bit ){
163 archive( CEREAL_NVP_("polymorphic_name", type_string) );
164 }
165 }else{
166 std::uint32_t nameid;
167 archive( CEREAL_NVP_("polymorphic_id", nameid) );
168 std::string loaded_string;
169 if( nameid & cereal::detail::msb_32bit ){
170 archive( CEREAL_NVP_("polymorphic_name", loaded_string) );
171 archive.registerPolymorphicName(nameid, loaded_string);
172 }else{
173 loaded_string = archive.getPolymorphicName(nameid);
174 }
175 auto cloudpickle=pybind11::module::import("cloudpickle");
176 if constexpr (traits::text_archive<Archive>){
177 self=cloudpickle.attr("loads")(pybind11::bytes(cereal::base64::decode(loaded_string)));
178 }else{
179 self=cloudpickle.attr("loads")(pybind11::bytes(loaded_string));
180 }
181 }
182 }else{
183 if constexpr (traits::input_archive<Archive>){
184 switch(kind){
185 case PyObjectTypeKind::none :
186 self=pybind11::type::of(pybind11::none());
187 break;
188 case PyObjectTypeKind::type :
189 self=pybind11::type::of(pybind11::handle((PyObject*)(&PyBool_Type)));
190 break;
191 case PyObjectTypeKind::boolean :
192 self=pybind11::type::of(pybind11::bool_());
193 break;
194 case PyObjectTypeKind::integer :
195 self=pybind11::type::of(pybind11::int_());
196 break;
197 case PyObjectTypeKind::float_ :
198 self=pybind11::type::of(pybind11::float_());
199 break;
200 case PyObjectTypeKind::bytearray :
201 self=pybind11::type::of(pybind11::bytearray());
202 break;
203 case PyObjectTypeKind::bytes :
204 self=pybind11::type::of(pybind11::bytes());
205 break;
206 case PyObjectTypeKind::string :
207 self=pybind11::type::of(pybind11::str());
208 break;
209 case PyObjectTypeKind::set :
210 self=pybind11::type::of(pybind11::frozenset(pybind11::set()));
211 break;
212 case PyObjectTypeKind::mutable_set :
213 self=pybind11::type::of(pybind11::set());
214 break;
215 case PyObjectTypeKind::sequence :
216 self=pybind11::type::of(pybind11::tuple());
217 break;
218 case PyObjectTypeKind::mutable_sequence :
219 self=pybind11::type::of(pybind11::list());
220 break;
221 case PyObjectTypeKind::mutable_mapping :
222 self=pybind11::type::of(pybind11::dict());
223 break;
224 case PyObjectTypeKind::numpy_array :
225 self=pybind11::type::of(pybind11::array());
226 break;
227 default :
228 break;
229 }
230 }
231 }
232 if(
233 kind==PyObjectTypeKind::set
234 || kind==PyObjectTypeKind::mutable_set
235 || kind==PyObjectTypeKind::sequence
236 || kind==PyObjectTypeKind::mutable_sequence
237 ){
238 archive(cereal::make_nvp("sequence",sequence));
239 }else{
240 if constexpr (traits::input_archive<Archive>){
241 sequence.clear();
242 }
243 }
244 if(
245 kind==PyObjectTypeKind::mapping
246 || kind==PyObjectTypeKind::mutable_mapping
247 ){
248 archive(cereal::make_nvp("str_key_only",str_key_only));
249 if(str_key_only){
250 archive(cereal::make_nvp("mapping_str_key_only",mapping_str_key_only));
251 }else{
252 archive(cereal::make_nvp("mapping_generic",mapping_generic));
253 }
254 }else{
255 if constexpr (traits::input_archive<Archive>){
256 mapping_str_key_only.clear();
257 mapping_generic.clear();
258 }
259 }
260 if(kind==PyObjectTypeKind::numpy_array){
261 archive(cereal::make_nvp("dtype",dtype));
262 archive(cereal::make_nvp("is_col_major",is_col_major));
263 }
264 }

287template<traits::pyobject Object>
288bool PYBIND11_EXPORT is_basic_type_object(const Object& obj){
289 if(obj.ptr()==nullptr || obj.is_none()){
290 return true;
291 }
292 if constexpr (std::derived_from<Object,pybind11::none>){
293 return true;
294 }else if constexpr (std::derived_from<Object,pybind11::type>){
295 return true;
296 }else if constexpr (std::derived_from<Object,pybind11::bool_>){
297 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyBool_Type);
298 }else if constexpr (std::derived_from<Object,pybind11::int_>){
299 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyLong_Type);
300 }else if constexpr (std::derived_from<Object,pybind11::float_>){
301 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyFloat_Type);
302 }else if constexpr (std::derived_from<Object,pybind11::bytearray>){
303 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyByteArray_Type);
304 }else if constexpr (std::derived_from<Object,pybind11::bytes>){
305 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyBytes_Type);
306 }else if constexpr (std::derived_from<Object,pybind11::str>){
307 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyUnicode_Type);
308 }else if constexpr (std::derived_from<Object,pybind11::array>){
309 return pybind11::type::of(obj).ptr() == (PyObject*)(pybind11::detail::npy_api::get().PyArray_Type_);
310 }else if constexpr (std::derived_from<Object,pybind11::set>){
311 return pybind11::type::of(obj).ptr() == (PyObject*)(&PySet_Type);
312 }else if constexpr (std::derived_from<Object,pybind11::frozenset>){
313 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyFrozenSet_Type);
314 }else if constexpr (std::derived_from<Object,pybind11::list>){
315 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyList_Type);
316 }else if constexpr (std::derived_from<Object,pybind11::tuple>){
317 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyTuple_Type);
318 }else if constexpr (std::derived_from<Object,pybind11::sequence>){
319 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyList_Type) || pybind11::type::of(obj).ptr() == (PyObject*)(&PyTuple_Type);
320 }else if constexpr (std::derived_from<Object,pybind11::dict>){
321 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyDict_Type);
322 }else{
323 // それ以外はisinstanceで判別
324 if(PyType_Check(obj.ptr())){
325 return true;
326 }else if(pybind11::isinstance<nlohmann::json>(obj)){
327 return pybind11::type::of(obj).is(pybind11::type::of<nlohmann::json>());
328 }else if(pybind11::isinstance<nlohmann::ordered_json>(obj)){
329 return pybind11::type::of(obj).is(pybind11::type::of<nlohmann::ordered_json>());
330 }else if(
331 pybind11::hasattr(obj,"_allow_cereal_serialization_in_cpp")
332 && obj.attr("_allow_cereal_serialization_in_cpp").template cast<bool>()
333 ){
334 return false;
335 }else if(pybind11::isinstance<pybind11::bool_>(obj)){
336 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyBool_Type);
337 }else if(pybind11::isinstance<pybind11::int_>(obj)){
338 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyLong_Type);
339 }else if(pybind11::isinstance<pybind11::float_>(obj)){
340 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyFloat_Type);
341 }else if(pybind11::isinstance<pybind11::bytearray>(obj)){
342 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyByteArray_Type);
343 }else if(pybind11::isinstance<pybind11::bytes>(obj)){
344 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyBytes_Type);
345 }else if(pybind11::isinstance<pybind11::str>(obj)){
346 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyUnicode_Type);
347 }else if(pybind11::isinstance<pybind11::array>(obj)){
348 return pybind11::type::of(obj).ptr() == (PyObject*)(pybind11::detail::npy_api::get().PyArray_Type_);
349 }else if(pybind11::isinstance<pybind11::set>(obj)){
350 return pybind11::type::of(obj).ptr() == (PyObject*)(&PySet_Type);
351 }else if(pybind11::isinstance<pybind11::frozenset>(obj)){
352 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyFrozenSet_Type);
353 }else if(pybind11::isinstance<pybind11::list>(obj)){
354 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyList_Type);
355 }else if(pybind11::isinstance<pybind11::tuple>(obj)){
356 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyTuple_Type);
357 }else if(pybind11::isinstance<pybind11::dict>(obj)){
358 return pybind11::type::of(obj).ptr() == (PyObject*)(&PyDict_Type);
359 }else{
360 return false;
361 }

364template<traits::pyobject Object>
365PyObjectTypeKind PYBIND11_EXPORT get_type_kind_of_object(const Object& obj){
366 // まずはC++側の型で判別可能なら中身がnoneでも型情報から判定
367 if constexpr (std::derived_from<Object,pybind11::none>){
368 return PyObjectTypeKind::none;
369 }else if constexpr (std::derived_from<Object,pybind11::type>){
370 return PyObjectTypeKind::type;
371 }else if constexpr (std::derived_from<Object,pybind11::bool_>){
372 return PyObjectTypeKind::boolean;
373 }else if constexpr (std::derived_from<Object,pybind11::int_>){
374 return PyObjectTypeKind::integer;
375 }else if constexpr (std::derived_from<Object,pybind11::float_>){
376 return PyObjectTypeKind::float_;
377 }else if constexpr (std::derived_from<Object,pybind11::bytearray>){
378 return PyObjectTypeKind::bytearray;
379 }else if constexpr (std::derived_from<Object,pybind11::bytes>){
380 return PyObjectTypeKind::bytes;
381 }else if constexpr (std::derived_from<Object,pybind11::str>){
382 return PyObjectTypeKind::string;
383 }else if constexpr (std::derived_from<Object,pybind11::array>){
384 return PyObjectTypeKind::numpy_array;
385 }else if constexpr (std::derived_from<Object,pybind11::set>){
386 return PyObjectTypeKind::mutable_set;
387 }else if constexpr (std::derived_from<Object,pybind11::frozenset>){
388 return PyObjectTypeKind::set;
389 }else if constexpr (std::derived_from<Object,pybind11::list>){
390 return PyObjectTypeKind::mutable_sequence;
391 }else if constexpr (std::derived_from<Object,pybind11::tuple>){
392 return PyObjectTypeKind::sequence;
393 }else if constexpr (std::derived_from<Object,pybind11::sequence>){
394 if(pybind11::type::of(obj).ptr() == (PyObject*)(&PyTuple_Type)){
395 return PyObjectTypeKind::sequence;
396 }else{
397 return PyObjectTypeKind::mutable_sequence;
398 }
399 }else if constexpr (std::derived_from<Object,pybind11::dict>){
400 return PyObjectTypeKind::mutable_mapping;
401 }else{
402 // それ以外はisinstanceで判別
403 if(obj.ptr()==nullptr || obj.is_none()){
404 return PyObjectTypeKind::none;
405 }else if(PyType_Check(obj.ptr())){
406 return PyObjectTypeKind::type;
407 }else if(pybind11::isinstance<nlohmann::json>(obj)){
408 return PyObjectTypeKind::nl_json;
409 }else if(pybind11::isinstance<nlohmann::ordered_json>(obj)){
410 return PyObjectTypeKind::nl_ordered_json;
411 }else if(
412 pybind11::hasattr(obj,"_allow_cereal_serialization_in_cpp")
413 && obj.attr("_allow_cereal_serialization_in_cpp").template cast<bool>()
414 ){
415 return PyObjectTypeKind::cpp_type;
416 }else if(PyTypeInfo::isBoolean(pybind11::type::of(obj))){
417 return PyObjectTypeKind::boolean;
418 }else if(PyTypeInfo::isIntegral(pybind11::type::of(obj))){
419 return PyObjectTypeKind::integer;
420 }else if(PyTypeInfo::isReal(pybind11::type::of(obj))){
421 return PyObjectTypeKind::float_;
422 }else if(pybind11::isinstance<pybind11::bytearray>(obj)){
423 return PyObjectTypeKind::bytearray;
424 }else if(pybind11::isinstance<pybind11::bytes>(obj)){
425 return PyObjectTypeKind::bytes;
426 }else if(pybind11::isinstance<pybind11::str>(obj)){
427 return PyObjectTypeKind::string;
428 }else if(pybind11::isinstance<pybind11::array>(obj)){
429 return PyObjectTypeKind::numpy_array;
430 }else if(PyTypeInfo::isMutableSet(pybind11::type::of(obj))){
431 return PyObjectTypeKind::mutable_set;
432 }else if(PyTypeInfo::isSet(pybind11::type::of(obj))){
433 return PyObjectTypeKind::set;
434 }else if(PyTypeInfo::isMutableSequence(pybind11::type::of(obj))){
435 return PyObjectTypeKind::mutable_sequence;
436 }else if(PyTypeInfo::isSequence(pybind11::type::of(obj))){
437 return PyObjectTypeKind::sequence;
438 }else if(PyTypeInfo::isMutableMapping(pybind11::type::of(obj))){
439 return PyObjectTypeKind::mutable_mapping;
440 }else{
441 return PyObjectTypeKind::other;
442 }
443 }
