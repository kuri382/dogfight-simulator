20// concept version of nlohmann json's traits

22template<typename T>
23using is_basic_json = nl::detail::is_basic_json<get_true_value_type_t<T>>;
24template<typename T>
25concept basic_json =is_basic_json<T>::value;
26
27
28template<typename BasicJsonType, typename T>
29concept has_to_json_by_default = requires (BasicJsonType& j, T&& val) {
30 ::nl::to_json(j, std::forward<T>(val));
31};
32template<typename BasicJsonType, typename T>
33concept has_from_json_by_default = requires (BasicJsonType&& j, T& val) {
34 ::nl::from_json(std::forward<BasicJsonType>(j), val);
35};
36template<typename BasicJsonType, typename T>
37concept has_non_default_from_json_by_default = requires (BasicJsonType&& j, nl::detail::identity_tag<T>& tag) {
38 ::nl::from_json(std::forward<BasicJsonType>(j), tag);
39};
40template<typename BasicJsonType, typename T>
41concept is_jsonable_by_default = has_to_json_by_default<BasicJsonType,T> && (has_from_json_by_default<BasicJsonType,T> || has_non_default_from_json_by_default<BasicJsonType,T>);
42
43template<typename BasicJsonType, typename T>
44concept has_to_json = std::conjunction_v<
45 std::negation<nl::detail::is_basic_json<get_true_value_type_t<T>>>,
46 nl::detail::has_to_json<BasicJsonType,nl::detail::uncvref_t<T>>
47>;
48template<typename BasicJsonType, typename T>
49concept has_from_json = std::conjunction_v<
50 std::negation<nl::detail::is_basic_json<get_true_value_type_t<T>>>,
51 nl::detail::has_from_json<BasicJsonType,nl::detail::uncvref_t<T>>
52>;
53template<typename BasicJsonType, typename T>
54concept has_non_default_from_json = std::conjunction_v<
55 std::negation<nl::detail::is_basic_json<get_true_value_type_t<T>>>,
56 nl::detail::has_non_default_from_json<BasicJsonType,nl::detail::uncvref_t<T>>
57>;
58template<typename BasicJsonType, typename T>
59concept is_jsonable = basic_json<T> || (has_to_json<BasicJsonType,T> && (has_from_json<BasicJsonType,T> || has_non_default_from_json<BasicJsonType,T>));
