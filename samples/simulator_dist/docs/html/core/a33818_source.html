54 std::map<std::string,double> atmosphere(const double &h){
55 std::map<std::string,double> ret;
56 double gamma=1.4;
57 double R=287.0531;
58 double HAL[]={0.,11000.,20000.0,32000.0,47000.0,51000.0,71000.0,80000.0};
59 double LR[]={-0.0065,0.0,0.001,0.0028,0.0,-0.0028,-0.002,0.0};
60 double T0[]={288.15,216.65,216.65,228.65,270.65,270.65,214.65,196.65};
61 double P0[]={101325.0,22632.0,5474.9,868.02,110.91,66.939,3.9564,0.88627};
62 int k=7;
63 for(int i=1;i<=7;++i){
64 if(h<HAL[i]){
65 k=i-1;
66 break;
67 }
68 }
69 double T=T0[k]+LR[k]*(h-HAL[k]);
70 double a=sqrt(T*gamma*R);
71 double P,rho,dRhodH,dPdH;
72 if(LR[k]!=0.0){
73 double factor=gravity/(-LR[k]*R);
74 P=P0[k]*pow(T/T0[k],factor);
75 dPdH=P0[k]*factor/T0[k]*pow(T/T0[k],factor-1.);
76 }else{
77 P=P0[k]*exp(gravity*(HAL[k]-h)/(R*T0[k]));
78 dPdH=P*(-gravity/R/T0[k]);
79 }
80 rho=P/(R*T);
81 dRhodH=(dPdH/T-LR[k]*P/(T*T))/R;
82 double bs=1.458e-6;//[kg/m/s/√k]
83 double S=110.4;//[K]
84 double nu=bs*pow(T,1.5)/(T+S);
85 ret["T"]=T;
86 ret["a"]=a;
87 ret["P"]=P;
88 ret["rho"]=rho;
89 ret["nu"]=nu;
90 ret["dRhodH"]=dRhodH;
91 ret["dTdH"]=LR[k];
92 ret["dPdH"]=dPdH;
93 ret["dadH"]=0.5*gamma*R*LR[k]/a;
94 return ret;
95 }
