21void PureFlatCRS::initialize(){
22 BaseType::initialize();
23 // 座標軸の設定
24 if(instanceConfig.contains("axisOrder")){
25 axisOrder=instanceConfig.at("axisOrder");
26 }else if(modelConfig.contains("axisOrder")){
27 axisOrder=modelConfig.at("axisOrder");
28 }else{
29 axisOrder="NED";
30 }
31 std::transform(axisOrder.begin(),axisOrder.end(),axisOrder.begin(),
32 [](unsigned char c){ return std::toupper(c);}
33 );
34 assert(checkTopocentricCartesianAxisOrder(axisOrder));
35
36 //原点の設定
37 givenByEllipsoidalOrigin=false;
38 givenByCartesianOrigin=false;
39 ellipsoidalOrigin=Eigen::Vector3d::Zero(); // in lat[deg], lon[deg], alt[m]
40 cartesianOrigin=Eigen::Vector3d::Zero(); // in X[m], Y[m], Z[m]
41 if(instanceConfig.contains("cartesianOrigin")){
42 cartesianOrigin=instanceConfig.at("cartesianOrigin");
43 givenByCartesianOrigin=true;
44 }else if(modelConfig.contains("cartesianOrigin")){
45 cartesianOrigin=modelConfig.at("cartesianOrigin");
46 givenByCartesianOrigin=true;
47 }else if(instanceConfig.contains("ellipsoidalOrigin")){
48 ellipsoidalOrigin=instanceConfig.at("ellipsoidalOrigin");
49 givenByEllipsoidalOrigin=true;
50 }else if(modelConfig.contains("ellipsoidalOrigin")){
51 ellipsoidalOrigin=modelConfig.at("ellipsoidalOrigin");
52 givenByEllipsoidalOrigin=true;
53 }
54}

163Quaternion PureFlatCRS::getQuaternionTo(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS) const{
164 if(!isValid()){validate();}
165 if(!dstCRS){
166 throw std::runtime_error("dstCRS is nullptr!");
167 }
168 if(this==std::dynamic_pointer_cast<PureFlatCRS>(dstCRS).get()){
169 return Quaternion(1,0,0,0);
170 }
171 if(!isTransformableTo(dstCRS)){
172 throw std::runtime_error(getFactoryClassName()+" can't be transformed to "+dstCRS->getFactoryClassName());
173 }
174 if(auto derived=std::dynamic_pointer_cast<DerivedCRS>(dstCRS)){
175 if(this!=std::dynamic_pointer_cast<PureFlatCRS>(derived->getNonDerivedBaseCRS()).get()){
176 throw std::runtime_error("PureFlatCRS can calculate quaternion only to/from AffineCRS or TopocentricCRS whose non-derived base is it.");
177 }
178 if(derived->isProjected()){
179 throw std::runtime_error("Quaternion between a projected and a non-projected CRS cannot be calculated.");
180 }
181 return derived->getQuaternionFrom(location,time,non_const_this());
182 }else{
183 throw std::runtime_error("PureFlatCRS can calculate quaternion only to/from AffineCRS or TopocentricCRS whose non-derived base is it.");
184 }

209double PureFlatCRS::getHeight(const Eigen::Vector3d& location, const Time& time) const{
210 //楕円体高度(幾何高度)を返す。ellipsoid (geometric) height
211 if(!isValid()){validate();}
212 for(int i=0; i<3; ++i){
213 if(axisOrder.at(i)=='U'){
214 return location(i);
215 }else if(axisOrder.at(i)=='D'){
216 return -location(i);
217 }
218 }
219 //コンストラクタでチェックしているので派生クラス等で後から書き換えていなければここには到達しない。
220 throw std::runtime_error("'"+axisOrder+"' is an invalid axisOrder.");
221}
