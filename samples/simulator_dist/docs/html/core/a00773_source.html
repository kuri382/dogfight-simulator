67 Vertex add_instance(const std::shared_ptr<Instance>& instance){
68 assert(instance);
69 auto eid=identify(instance);
70 assert(reverse_map.find(eid)==reverse_map.end());
71
72 auto ret=add_vertex({instance},graph);
73 reverse_map[eid]=ret;
74 vertex_to_index[ret]=topo_order.size();
75 topo_order.push_back(ret);
76 return ret;

79 void remove_instance(const std::shared_ptr<Instance>& instance){
80 assert(instance);
81 auto eid=identify(instance);
82 auto found=reverse_map.find(eid);
83 assert(found!=reverse_map.end());
84 remove_vertex(found->second,graph);
85 auto affected=topo_order.erase(std::find(topo_order.begin(),topo_order.end(),found->second));
86 reverse_map.erase(found);
87 for(auto&& v : std::ranges::subrange(affected,topo_order.end())){
88 vertex_to_index[v]--;
89 }
90 }

142 void reorder(){
143 topo_order.clear();
144
145 std::map<Vertex,boost::default_color_type> color_map_storage;
146 boost::associative_property_map<decltype(color_map_storage)> cmap(color_map_storage);
147 boost::topological_sort(graph,std::front_inserter(topo_order),color_map(cmap));
148 sorted=true;
149 std::size_t i=0;
150 for(auto&& v : topo_order){
151 vertex_to_index[v]=i;
152 ++i;
153 }
154 }

176 void online_reorder(const Vertex& src,const Vertex& tgt){
177 // ソート済の状態から、srcからtgtへのEdgeを1本追加して再ソートを行う。
178 //
179 // [Kavitha 2007] Kavitha, T. et al. "Faster Algorithms for Online Topological Ordering." arXiv:0711.0251
180 if(vertex_to_index[src]<vertex_to_index[tgt]){
181 sorted=true;
182 return;
183 }
184
185 using i_v_pair=std::pair<std::size_t,Vertex>;
186 auto i_less=[](const i_v_pair& lhs, const i_v_pair& rhs){
187 return lhs.first<rhs.first;
188 };
189 auto i_greater=[](const i_v_pair& lhs, const i_v_pair& rhs){
190 return lhs.first>rhs.first;
191 };
192 std::set<std::size_t> i_set;
193 std::set<i_v_pair,decltype(i_less)> anc(i_less),des(i_less);
194
195 boost::heap::fibonacci_heap<
196 i_v_pair,
197 boost::heap::compare<decltype(i_less)>
198 > Fu(i_less);//最大ヒープ
199 boost::heap::fibonacci_heap<
200 i_v_pair,
201 boost::heap::compare<decltype(i_greater)>
202 > Fv(i_greater);//最小ヒープ
203 std::unordered_set<Vertex> Fu_visited_set;
204 std::unordered_set<Vertex> Fv_visited_set;
205
206 std::size_t ma=in_degree(src,graph);
207 std::size_t md=out_degree(tgt,graph);
208
209 double logn=log(topo_order.size());
210
211 i_v_pair x{vertex_to_index[src],src}, y{vertex_to_index[tgt],tgt};
212 anc.insert(x);
213 i_set.insert(x.first);
214 des.insert(y);
215 i_set.insert(y.first);
216
217 std::set<i_v_pair,decltype(i_less)> toBeShifted1(i_less),toBeShifted2(i_less);
218
219 while(true){
220 bool balanced = ma<=md ? md-ma<=anc.size()*logn : ma-md<=des.size()*logn;
221 if(balanced){
222 // visit(x)
223 for(auto&& w : get_predecessors(x.second)){
224 if(!Fu_visited_set.contains(w)){
225 if(Fv_visited_set.contains(w)){
226 std::ostringstream oss;
227 oss<<"cycle detected while adding an edge ("<<stringify(get_instance(src))<<" -> "<<stringify(get_instance(tgt))<<"). cycle=[";
228 oss<<stringify(get_instance(w))<<" -> ";
229 oss<<stringify(get_instance(x.second))<<" -> { ";
230 for(auto&& fu_visited : Fu_visited_set){
231 oss<<stringify(get_instance(fu_visited))<<" , ";
232 }
233 oss<<" } -> { ";
234 for(auto&& fv_visited : Fv_visited_set){
235 oss<<stringify(get_instance(fv_visited))<<" , ";
236 }
237 oss<<" }";
238 oss<<"]";
239 throw std::runtime_error(oss.str());
240 }
241 Fu.push({vertex_to_index[w],w});
242 Fu_visited_set.insert(w);
243 }
244 }
245 // visit(y)
246 for(auto&& w : get_successors(y.second)){
247 if(!Fv_visited_set.contains(w)){
248 if(Fu_visited_set.contains(w)){
249 std::ostringstream oss;
250 oss<<"cycle detected while adding an edge ("<<stringify(get_instance(src))<<" -> "<<stringify(get_instance(tgt))<<"). cycle=[";
251 oss<<stringify(get_instance(w))<<" -> ";
252 oss<<stringify(get_instance(x.second))<<" -> { ";
253 for(auto&& fu_visited : Fu_visited_set){
254 oss<<stringify(get_instance(fu_visited))<<" , ";
255 }
256 oss<<" } -> { ";
257 for(auto&& fv_visited : Fv_visited_set){
258 oss<<stringify(get_instance(fv_visited))<<" , ";
259 }
260 oss<<" }";
261 oss<<"]";
262 throw std::runtime_error(oss.str());
263 }
264 Fv.push({vertex_to_index[w],w});
265 Fv_visited_set.insert(w);
266 }
267 }
268 // extract-max(Fu)
269 if(Fu.size()>0){
270 x=Fu.top();
271 anc.insert(x);
272 i_set.insert(x.first);
273 Fu.pop();
274 }else{
275 // done
276 // yの直前にancとdesを挿入
277 for(std::size_t i=*i_set.begin()+1;i<y.first;++i){
278 if(!i_set.contains(i)){
279 i_v_pair z{i,topo_order[i]};
280 toBeShifted1.insert(z);
281 i_set.insert(i);
282 }
283 }
284 for(std::size_t i=y.first+1;i<topo_order.size();++i){
285 if(!i_set.contains(i)){
286 i_v_pair z{i,topo_order[i]};
287 toBeShifted2.insert(z);
288 i_set.insert(i);
289 }
290 }
291 break;
292 }
293 // extract-min(Fv)
294 if(Fv.size()>0){
295 y=Fv.top();
296 des.insert(y);
297 i_set.insert(y.first);
298 Fv.pop();
299 }else{
300 // done
301 // xの直後にancとdesを挿入
302 for(std::size_t i=*i_set.begin()+1;i<x.first;++i){
303 if(!i_set.contains(i)){
304 i_v_pair z{i,topo_order[i]};
305 toBeShifted1.insert(z);
306 i_set.insert(i);
307 }
308 }
309 for(std::size_t i=x.first+1;i<topo_order.size();++i){
310 if(!i_set.contains(i)){
311 i_v_pair z{i,topo_order[i]};
312 toBeShifted2.insert(z);
313 i_set.insert(i);
314 }
315 }
316 break;
317 }
318 if(x.first<y.first){
319 // done
320 // xの直後にancとdesを挿入
321 for(std::size_t i=*i_set.begin()+1;i<x.first;++i){
322 if(!i_set.contains(i)){
323 i_v_pair z{i,topo_order[i]};
324 toBeShifted1.insert(z);
325 i_set.insert(i);
326 }
327 }
328 for(std::size_t i=x.first+1;i<topo_order.size();++i){
329 if(!i_set.contains(i)){
330 i_v_pair z{i,topo_order[i]};
331 toBeShifted2.insert(z);
332 i_set.insert(i);
333 }
334 }
335 break;
336 }else{
337 ma+=in_degree(x.second,graph);
338 md+=out_degree(y.second,graph);
339 }
340 }else if(ma<md){
341 // visit(x)
342 for(auto&& w : get_predecessors(x.second)){
343 if(!Fu_visited_set.contains(w)){
344 if(Fv_visited_set.contains(w)){
345 std::ostringstream oss;
346 oss<<"cycle detected while adding an edge ("<<stringify(get_instance(src))<<" -> "<<stringify(get_instance(tgt))<<"). cycle=[";
347 oss<<stringify(get_instance(w))<<" -> ";
348 oss<<stringify(get_instance(x.second))<<" -> { ";
349 for(auto&& fu_visited : Fu_visited_set){
350 oss<<stringify(get_instance(fu_visited))<<" , ";
351 }
352 oss<<" } -> { ";
353 for(auto&& fv_visited : Fv_visited_set){
354 oss<<stringify(get_instance(fv_visited))<<" , ";
355 }
356 oss<<" }";
357 oss<<"]";
358 throw std::runtime_error(oss.str());
359 }
360 Fu.push({vertex_to_index[w],w});
361 Fu_visited_set.insert(w);
362 }
363 }
364 // extract-max(Fu)
365 if(Fu.size()>0){
366 x=Fu.top();
367 anc.insert(x);
368 i_set.insert(x.first);
369 Fu.pop();
370 }else{
371 // done
372 // yの直前にancとdesを挿入
373 for(std::size_t i=*i_set.begin()+1;i<y.first;++i){
374 if(!i_set.contains(i)){
375 i_v_pair z{i,topo_order[i]};
376 toBeShifted1.insert(z);
377 i_set.insert(i);
378 }
379 }
380 for(std::size_t i=y.first+1;i<topo_order.size();++i){
381 if(!i_set.contains(i)){
382 i_v_pair z{i,topo_order[i]};
383 toBeShifted2.insert(z);
384 i_set.insert(i);
385 }
386 }
387 break;
388 }
389 if(x.first<y.first){
390 // done
391 // xの直後にancとdesを挿入
392 for(std::size_t i=*i_set.begin()+1;i<x.first;++i){
393 if(!i_set.contains(i)){
394 i_v_pair z{i,topo_order[i]};
395 toBeShifted1.insert(z);
396 i_set.insert(i);
397 }
398 }
399 for(std::size_t i=x.first+1;i<topo_order.size();++i){
400 if(!i_set.contains(i)){
401 i_v_pair z{i,topo_order[i]};
402 toBeShifted2.insert(z);
403 i_set.insert(i);
404 }
405 }
406 break;
407 }else{
408 ma+=in_degree(x.second,graph);
409 }
410 }else{
411 // visit(y)
412 for(auto&& w : get_successors(y.second)){
413 if(!Fv_visited_set.contains(w)){
414 if(Fu_visited_set.contains(w)){
415 std::ostringstream oss;
416 oss<<"cycle detected while adding an edge ("<<stringify(get_instance(src))<<" -> "<<stringify(get_instance(tgt))<<"). cycle=[";
417 oss<<stringify(get_instance(w))<<" -> ";
418 oss<<stringify(get_instance(x.second))<<" -> { ";
419 for(auto&& fu_visited : Fu_visited_set){
420 oss<<stringify(get_instance(fu_visited))<<" , ";
421 }
422 oss<<" } -> { ";
423 for(auto&& fv_visited : Fv_visited_set){
424 oss<<stringify(get_instance(fv_visited))<<" , ";
425 }
426 oss<<" }";
427 oss<<"]";
428 throw std::runtime_error(oss.str());
429 }
430 Fv.push({vertex_to_index[w],w});
431 Fv_visited_set.insert(w);
432 }
433 }
434 // extract-min(Fv)
435 if(Fv.size()>0){
436 y=Fv.top();
437 des.insert(y);
438 i_set.insert(y.first);
439 Fv.pop();
440 }else{
441 // done
442 // xの直後にancとdesを挿入
443 for(std::size_t i=*i_set.begin()+1;i<x.first;++i){
444 if(!i_set.contains(i)){
445 i_v_pair z{i,topo_order[i]};
446 toBeShifted1.insert(z);
447 i_set.insert(i);
448 }
449 }
450 for(std::size_t i=x.first+1;i<topo_order.size();++i){
451 if(!i_set.contains(i)){
452 i_v_pair z{i,topo_order[i]};
453 toBeShifted2.insert(z);
454 i_set.insert(i);
455 }
456 }
457 break;
458 }
459 if(x.first<y.first){
460 // done
461 // yの直前にancとdesを挿入
462 for(std::size_t i=*i_set.begin()+1;i<y.first;++i){
463 if(!i_set.contains(i)){
464 i_v_pair z{i,topo_order[i]};
465 toBeShifted1.insert(z);
466 i_set.insert(i);
467 }
468 }
469 for(std::size_t i=y.first+1;i<topo_order.size();++i){
470 if(!i_set.contains(i)){
471 i_v_pair z{i,topo_order[i]};
472 toBeShifted2.insert(z);
473 i_set.insert(i);
474 }
475 }
476 break;
477 }else{
478 md+=out_degree(y.second,graph);
479 }
480 }
481 }
482
483 auto it=i_set.begin();
484 for(auto&& w : toBeShifted1){
485 vertex_to_index[w.second]=*it;
486 topo_order[*it]=w.second;
487 ++it;
488 }
489 for(auto&& w : anc){
490 vertex_to_index[w.second]=*it;
491 topo_order[*it]=w.second;
492 ++it;
493 }
494 for(auto&& w : des){
495 vertex_to_index[w.second]=*it;
496 topo_order[*it]=w.second;
497 ++it;
498 }
499 for(auto&& w : toBeShifted2){
500 vertex_to_index[w.second]=*it;
501 topo_order[*it]=w.second;
502 ++it;
503 }
504 sorted=true;
505 }

637 template<class Archive>
638 void save(Archive & archive) const{
639 std::vector<std::shared_ptr<Instance>> instances;
640 instances.reserve(topo_order.size());
641 for(auto&& v : topo_order){
642 instances.push_back(get_instance(v));
643 }
644 std::vector<std::pair<std::size_t,std::size_t>> dependencies;
645 auto es=edges(graph);
646 for(auto&& e : std::ranges::subrange(es.first,es.second)){
647 dependencies.push_back({
648 vertex_to_index.at(source(e,graph)),
649 vertex_to_index.at(target(e,graph))
650 });
651 }
652
653 archive(
654 CEREAL_NVP(instances),
655 CEREAL_NVP(dependencies),
656 CEREAL_NVP(sorted),
657 CEREAL_NVP(auto_reorder)
658 );
659 }

661 template<class Archive>
662 void load(Archive & archive){
663 auto_reorder=false;
664
665 std::vector<std::shared_ptr<Instance>> instances;
666 archive(
667 CEREAL_NVP(instances)
668 );
669
670 clear();
671 for(auto&& instance : instances){
672 add_instance(instance);
673 }
674
675 std::vector<std::pair<std::size_t,std::size_t>> dependencies;
676 archive(
677 CEREAL_NVP(dependencies)
678 );
679
680 for(auto&& [pre,suc] : dependencies){
681 add_dependency(topo_order[pre],topo_order[suc]);
682 }
683
684 archive(
685 CEREAL_NVP(sorted),
686 CEREAL_NVP(auto_reorder)
687 );
688 }
