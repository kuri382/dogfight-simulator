8 * @par 使用方法

11 * 記述方法はpybind11/detail/init.hを参考にされたい。
12 *
13 * Python側への公開時にはpybind11::initの代わりに、cls.def(asrc::core::py_init<...>(...));

15 */

24template<typename T, typename... Args>
25struct py_init_constructor_impl {
26 template <typename Class, typename... Extra>
27 requires ( std::same_as<T,typename Class::type> )
28 void execute(Class &cl, const Extra &...extra) const{

30 }
31};
