22bool checkSphericalAxisOrder(const std::string& order){
23 // Azimuth, Elevation, Distance(Range)の3種類の軸を含んでいるかどうかを判定する。
24 if(order.size()!=3){
25 return false;
26 }
27 bool found[3] = {false,false,false}; // Azimuth, Elevation, Distance(Range)
28 for(int i=0; i<3; ++i){
29 if(order.at(i)=='A'){ // Azimuth
30 found[0]=true;
31 }else if(order.at(i)=='E'){ // Elevation
32 found[1]=true;
33 }else if(order.at(i)=='D'){ // Distance
34 found[2]=true;
35 }else if(order.at(i)=='R'){ // Range
36 found[2]=true;
37 }
38 }
39 return found[0] && found[1] && found[2];
40}
41bool checkBodyCartesianAxisOrder(const std::string& order){
42 // FWD/AFT, STBD/PORT, UP/DOWNの3種類の軸を含んでいるかどうかを判定する。
43 // 左右を表すSTBD/PORTはRIGHT(R)/LEFT(L)も許容する。
44 if(order.size()!=3){
45 return false;
46 }
47 bool found[3] = {false,false,false}; // FWD/AFT, STBD/PORT, UP/DOWN
48 for(int i=0; i<3; ++i){
49 if(order.at(i)=='F'){ // FWD
50 found[0]=true;
51 }else if(order.at(i)=='A'){ // AFT
52 found[0]=true;
53 }else if(order.at(i)=='S'){ // STBD(RIGHT)
54 found[1]=true;
55 }else if(order.at(i)=='P'){ // PORT(LEFT)
56 found[1]=true;
57 }else if(order.at(i)=='R'){ // STBD(RIGHT)
58 found[1]=true;
59 }else if(order.at(i)=='L'){ // PORT(LEFT)
60 found[1]=true;
61 }else if(order.at(i)=='U'){ // UP
62 found[2]=true;
63 }else if(order.at(i)=='D'){ // DOWN
64 found[2]=true;
65 }
66 }
67 return found[0] && found[1] && found[2];

92Eigen::Matrix3d getAxisSwapRotationMatrix(const std::string& srcOrder_, const std::string& dstOrder_){
93 static const std::map<char,char> bodyCartesianReverseAxis={
94 {'F','A'},
95 {'A','F'},
96 {'S','P'},
97 {'P','S'},
98 {'U','D'},
99 {'D','U'}
100 };
101 static const std::map<char,char> topocentricReverseAxis={
102 {'N','S'},
103 {'S','N'},
104 {'E','W'},
105 {'W','E'},
106 {'U','D'},
107 {'D','U'}
108 };
109 std::string srcOrder=srcOrder_;
110 std::string dstOrder=dstOrder_;
111 if(checkBodyCartesianAxisOrder(srcOrder)){
112 if(!checkBodyCartesianAxisOrder(dstOrder)){
113 throw std::runtime_error("'"+srcOrder+"' and '"+dstOrder+"' are not same type of axisOrder.");
114 }
115 //RをSに、LをPに変換する。
116 for(int i=0; i<3; ++i){
117 if(srcOrder.at(i)=='R'){
118 srcOrder.at(i)='S';
119 }else if(srcOrder.at(i)=='L'){
120 srcOrder.at(i)='P';
121 }
122 if(dstOrder.at(i)=='R'){
123 dstOrder.at(i)='S';
124 }else if(dstOrder.at(i)=='L'){
125 dstOrder.at(i)='P';
126 }
127 }
128 Eigen::Matrix3d ret=Eigen::Matrix3d::Zero();
129 for(int i=0; i<3; ++i){
130 for(int j=0; j<3; ++j){
131 if(dstOrder.at(i)==srcOrder.at(j)){
132 ret(i,j)=1;
133 }else if(dstOrder.at(i)==bodyCartesianReverseAxis.at(srcOrder.at(j))){
134 ret(i,j)=-1;
135 }
136 }
137 }
138 return ret;
139 }else if(checkTopocentricCartesianAxisOrder(srcOrder)){
140 if(!checkTopocentricCartesianAxisOrder(srcOrder)){
141 throw std::runtime_error("'"+srcOrder+"' and '"+dstOrder+"' are not same type of axisOrder.");
142 }
143 Eigen::Matrix3d ret=Eigen::Matrix3d::Zero();
144 for(int i=0; i<3; ++i){
145 for(int j=0; j<3; ++j){
146 if(dstOrder.at(i)==srcOrder.at(j)){
147 ret(i,j)=1;
148 }else if(dstOrder.at(i)==topocentricReverseAxis.at(srcOrder.at(j))){
149 ret(i,j)=-1;
150 }
151 }
152 }
153 return ret;
154 }else{
155 throw std::runtime_error("Given axisOrder '"+srcOrder+"' is not valid as bodyCartesian or topocentric axisOrder.");
156 }

572 auto intermediateCRS = getNonDerivedBaseCRS()->getIntermediateCRS();
573 if(non_const_this() == intermediateCRS){
574 // 高度オフセットの計算(ENU経由)
575 auto qToSelfFromENU=getEquivalentQuaternionToTopocentricCRS(origin,time,"ENU").conjugate();
576 Eigen::Vector3d actualOriginInSelf;
577 if(onSurface){
578 auto altOffset=qToSelfFromENU.transformVector(
579 Eigen::Vector3d(0,0,getHeight(origin,time))
580 );// [Self <- ENU]
581 actualOriginInSelf=origin-fromCartesian(altOffset);
582 }else{
583 actualOriginInSelf=origin;
584 }
585 // クォータニオンの計算
586 Quaternion qToSelfFromTopo=getEquivalentQuaternionToTopocentricCRS(origin,time,axisOrder).conjugate();
587
588 // 変換の実行(MotionStateを利用)
589 MotionState topoMotionState(
590 non_const_this(),
591 time,
592 actualOriginInSelf,
593 Eigen::Vector3d::Zero(),
594 Eigen::Vector3d::Zero(),
595 qToSelfFromTopo,
596 "FSD"
597 );
598 if(coordinateType==CoordinateType::POSITION_ABS){
599 return topoMotionState.absBtoP(value);
600 }else if(coordinateType==CoordinateType::POSITION_REL){
601 return topoMotionState.relBtoP(value);
602 }else if(coordinateType==CoordinateType::DIRECTION){
603 return topoMotionState.dirBtoP(value);
604 }else if(coordinateType==CoordinateType::VELOCITY){
605 return topoMotionState.velBtoP(value,location);
606 }else if(coordinateType==CoordinateType::ANGULAR_VELOCITY){
607 return topoMotionState.omegaBtoP(value);
608 }else{
609 throw std::runtime_error("Invalide coordinate type.");
610 }
611 }else{
612 auto originInIntermediate = transformTo(origin,origin,time,intermediateCRS,CoordinateType::POSITION_ABS);
613 auto valueInIntermediate = intermediateCRS->transformFromTopocentricCRS(
614 value,
615 location,
616 time,
617 coordinateType,

619 axisOrder,
620 onSurface
621 );

623 location,
624 location,
625 time,
626 CoordinateType::POSITION_ABS,
627 originInIntermediate(),
628 axisOrder,
629 onSurface
630 );
631 return transformFrom(valueInIntermediate,locationInIntermediate,time,intermediateCRS,coordinateType)();
632 }
