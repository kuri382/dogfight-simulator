84 if(!_base){

88 }

155Coordinate DerivedCRS::transformTo(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS, const CoordinateType& coordinateType) const{
156 if(!isValid()){validate();}
157 if(!dstCRS){
158 throw std::runtime_error("dstCRS is nullptr!");
159 }
160 if(this==std::dynamic_pointer_cast<DerivedCRS>(dstCRS).get()){
161 return std::move(Coordinate(
162 value,
163 location,
164 dstCRS,
165 time,
166 coordinateType
167 ));
168 }
169 if(!isTransformableTo(dstCRS)){
170 throw std::runtime_error(getFactoryClassName()+" can't be transformed to "+dstCRS->getFactoryClassName());
171 }
172 if(dstCRS==getBaseCRS()){
173 return transformToBaseCRS(value,location,time,coordinateType);
174 }else if(dstCRS==getNonDerivedBaseCRS()){
175 return transformToNonDerivedBaseCRS(value,location,time,coordinateType);
176 }else if(dstCRS==getIntermediateCRS()){
177 return transformToIntermediateCRS(value,location,time,coordinateType);
178 }else{
179 auto base=getBaseCRS();
180 auto valueInBase=transformToBaseCRS(value,location,time,coordinateType);
181 return dstCRS->transformFrom(
182 valueInBase(),
183 valueInBase.getLocation(),
184 time,
185 base,
186 coordinateType
187 );
188 }

190Coordinate DerivedCRS::transformFrom(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& srcCRS, const CoordinateType& coordinateType) const{
191 if(!isValid()){validate();}
192 if(!srcCRS){
193 throw std::runtime_error("srcCRS is nullptr!");
194 }
195 if(this==std::dynamic_pointer_cast<DerivedCRS>(srcCRS).get()){
196 return std::move(Coordinate(
197 value,
198 location,
199 srcCRS,
200 time,
201 coordinateType
202 ));
203 }
204 if(!isTransformableFrom(srcCRS)){
205 throw std::runtime_error(getFactoryClassName()+" can't be transformed from "+srcCRS->getFactoryClassName());
206 }
207 if(srcCRS==getBaseCRS()){
208 return transformFromBaseCRS(value,location,time,coordinateType);
209 }else if(srcCRS==getNonDerivedBaseCRS()){
210 return transformFromNonDerivedBaseCRS(value,location,time,coordinateType);
211 }else if(srcCRS==getIntermediateCRS()){
212 return transformFromIntermediateCRS(value,location,time,coordinateType);
213 }else{
214 auto base=getBaseCRS();
215 auto valueInBase=base->transformFrom(value,location,time,srcCRS,coordinateType);
216 return transformFromBaseCRS(
217 valueInBase(),
218 valueInBase.getLocation(),
219 time,
220 coordinateType

363void DerivedCRS::setBaseCRS(const std::shared_ptr<CoordinateReferenceSystem>& newBaseCRS) const{
364 if(!isValid()){validate();}
365 if(_base==newBaseCRS){
366 return;
367 }
368 if(_base){
369 removeDerivedCRS(_base,non_const_this());
370 }
371 if(!newBaseCRS){
372 throw std::runtime_error("The given base CRS is not a valid object.");
373 }

375 if(auto derived = std::dynamic_pointer_cast<DerivedCRS>(_base)){
376 _nonDerivedBase=derived->getNonDerivedBaseCRS();
