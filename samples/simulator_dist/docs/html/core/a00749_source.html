121 template <class Archive, typename Scalar, int ExtraFlags>
122 inline void save_numpy_array_t_without_swap( Archive & ar, const ::pybind11::array_t<Scalar,ExtraFlags>& m ){
123 namespace py=pybind11;
124
125 const bool isRowMajorType = is_row_major(m);
126 const bool isColMajorType = is_col_major(m);
127 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
128
129 py::buffer_info info=m.request(false);
130 if(info.ndim==1){
131 ar( ::cereal::make_size_tag( (::cereal::size_type)(info.size) ) );
132 if constexpr (cereal::traits::is_output_serializable<cereal::BinaryData<const Scalar*>, Archive>::value) {
133 if(info.strides[0]==info.itemsize){
134 ar( ::cereal::binary_data( m.data(), info.size * info.itemsize ) );
135 return;
136 }
137 }
138 auto ref=m.template unchecked<1>();
139 for(ssize_t i=0;i<info.size;++i){
140 ar(ref[i]);
141 }
142 }else{
143 // RowMajorは最後の軸に沿って連続、最初の軸が最外周
144 // ColMajorは最初の軸に沿って連続、最後の軸が最外周
145 const ssize_t outerDimIndex = (isRowMajorType ? 0 : info.ndim-1);
146 ar( ::cereal::make_size_tag( (::cereal::size_type)(info.shape[outerDimIndex]) ) );
147 for(ssize_t i=0;i<info.shape[outerDimIndex];++i){
148 if(isRowMajorType){
149 auto sub=::pybind11::array_t<Scalar,py::array::c_style>::ensure(m[py::make_tuple(i,py::ellipsis())]);
150 ar(inner_numpy_array_t_wrapper(sub));
151 }else{
152 auto sub=::pybind11::array_t<Scalar,py::array::f_style>::ensure(m[py::make_tuple(py::ellipsis(),i)]);
153 ar(inner_numpy_array_t_wrapper(sub));
154 }
155 }
156 }

159 template <class Archive, typename Scalar, int ExtraFlags>
160 inline void load_numpy_array_t_without_swap( Archive & ar, ::pybind11::array_t<Scalar,ExtraFlags> & m , bool noResize){
161 namespace py=pybind11;
162
163 const bool isRowMajorType = is_row_major(ExtraFlags);
164 const bool isColMajorType = is_col_major(ExtraFlags);
165 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
166
167 if(!m.writeable()){
168 throw cereal::Exception("Destination is not a writeable py::array.");
169 }
170
171 py::buffer_info info=m.request(true);
172 if(info.ndim==1){
173 cereal::size_type size;
174 ar( cereal::make_size_tag( size ) );
175 if(m.owndata() && !noResize){
176 try{
177 m.resize({(ssize_t)size},false);
178 }catch(std::exception& ex){
179 assert(info.size==size);
180 }
181 }else{
182 assert(info.size==size);
183 }
184 if constexpr ( cereal::traits::is_input_serializable<cereal::BinaryData<Scalar*>, Archive>::value ) {
185 if(info.strides[0]==info.itemsize){
186 ar( ::cereal::binary_data( m.mutable_data(), size * info.itemsize ) );
187 return;
188 }
189 }
190 auto ref=m.template mutable_unchecked<1>();
191 for(ssize_t i=0;i<size;++i){
192 ar(ref[i]);
193 }
194 }else{
195 // RowMajorは最後の軸に沿って連続、最初の軸が最外周
196 // ColMajorは最初の軸に沿って連続、最後の軸が最外周
197 const ssize_t outerDimIndex = (isRowMajorType ? 0 : info.ndim-1);
198
199 cereal::size_type outerSize;
200 ar( ::cereal::make_size_tag( outerSize ) );
201 for(ssize_t i=0; i<outerSize; ++i){
202 if(i==0){
203 ::pybind11::array_t<Scalar,ExtraFlags> sub(std::vector<ssize_t>(info.ndim-1,1));
204 auto wrapped=inner_numpy_array_t_wrapper(sub,false);
205 ar(wrapped);
206 if(m.owndata() && !noResize){
207 std::vector<ssize_t> newShape(sub.request(true).shape);
208 if(isRowMajorType){
209 newShape.insert(newShape.begin(),outerSize);
210 }else{
211 newShape.push_back(outerSize);
212 }
213 try{
214 // resizeはstorage orderが無視されるため新たなarrayの代入とする。
215 m=::pybind11::array_t<Scalar,ExtraFlags>(newShape);
216 }catch(...){
217 assert(info.shape[outerDimIndex]==outerSize);
218 }
219 }else{
220 assert(info.shape[outerDimIndex]==outerSize);
221 }
222 if(isRowMajorType){
223 m[py::make_tuple(i,py::ellipsis())]=sub;
224 }else{
225 m[py::make_tuple(py::ellipsis(),i)]=sub;
226 }
227 }else{
228 if(isRowMajorType){
229 ::pybind11::array_t<Scalar,py::array::c_style> sub(m[py::make_tuple(i,py::ellipsis())]);
230 auto wrapped=inner_numpy_array_t_wrapper(sub,true);
231 ar(wrapped);
232 }else{
233 ::pybind11::array_t<Scalar,py::array::f_style> sub(m[py::make_tuple(py::ellipsis(),i)]);
234 auto wrapped=inner_numpy_array_t_wrapper(sub,true);
235 ar(wrapped);
236 }
237 }
238 }
239 }

256 template <class Archive, typename Scalar, int ExtraFlags>
257 inline void save_numpy_array_t( Archive & ar, const ::pybind11::array_t<Scalar,ExtraFlags>& m )
258 {
259 namespace py=pybind11;
260
261 const bool isRowMajorType = is_row_major(m);
262 const bool isColMajorType = is_col_major(m);
263 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
264
265 py::buffer_info info=m.request(false);
266
267 constexpr bool always_rowmajor = is_text_archive && serialize_eigen_as_text_always_rowmajor;
268 constexpr bool save_ndim = !traits::same_archive_as<Archive,::asrc::core::util::NLJSONOutputArchive>;
269 constexpr bool use_nvp = (!always_rowmajor || save_ndim);
270
271 if constexpr (save_ndim){
272 // NLJSONArchive以外はdtypeとndimが既知でないと読み込めないので出力しておく必要あり
273 save_dtype_helper<Scalar>(ar);
274 ar(::cereal::make_nvp("ndim",info.ndim));
275 }
276
277 if constexpr (!always_rowmajor){
278 // つねにRowMajorを強制するのでなければLayOut情報の出力が必要
279 ar(::cereal::make_nvp("isRowMajorSrc",isRowMajorType));
280 }
281
282 if constexpr (always_rowmajor){
283 if(!isRowMajorType){
284 // ColMajorをRowMajorとして出力
285 // おそらくswapしたtensorを複製した方が速い
286 auto swapped=py::array_t<Scalar,py::array::c_style>::ensure(m);
287 if constexpr (use_nvp){
288 ar(::cereal::make_nvp("value",inner_numpy_array_t_wrapper(swapped)));
289 }else{
290 save_numpy_array_t_without_swap(ar,swapped);
291 }
292 return;
293 }
294 }
295 // as-is
296 if constexpr (use_nvp){
297 auto non_const=py::reinterpret_borrow<py::array_t<Scalar,ExtraFlags>>(m);
298 ar(::cereal::make_nvp("value",inner_numpy_array_t_wrapper(non_const)));
299 }else{
300 save_numpy_array_t_without_swap(ar,m);
301 }
302 }

304 template <class Archive, typename Scalar, int ExtraFlags>
305 inline void load_numpy_array_t( Archive & ar, ::pybind11::array_t<Scalar,ExtraFlags> & m )
306 {
307 namespace py=pybind11;
308
309 const bool isRowMajorType = is_row_major(m);
310 const bool isColMajorType = is_col_major(m);
311 const bool isForcecast = is_forcecast(m);
312 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
313
314 if(!m.writeable()){
315 throw cereal::Exception("Destination is not a writeable py::array.");
316 }
317
318 constexpr bool always_rowmajor = is_text_archive && serialize_eigen_as_text_always_rowmajor;
319 constexpr bool save_ndim = !traits::same_archive_as<Archive,::asrc::core::util::NLJSONInputArchive>;
320 constexpr bool use_nvp = (!always_rowmajor || save_ndim);
321
322 ssize_t srcNdim;
323 if constexpr (save_ndim){
324 // NLJSONArchive以外は階数(ndim)をArchiveから抽出
325 ar(::cereal::make_nvp("ndim",srcNdim));
326 }else{
327 // NLJSONArchiveはjsonの形から推論
328 const nl::ordered_json& j=ar.getCurrentNode();
329 assert(j.is_array() && j.size()>0);
330
331 srcNdim=extract_ndim_helper(j);
332 }
333
334 bool isRowMajorSrc;
335 if constexpr (!always_rowmajor){
336 // つねにRowMajorを強制するのでなければLayOutを抽出
337 ar(::cereal::make_nvp("isRowMajorSrc",isRowMajorSrc));
338 }else{
339 isRowMajorSrc=true;
340 }
341
342 py::buffer_info info=m.request(true);
343
344 if(isForcecast || isRowMajorType!=isRowMajorSrc || (srcNdim>1 && isColMajorType!=!isRowMajorSrc) || info.ndim!=srcNdim){
345 // forcecastの場合又はLayoutか階数が異なる場合はそれらを合わせた新規オブジェクトに読み込んでから代入
346 if(isRowMajorSrc){
347 py::array_t<Scalar,py::array::c_style> tmp{std::vector<ssize_t>(srcNdim,1)};
348 if constexpr (use_nvp){
349 ar(::cereal::make_nvp("value",inner_numpy_array_t_wrapper(tmp,false)));
350 }else{
351 load_numpy_array_t_without_swap(ar,tmp,false);
352 }
353 m=tmp;
354 }else{
355 py::array_t<Scalar,py::array::f_style> tmp{std::vector<ssize_t>(srcNdim,1)};
356 if constexpr (use_nvp){
357 ar(::cereal::make_nvp("value",inner_numpy_array_t_wrapper(tmp,false)));
358 }else{
359 load_numpy_array_t_without_swap(ar,tmp,false);
360 }
361 m=tmp;
362 }
363 }else{
364 // as-is
365 if constexpr (use_nvp){
366 ar(::cereal::make_nvp("value",inner_numpy_array_t_wrapper(m,false)));
367 }else{
368 load_numpy_array_t_without_swap(ar,m,false);
369 }
370 }
