164double SimpleFighterJetEngine::trimPower(double tgtThrust,double alt,double Mach){
165 double thrustMin=calcThrust(0.0,alt,Mach);
166 double thrustMax=calcThrust(1.0,alt,Mach);
167 auto sub=[&](double pCmd){
168 return calcThrust(pCmd,alt,Mach)-tgtThrust;
169 };
170 double trimmed;
171 if(thrustMin>=tgtThrust){
172 trimmed=0.0;
173 }else if(thrustMax<=tgtThrust){
174 trimmed=1.0;
175 }else{
176 boost::math::tools::eps_tolerance<double> tol(16);
177 boost::uintmax_t max_iter=10;
178 try{
179 auto result=boost::math::tools::toms748_solve(sub,0.0,1.0,tol,max_iter);
180 trimmed=(result.first+result.second)/2;
181 }catch(std::exception& e){
182 std::cout<<"at trimPower"<<std::endl;
183 DEBUG_PRINT_EXPRESSION(sub(0.0))
184 DEBUG_PRINT_EXPRESSION(sub(1.0))
185 DEBUG_PRINT_EXPRESSION(tgtThrust)
186 DEBUG_PRINT_EXPRESSION(alt)
187 DEBUG_PRINT_EXPRESSION(Mach)
188 throw e;
189 }
190 }
191 return trimmed;

213 Eigen::MatrixXd x(1,2);
214 std::pair<Eigen::Index,double> tmp=_search1d(points[1],m2ft(alt));
215 double alt_h=points[1](tmp.first+1);
216 double alt_l=points[1](tmp.first);

231double SimpleFighterJetEngine::calcdTdM(double power_,double alt,double rmach){
232 Eigen::MatrixXd x(1,2);
233 std::pair<Eigen::Index,double> tmp=_search1d(points[0],rmach);
234 double rmach_h=points[0](tmp.first+1);
235 double rmach_l=points[0](tmp.first);

238 double rmachEx_l=pointsEx[0](tmp.first);

244 x<<rmach_h,m2ft(alt);

248 double tmax_h=interpn(pointsEx,tmaxTable,x)(0);
249 if(power_<0.5){
250 return lbf2N(2*power_*((tmil_h-tmil_l)-(tmin_h-tmin_l))/(rmach_h-rmach_l));

288 double alt=p->motion.getHeight();

295 Eigen::VectorXd f(1);

391 auto eng=getShared<SimpleFighterJetEngine>(engine);

509 x<<rad2deg(aoa_);
510 points.push_back(aoaTable);

514double CoordinatedFighter::cxa(double aoa_){
515 std::vector<Eigen::VectorXd> points;
516 Eigen::MatrixXd x(1,1);
517 x<<rad2deg(aoa_);
518 points.push_back(aoaTable);
519 double de_=interpn(points,trimmedDe,x)(0);
520 double ddeda_=interpgradn(points,trimmedDe,x)(0,0);
521 points.insert(points.begin(),deTable);
522 x=Eigen::MatrixXd::Zero(1,2);
523 x<<rad2deg(de_),rad2deg(aoa_);
524 Eigen::MatrixXd grad=interpgradn(points,cxTable,x);
525 double ret=(grad(1,0)+grad(0,0)*ddeda_)*(180/M_PI);
526 return ret;

528double CoordinatedFighter::cza(double aoa_){

533 double ddeda_=interpgradn(points,trimmedDe,x)(0,0);

538double CoordinatedFighter::cl(double aoa_){

541double CoordinatedFighter::cd(double aoa_,double mach_){
542 return -cx(aoa_)*cos(aoa_)-cz(aoa_)*sin(aoa_)+cdw(mach_);

544double CoordinatedFighter::cdw(double mach_){
545 //wave drag is based on [3] and [4].
546 //coefficients are obtained by curve fitting with some points.
547 double cdw0=cdwTable(0);
548 double kdwm=cdwTable(1);
549 double kdw=cdwTable(2);
550 double dm=cdwTable(3);
551 double fm=1.0/(1.0+exp(-8*(mach_-(1.0-dm/2.0))/dm));
552 return fm*cdw0*kdw/pow(pow(pow(mach_-kdwm,2)-1.0,2)+pow(kdw,4),0.25);
553}
554Eigen::Vector3d CoordinatedFighter::calcAero(double aoa_,double mach_){
555 /*
556 "Coordinated" flight assumes de=da=dr==0.
557 Angle of attack is only considered for aerodynamic force and the aircraft attitude neglects it.
558 CX=-CD=Cx()*cos()+Cz()*sin()
559 CZ=-CL=Cz()*cos()-Cx()*sin()
560 Polynomial coefficients of Cx and Cz is based on [2] and the lift is maximized when =0.66449rad
561 */
562 double Cx=cx(aoa_);
563 double Cz=cz(aoa_);
564 return Eigen::Vector3d(Cx*cos(aoa_)+Cz*sin(aoa_)-cdw(mach_),0.0,Cz*cos(aoa_)-Cx*sin(aoa_));
565}
566void CoordinatedFighter::calcMotion(double dt){
567 motion_prev=motion;
568 Quaternion qPrev=motion.q;
569 Eigen::Vector3d gravityVector = motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false)*m*gravity; //TODO
570 double alt=motion.getHeight();
571 double V=motion.vel.norm();
572 std::map<std::string,double> atm=atmosphere(alt);
573 double Mach=V/atm["a"];
574 double dp=atm["rho"]*V*V/2;
575 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
576 double thrust=getThrust();
577 double aoa=ctrl.at("aoa");
578 double dRoll=ctrl.at("dRoll");
579 Eigen::Vector3d rollAx=motion.vel()/V;
580 //Roll rotation(Immidiate)
581 if(abs(dRoll)>1e-8){
582 Quaternion dq1=Quaternion::fromAngle(rollAx,dRoll);
583 motion.setQ(dq1*motion.q);
584 }else{
585 Eigen::VectorXd dq1=motion.q.dqdwi()*rollAx*dRoll;
586 motion.setQ((motion.q+Quaternion(dq1)).normalized());
587 }
588 //position and velocity update
589 Eigen::Vector3d CFb=calcAero(aoa,Mach);
590 Eigen::Vector3d F=relBtoP(Eigen::Vector3d(thrust*cos(aoa),0.,-thrust*sin(aoa))+CFb*(dp*S))+gravityVector;
591 Eigen::Vector3d accel=F/m;
592 double accelScalar=accel.dot(rollAx);
593 Eigen::Vector3d omegaV=rollAx.cross(accel/V);
594 double rotAngle=omegaV.norm()*dt;
595 if(rotAngle<1e-8){
596 motion.setPos(motion.pos()+motion.vel()*dt+accel*dt*dt/2.);
597 motion.setVel(motion.vel()+accel*dt);
598 Eigen::VectorXd dq2=motion.q.dqdwi()*omegaV*dt;
599 motion.setQ((motion.q+Quaternion(dq2)).normalized());
600 }else{
601 Eigen::Vector3d ex0=motion.vel()/V;
602 Eigen::Vector3d ey0=omegaV.cross(ex0).normalized();
603 double wt=rotAngle;
604 double acc=accelScalar;
605 Eigen::Vector3d vn=(ex0*cos(wt)+ey0*sin(wt)).normalized();
606 motion.setVel((V+acc*dt)*vn);
607 Eigen::Vector3d dr=V*dt*(ex0*sinc(wt)+ey0*oneMinusCos_x2(wt)*wt)+acc*dt*dt*(ex0*(sinc(wt)-oneMinusCos_x2(wt))+ey0*wt*(sincMinusOne_x2(wt)+oneMinusCos_x2(wt)));
608 motion.setPos(motion.pos()+dr);
609 accel=(motion.vel()-vel_prev())/dt;
610 Quaternion dq2=Quaternion::fromAngle(omegaV/(rotAngle/dt),rotAngle);
611 motion.setQ(dq2*motion.q);
612 }
613 Quaternion dq=motion.q*qPrev.conjugate();
614 Eigen::Vector3d ax=motion.q.vec();
615 double sn=ax.norm();
616 if(sn>0){ax/=sn;}
617 motion.setOmega(ax*atan2(sn,dq.w())+rollAx*dRoll/dt);
618}

705 }

711 if(Eigen::Vector3d(0,ey.dot(down),zz).norm() >= p->sideGLimit){
712 if(zz>=0){
713 tgtCz=p->m*util::gravity*(sqrt(p->sideGLimit-ey.dot(down)*ey.dot(down))-ez.dot(down));
714 }else{
715 tgtCz=-p->m*util::gravity*(sqrt(p->sideGLimit-ey.dot(down)*ey.dot(down))-ez.dot(down));
716 }
717 aoa_=p->trimAoa(tgtCz);
718 }
719 return {{"pCmd",pCmd_},{"aoa",aoa_},{"dRoll",dRoll_}};

726 Eigen::Vector3d d=p->motion.vel()/V;
727 std::map<std::string,double> atm=atmosphere(alt);
728 double Mach=V/atm["a"];
729 double dp=atm["rho"]*V*V/2;
730 Eigen::Vector3d gravityVector = p->motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false)*p->m*gravity; //TODO
731 Eigen::Vector3d gravityVectorVert=gravityVector-(gravityVector.dot(d))*d;//==g.dot(n)+g.dot(b) for any
732 Eigen::Vector3d b,n;
733 auto eng=getShared<SimpleFighterJetEngine>(p->engine);
734 double Tmin=eng->calcThrust(0.0,alt,Mach);
735 double Tmax=eng->calcThrust(1.0,alt,Mach);

737 double dstTd,dstAoa;//turn command
738 int velCmdType,turnCmdType;
739 if(cmd.contains("dstV")){
740 velCmdType=0;
741 double dstV=cmd.at("dstV");

743 }else if(cmd.contains("dstAccel")){

757 }

763 }
