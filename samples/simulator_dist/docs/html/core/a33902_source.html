196Eigen::Vector3d Coordinate::toSpherical() const{
197 if(crs){
198 return crs->toSpherical(value);
199 }else{
200 //親CRSの指定が無い場合はNEDのTopocentricとみなす。
201 //NED -> (Azimuth Elevation Range)
202 double xy=sqrt(value(0)*value(0)+value(1)*value(1));
203 if(xy==0){
204 return Eigen::Vector3d(
205 0,
206 value(2)>0 ? -M_PI_2 : M_PI_2,
207 abs(value(2))
208 );
209 }else{
210 double el=atan2(-value(2),xy);
211 return Eigen::Vector3d(
212 atan2(value(1),value(0)),
213 el,
214 sqrt(xy*xy+value(2)*value(2))
215 );
216 }
217 }
218}
