126void MotionState::validate() const{
127 Eigen::Vector3d fwdInNED;
128 if(crs){
129 std::shared_ptr<CoordinateReferenceSystem> intermediateCRS=crs->getNonDerivedBaseCRS()->getIntermediateCRS(); //Topocentricの基準となる地球固定座標系(ECEF or PureFlat)
130 Quaternion qToNEDFromIntermediate=intermediateCRS->getEquivalentQuaternionToTopocentricCRS(pos(intermediateCRS),time,"NED");//[NED <- intermediate]
131 Quaternion qToIntermediateFromParent=crs->getQuaternionTo(pos(),time,intermediateCRS);//[intermediate <- parent]
132
133 auto qToNEDFromBody=qToNEDFromIntermediate*qToIntermediateFromParent*q;//[NED <- this]
134 fwdInNED=(
135 qToNEDFromBody* //[NED <- this]
136 getAxisSwapQuaternion("FSD",axisOrder) //[this <- FSD(body)]
137 ).transformVector(Eigen::Vector3d(1,0,0));
138 }else{
139 //親CRSの指定が無い場合はNEDのTopocentricとみなす。
140 fwdInNED=(
141 q* //[parent(NED) <- this]
142 getAxisSwapQuaternion("FSD",axisOrder) //[this <- FSD(body)]
143 ).transformVector(Eigen::Vector3d(1,0,0));
144 }
145 double xy=sqrt(fwdInNED(0)*fwdInNED(0)+fwdInNED(1)*fwdInNED(1));
146 double newEl,newAz;
147 if(xy==0){
148 newEl=fwdInNED(2)>0 ? -M_PI_2 : M_PI_2;
149 newAz=_az;
150 }else{
151 newEl=atan2(-fwdInNED(2),xy);
152 newAz=atan2(fwdInNED(1),fwdInNED(0));
153 }
154 Eigen::Vector3d fwdH(cos(newAz),sin(newAz),0);
155 Eigen::Vector3d stbdH,downH;
156 stbdH<<-sin(newAz),cos(newAz),0;
157 downH<<0,0,1;
158 qToFSDFromNED=Quaternion::fromBasis(fwdH,stbdH,downH).conjugate();//[FSD(topo) <- NED]
159 _az=newAz;
160 _el=newEl;
161 if(crs){
162 alt=crs->getHeight(pos(),time);
163 crsUpdateCount=crs->getUpdateCount();
164 }else{
165 alt=-pos(2); //親座標系が未指定の場合は暗黙に3軸目が下向き正の高度に相当するものと解釈する。
166 }
167 _isValid=true;
168}

174Eigen::Vector3d MotionState::toSpherical(const Eigen::Vector3d& v) const{
175 if(crs){
176 return crs->toSpherical(v);
177 }else{
178 //親CRSの指定が無い場合はNEDのTopocentricとみなす。
179 //NED -> (Azimuth Elevation Range)
180 double xy=sqrt(v(0)*v(0)+v(1)*v(1));
181 if(xy==0){
182 return Eigen::Vector3d(
183 0,
184 v(2)>0 ? -M_PI_2 : M_PI_2,
185 abs(v(2))
186 );
187 }else{
188 double el=atan2(-v(2),xy);
189 return Eigen::Vector3d(
190 atan2(v(1),v(0)),
191 el,
192 sqrt(xy*xy+v(2)*v(2))
193 );
194 }
195 }

776void MotionState::setCRS(const std::shared_ptr<CoordinateReferenceSystem>& newCRS,bool transform){
777 //CRSを再設定する。座標変換の有無を第2引数で指定する。
778 //元のCRSが存在しなかった場合は変換なし元の値を用いる。
779 if(!newCRS){
780 throw std::runtime_error("Do not give nullptr as a new CRS of MotionState!");
781 }
782 if(transform && bool(crs)){
783 _q=crs->transformQuaternionTo(q.conjugate(),pos,newCRS).conjugate();
784 }
785 _pos.setCRS(newCRS,transform && bool(crs));
786 _vel.setCRS(newCRS,transform && bool(crs));
787 _omega.setCRS(newCRS,transform && bool(crs));
788 crs=newCRS;
789 validate();
790}
791MotionState MotionState::transformTo(const std::shared_ptr<CoordinateReferenceSystem>& newCRS) const{
792 if(crs==newCRS){
793 //同一ならバイパス
794 return *this;
795 }
796 if(!crs){
797 throw std::runtime_error("MotionState without CRS cannot be converted to another CRS.");
798 }
799 if(!newCRS){
800 throw std::runtime_error("Do not give nullptr as new parent CRS of MotionState!");
801 }
802 return std::move(MotionState(
803 newCRS,
804 time,
805 pos.transformTo(newCRS)(),
806 vel.transformTo(newCRS)(),
807 omega.transformTo(newCRS)(),
808 crs->transformQuaternionTo(q.conjugate(),pos,newCRS).conjugate(),
809 axisOrder

837//(9)時刻情報を用いた外挿
838MotionState MotionState::extrapolate(const double &dt) const{
839 auto cartesianOmega=omega.toCartesian();
840 double w=cartesianOmega.norm();
841 Quaternion qAft;
842 if(w*dt>1e-6){
843 Quaternion dq=Quaternion::fromAngle(cartesianOmega/w,w*dt);
844 qAft=(dq*q).normalized();
845 }else{
846 Eigen::VectorXd dq=q.dqdwi()*cartesianOmega*dt;
847 qAft=(q+Quaternion(dq)).normalized();
848 }
849 MotionState ret=MotionState(
850 crs,
851 time+dt,
852 pos.toCartesian()+vel.toCartesian()*dt,
853 vel.toCartesian(),
854 cartesianOmega,
855 qAft,
856 axisOrder
857 );
858 return ret;
859}
860MotionState MotionState::extrapolateTo(const Time &dstTime) const{
861 return extrapolate(dstTime-time);
862}
