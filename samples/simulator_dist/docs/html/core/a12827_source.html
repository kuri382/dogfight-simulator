40def get_plugin_directories(req):
41 ret = []
42 for p in req:
43 import importlib
44 try:
45 m = importlib.import_module(p)
46 ret.append(os.path.abspath(os.path.dirname(m.__file__)).replace(os.path.sep,'/'))
47 except ImportError as e:
48 print(f"Required plugin {p} could not be imported!")
49 raise e
50 return ";".join(ret)
