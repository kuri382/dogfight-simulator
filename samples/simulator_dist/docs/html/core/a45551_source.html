88void MassPointFighter::calcMotion(double dt){
89 motion_prev=motion;
90 double V=motion.vel.norm();
91 nl::json ctrl=controllers["FlightController"].lock()->commands["motion"];
92 Eigen::Vector3d omegaB=ctrl.at("omegaB");
93 double accel=getThrust()/m;
94 motion.setOmega(relBtoP(omegaB));
95 double theta=motion.omega.norm()*dt;
96 if(theta>1e-6){
97 Eigen::Vector3d ax=motion.omega.normalized()();
98 Quaternion dq=Quaternion::fromAngle(ax,theta);
99 motion.setQ(dq*motion.q);
100 }else{
101 Eigen::VectorXd dq=motion.q.dqdwi()*motion.omega()*dt;
102 motion.setQ((motion.q+Quaternion(dq)).normalized());
103 }
104 double Vaft=std::max(vMin,std::min(vMax,V+accel*dt));
105 motion.setVel(relBtoP(Eigen::Vector3d(Vaft,0,0)));
106 motion.setPos(motion.pos()+motion.vel()*dt);
107}

150nl::json MassPointFlightController::calcDirect(const nl::json &cmd){
151 auto p=getShared<MassPointFighter>(parent);
152 auto roll=std::clamp(cmd.at("roll").get<double>(),-1.,1.)*p->rollMax;
153 auto pitch=std::clamp(cmd.at("pitch").get<double>(),-1.,1.)*p->pitchMax;
154 auto yaw=std::clamp(cmd.at("yaw").get<double>(),-1.,1.)*p->yawMax;
155 double pCmd;
156 if(cmd.contains("throttle")){//0〜1
157 pCmd=std::clamp(cmd.at("throttle").get<double>(),0.,1.);
158 }else if(cmd.contains("accel")){//-1〜+1
159 pCmd=(1+std::clamp(cmd.at("accel").get<double>(),-1.,1.))*0.5;
160 }else{
161 throw std::runtime_error("Either 'throttle' or 'accel' must be in cmd keys.");
162 }
163 return {
164 {"omegaB",Eigen::Vector3d(roll,pitch,yaw)},
165 {"pCmd",pCmd}
166 };

168nl::json MassPointFlightController::calcFromDirAndVel(const nl::json &cmd){
169 auto p=getShared<MassPointFighter>(parent);
170 auto eng=getShared<IdealDirectPropulsion>(p->engine);
171 double Tmin=eng->tMin;
172 double Tmax=eng->tMax;
173 double pCmd;
174 if(cmd.contains("dstV")){
175 double V=p->motion.vel.norm();
176 double dstV=cmd.at("dstV");
177 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
178 double dstAccel=(dstV-V)/dt;
179 double aMin=Tmin/p->m;
180 double aMax=Tmax/p->m;
181 pCmd=std::clamp((dstAccel-aMin)/(aMax-aMin),0.0,1.0);
182 }else if(cmd.contains("dstAccel")){
183 double dstAccel=cmd.at("dstAccel");
184 double aMin=Tmin/p->m;
185 double aMax=Tmax/p->m;
186 pCmd=std::clamp((dstAccel-aMin)/(aMax-aMin),0.0,1.0);
187 }else if(cmd.contains("dstThrust")){
188 double dstT=cmd.at("dstThrust");
189 pCmd=std::clamp((dstT-Tmin)/(Tmax-Tmin),0.0,1.0);
190 }else if(cmd.contains("dstThrottle")){
191 double dstT=cmd.at("dstThrottle");
192 pCmd=std::clamp(dstT,0.0,1.0);
193 }else{
194 throw std::runtime_error("Only one of dstV, dstAccel, dstThrust or dstThrottle is acceptable.");
195 }
196
197 Eigen::Vector3d omegaB;
198 if(cmd.contains("dstDir")){
199 Eigen::Vector3d dstDir=cmd.at("dstDir");
200 if(cmd.contains("dstAlt")){
201 dstDir=altitudeKeeper(p->motion,dstDir,cmd.at("dstAlt").get<double>());
202 }
203 double V=p->motion.vel.norm();
204 Eigen::Vector3d currentDir=p->motion.vel()/V;
205 dstDir.normalize();
206 Eigen::Vector3d rotationAxis=currentDir.cross(dstDir);
207 auto ex=p->dirBtoP(Eigen::Vector3d(1,0,0));
208 auto ey=p->dirBtoP(Eigen::Vector3d(0,1,0));
209 auto ez=p->dirBtoP(Eigen::Vector3d(0,1,1));
210 auto down=p->motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false);
211 auto horizontalY=down.cross(ex);
212 double snTheta=std::min(1.0,rotationAxis.norm());
213 double csTheta=std::clamp(currentDir.dot(dstDir),-1.0,1.0);
214 double eTheta=atan2(snTheta,csTheta);
215 double eps=1e-4;
216 double dt=p->interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
217 if(snTheta<eps){
218 //Almost no rotation or 180deg rotation
219 rotationAxis=currentDir.cross(down.cross(currentDir)).normalized();
220 }else{
221 rotationAxis/=snTheta;
222 }
223 Quaternion dq=Quaternion::fromAngle(rotationAxis,eTheta);
224 Quaternion q1=dq*p->motion.q;
225 ex=q1.transformVector(Eigen::Vector3d(1,0,0));// == dstDir
226 ey=q1.transformVector(Eigen::Vector3d(0,1,0));
227 double sinRoll=ex.dot(ey.cross(rotationAxis));
228 double cosRoll=ey.dot(rotationAxis);
229 double roll=atan2(sinRoll,cosRoll);
230 if(abs(roll)<=M_PI_2){
231 dq=Quaternion::fromAngle(ex,roll)*dq;
232 }else{
233 dq=Quaternion::fromAngle(ex,M_PI-roll)*dq;
234 }
235 if(dq.w()<0){
236 dq=-dq;
237 }
238 double cosRot=dq.w();
239 double sinRot=dq.vec().norm();
240 Eigen::Vector3d ax=dq.vec().normalized();
241 omegaB=p->relPtoB(ax*atan2(sinRot,cosRot)*2/dt);
242 omegaB*=std::min(std::min(1.0,p->rollMax/abs(omegaB(0))),std::min(p->pitchMax/abs(omegaB(1)),p->yawMax/abs(omegaB(2))));
243 }else if(cmd.contains("dstTurnRate")){
244 omegaB=cmd.at("dstTurnRate");
245 throw std::runtime_error("Only one of dstDir, or dstTurnRate is acceptable.");
246 }
247
248 return {
249 {"omegaB",omegaB},
250 {"pCmd",pCmd}
251 };
252}
