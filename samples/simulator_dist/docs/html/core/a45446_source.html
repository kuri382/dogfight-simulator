88nl::json ConfigDispatcherElement::get(ConfigDispatcher& dispatcher,const std::string& instance,const int& index){
89 nl::json ret;
90 std::string type=config.at("type");
91 if(instance!="" && instances.count(instance)>0){
92 ret=instances[instance];
93 }else{
94 if(type=="alias"){
95 assert(dispatcher.aliases.count(config.at("alias"))>0);
96 std::shared_ptr<ConfigDispatcherElement> element=dispatcher.aliases.at(config.at("alias"));
97 std::string sub_instance="";
98 if(config.contains("instance") && config.at("instance").is_string()){
99 sub_instance=config.at("instance");
100 }
101 int sub_index=-1;
102 if(config.contains("index") && config.at("index").is_number_integer()){
103 sub_index=config.at("index");
104 }
105 ret=element->get(dispatcher,sub_instance,sub_index);
106 }else if(type=="broadcast"){
107 nl::json element=config.at("element");
108 std::size_t number;
109 bool dispatchAfterBroadcast=false;
110 if(config.contains("dispatchAfterBroadcast")){
111 dispatchAfterBroadcast=config.at("dispatchAfterBroadcast");
112 }
113 std::vector<std::string> names;
114 if(config.contains("names") && config.at("names").is_array()){
115 nl::json j_names=config.at("names");
116 number=j_names.size();
117 for(std::size_t i=0;i<number;++i){
118 names.push_back(j_names.at(i));
119 }
120 }else{
121 number=config.at("number");
122 for(std::size_t i=0;i<number;++i){
123 names.push_back("Element"+std::to_string(i+1));
124 }
125 }
126 ret={
127 {"type","group"},
128 {"names",names},
129 {"elements",nl::json::array()},
130 {"overrider",nl::json::array()},
131 {"instance",nullptr},
132 {"index",-1}
133 };
134 if(dispatchAfterBroadcast){
135 for(std::size_t i=0;i<number;++i){
136 ret["elements"].push_back(dispatcher.get(element));
137 }
138 }else{
139 nl::json dispatched=dispatcher.get(element);
140 for(std::size_t i=0;i<number;++i){
141 ret["elements"].push_back(dispatched);
142 }
143 }
144 }else if(type=="choice"){
145 nl::json weights=config.at("weights");
146 nl::json candidates=config.at("candidates");
147 std::discrete_distribution<std::size_t> dist(weights.begin(),weights.end());
148 ret=dispatcher.get(candidates[dist(dispatcher.randomGen)]);
149 }else if(type=="concatenate"){
150 nl::json elements=config.at("elements");
151 assert(elements.is_array());
152 ret={
153 {"type","group"},
154 {"names",nl::json::array()},
155 {"elements",nl::json::array()},
156 {"overrider",nl::json::array()},
157 {"instance",nullptr},
158 {"index",-1}
159 };
160 std::size_t totalCount=0;
161 for(std::size_t i=0;i<elements.size();++i){
162 nl::json sub=dispatcher.get(elements.at(i));
163 if(sub.at("type")=="group"){
164 for(std::size_t j=0;j<sub.at("elements").size();++j){
165 ret["names"].push_back(sub.at("names").at(j));
166 ret["elements"].push_back(sub.at("elements").at(j));
167 totalCount++;
168 }
169 }else{
170 ret["names"].push_back("Element"+std::to_string(totalCount+1));
171 ret["elements"].push_back(sub);
172 }
173 }
174 }else if(type=="direct"){
175 ret=config;
176 }else if(type=="group"){
177 nl::json elements=config.at("elements");
178 assert(elements.is_array());
179 std::string order="fixed";
180 if(config.contains("order")){
181 if(config.at("order").is_string()){
182 order=config.at("order");
183 }
184 }
185 std::vector<std::string> names;
186 if(config.contains("names") && config.at("names").is_array()){
187 nl::json j_names=config.at("names");
188 for(std::size_t i=0;i<j_names.size();++i){
189 names.push_back(j_names.at(i));
190 }
191 }else{
192 for(std::size_t i=0;i<elements.size();++i){
193 names.push_back("Element"+std::to_string(i+1));
194 }
195 }
196 assert(elements.size()==names.size());
197 std::vector<std::size_t> indices(elements.size());
198 for(std::size_t i=0;i<elements.size();++i){indices[i]=i;}
199 if(order=="shuffled"){
200 std::shuffle(indices.begin(),indices.end(),dispatcher.randomGen);
201 }
202 ret={
203 {"type","group"},
204 {"names",names},
205 {"elements",nl::json::array()},
206 {"overrider",nl::json::array()},
207 {"instance",nullptr},
208 {"index",-1}
209 };
210 for(std::size_t i=0;i<elements.size();++i){
211 ret["elements"].push_back(dispatcher.get(elements.at(indices[i])));
212 }
213 }else if(type=="none"){
214 ret={
215 {"type","none"}
216 };
217 }else{
218 throw std::runtime_error("unknown type.");
219 }
220 if(config.at("overrider").size()>0){
221 for(auto& e:config.at("overrider")){
222 nl::json value=ConfigDispatcherElement(e).get(dispatcher);
223 if(ret.at("type")=="group"){
224 assert(value.at("type")=="group" && ret.at("elements").size()==value.at("elements").size());
225 for(std::size_t i=0;i<ret.at("elements").size();++i){
226 if(ret.at("elements").at(i).at("type")!="none"){
227 ret.at("elements").at(i).merge_patch(value.at("elements").at(i));
228 }
229 }
230 }else if(ret.at("type")=="none"){
231 //ret={{"type","none"}};
232 }else{
233 ret.merge_patch(value);
234 }
235 }
236 }
237 if(instance!=""){
238 instances[instance]=ret;
239 }
240 }
241 if(index>=0 && ret.at("type")=="group" && index<ret.at("elements").size()){
242 return ret.at("elements").at(index);
243 }else{
244 return ret;
245 }

282nl::json ConfigDispatcher::sanitize(const nl::json& src){
283 if(src.is_object()){
284 if(src.contains("type")){
285 if(src.at("type")=="direct"){
286 return sanitize(src.at("value"));
287 }else if(src.at("type")=="group"){
288 nl::json ret=nl::json::object();
289 ret["type"]="group";
290 for(std::size_t i=0;i<src["names"].size();++i){
291 ret[src["names"][i].get<std::string>()]=sanitize(src["elements"][i]);
292 }
293 return ret;
294 }else if(src.at("type")=="none"){
295 return nl::json(nullptr);
296 }else{
297 return src;
298 }
299 }else{
300 nl::json ret=nl::json::object();
301 for(auto& e:src.items()){
302 auto tmp=sanitize(e.value());
303 if(!tmp.is_null()){
304 ret[e.key()]=tmp;
305 }
306 }
307 return ret;
308 }
309 }else{
310 return src;
311 }

316std::map<std::string,nl::json> RecursiveJsonExtractor::sub(const nl::json& node,const std::string& path,std::function<bool(const nl::json&)> checker){
317 std::map<std::string,nl::json> ret;
318 if(checker(node)){
319 if(path=="/"){
320 ret["/"]=node;
321 }else{
322 ret[path.substr(0,path.size()-1)]=node;
323 }
324 }else{
325 if(node.is_object()){
326 for(auto& e:node.items()){
327 auto subvec=sub(e.value(),path+e.key()+"/",checker);
328 for(auto& s:subvec){
329 ret[s.first]=s.second;
330 }
331 }
332 }else if(node.is_array()){
333 int i=0;
334 for(auto& e:node){
335 auto subvec=sub(e,path+"["+std::to_string(i)+"]/",checker);
336 for(auto& s:subvec){
337 ret[s.first]=s.second;
338 }
339 ++i;
340 }
341 }else{
342 //do nothing
343 }
344 }
345 return std::move(ret);
346}
