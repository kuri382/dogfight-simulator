81bool CommunicationBuffer::requestInvitation(const std::shared_ptr<Asset>& asset){
82 bool ret=false;
83 for(auto&& query:inviteOnRequest){
84 std::regex re;
85 bool typeMatch;
86 if(query.find("PhysicalAsset:")==0){
87 re=std::regex(query.substr(14));
88 typeMatch=util::isinstance<PhysicalAsset>(asset);
89 }else if(query.find("Controller:")==0){
90 re=std::regex(query.substr(11));
91 typeMatch=util::isinstance<Controller>(asset);
92 }else if(query.find("Agent:")==0){
93 re=std::regex(query.substr(6));
94 typeMatch=util::isinstance<Agent>(asset);
95 }else{
96 throw std::runtime_error("Invalid query for CommunicationBuffer's inviteOnRequest. query="+query);
97 }
98 if(typeMatch && std::regex_match(asset->getFullName(),re)){
99 invitationRequested.emplace_back(asset);
100 ret=true;
101 break;
102 }
103 }
104 if(ret && validated){
105 //初期化後の追加の場合
106 participants[asset->getFullName()]=asset;
107 asset->addCommunicationBuffer(this->shared_from_this());
108 }
109 return ret;

128void CommunicationBuffer::validateSub(const std::string& query){
129 if(query.find("PhysicalAsset:")==0){
130 std::regex re(query.substr(14));
131 for(auto&& e:manager.lock()->getAssets([&re](const std::shared_ptr<const PhysicalAsset>& asset)->bool {
132 return std::regex_match(asset->getFullName(),re);
133 })){
134 participants[e.lock()->getFullName()]=e;
135 }
136 }else if(query.find("Controller:")==0){
137 std::regex re(query.substr(11));
138 for(auto&& e:manager.lock()->getControllers([&re](const std::shared_ptr<const Controller>& cont)->bool {
139 return std::regex_match(cont->getFullName(),re);
140 })){
141 participants[e.lock()->getFullName()]=e;
142 }
143 }else if(query.find("Agent:")==0){
144 std::regex re(query.substr(6));
145 for(auto&& e:manager.lock()->getAgents([&re](const std::shared_ptr<const Agent>& agent)->bool {
146 return std::regex_match(agent->getFullName(),re);
147 })){
148 participants[e.lock()->getFullName()]=e;
149 }
150 }else{
151 throw std::runtime_error("Invalid query for CommunicationBuffer's participants. name="+name+", query="+query);
152 }
