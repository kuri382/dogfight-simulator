292 def drawCone(self, f, L, angle, N):
293 """ fの正面を起点に距離L、頂角angleの球面錐を描画する。
294 色設定はこの関数の外で行う。
295 Lはm、angleはradで指定する。
296 """
297 ex=np.array(f["ex"])
298 ey=np.array(f["ey"])
299 ez=np.array(f["ez"])
300 xy=sqrt(ex[0]*ex[0]+ex[1]*ex[1])
301 pitchAngle=atan2(-ex[2],xy)
302 eyHorizontal=np.cross(np.array([0.,0.,1.]),ex)
303 sn=np.linalg.norm(eyHorizontal)
304 if(abs(sn)<1e-6):
305 eyHorizontal=np.array([0.,1.,0.])
306 else:
307 eyHorizontal/=sn
308 exHorizontal=np.cross(eyHorizontal,np.array([0.,0.,1.]))
309 exHorizontal/=np.linalg.norm(exHorizontal)
310 cs=np.dot(ex,exHorizontal)
311 if self.draw_xy:
312 #XY平面プラス側
313 psxy=[self.simToReg(f["pos"])]+[self.simToReg(
314 f["pos"]+L*(cos(-angle+angle*2*i/N)*exHorizontal+sin(-angle+angle*2*i/N)*eyHorizontal))
315 for i in range(N+1)]
316 psxy=polygonRegionCut(psxy,[0.0,1.0],0)
317 glBegin(GL_TRIANGLE_FAN)
318 for p in psxy:
319 tmp = Draw2D.convertPos(self.regToSur(p)[0])
320 glVertex2d(tmp[0], tmp[1])
321 glEnd()
322 #XY平面マイナス側
323 psxy=[self.simToReg(f["pos"])]+[self.simToReg(
324 f["pos"]-L*(cos(-angle+angle*2*i/N)*exHorizontal*abs(sin(pitchAngle))+sin(-angle+angle*2*i/N)*eyHorizontal))
325 for i in range(N+1)]
326 psxy=polygonRegionCut(psxy,[0.0,1.0],0)
327 glBegin(GL_TRIANGLE_FAN)
328 for p in psxy:
329 tmp = Draw2D.convertPos(self.regToSur(p)[0])
330 glVertex2d(tmp[0], tmp[1])
331 glEnd()
332 exInYZ=(ex*np.array([0,1,1]))
333 sn=np.linalg.norm(exInYZ)
334 if(sn<1e-6):
335 exInYZ=np.array([0.,1.,0.])
336 absSinYaw=1.0
337 eyz=np.array([0.,0.,1.])
338 else:
339 exInYZ=exInYZ/sn
340 eyz=np.cross(exInYZ,np.array([1.,0.,0.]))
341 absSinYaw=abs(asin(ex[0]))
342 if self.draw_yz:
343 #YZ平面プラス側
344 psyz=[self.simToReg(f["pos"])]+[self.simToReg(
345 f["pos"]+L*(cos(-angle+angle*2*i/N)*exInYZ+sin(-angle+angle*2*i/N)*eyz))
346 for i in range(N+1)]
347 psyz=polygonRegionCut(psyz,[0.0,1.0],2)
348 glBegin(GL_TRIANGLE_FAN)
349 for p in psyz:
350 tmp = Draw2D.convertPos(self.regToSur(p)[1])
351 glVertex2d(tmp[0], tmp[1])
352 glEnd()
353 #YZ平面マイナス側
354 psyz=[self.simToReg(f["pos"])]+[self.simToReg(
355 f["pos"]-L*(cos(-angle+angle*2*i/N)*exInYZ*absSinYaw+sin(-angle+angle*2*i/N)*eyz))
356 for i in range(N+1)]
357 psyz=polygonRegionCut(psyz,[0.0,1.0],2)
358 glBegin(GL_TRIANGLE_FAN)
359 for p in psyz:
360 tmp = Draw2D.convertPos(self.regToSur(p)[1])
361 glVertex2d(tmp[0], tmp[1])
362 glEnd()

364 def getPolicyNameToDisplay(self, frame: dict) -> str:
365 """各陣営のAgentを表す名称を取得するもの。
366 デフォルトではframe内で最初に見つかった所属Agentのpolicy名となる。
367 派生クラスで適宜カスタマイズすること。
368
369 Args:
370 frame (dict): 現在のフレームデータ
371
372 Returns:
373 str: Agent名表示用の文字列
374 """
375 ruler=frame["ruler"]
376 policyName = {}
377 for team in [ruler["westSider"], ruler["eastSider"]]:
378 policyName[team]=""
379 for agentFullName,agent in frame["agents"].items():
380 if agent["team"]==team:
381 _,_,policy=agentFullName.split(":")
382 if policy.startswith("Policy_"):
383 policyName[team]=policy[7:]
384 else:
385 policyName[team]=policy
386 break
387 return policyName

415 def calcRegion(self,frame):
416 """x,y,z軸の描画対象範囲を計算する。
417 """
418 ruler=frame["ruler"]
419 self.fgtrRegion=[
420 np.min(np.r_[[self.transformToLocalCRS(f["pos"]) for f in frame["fighters"].values()]],0),
421 np.max(np.r_[[self.transformToLocalCRS(f["pos"]) for f in frame["fighters"].values()]],0)
422 ]
423 ruler=frame["ruler"]
424 self.fieldRegion=[
425 np.array([-ruler["dOut"],-ruler["dLine"],-ruler["hLim"]]),
426 np.array([ruler["dOut"],ruler["dLine"],0])
427 ]
428 self.bothRegion=[
429 np.min(np.r_[[self.fgtrRegion[i],self.fieldRegion[i]]],0) for i in range(2)
430 ]
431 xyAspect=self.h_xy*Draw2D.height/(Draw2D.width*(1-self.w_margin*2))
432 minR=np.array([(self.bothRegion[0][i] if self.regionType[i]=='fit' else self.fieldRegion[0][i]) for i in range(3)])
433 maxR=np.array([(self.bothRegion[1][i] if self.regionType[i]=='fit' else self.fieldRegion[1][i]) for i in range(3)])
434 mid=(minR+maxR)/2.0
435 delta=(maxR-minR)/2.0
436 minR=mid-delta*(self.fieldMargin+np.array([1,1,1]))
437 maxR=mid+delta*(self.fieldMargin+np.array([1,1,1]))
438 if(self.xyScaleType=='same'):
439 if(xyAspect*(maxR[1]-minR[1])>(maxR[0]-minR[0])):
440 mid=(maxR[0]+minR[0])/2.0
441 delta=xyAspect*(maxR[1]-minR[1])/2.0
442 minR[0]=mid-delta
443 maxR[0]=mid+delta
444 elif(xyAspect*(maxR[1]-minR[1])<(maxR[0]-minR[0])):
445 mid=(maxR[1]+minR[1])/2.0
446 delta=(maxR[0]-minR[0])/xyAspect/2.0
447 minR[1]=mid-delta
448 maxR[1]=mid+delta
449 #slide
450 slide=np.array([0.0,0.0,0.0])
451 if self.slideRegion:
452 for i in range(3):
453 if self.fgtrRegion[0][i] < minR[i]:
454 slide[i] -= minR[i] - self.fgtrRegion[0][i]
455 if self.fgtrRegion[1][i] > maxR[i]:
456 slide[i] += self.fgtrRegion[1][i] - maxR[i]
457 self.region=[minR+slide,maxR+slide]
458 def transformToLocalCRS(self, value, coordinateType: CoordinateType=CoordinateType.POSITION_ABS):
459 if isinstance(value, Coordinate):
460 return value(self.crs)
461 else:
462 return self.crs.transformFrom(value, self.time, self.rootCRS, coordinateType)()
463 def simToReg(self, sim, needTransformation: bool=True):#returns (xy,yz)
464 """シミュレーション中の基準座標系で表された位置simを、
465 x,y,z各軸を描画範囲で正規化し0〜1にして返す。
466 """
467 if needTransformation:
468 sim = self.transformToLocalCRS(sim)
469 return (sim-self.region[0])/(self.region[1]-self.region[0])
470 def regToSur(self,reg):#'xy' or 'yz'
471 """各軸正規化された位置regを、
472 xy図、yz図上で対応する位置に表示されるような、画面全体のxy座標に変換して返す。
473 """
474 xy=np.array([
475 Draw2D.width*(self.w_margin+reg[1]*(1.0-2*self.w_margin)),
476 Draw2D.height*((self.h_margin*2+self.h_yz)+reg[0]*(self.h_xy))
477 ])
478 yz=np.array([
479 Draw2D.width*(self.w_margin+reg[1]*(1.0-2*self.w_margin)),
480 Draw2D.height*(self.h_margin+(1.0-reg[2])*self.h_yz)
481 ])
482 return np.array([xy,yz])
483 def makeGrid(self,interval):
484 """戦域の区切り線を描く。
485 """
486 lower=np.ceil(self.region[0]/interval)
487 upper=np.floor(self.region[1]/interval)
488 cnt=upper-lower
489 for i in range(int(lower[0]),int(upper[0])+1):
490 d=(i*interval[0]-self.region[0][0])/(self.region[1][0]-self.region[0][0])
491 p=[self.regToSur(x)[0] for x in [[d,0,0],[d,1,1]]]
492 drawLine2D(p[0][0],p[0][1],p[1][0],p[1][1])
493 for i in range(int(lower[1]),int(upper[1])+1):
494 d=(i*interval[1]-self.region[0][1])/(self.region[1][1]-self.region[0][1])
495 p=[self.regToSur(x) for x in [[0,d,0],[1,d,1]]]
496 drawLine2D(p[0][0][0],p[0][0][1],p[1][0][0],p[1][0][1])
