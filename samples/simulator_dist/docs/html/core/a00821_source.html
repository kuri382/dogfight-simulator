16template<class T>
17requires ( std::is_enum_v<T> )
18struct def_common_serialization<T>{
19 static constexpr bool op_enable_if_hook = true; // これによりcls.defのオーバーロードが実体化され、以下のexecuteを呼べるようになる
20
21 template <typename Class, typename... Extra>
22 void execute(Class &cl, const Extra &...extra) const{
23 cl
24 .def("to_json",[](T& v){return ::asrc::core::util::enumToJson<T,nl::ordered_json>(v);})
25 .def("load_from_json",[](T& v,const nl::ordered_json& j){j.get_to(v);})
26 .def_static("from_json",[](const nl::ordered_json& j){return ::asrc::core::util::jsonToEnum<T>(j);})
27 .def("to_binary",[](T& v){return ::asrc::core::util::enumToBinary(v);})
28 .def("load_from_binary",[](T& v,const std::string& j){v=::asrc::core::util::binaryToEnum<T>(j);})
29 .def_static("from_binary",[](const std::string& j){return ::asrc::core::util::binaryToEnum<T>(j);})
30 .def("save",&::asrc::core::util::save_func_exposed_to_python<T>)
31 .def("load",&::asrc::core::util::load_func_exposed_to_python<T>)
32 .def_static("static_load",&::asrc::core::util::static_load_func_exposed_to_python<T>)
33 .def_property_readonly_static("_allow_cereal_serialization_in_cpp",[](pybind11::object /* self */){return true;})
34 ;
35 }
36};

38/**
39 * @brief enum classを数値,文字列, nl::basic_json のいずれからでもインスタンス化可能な形でPython側に公開する。
40 *
41 * @details
42 * @code{.cpp}
43 * ::asrc::core::expose_enum_value_helper(
44 * ::asrc::core::expose_enum_class<EnumType>(module,"EnumType")
45 * ,"value1"
46 * ,"value2"
47 * ,"value3"
48 * );
49 * @endcode
50 * のように用いる。
51 */
52template<typename T,typename Parent, typename ... Extras>
53requires ( std::is_enum_v<T> )
54pybind11::enum_<T> expose_enum_class(Parent& m, const char* className, Extras&& ... extras){
55 auto ret=pybind11::enum_<T>(m,className,std::forward<Extras>(extras)...);
56 ret
57 .def(pybind11::init(&::asrc::core::util::strToEnum<T>))
58 .def(pybind11::init(&::asrc::core::util::jsonToEnum<T>))
59 .def(def_common_serialization<T>())
60 ;
61 return std::move(ret);
62
63}
