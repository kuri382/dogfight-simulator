23void TopocentricCRS::initialize(){
24 BaseType::initialize();
25 // 原点の設定
26 // cartesianOriginは原点のbaseCRSにおける座標。
27 givenByEllipsoidalOrigin=false;
28 givenByCartesianOrigin=false;
29 givenByMotionState=false;
30 ellipsoidalOrigin=Eigen::Vector3d::Zero(); // in lat[deg], lon[deg], alt[m]
31 cartesianOrigin=Eigen::Vector3d::Zero(); // in X[m], Y[m], Z[m]
32 if(instanceConfig.contains("cartesianOrigin")){
33 // baseがGeodeticの場合は対応するECEF上での原点位置、
34 // Affineの場合はbaseCRS上での原点位置として扱う。
35 cartesianOrigin=instanceConfig.at("cartesianOrigin");
36 givenByCartesianOrigin=true;
37 }else if(modelConfig.contains("cartesianOrigin")){
38 // baseがGeodeticの場合は対応するECEF上での原点位置、
39 // Affineの場合はbaseCRS上での原点位置として扱う。
40 cartesianOrigin=modelConfig.at("cartesianOrigin");
41 givenByCartesianOrigin=true;
42 }else if(instanceConfig.contains("ellipsoidalOrigin")){
43 // baseがGeodeticの場合のみ有効
44 ellipsoidalOrigin=instanceConfig.at("ellipsoidalOrigin");
45 givenByEllipsoidalOrigin=true;
46 }else if(modelConfig.contains("ellipsoidalOrigin")){
47 ellipsoidalOrigin=modelConfig.at("ellipsoidalOrigin");
48 givenByEllipsoidalOrigin=true;
49 }else if(instanceConfig.contains("motion")){
50 // baseがGeographicの場合は無効
51 // baseCRS上での原点位置及び座標軸の回転情報として扱う。
52 internalMotionState=instanceConfig.at("motion");
53 givenByMotionState=true;
54 }else if(modelConfig.contains("motion")){
55 internalMotionState=modelConfig.at("motion");
56 givenByMotionState=true;
57 }else{
58 throw std::runtime_error("TopocentricCRS requires 'cartesianOrigin', 'ellipsoidalOrigin', or 'motion' for instantiation.");
59 }
60 if(instanceConfig.contains("onSurface")){
61 onSurface=instanceConfig.at("onSurface");
62 }else if(modelConfig.contains("onSurface")){
63 onSurface=modelConfig.at("onSurface");
64 }else{
65 onSurface=false;
66 }
67 lastAZ=0;
68}

92void TopocentricCRS::validateImpl(bool isFirstTime) const{
93 this->DerivedCRS::validateImpl(isFirstTime);
94 if(isFirstTime || !hasBeenValidated()){
95 // baseCRSの種別チェックは省略(Topocentricは任意のCRSをbaseにしてよい)
96
97 // originの指定方法チェック
98 auto baseCRS = getBaseCRS();
99 if(givenByEllipsoidalOrigin){
100 // baseがGeodeticの場合のみ有効
101 if(!std::dynamic_pointer_cast<GeodeticCRS>(baseCRS)){
102 throw std::runtime_error("'ellipsoidalOrigin' for TopocentricCRS config is valid only if baseCRS is GeodeticCRS.");
103 }
104 }else if(givenByMotionState){
105 // baseがGeographicの場合は無効
106 // baseCRS上での原点位置及び座標軸の回転情報として扱う。
107 if(std::dynamic_pointer_cast<GeographicCRS>(baseCRS)){
108 throw std::runtime_error("'motion' in TopocentricCRS config is valid only if baseCRS is not GeographicCRS.");
109 }
110 if(internalMotionState.getCRS()){
111 // 有効なCRSを持っていれば変換
112 cartesianOrigin=internalMotionState.pos(baseCRS);
113 }else{
114 // 持っていなければ直接値を取得
115 cartesianOrigin=internalMotionState.pos();
116 }
117 }
118 checkCoordinateSystemType();
119 checkAxisOrder();
120 internalMotionState.setCRS(getIntermediateCRS(),true);
121 }
122 // internalMotionState(のposとq)を更新する
123 calculateMotionStates();
124}

168Coordinate TopocentricCRS::transformTo(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS, const CoordinateType& coordinateType) const{
169 if(!isValid()){validate();}
170 if(!dstCRS){
171 throw std::runtime_error("dstCRS is nullptr!");
172 }
173 if(this==std::dynamic_pointer_cast<TopocentricCRS>(dstCRS).get()){
174 return std::move(Coordinate(
175 value,
176 location,
177 dstCRS,
178 time,
179 coordinateType
180 ));
181 }
182 if(!isTransformableTo(dstCRS)){
183 throw std::runtime_error(getFactoryClassName()+" can't be transformed to "+dstCRS->getFactoryClassName());
184 }
185 if(dstCRS==getIntermediateCRS()){
186 return transformToIntermediateCRS(value,location,time,coordinateType);
187 }else if(dstCRS==getNonDerivedBaseCRS()){
188 return transformToNonDerivedBaseCRS(value,location,time,coordinateType);
189 }else if(dstCRS==getBaseCRS()){
190 return transformToBaseCRS(value,location,time,coordinateType);
191 }else{
192 return dstCRS->transformFrom(transformToIntermediateCRS(value,location,time,coordinateType));
193 }

222Quaternion TopocentricCRS::getQuaternionTo(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS) const{
223 if(!isValid()){validate();}
224 if(!dstCRS){
225 throw std::runtime_error("dstCRS is nullptr!");
226 }
227 if(this==std::dynamic_pointer_cast<TopocentricCRS>(dstCRS).get()){
228 return Quaternion(1,0,0,0);
229 }
230 if(!isTransformableTo(dstCRS)){
231 throw std::runtime_error(getFactoryClassName()+" can't be transformed to "+dstCRS->getFactoryClassName());
232 }
233 // MotionState::qは左からqをかけてderived座標をbase座標に移す
234 // v_base = q.transformVector(v_derived)
235 auto qToIntermediateCRSFromThis=internalMotionState.q;
236 if(getIntermediateCRS()==dstCRS){
237 return qToIntermediateCRSFromThis;
238 }else{
239 auto locationInIntermediate=transformToIntermediateCRS(location,location,time,CoordinateType::POSITION_ABS)();
240 auto qToDstFromIntermediateCRS=getIntermediateCRS()->getQuaternionTo(locationInIntermediate,time,dstCRS);
241 return qToDstFromIntermediateCRS*qToIntermediateCRSFromThis;
242 }
243}
244Quaternion TopocentricCRS::getQuaternionFrom(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& srcCRS) const{
245 if(!isValid()){validate();}
246 if(!srcCRS){
247 throw std::runtime_error("srcCRS is nullptr!");
248 }
249 if(this==std::dynamic_pointer_cast<TopocentricCRS>(srcCRS).get()){
250 return Quaternion(1,0,0,0);
251 }
252 if(!isTransformableFrom(srcCRS)){
253 throw std::runtime_error(getFactoryClassName()+" can't be transformed from "+srcCRS->getFactoryClassName());
254 }
255 // MotionState::qは左からqをかけてderived座標をbase座標に移す
256 // v_base = q.transformVector(v_derived)
257 auto qToThisFromIntermediateCRS=internalMotionState.q.conjugate();
258 if(getIntermediateCRS()==srcCRS){
259 return qToThisFromIntermediateCRS;
260 }else{
261 auto qToIntermediateCRSFromSrc=getIntermediateCRS()->getQuaternionFrom(location,time,srcCRS);
262 return qToThisFromIntermediateCRS*qToIntermediateCRSFromSrc;
263 }

306Coordinate TopocentricCRS::transformToIntermediateCRS(const Eigen::Vector3d& value_,const Eigen::Vector3d& location_, const Time& time, const CoordinateType& coordinateType) const{
307 if(!isValid()){validate();}
308 Eigen::Vector3d ret;
309 auto value=toCartesian(value_);
310 auto location=toCartesian(location_);
311 if(coordinateType==CoordinateType::POSITION_ABS){
312 ret=internalMotionState.absBtoP(value);
313 }else if(coordinateType==CoordinateType::POSITION_REL){
314 ret=internalMotionState.relBtoP(value);
315 }else if(coordinateType==CoordinateType::DIRECTION){
316 ret=internalMotionState.dirBtoP(value);
317 }else if(coordinateType==CoordinateType::VELOCITY){
318 ret=internalMotionState.velBtoP(value,location);
319 }else if(coordinateType==CoordinateType::ANGULAR_VELOCITY){
320 ret=internalMotionState.omegaBtoP(value);
321 }else{
322 throw std::runtime_error("Invalide coordinate type.");
323 }
324 return Coordinate(
325 ret,
326 internalMotionState.absBtoP(location),
327 getIntermediateCRS(),
328 time,
329 coordinateType
330 );

383Eigen::Vector3d TopocentricCRS::sphericalToCartesian(const Eigen::Vector3d& v) const{
384 if(checkTopocentricCartesianAxisOrder(getAxisOrder())){
385 // (Azimuth Elevation Range) -> NED -> axisOrder
386 Eigen::Vector3d ned(
387 v(2)*cos(v(0))*cos(v(1)),
388 v(2)*sin(v(0))*cos(v(1)),
389 -v(2)*sin(v(1))
390 );
391 return getAxisSwapQuaternion("NED",getAxisOrder()).transformVector(ned);
392 }else{// if(checkBodyCartesianAxisOrder(getAxisOrder()){
393 // (Azimuth Elevation Range) -> FSD -> axisOrder
394 Eigen::Vector3d fsd(
395 v(2)*cos(v(0))*cos(v(1)),
396 v(2)*sin(v(0))*cos(v(1)),
397 -v(2)*sin(v(1))
398 );
399 return getAxisSwapQuaternion("FSD",getAxisOrder()).transformVector(fsd);
400 }
401}
402Eigen::Vector3d TopocentricCRS::cartesianToSpherical(const Eigen::Vector3d& v) const{
403 if(checkTopocentricCartesianAxisOrder(getAxisOrder())){
404 // axisOrder -> NED -> (Azimuth Elevation Range)
405 auto ned=getAxisSwapQuaternion(getAxisOrder(),"NED").transformVector(v);
406 double xy=sqrt(ned(0)*ned(0)+ned(1)*ned(1));
407 if(xy==0){
408 return Eigen::Vector3d(
409 0,
410 ned(2)>0 ? -M_PI_2 : M_PI_2,
411 abs(ned(2))
412 );
413 }else{
414 double el=atan2(-ned(2),xy);
415 return Eigen::Vector3d(
416 atan2(ned(1),ned(0)),
417 el,
418 sqrt(xy*xy+ned(2)*ned(2))
419 );
420 }
421 }else{// if(checkBodyCartesianAxisOrder(getAxisOrder()){
422 // axisOrder -> FSD -> (Azimuth Elevation Range)
423 auto fsd=getAxisSwapQuaternion(getAxisOrder(),"FSD").transformVector(v);
424 double xy=sqrt(fsd(0)*fsd(0)+fsd(1)*fsd(1));
425 if(xy==0){
426 return Eigen::Vector3d(
427 0,
428 fsd(2)>0 ? -M_PI_2 : M_PI_2,
429 abs(fsd(2))
430 );
431 }else{
432 double el=atan2(-fsd(2),xy);
433 return Eigen::Vector3d(
434 atan2(fsd(1),fsd(0)),
435 el,
436 sqrt(xy*xy+fsd(2)*fsd(2))
437 );
438 }
439 }

458void TopocentricCRS::calculateMotionStates() const{
459 auto baseCRS = getBaseCRS();
460 auto intermediateCRS = getIntermediateCRS();
461 auto nonDerivedBaseCRS = getNonDerivedBaseCRS();
462
463 Eigen::Vector3d cartesianOriginInIntermediateCRS = cartesianOrigin;
464 if(auto derived = std::dynamic_pointer_cast<DerivedCRS>(baseCRS)){
465 // baseがDerivedCRSの場合には、cartesianOriginをbaseCRSにおける原点位置と解釈する。
466 // ellipsoidalOriginはDerivedCRSをbaseに指定する意味がないので無視する。
467 cartesianOriginInIntermediateCRS = derived->transformTo(
468 cartesianOrigin,
469 cartesianOrigin,
470 internalMotionState.time,
471 intermediateCRS,
472 CoordinateType::POSITION_ABS
473 )();
474 }
475
476 // ENUへの変換
477 if(auto geod = std::dynamic_pointer_cast<GeodeticCRS>(intermediateCRS)){
478 if(givenByEllipsoidalOrigin){
479 // こちらの分岐に入るときはbase自身がGeodeticCRS
480 cartesianOriginInIntermediateCRS = geod->transformGeographicCoordinateToECEF(ellipsoidalOrigin,CoordinateType::POSITION_ABS);
481 cartesianOrigin = cartesianOriginInIntermediateCRS;
482 }else{
483 ellipsoidalOrigin = geod->transformECEFCoordinateToGeographicCRS(cartesianOriginInIntermediateCRS,CoordinateType::POSITION_ABS);
484 }
485 // EPSG:9836
486 double lat_0=deg2rad(ellipsoidalOrigin(0));
487 double lon_0=deg2rad(ellipsoidalOrigin(1));
488 Eigen::Matrix3d Rt({
489 {-sin(lon_0), -sin(lat_0)*cos(lon_0), cos(lat_0)*cos(lon_0)},
490 { cos(lon_0), -sin(lat_0)*sin(lon_0), cos(lat_0)*sin(lon_0)},
491 { 0, cos(lat_0), sin(lat_0)}
492 });
493 // pos_ecef = Rt * pos_topo + cartesianOriginInIntermediateCRS
494 // pos_base = this.q.transformVector(pos_this) + this.pos
495 internalMotionState.setPos(cartesianOriginInIntermediateCRS);
496 internalMotionState.setQ(Quaternion::fromRotationMatrix(Rt));
497 }else if(auto flat = std::dynamic_pointer_cast<PureFlatCRS>(intermediateCRS)){
498 // PureFlatからの変換はPureFlatのaxisOrderを考慮する必要がある。
499 internalMotionState.setPos(cartesianOriginInIntermediateCRS);
500 internalMotionState.setQ(getAxisSwapQuaternion("ENU",flat->getAxisOrder()));
501 }else{
502 throw std::runtime_error("Non-derived (top-level) base CRS of TopocentricCRS must be Geodetic or PureFlat CRS.");
503 }
504
505 //高度方向のオフセット
506 if(onSurface){
507 auto altOffset=internalMotionState.q.transformVector(
508 Eigen::Vector3d(0,0,intermediateCRS->getHeight(internalMotionState.pos(),internalMotionState.time))
509 );// [intermediateCRS<-ENU]
510 internalMotionState.setPos(internalMotionState.pos()-altOffset);
511 }
512
513 //ENUから指定の座標軸への変換
514 if(checkTopocentricCartesianAxisOrder(getAxisOrder())){
515 if(getAxisOrder()!="ENU"){
516 internalMotionState.setQ( //[intermediateCRS <- axisOrder] =
517 internalMotionState.q * // [intermediateCRS<-ENU] *
518 getAxisSwapQuaternion(getAxisOrder(),"ENU") // [ENU <- axisOrder]
519 );
520 }
521 }else{// if(checkBodyCartesianAxisOrder(getAxisOrder()){
522 if(auto affine = std::dynamic_pointer_cast<AffineCRS>(baseCRS)){
523 // baseのFWD方向のベクトルを水平面に射影した方向を「前方」に相当する基準軸Fとした局所水平座標系
524 // 「左方」に相当するP=U×Fの3軸を導出する必要がある。
525 Eigen::Vector3d fwdInBase=getAxisSwapQuaternion("FPU",affine->getAxisOrder()).transformVector(Eigen::Vector3d(1,0,0));
526 Eigen::Vector3d fwdInIntermediateCRS=affine->transformTo(
527 fwdInBase,
528 Eigen::Vector3d::Zero(),
529 internalMotionState.time,
530 intermediateCRS,
531 CoordinateType::DIRECTION
532 )();
533 Eigen::Vector3d fwdInENU=internalMotionState.relPtoB(fwdInIntermediateCRS);
534 double xy=sqrt(fwdInENU(0)*fwdInENU(0)+fwdInENU(1)*fwdInENU(1));
535 if(xy>0){
536 lastAZ=atan2(fwdInENU(1),fwdInENU(0));
537 } // xy==0の場合は前回の値を使う。
538 fwdInENU=Eigen::Vector3d(cos(lastAZ),sin(lastAZ),0);
539 Eigen::Vector3d portInENU(-sin(lastAZ),cos(lastAZ),0);
540 Eigen::Vector3d upInENU(0,0,1);
541 auto qFPUtoENU=Quaternion::fromBasis(fwdInENU,portInENU,upInENU); // [ENU <- FPU]
542
543 internalMotionState.setQ( //[intermediateCRS <- axisOrder] =
544 internalMotionState.q * //[intermediateCRS <- ENU] *
545 qFPUtoENU.conjugate() * //[ENU <- FPU] *
546 getAxisSwapQuaternion(getAxisOrder(),"FPU") //[FPU <- axisOrder] *
547 );
548 }else{

550 }

556Quaternion TopocentricCRS::getEquivalentQuaternionToTopocentricCRS(const Eigen::Vector3d& origin, const Time& time, const std::string& axisOrder_) const{
557 //CRSとして生成せずにクォータニオンとして得る。
558 std::string dstAxisOrder=axisOrder_;
559 std::transform(dstAxisOrder.begin(),dstAxisOrder.end(),dstAxisOrder.begin(),
560 [](unsigned char c){ return std::toupper(c);}
561 );
562 auto intermediateCRS = getIntermediateCRS();
563 if(!intermediateCRS){
564 throw std::runtime_error("intermediateCRS is nullptr!");
565 }
566 Coordinate cartesianOriginInIntermediateCRS = transformToIntermediateCRS(origin,origin,time,CoordinateType::POSITION_ABS);
567 if(checkTopocentricCartesianAxisOrder(dstAxisOrder)){
568 return (
569 intermediateCRS->getEquivalentQuaternionToTopocentricCRS(cartesianOriginInIntermediateCRS,dstAxisOrder) * //[new Topo <- IntermediateCRS]
570 getQuaternionFrom(cartesianOriginInIntermediateCRS).conjugate() //[IntermediateCRS <- this]
571 );
572 }else{
573 throw std::runtime_error("Body-oriented TopocentricCRS cannot be created from TopocentricCRS.");
574 }
575}
