52std::pair<Eigen::Vector3d,Eigen::Vector3d> PropNav::calc(const Eigen::Vector3d &rs,const Eigen::Vector3d &vs,const Eigen::Vector3d &rt,const Eigen::Vector3d &vt){
53 Eigen::Vector3d rr=rt-rs;
54 Eigen::Vector3d vr=vt-vs;
55 double R2=rr.squaredNorm();
56 if(R2==0){
57 //本来は1/|rr|のスケールで発散するが、rrの向きが不定となる|rr|=0のときに限り、rrとvsが平行とみなすことによって指令値を0とする。
58 return std::make_pair(Eigen::Vector3d(0,0,0),Eigen::Vector3d(0,0,0));
59 }
60 double Vs2=vs.squaredNorm();
61 if(Vs2==0){
62 //vsの向きが不定となる|vs|=0のときに限り、vsとomegaが平行とみなすことによって指令値を0とする。
63 //ただし、現状のMissileクラスの運動モデルの実装の方が|v|=0に対応していないためこのような状況下では使用されない。
64 return std::make_pair(Eigen::Vector3d(0,0,0),Eigen::Vector3d(0,0,0));
65 }
66 Eigen::Vector3d omega=rr.cross(vr)/R2;
67 Eigen::Vector3d accel=gain*omega.cross(vs);
68 return std::make_pair(accel,vs.cross(accel)/Vs2);
69}

323void Missile::calcQ(){
324 Eigen::Vector3d ex=motion.vel().normalized();
325 Eigen::Vector3d ez=motion.dirHtoP(Eigen::Vector3d(0,0,1),Eigen::Vector3d(0,0,0),"NED",false);
326 Eigen::Vector3d ey=ez.cross(ex);
327 double Y=ey.norm();
328 if(Y<1e-8){
329 ey=ex.cross(Eigen::Vector3d(1,0,0)).normalized();
330 }else{
331 ey/=Y;
332 }
333 ez=ex.cross(ey);
334 motion.setQ(Quaternion::fromBasis(ex,ey,ez));
335}
336double Missile::calcRange(double vs,double hs,double vt,double ht,double obs,double aa){
337 double r0=30000.0;
338 double r1=300000.0;
339 double minR=1.0;
340 while(!calcRangeSub(vs,hs,vt,ht,obs,aa,r0)){
341 r1=r0;
342 r0*=0.5;
343 if(r0<minR){
344 return minR;
345 }
346 }
347 while(calcRangeSub(vs,hs,vt,ht,obs,aa,r1)){
348 r0=r1;
349 r1*=2;
350 }
351 double rm;
352 while((r1-r0)>1.0){
353 rm=(r0+r1)*0.5;
354 if(calcRangeSub(vs,hs,vt,ht,obs,aa,rm)){
355 r0=rm;
356 }else{
357 r1=rm;
358 }
359 }
360 return (r0+r1)*0.5;
361}
362bool Missile::calcRangeSub(double vs,double hs,double vt,double ht,double obs,double aa,double r){
363 Eigen::Vector3d posBef=motion.pos();
364 Eigen::Vector3d velBef=motion.vel();
365 Eigen::Vector3d omegaBef=motion.omega();
366 Quaternion qBef=motion.q;
367 Time timeBef=motion.time;
368 MotionState motion_prevBef=motion_prev;
369 bool isBoundToParentBef=isBoundToParent;
370 if(isBoundToParentBef){
371 unbindFromParent();
372 }
373 double dt=interval[SimPhase::BEHAVE]*manager->getBaseTimeStep();
374 motion.setPos(motion.absHtoP(Eigen::Vector3d(0,0,-hs),"NED",true));
375 motion.setVel(motion.velHtoP(Eigen::Vector3d(cos(obs),-sin(obs),0)*std::max(vs,1e-3),Eigen::Vector3d(-r,0,-hs),"NED",true));
376 motion.setOmega(Eigen::Vector3d(0,0,0));
377 calcQ();
378 Eigen::Vector3d tpos=motion.absHtoP(Eigen::Vector3d(r,0,-ht),"NED",true);
379 Eigen::Vector3d tvel=motion.velHtoP(Eigen::Vector3d(cos(aa),sin(aa),0)*vt,Eigen::Vector3d(0,0,-ht),"NED",true);
380 Eigen::Vector3d tpos_prev=tpos-dt*tvel;
381 motion_prev.setPos(motion.pos()-dt*motion.vel());
382 motion_prev.setVel(motion.vel());
383 double t=0;
384 bool finished=false;
385 bool hit=false;
386 auto propNav=getShared<PropNav>(controllers["Navigator"]);
387 bool hasLaunchedBef=hasLaunched;
388 hasLaunched=true;
389 while(!finished){
390 commands={
391 {"Navigator",{
392 {"rs",motion.pos()},
393 {"vs",motion.vel()},
394 {"rt",tpos},
395 {"vt",tvel}
396 }}
397 };
398 propNav->control();
399 calcMotion(t,dt);
400 tpos_prev=tpos;
401 tpos+=dt*tvel;
402 t+=dt;
403 if(hitCheck(tpos,tpos_prev)){
404 finished=true;
405 hit=true;
406 }else if(checkDeactivateCondition(t)){
407 finished=true;
408 hit=false;
409 }
410 }
411 if(isBoundToParentBef){
412 bindToParent();
413 }
414 motion.setPos(posBef);
415 motion.setVel(velBef);
416 motion.setOmega(omegaBef);
417 motion.setTime(timeBef);
418 motion.setQ(qBef);
419 motion_prev=motion_prevBef;
420 hasLaunched=hasLaunchedBef;
421 return hit;
422}

430double Missile::getRmax(const Eigen::Vector3d &rs_,const Eigen::Vector3d &vs_,const Eigen::Vector3d &rt_,const Eigen::Vector3d &vt_,const std::shared_ptr<CoordinateReferenceSystem>& crs){
431 //When aa is omitted, current aa is used.
432 Eigen::Vector3d rs,rt,vs,vt;
433 if(crs){
434 Time t = manager ? manager->getTime() : Time();
435 MotionState shooterMotion(crs,t,rs_,vs_,Eigen::Vector3d::Zero(),Quaternion(1,0,0,0),"FSD");
436 rs=shooterMotion.absPtoH(rs_,rs_,"NED",true);//0,0,-alt
437 rt=shooterMotion.absPtoH(rt_,rt_,"NED",true);
438 vs=shooterMotion.velPtoH(vs_,rs_,"NED",true);
439 vt=shooterMotion.velPtoH(vt_,rt_,"NED",true);
440 }else{
441 rs=rs_;
442 rt=rt_;
443 vs=vs_;
444 vt=vt_;
445 }
446 double Vs=vs.norm();
447 double hs=-rs(2);
448 double Vt=vt.norm();
449 double ht=-rt(2);
450 Eigen::Vector3d dr=Eigen::Vector3d(rt(0)-rs(0),rt(1)-rs(1),0);
451 Eigen::Vector3d vsh=Eigen::Vector3d(vs(0),vs(1),0);
452 Eigen::Vector3d vth=Eigen::Vector3d(vt(0),vt(1),0);
453 double Rh=dr.norm();
454 double Vsh=vsh.norm();
455 double Vth=vth.norm();
456 double obs,aa;
457 if(Vsh==0 || Rh==0){
458 obs=0.0;
459 }else{
460 obs=acos(std::min(1.,std::max(-1.,dr.dot(vsh)/(Rh*Vsh))));
461 }
462 if(Vth==0 || Rh==0){
463 aa=0.0;
464 }else{
465 aa=acos(std::min(1.,std::max(-1.,dr.dot(vth)/(Rh*Vth))));
466 }
467 if(obs>1e-8){
468 bool sameSide=(vsh.cross(dr)).dot(vth.cross(dr))>=0;
469 if(!sameSide){
470 aa=-aa;
471 }
472 }
473 Eigen::MatrixXd arg(1,6);
474 arg<<Vs,hs,Vt,ht,obs,aa;
475 return interpn(rangeTablePoints,rangeTable,arg)(0);
476}

517void Missile::makeRangeTable(const std::string& dstPath){
518 py::gil_scoped_acquire acquire;
519 std::cout<<"makeRangeTable start"<<std::endl;
520 int nvs=7;
521 int nhs=6;
522 int nvt=7;
523 int nht=6;
524 int nobs=7;
525 int naa=13;
526 Eigen::VectorXd vs=Eigen::VectorXd::LinSpaced(nvs,0.0,600.0);
527 Eigen::VectorXd hs=Eigen::VectorXd::LinSpaced(nhs,0.0,20000.0);
528 Eigen::VectorXd vt=Eigen::VectorXd::LinSpaced(nvt,0.0,600.0);
529 Eigen::VectorXd ht=Eigen::VectorXd::LinSpaced(nht,0.0,20000.0);
530 Eigen::VectorXd obs=Eigen::VectorXd::LinSpaced(nobs,0.0,M_PI);
531 Eigen::VectorXd aa=Eigen::VectorXd::LinSpaced(naa,-M_PI,M_PI);
532 int numDataPoints=nvs*nhs*nht*nvt*nobs*naa;
533 Eigen::MatrixXd indices=Eigen::MatrixXd::Zero(6,numDataPoints);//ColMajor
534 Eigen::MatrixXd args=Eigen::MatrixXd::Zero(6,numDataPoints);//ColMajor
535 std::cout<<"create indices..."<<std::endl;
536 int idx=0;
537 for(int aa_i=0;aa_i<naa;aa_i++){
538 for(int obs_i=0;obs_i<nobs;obs_i++){
539 for(int ht_i=0;ht_i<nht;ht_i++){
540 for(int vt_i=0;vt_i<nvt;vt_i++){
541 for(int hs_i=0;hs_i<nhs;hs_i++){
542 for(int vs_i=0;vs_i<nvs;++vs_i){
543 args.block(0,idx,6,1)<<vs(vs_i),hs(hs_i),vt(vt_i),ht(ht_i),obs(obs_i),aa(aa_i);
544 idx++;
545 }
546 }
547 }
548 }
549 }
550 }
551 std::cout<<"create indices done."<<std::endl;
552 int numProcess=std::thread::hardware_concurrency();
553 std::vector<std::thread> th;
554 std::vector<std::future<Eigen::VectorXd>> f;
555 std::vector<int> begins;
556 int numPerProc=numDataPoints/numProcess;
557 std::cout<<"run in "<<numProcess<<" processes. total="<<numDataPoints<<", perProc="<<numPerProc<<std::endl;
558 std::mutex mtx;
559 std::vector<std::shared_ptr<Missile>> msls;
560 for(int i=0;i<numProcess;++i){
561 nl::json dummy_ic=instanceConfig;
562 dummy_ic["entityFullName"]=getFullName()+"/for_range_table_"+std::to_string(i+1);
563 msls.push_back(createUnmanagedEntityByClassName<Missile>(isEpisodic(),"PhysicalAsset",getFactoryClassName(),modelConfig,dummy_ic));
564 std::promise<Eigen::VectorXd> p;
565 f.emplace_back(p.get_future());
566 Eigen::MatrixXd subargs=args.block(0,numPerProc*i,6,(i!=numProcess-1 ? numPerProc : numDataPoints-numPerProc*i));
567 th.emplace_back(makeRangeTableSub,std::ref(mtx),std::move(p),msls[i],subargs);
568 }
569 std::cout<<"waiting calculation"<<std::endl;
570 std::vector<Eigen::VectorXd> rangeTablePoints_={vs,hs,vt,ht,obs,aa};
571 Eigen::VectorXd returned(numDataPoints);
572 for(int i=0;i<numProcess;++i){
573 try{
574 returned.block(numPerProc*i,0,(i!=numProcess-1 ? numPerProc : numDataPoints-numPerProc*i),1)=f[i].get();
575 }catch(std::exception& ex){
576 {
577 std::lock_guard<std::mutex> lock(mtx);
578 std::cout<<"exception in proc("<<i<<"): "<<ex.what()<<std::endl;
579 }
580 }
581 }
582 Eigen::Tensor<double,6> rangeTable_=Eigen::TensorMap<Eigen::Tensor<double,6>>(returned.data(),nvs,nhs,nvt,nht,nobs,naa);
583 for(int i=0;i<numProcess;++i){
584 th[i].join();
585 }
