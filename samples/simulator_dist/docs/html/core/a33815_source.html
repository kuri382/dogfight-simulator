222nl::json SimulationManager::parseConfig(const nl::json& config_){
223 nl::json ret=nl::json::object();
224 if(config_.is_object()){
225 ret=config_;
226 }else if(config_.is_array()){
227 for(auto& e:config_){
228 if(e.is_object()){
229 ret.merge_patch(e);
230 }else if(e.is_string()){
231 std::ifstream ifs(e.get<std::string>());

837std::uint64_t SimulationManager::getPossibleNextTickCount(const SimPhase& phase) const{
838 // 現在の処理状況(tickCountと直近のphase)において、
839 // 引数で与えられたphaseが次に実行される可能性のある時刻(tick)を返す。
840 // phaseの処理中には呼び出さないこと。
841 switch(phase){
842 case SimPhase::PERCEIVE:
843 //inReset=trueを除くとつねにtickCount+1
844 return tickCount+1;
845 case SimPhase::CONTROL:
846 if(lastAssetPhase==SimPhase::CONTROL || lastAssetPhase==SimPhase::BEHAVE){
847 return tickCount+1;
848 }else{
849 //NONE,VALIDATE,PERCEIVE
850 return tickCount;
851 }
852 case SimPhase::BEHAVE:
853 if(lastAssetPhase==SimPhase::BEHAVE){
854 return tickCount+1;
855 }else{
856 //NONE,VALIDATE,PERCEIVE,CONTROL
857 return tickCount;
858 }
859 case SimPhase::AGENT_STEP:
860 if(lastAssetPhase==SimPhase::CONTROL || lastAssetPhase==SimPhase::BEHAVE){
861 return tickCount+1;
862 }else{
863 //NONE,VALIDATE,PERCEIVE
864 return tickCount;
865 }
866 case SimPhase::ON_INNERSTEP_BEGIN:
867 if(lastAssetPhase==SimPhase::PERCEIVE){
868 if(lastGroup1CallbackPhase==SimPhase::ON_INNERSTEP_BEGIN){
869 return tickCount+1;
870 }else{
871 return tickCount;
872 }
873 }else{
874 return tickCount+1;
875 }
876 case SimPhase::ON_INNERSTEP_END:
877 if(lastAssetPhase==SimPhase::PERCEIVE){
878 // onInnerStepEndはつねにonInnerStepBeginと対になって実行される
879 if(lastGroup1CallbackPhase==SimPhase::ON_INNERSTEP_BEGIN){
880 return tickCount;
881 }else{
882 return tickCount+1;
883 }
884 }else{
885 return tickCount+1;
886 }
887 default: // non-periodic phase
888 return tickCount;
889 }

1689void SimulationManager::runAssetPhaseFunc(const SimPhase& phase){
1690 //後の処理で各phaseごとの関数呼び出し形式を共通化しておく
1691 static const std::map<SimPhase,std::function<void(const std::shared_ptr<Asset>&)>> phaseFunc={
1692 {SimPhase::PERCEIVE,[](const std::shared_ptr<Asset>& a){a->perceive(false);}},
1693 {SimPhase::CONTROL,[](const std::shared_ptr<Asset>& a){a->control();}},
1694 {SimPhase::BEHAVE,[](const std::shared_ptr<Asset>& a){a->behave();}}
1695 };
1696 assert(phaseFunc.count(phase)>0);
1697 //Asset::kill()の待機リストを初期化
1698 killRequests.clear();
1699 //phaseFunc実行中のエラー監視用フラグ。子スレッドからの例外送出を拾わないため必要。
1700 std::atomic_bool success=true;
1701
1702 updateProcessQueueIfNeeded(phase);
1703 //この時刻のこのphaseで処理すべきAssetの範囲を取得する
1704 auto itr = asset_process_queue[phase].find(tickCount);
1705 if(itr==asset_process_queue[phase].end()){
1706 //この時刻の処理対象なし。
1707 calcNextTick(phase);
1708 return;
1709 }
1710 auto toBeProcessed = std::move(itr->second);
1711 asset_process_queue[phase].erase(itr);
1712 if(numThreads==1){
1713 //シングルスレッドの場合、先頭から順番に処理すればよい
1714 std::chrono::system_clock::time_point before,after;
1715 std::chrono::microseconds difference;
1716 std::string phaseName=enumToStr(phase);
1717 for(auto&& [order,asset] : toBeProcessed){
1718 std::string name=asset->getFullName();
1719 if(asset->isAlive()){
1720 try{
1721 if(measureTime){
1722 before=std::chrono::system_clock::now();
1723 }
1724 phaseFunc.at(phase)(asset);
1725 if(measureTime){
1726 after=std::chrono::system_clock::now();
1727 difference=std::chrono::duration_cast<std::chrono::microseconds>(after-before);
1728 maxProcessTime[phaseName][name]=std::max<std::uint64_t>(maxProcessTime[phaseName][name],difference.count());
1729 minProcessTime[phaseName][name]=std::min<std::uint64_t>(minProcessTime[phaseName][name],difference.count());
1730 processCount[phaseName][name]++;
1731 meanProcessTime[phaseName][name]+=((double)(difference.count())-meanProcessTime[phaseName][name])/processCount[phaseName][name];
1732 }
1733 }catch(std::exception& ex){
1734 std::cout<<"asset."<<phaseName<<"() failed. asset="<<name<<std::endl;
1735 std::cout<<ex.what()<<std::endl;
1736 std::cout<<"w="<<worker_index()<<","<<"v="<<vector_index()<<std::endl;
1737 success=false;
1738 throw ex;
1739 }
1740 }
1741 }
1742 }else{
1743 //並列化する場合、処理できるようになったものから処理を実施
1744 py::gil_scoped_release release;
1745 std::set<std::size_t> waitings;
1746 for(auto&& index : toBeProcessed | std::views::keys){
1747 waitings.insert(index);
1748 }
1749
1750 //処理状況監視用の変数の初期化
1751 std::map<std::size_t,std::atomic_bool> futures;//各Assetの処理状況を格納する。終了時にtrueとなる。
1752 std::map<std::size_t,std::vector<std::size_t>> checklists;//各Assetが依存するAssetのindexリスト。trueになった後に何度も確認してしまわないように要素を削除しながら回す。
1753 for(auto&& index : waitings){
1754 futures[index]=false;
1755 checklists[index];
1756 for(auto&& predecessor : asset_dependency_graph[phase].get_predecessors_by_index(index)){
1757 if(toBeProcessed.contains(predecessor)){
1758 checklists[index].push_back(predecessor);
1759 }
1760 }
1761 }
1762
1763 //処理を順番に実行する
1764 auto wit=waitings.begin();
1765 while(wit!=waitings.end()){
1766 std::size_t index=*wit;
1767 bool ready=true;
1768 auto dit=checklists[index].begin();
1769 while(dit!=checklists[index].end()){
1770 if(futures.find(*dit)==futures.end() || futures[*dit]){
1771 dit=checklists[index].erase(dit);
1772 }else{
1773 ready=false;
1774 dit=checklists[index].end();
1775 }
1776 }
1777 if(ready){
1778 auto asset=toBeProcessed.at(index);
1779 std::string name=asset->getFullName();
1780 if(asset->isAlive()){
1781 int w=worker_index();
1782 int v=vector_index();
1783 pool.detach_task([&futures,&success,phase,index,name,asset,w,v,this]{
1784 std::chrono::system_clock::time_point before,after;
1785 std::chrono::microseconds difference;
1786 try{
1787 if(measureTime){
1788 before=std::chrono::system_clock::now();
1789 }
1790 phaseFunc.at(phase)(asset);
1791 if(measureTime){
1792 std::string phaseName=enumToStr(phase);
1793 after=std::chrono::system_clock::now();
1794 difference=std::chrono::duration_cast<std::chrono::microseconds>(after-before);
1795 maxProcessTime.at(phaseName).at(name)=std::max<std::uint64_t>(maxProcessTime.at(phaseName).at(name),difference.count());
1796 minProcessTime.at(phaseName).at(name)=std::min<std::uint64_t>(minProcessTime.at(phaseName).at(name),difference.count());
1797 processCount.at(phaseName).at(name)++;
1798 meanProcessTime.at(phaseName).at(name)+=((double)(difference.count())-meanProcessTime.at(phaseName).at(name))/processCount.at(phaseName).at(name);
1799 }
1800 }catch(std::exception& ex){
1801 std::cout<<"asset."<<enumToStr(phase)<<"() failed. asset="<<name<<std::endl;
1802 std::cout<<ex.what()<<std::endl;
1803 std::cout<<"w="<<w<<","<<"v="<<v<<std::endl;
1804 success=false;
1805 throw ex;
1806 }
1807 futures[index]=true;
1808 });
1809 }else{
1810 //not alive
1811 futures[index]=true;
1812 }
1813 wit=waitings.erase(wit);
1814 }else{
1815 ++wit;
1816 }
1817 if(wit==waitings.end()){
1818 wit=waitings.begin();
1819 }
1820 if(!success){
1821 throw std::runtime_error("SimulationManager::runAssetPhaseFunc("+enumToStr(phase)+") failed.");
1822 }
1823 }
1824 pool.wait();
1825 }
1826 lastAssetPhase=phase;
1827 killAndRemoveAssets();
1828 setupNewEntities();
1829 //次回処理時刻の算出
1830 for(auto&& [index,asset] : toBeProcessed){
1831 calcNextTick(phase,asset);
1832 }
1833 calcNextTick(phase);
1834}
1835void SimulationManager::runCallbackPhaseFunc(const SimPhase& phase,bool group1){
1836 static const std::map<SimPhase,std::function<void(const std::shared_ptr<Callback>&)>> phaseFunc={
1837 {SimPhase::ON_GET_OBSERVATION_SPACE,[](const std::shared_ptr<Callback>& c){c->onGetObservationSpace();}},

1839 {SimPhase::ON_MAKE_OBS,[](const std::shared_ptr<Callback>& c){c->onMakeObs();}},
1840 {SimPhase::ON_DEPLOY_ACTION,[](const std::shared_ptr<Callback>& c){c->onDeployAction();}},
1841 {SimPhase::ON_EPISODE_BEGIN,[](const std::shared_ptr<Callback>& c){c->onEpisodeBegin();}},
1842 {SimPhase::ON_VALIDATION_END,[](const std::shared_ptr<Callback>& c){c->onValidationEnd();}},
1843 {SimPhase::ON_STEP_BEGIN,[](const std::shared_ptr<Callback>& c){c->onStepBegin();}},
1844 {SimPhase::ON_INNERSTEP_BEGIN,[](const std::shared_ptr<Callback>& c){c->onInnerStepBegin();}},
1845 {SimPhase::ON_INNERSTEP_END,[](const std::shared_ptr<Callback>& c){c->onInnerStepEnd();}},
1846 {SimPhase::ON_STEP_END,[](const std::shared_ptr<Callback>& c){c->onStepEnd();}},
1847 {SimPhase::ON_EPISODE_END,[](const std::shared_ptr<Callback>& c){c->onEpisodeEnd();}}
1848 };
1849 assert(phaseFunc.count(phase)>0);
1850 updateProcessQueueIfNeeded(phase);
1851 auto& group_queue = group1 ? group1_callback_process_queue.at(phase) : group2_callback_process_queue.at(phase);
1852 std::chrono::system_clock::time_point before,after;
1853 std::chrono::microseconds difference;
1854 std::string phaseName=enumToStr(phase);
1855 killRequests.clear();
1856 if(phase==SimPhase::ON_GET_OBSERVATION_SPACE ||
1857 phase==SimPhase::ON_GET_ACTION_SPACE ||
1858 phase==SimPhase::ON_MAKE_OBS ||
1859 phase==SimPhase::ON_DEPLOY_ACTION ||
1860 phase==SimPhase::ON_EPISODE_BEGIN ||
1861 phase==SimPhase::ON_EPISODE_END ||
1862 phase==SimPhase::ON_VALIDATION_END ||
1863 phase==SimPhase::ON_STEP_BEGIN ||
1864 phase==SimPhase::ON_STEP_END
1865 ){
1866 // 周期処理ではないもの
1867 for(auto&& callback : callback_dependency_graph[phase].get_range()){
1868 if(group1 != (isinstance<Ruler>(callback) || isinstance<Reward>(callback))){
1869 continue;
1870 }
1871 std::string name=callback->getFullName();
1872 try{
1873 if(measureTime){
1874 before=std::chrono::system_clock::now();
1875 }
1876 phaseFunc.at(phase)(callback);
1877 if(measureTime){
1878 after=std::chrono::system_clock::now();
1879 difference=std::chrono::duration_cast<std::chrono::microseconds>(after-before);
1880 maxProcessTime[phaseName][name]=std::max<std::uint64_t>(maxProcessTime[phaseName][name],difference.count());
1881 minProcessTime[phaseName][name]=std::min<std::uint64_t>(minProcessTime[phaseName][name],difference.count());
1882 processCount[phaseName][name]++;
1883 meanProcessTime[phaseName][name]+=((double)(difference.count())-meanProcessTime[phaseName][name])/processCount[phaseName][name];
1884 }
1885 if(phase==SimPhase::ON_STEP_END && callback==ruler){
1886 //RulerのonStepEndの後、Rewardの計算前にdonesを処理してagentsToActを上書き
1887 gatherDones();
