27// try_load
28// Archive::containsで読み込み前にkeyの存否判定ができる場合は、存在する場合のみ値を読み込み、存否判定結果を返す。
29// 存否判定ができないArchiveの場合は通常どおり値を読み込むもうとし、Archiveクラス依存の例外処理に委ねる。
30template<traits::input_archive Archive, class KeyType, class ValueType>
31bool try_load(Archive& ar, KeyType&& key, ValueType& value){
32 if constexpr (traits::archive_has_contains<Archive, KeyType>) {
33 if(!ar.contains(std::forward<KeyType>(key))){
34 return false;
35 }
36 }
37 ar(cereal::make_nvp(std::forward<KeyType>(key),value));
38 return true;
39}
