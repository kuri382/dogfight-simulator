68Eigen::Vector3d getOrthogonalVector(const Eigen::Vector3d &v){
69 double n=v.norm();
70 if(n==0){
71 return Eigen::Vector3d(1,0,0);
72 }
73 int idx;
74 v.maxCoeff(&idx);
75 int tmp=(idx+1)%3;

89Eigen::MatrixXd ArimotoPotter(const Eigen::MatrixXd& A, const Eigen::MatrixXd& B, const Eigen::MatrixXd& Q, const Eigen::MatrixXd& R){
90 Eigen::Index dim_x=A.rows();
91 Eigen::Index dim_u=B.cols();
92 Eigen::MatrixXd H=Eigen::MatrixXd::Zero(2*dim_x,2*dim_x);
93 H<<A,-B*R.inverse()*B.transpose(),-Q,-A.transpose();
94 Eigen::EigenSolver<Eigen::MatrixXd> eig(H);
95 Eigen::MatrixXcd eigvec=Eigen::MatrixXcd::Zero(2*dim_x,dim_x);
96 int j=0;
97 for(int i=0;i<2*dim_x;++i){
98 if(eig.eigenvalues()[i].real()<0.){
99 eigvec.col(j)=eig.eigenvectors().block(0,i,2*dim_x,1);
100 ++j;
101 }
102 }
103 Eigen::MatrixXcd Vs_1,Vs_2;
104 Vs_1=eigvec.block(0,0,dim_x,dim_x);
105 Vs_2=eigvec.block(dim_x,0,dim_x,dim_x);
106 return (Vs_2*Vs_1.inverse()).real();
107}

109Eigen::VectorXd RK4(const Eigen::VectorXd& x0,const Time& t0,double dt,std::function<Eigen::VectorXd (const Eigen::VectorXd& x,const Time& t)> deriv){
110 Eigen::VectorXd k1=deriv(x0,t0);
111 Eigen::VectorXd k2=deriv(x0+dt*0.5*k1, t0+dt*0.5);
112 Eigen::VectorXd k3=deriv(x0+dt*0.5*k2, t0+dt*0.5);
113 Eigen::VectorXd k4=deriv(x0+dt*k3, t0+dt);
114 return x0+dt/6.0*(k1+2.0*k2+2.0*k3+k4);
115}
