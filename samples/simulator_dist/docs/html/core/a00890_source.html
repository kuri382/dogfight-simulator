84//中身を指定してNameValuePair<U>を判定する。CV修飾と参照修飾を無視する。
85template<typename T, typename U>
86struct is_cereal_nvp_of_impl : std::false_type {};
87template<typename T, typename U>
88struct is_cereal_nvp_of_impl<cereal::NameValuePair<T>,U> : std::is_same<U,T> {};
89template<typename T, typename U>
90struct is_cereal_nvp_of_impl<cereal::NameValuePair<T&>,U> : std::is_same<U,T> {};
91template<typename T, typename U>
92struct is_cereal_nvp_of_impl<cereal::NameValuePair<T&&>,U> : std::is_same<U,T> {};
93template<typename T, typename U>
94struct is_cereal_nvp_of {
95 private:
96 using TT=std::remove_cv_t<std::remove_reference_t<T>>;
97 using UU=std::remove_reference_t<U>;
98 public:
99 static constexpr bool value=is_cereal_nvp_of_impl<TT,UU>::value;
100};
101template<typename T, typename U>
102concept cereal_nvp_of = cereal_nvp<T> && is_cereal_nvp_of<T,U>::value;
