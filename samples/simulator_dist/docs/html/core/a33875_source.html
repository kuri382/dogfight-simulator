229void Track3D::merge(){
230 // このクラスでは算術平均でマージするが、必要に応じて派生クラスでオーバーライドすること。
231 if(buffer.size()>0){
232 Eigen::Vector3d meanPos(0,0,0);
233 Eigen::Vector3d meanVel(0,0,0);
234 for(auto& e:buffer){
235 Track3D extrapolated=e.transformTo(crs);
236 extrapolated.updateByExtrapolation(time-e.time);
237 meanPos+=extrapolated.pos();
238 meanVel+=extrapolated.vel();
239 }
240 meanPos/=(buffer.size()+1.0);
241 meanVel/=(buffer.size()+1.0);
242 _pos.setValue(meanPos);
243 _vel.setLocation(meanPos);
244 _vel.setValue(meanVel);
245 clearBuffer();
246 }

314Track2D::Track2D(const std::weak_ptr<PhysicalAsset>& truth_,const std::shared_ptr<CoordinateReferenceSystem>& crs_,const Eigen::Vector3d &origin_)
315:TrackBase(truth_,crs_),
316 _origin(origin_,crs,time,CoordinateType::POSITION_ABS),
317 _dir(Eigen::Vector3d(1,0,0),origin_,crs,time,CoordinateType::DIRECTION),
318 _omega(Eigen::Vector3d::Zero(),origin_,crs,time,CoordinateType::ANGULAR_VELOCITY),
319 origin(_origin),dir(_dir),omega(_omega){
320 if(truth!=boost::uuids::nil_uuid()){
321 if(crs){
322 _origin=Coordinate(origin_,crs,time,CoordinateType::POSITION_ABS);
323 Eigen::Vector3d dr=truth_.lock()->pos(crs)-origin_;
324 double R=dr.norm();
325 _dir=Coordinate(dr/R,origin_,crs,time,CoordinateType::DIRECTION);
326 _omega=Coordinate(dir().cross(truth_.lock()->vel(crs))/R,origin_,crs,time,CoordinateType::ANGULAR_VELOCITY);
327 }else{
328 _origin=Coordinate(origin_,crs,time,CoordinateType::POSITION_ABS);
329 Eigen::Vector3d dr=truth_.lock()->pos()-origin_;
330 double R=dr.norm();
331 _dir=Coordinate(dr/R,origin_,crs,time,CoordinateType::DIRECTION);
332 _omega=Coordinate(dir().cross(truth_.lock()->vel())/R,origin_,crs,time,CoordinateType::ANGULAR_VELOCITY);
333 }
334 }

445void Track2D::merge(){
446 // このクラスでは算術平均でマージするが、必要に応じて派生クラスでオーバーライドすること。
447 // 例えば、観測点が複数ヶ所にわたる場合には明らかに不適切である。
448 if(buffer.size()>0){
449 Eigen::Vector3d meanOrigin(0,0,0);
450 Eigen::Vector3d meanDir(0,0,0);
451 Eigen::Vector3d meanOmega(0,0,0);
452 for(auto& e:buffer){
453 Track2D extrapolated=e.transformTo(crs);
454 extrapolated.updateByExtrapolation(time-e.time);
455 meanDir+=extrapolated.dir();
456 meanOrigin+=extrapolated.origin();
457 meanOmega+=extrapolated.omega();
458 }
459 meanDir=(meanDir/(buffer.size()+1.0)).normalized();
460 meanOrigin/=(buffer.size()+1.0);
461 meanOmega/=(buffer.size()+1.0);
462 _origin.setValue(meanOrigin);
463 _dir.setLocation(meanOrigin);
464 _omega.setLocation(meanOrigin);
465 _dir.setValue(meanDir);
466 _omega.setValue(meanOmega);
467 clearBuffer();
468 }

479void Track2D::updateByExtrapolation(const double& dt){
480 // このクラスではcrs上の見かけの位置、角速度で外挿するが、必要に応じて派生クラスでオーバーライドすること。
481 // あるいは、慣性座標系等の外挿しても問題ないcrsに変換してから外挿すること
482 double w=omega.norm();
483 if(w>0){
484 Quaternion q=Quaternion::fromAngle(omega()/w,w*dt);
485 _dir.setValue(q.transformVector(dir()));
486 }
487}
488Track2D Track2D::extrapolate(const double& dt){
489 // このクラスではcrs上の見かけの位置、角速度で外挿するが、必要に応じて派生クラスでオーバーライドすること。
490 // あるいは、慣性座標系等の外挿しても問題ないcrsに変換してから外挿すること
491 double w=omega.norm();
492 if(w>0){
493 Quaternion q=Quaternion::fromAngle(omega()/w,w*dt);
494 return Track2D(truth,crs,time+dt,q.transformVector(dir()),origin(),omega());
495 }else{
496 return Track2D(truth,crs,time+dt,dir(),origin(),omega());
497 }
