17void Agent::initialize(){
18 BaseType::initialize();
19 name=instanceConfig.at("name");
20 type=instanceConfig.at("type");
21 model=instanceConfig.at("model");
22 policy=instanceConfig.at("policy");
23 instanceConfig.at("parents").get_to(parents);
24 commands=nl::json::object();
25 observables=nl::json::object();
26 for(auto &&e:parents){
27 commands[e.second->getFullName()]=nl::json::object();
28 observables[e.second->getFullName()]={{"decision",nl::json::object()}};
29 }
30 if(modelConfig.contains("interval")){
31 nl::json intervals=modelConfig.at("interval");
32 std::string intervalUnit="tick";
33 if(intervals.contains("unit")){
34 intervalUnit=intervals.at("unit");
35 }
36 if(intervalUnit=="time"){
37 double dt=manager->getBaseTimeStep();
38 interval[SimPhase::AGENT_STEP]=std::max<std::uint64_t>(1,std::lround(getValueFromJsonKRD(intervals,"step",randomGen,manager->getDefaultAgentStepInterval()*dt)/dt));
39 }else if(intervalUnit=="tick"){
40 interval[SimPhase::AGENT_STEP]=getValueFromJsonKRD(intervals,"step",randomGen,manager->getDefaultAgentStepInterval());
41 }else{
42 throw std::runtime_error("Only \"tick\" or \"time\" can be designated as the unit of interval."+intervalUnit);
43 }
44 }else{
45 interval[SimPhase::AGENT_STEP]=manager->getDefaultAgentStepInterval();
46 }
47 firstTick[SimPhase::AGENT_STEP]=0;
48 dontExpose=getValueFromJsonKD(modelConfig,"dontExpose",false);
49 if(dontExpose){
50 firstTick[SimPhase::AGENT_STEP]=std::numeric_limits<std::uint64_t>::max();
51 interval[SimPhase::AGENT_STEP]=std::numeric_limits<std::uint64_t>::max();
52 }
53}

321void ExpertWrapper::control(){
322 std::uint64_t now=manager->getTickCount();
323 if(expert->isTickToRunPhaseFunc(SimPhase::CONTROL,now)){
324 expert->control();
325 }
326 observables=expert->observables;
327 nl::json decisions=nl::json::object();
328 for(auto&&e:expert->observables.items()){
329 decisions[e.key()]=e.value().at("decision");
330 }
331 if(imitator->isTickToRunPhaseFunc(SimPhase::CONTROL,now)){
332 if(!hasImitatorDeployed){
333 py::gil_scoped_acquire acquire;
334 imitatorAction=imitator->convertActionFromAnother(decisions,expert->commands);
335 imitator->deploy(imitatorAction);
336 hasImitatorDeployed=true;
337 }
338 imitator->controlWithAnotherAgent(decisions,expert->commands);
339 }
340 if(whichOutput=="Expert"){
341 commands=expert->commands;
342 }else{
343 commands=imitator->commands;
344 }

373void MultiPortCombiner::makeChildren(){
374 BaseType::makeChildren();
375 std::map<std::string,nl::json> childrenConfigs=modelConfig.at("children");
376 std::map<std::string,nl::json> buffer;
377 std::string childName,childModel,childPort;
378 for(auto &&e:childrenConfigs){
379 std::string key=e.first;
380 nl::json value=e.second;
381 try{
382 childName=value.at("name").get<std::string>();
383 }catch(...){
384 childName="child_"+key;
385 }
386 childModel=value.at("model").get<std::string>();
387 try{
388 childPort=value.at("port").get<std::string>();
389 }catch(...){
390 childPort="0";
391 }
392 if(parents.count(key)>0){
393 //instanciate only when parents for the child actually exist.
394 if(buffer.count(childName)>0){
395 buffer[childName]["ports"][childPort]=key;
396 buffer[childName]["parents"][childPort]=parents[key];
397 }else{
398 buffer[childName]={
399 {"ports",{{childPort,key}}},
400 {"model",childModel},
401 {"parents",{{childPort,parents[key]}}}
402 };
403 }
404 ports[key]={
405 {"childName",childName},
406 {"childPort",childPort}
407 };
408 }
409 }
410 nl::json sub;
411 for(auto &&e:buffer){
412 std::string key=e.first;
413 nl::json value=e.second;
414 sub={
415 {"name",name+"/"+key},
416 {"seed",randomGen()},
417 {"parents",buffer[key]["parents"]},
418 {"type","Child"},
419 {"model",buffer[key]["model"]},
420 {"policy",policy},
421 {"entityFullName",name+"/"+key+":"+model+":"+policy}
422 };
423 children[key]=createUnmanagedEntity<Agent>(
424 isEpisodic(),
425 "Agent",
426 buffer[key]["model"],
427 sub
428 );
429 }
430 std::uint64_t stepInterval=children.begin()->second->getStepInterval();
431 for(auto &&e:children){
432 assert(e.second->getStepInterval()==stepInterval);
433 }
