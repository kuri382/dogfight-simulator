61void Ruler::onEpisodeBegin(){
62 teams.clear();
63 score.clear();
64 stepScore.clear();
65 nl::json j_teams;
66 try{
67 j_teams=modelConfig.at("teams");
68 }catch(...){
69 j_teams="All";
70 }
71 if(j_teams.is_null()){
72 j_teams="All";
73 }
74 if(j_teams.is_string()){
75 std::string team=j_teams;
76 if(team=="All"){
77 for(auto &&t:manager->getTeams()){
78 teams.push_back(t);
79 score[t]=0.0;
80 stepScore[t]=0.0;
81 }
82 }else{
83 teams.push_back(team);
84 score[team]=0.0;
85 stepScore[team]=0.0;
86 }
87 }else if(j_teams.is_array()){
88 teams=j_teams.get<std::vector<std::string>>();
89 for(auto &&t:teams){
90 score[t]=0.0;
91 stepScore[t]=0.0;
92 }
93 }else{
94 std::cout<<"modelConfig['teams']="<<j_teams<<std::endl;
95 throw std::runtime_error("Invalid designation of 'teams' for Ruler.");
96 }
97 dones.clear();
98 for(auto &&e:manager->getAgents()){
99 auto a=getShared(e);
100 dones[a->getName()]=!a->isAlive();
101 }
102 dones["__all__"]=false;
103 endReason=EndReason::NOTYET;
104 observables["endReason"]=endReason;
105}

123void Ruler::checkDone(){
124 dones.clear();
125 for(auto &&e:manager->getAgents()){
126 auto a=getShared(e);
127 dones[a->getName()]=!a->isAlive();
128 }
129 if(manager->getElapsedTime()>=maxTime){
130 int index=0;
131 double maxScore=0;
132 for(auto &&t:teams){
133 if(index==0 || maxScore<score[t]){
134 maxScore=score[t];
135 winner=t;
136 }
137 index++;
138 }
139 for(auto &e:dones){
140 e.second=true;
141 }
142 dones["__all__"]=true;
143 endReason=EndReason::TIMEUP;
144 return;
145 }
146 if(dones.size()==0){
147 dones["__all__"]=false;
148 return;
149 }
150 dones["__all__"]=true;
151 for(auto &&e:dones){
152 if(!e.second){
153 dones["__all__"]=false;
154 }
155 }
156 if(dones["__all__"]){
157 endReason=EndReason::NO_ONE_ALIVE;
158 }
159 observables["endReason"]=endReason;
160}
