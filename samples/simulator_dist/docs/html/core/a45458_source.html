26 auto fullName=getFullName();

34 if(instanceConfig.contains("parent")){

128void PhysicalAsset::setInitialMotionStateFromJson(const nl::json& j){
129 if(j.contains("motion")){
130 // MotionStateを直接指定する場合
131 motion.load_from_json(j.at("motion"));
132
133 //Body座標軸の修正
134 if(j.contains("bodyAxisOrder")){
135 bodyAxisOrder=j.at("bodyAxisOrder");
136 }else if(modelConfig.contains("bodyAxisOrder")){
137 bodyAxisOrder=modelConfig.at("bodyAxisOrder");
138 }else{
139 bodyAxisOrder=motion.axisOrder;
140 }
141 motion.setAxisOrder(bodyAxisOrder);
142
143 //親座標系の修正
144 std::shared_ptr<CoordinateReferenceSystem> parentCRS;
145 if(isBoundToParent){
146 //親に固定されている場合はそのBody座標系でMotionStateを管理する。
147 parentCRS=parent.lock()->getBodyCRS();
148 }else{
149 //親に固定されていない場合はSimulationManagerの基準座標系でMotionStateを管理する。
150 parentCRS=manager->getRootCRS();
151 }
152 motion.setCRS(parentCRS,true);
153 }else{
154 // 各状態量を個別に指定する場合
155 if(j.contains("bodyAxisOrder")){
156 bodyAxisOrder=j.at("bodyAxisOrder");
157 }else if(modelConfig.contains("bodyAxisOrder")){
158 bodyAxisOrder=modelConfig.at("bodyAxisOrder");
159 }else{
160 bodyAxisOrder="FSD";
161 }
162 motion.setTime(manager->getTime());
163 motion.setAxisOrder(bodyAxisOrder);
164 std::shared_ptr<CoordinateReferenceSystem> parentCRS;
165 if(isBoundToParent){
166 //親に固定されている場合はそのBody座標系でMotionStateを管理する。
167 parentCRS=parent.lock()->getBodyCRS();
168 }else{
169 //親に固定されていない場合はSimulationManagerの基準座標系でMotionStateを管理する。
170 parentCRS=manager->getRootCRS();
171 }
172 if(j.contains("crs")){
173 auto initialStateCRS=createOrGetEntity<CoordinateReferenceSystem>(j.at("crs"));
174 motion.setCRS(initialStateCRS,false);
175 }else{
176 //初期条件の座標系を指定しなかった場合は、親があれば親の座標系、無ければRulerのlocal座標系とみなす
177 if(parent.expired()){
178 motion.setCRS(manager->getRuler().lock()->getLocalCRS(),false);
179 }else{
180 motion.setCRS(parentCRS,false);
181 }
182 }
183 // ----座標計算ここから----
184 if(j.contains("pos")){
185 nl::json pos_j=getValueFromJsonKR(j,"pos",randomGen);
186 Eigen::Vector3d pos_v;
187 if(pos_j.is_array() && pos_j.size()==3){
188 pos_v=pos_j;
189 }else{
190 throw std::runtime_error("PhysicalAsset::setInitialMotionStateFromJson() failed. 'pos' should be 3-D vector.");
191 }
192 motion.setPos(pos_v);
193 }
194
195 double azInDeg=0.0;
196 if(j.contains("heading")){
197 azInDeg=getValueFromJsonKR(j,"heading",randomGen);
198 }else if(j.contains("azimuth")){
199 azInDeg=getValueFromJsonKR(j,"azimuth",randomGen);
200 }else if(j.contains("az")){
201 azInDeg=getValueFromJsonKR(j,"az",randomGen);
202 }
203 while(azInDeg>180){azInDeg-=360;}
204 while(azInDeg<=-180){azInDeg+=360;}
205 double az=deg2rad(azInDeg);
206 double cs=cos(az);
207 double sn=sin(az);
208 double cs_2=cos(az/2);
209 double sn_2=sin(az/2);
210 if(azInDeg==0){
211 cs=1;sn=0;
212 cs_2=1;sn_2=0;
213 }else if(azInDeg==90){
214 cs=0;sn=1;
215 cs_2=sqrt(2)/2;
216 sn_2=sqrt(2)/2;
217 }else if(azInDeg==180){
218 cs=-1;sn=0;
219 cs_2=0;sn_2=1;
220 }else if(azInDeg==-90){
221 cs=0;sn=-1;
222 cs_2=sqrt(2)/2;
223 sn_2=-sqrt(2)/2;
224 }
225 double elInDeg=0.0;
226 if(j.contains("pitch")){
227 elInDeg=getValueFromJsonKR(j,"pitch",randomGen);
228 }else if(j.contains("elevation")){
229 elInDeg=getValueFromJsonKR(j,"elevation",randomGen);
230 }else if(j.contains("el")){
231 elInDeg=getValueFromJsonKR(j,"el",randomGen);
232 }
233 while(elInDeg>180){elInDeg-=360;}
234 while(elInDeg<=-180){elInDeg+=360;}
235 double el=deg2rad(elInDeg);
236 double rollInDeg=0.0;
237 if(j.contains("roll")){
238 rollInDeg=getValueFromJsonKR(j,"roll",randomGen);
239 }else if(j.contains("bank")){
240 rollInDeg=getValueFromJsonKR(j,"bank",randomGen);
241 }
242 while(rollInDeg>180){rollInDeg-=360;}
243 while(rollInDeg<=-180){rollInDeg+=360;}
244 double roll=deg2rad(rollInDeg);
245
246 Quaternion q1=Quaternion::fromAngle(Eigen::Vector3d(0,0,1),az);
247 Quaternion q2=Quaternion::fromAngle(Eigen::Vector3d(0,1,0),el);
248 Quaternion q3=Quaternion::fromAngle(Eigen::Vector3d(1,0,0),roll);
249 Quaternion qToNEDFromFSD = q1*q2*q3;
250 Quaternion qToNEDFromBody = qToNEDFromFSD*getAxisSwapQuaternion("FSD",bodyAxisOrder); // [NED <- Body]
251 Quaternion qToParentFromNED = motion.getCRS()->getEquivalentQuaternionToTopocentricCRS(pos,"NED").conjugate(); // [Parent <- NED]
252 Quaternion q=qToParentFromNED*qToNEDFromBody; // [Parent <- Body]
253 motion.setQ(q);
254
255 Eigen::Vector3d vel_v;
256 if(j.contains("vel")){
257 nl::json vel_j=getValueFromJsonKR(j,"vel",randomGen);
258 if(vel_j.is_array() && vel_j.size()==3){
259 vel_v=vel_j;
260 }else if(vel_j.is_number()){
261 double V=vel_j;
262 double alphaInDeg=getValueFromJsonKRD(j,"alpha",randomGen,0.0);
263 while(alphaInDeg>180){alphaInDeg-=360;}
264 while(alphaInDeg<=-180){alphaInDeg+=360;}
265 double alpha=deg2rad(alphaInDeg);
266 double betaInDeg=getValueFromJsonKRD(j,"beta",randomGen,0.0);
267 while(betaInDeg>180){betaInDeg-=360;}
268 while(betaInDeg<=-180){betaInDeg+=360;}
269 double beta=deg2rad(betaInDeg);
270 Eigen::Vector3d velInFSD=V*Eigen::Vector3d(cos(alpha)*cos(beta),sin(beta),sin(alpha)*cos(beta));
271 Eigen::Vector3d velInNED=qToNEDFromFSD.transformVector(velInFSD);
272 vel_v=motion.velHtoP(
273 velInNED,
274 Eigen::Vector3d::Zero(),
275 "NED",
276 false
277 );
278 }else{
279 throw std::runtime_error("PhysicalAsset::setInitialMotionStateFromJson() failed. 'vel' should be 3-D vector or scalar value.");
280 }
281 }else{
282 vel_v=Eigen::Vector3d::Zero();
283 }
284 motion.setVel(vel_v);
285
286 Eigen::Vector3d omega_v;
287 if(j.contains("omegaB")){
288 omega_v=getValueFromJsonKR(j,"omegaB",randomGen);
289 omega_v=motion.omegaHtoP(
290 qToNEDFromBody.transformVector(omega_v),
291 Eigen::Vector3d::Zero(),
292 "NED",
293 false
294 );
295 }else if(j.contains("omegaP")){
296 omega_v=getValueFromJsonKR(j,"omegaP",randomGen);
297 }else{
298 omega_v=Eigen::Vector3d::Zero();
299 }
300 motion.setOmega(omega_v);
301 // ----座標計算ここまで----
302 motion.setCRS(parentCRS,true);
303 }
304 motion_prev=motion;
305 motion_prev.setVel(motion.vel());
306 motion_prev.setPos(motion.pos()-motion.vel()*(interval[SimPhase::BEHAVE]*manager->getBaseTimeStep()));
307}
