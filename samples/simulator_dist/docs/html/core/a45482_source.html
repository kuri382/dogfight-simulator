12void Callback::initialize(){
13 BaseType::initialize();
14 auto fullName=getFullName();
15 if(fullName.find("/")>=0){
16 group=fullName.substr(0,fullName.find("/"));
17 name=fullName.substr(fullName.find("/")+1);
18 }else{
19 group="";
20 name=fullName;
21 }
22 try{
23 acceptReconfigure=modelConfig.at("acceptReconfigure");
24 }catch(...){
25 acceptReconfigure=false;
26 }
27 firstTick[SimPhase::ON_EPISODE_BEGIN]=0;//meaningless but defined for same interface
28 firstTick[SimPhase::ON_VALIDATION_END]=0;//meaningless but defined for same interface
29 firstTick[SimPhase::ON_STEP_BEGIN]=0;//meaningless but defined for same interface
30 nl::json firstTicks=nl::json::object();
31 if(instanceConfig.contains("firstTick")){
32 firstTicks=instanceConfig.at("firstTick");
33 }else if(modelConfig.contains("firstTick")){
34 firstTicks=modelConfig.at("firstTick");
35 }
36 std::string firstTickUnit="tick";
37 if(firstTicks.contains("unit")){
38 firstTickUnit=firstTicks.at("unit");
39 }
40 if(firstTickUnit=="time"){
41 double dt=manager->getBaseTimeStep();
42 firstTick[SimPhase::ON_INNERSTEP_BEGIN]=std::lround(getValueFromJsonKRD(firstTicks,"value",randomGen,0.0)/dt);
43 }else if(firstTickUnit=="tick"){
44 firstTick[SimPhase::ON_INNERSTEP_BEGIN]=getValueFromJsonKRD(firstTicks,"value",randomGen,0);
45 }else{
46 throw std::runtime_error("Only \"tick\" or \"time\" can be designated as the unit of firstTick."+firstTickUnit);
47 }
48 firstTick[SimPhase::ON_STEP_END]=0;//meaningless but defined for same interface
49 firstTick[SimPhase::ON_EPISODE_END]=0;//meaningless but defined for same interface
50 firstTick[SimPhase::ON_GET_OBSERVATION_SPACE]=0;//meaningless but defined for same interface
51 firstTick[SimPhase::ON_GET_ACTION_SPACE]=0;//meaningless but defined for same interface
52 firstTick[SimPhase::ON_MAKE_OBS]=0;//meaningless but defined for same interface
53 firstTick[SimPhase::ON_DEPLOY_ACTION]=0;//meaningless but defined for same interface
54 interval[SimPhase::ON_EPISODE_BEGIN]=std::numeric_limits<std::uint64_t>::max();//meaningless but defined for same interface
55 interval[SimPhase::ON_VALIDATION_END]=std::numeric_limits<std::uint64_t>::max();//meaningless but defined for same interface
56 interval[SimPhase::ON_STEP_BEGIN]=0;//meaningless but defined for same interface
57 nl::json intervals=nl::json::object();
58 if(instanceConfig.contains("interval")){
59 intervals=instanceConfig.at("interval");
60 }else if(modelConfig.contains("interval")){
61 intervals=modelConfig.at("interval");
62 }
63 std::string intervalUnit="tick";
64 if(intervals.contains("unit")){
65 intervalUnit=intervals.at("unit");
66 }
67 if(intervalUnit=="time"){
68 double dt=manager->getBaseTimeStep();
69 interval[SimPhase::ON_INNERSTEP_BEGIN]=std::max<std::uint64_t>(1,std::lround(getValueFromJsonKRD(intervals,"value",randomGen,dt)/dt));
70 }else if(intervalUnit=="tick"){
71 interval[SimPhase::ON_INNERSTEP_BEGIN]=getValueFromJsonKRD(intervals,"value",randomGen,1);
72 }else{
73 throw std::runtime_error("Only \"tick\" or \"time\" can be designated as the unit of interval."+intervalUnit);
74 }
75 firstTick[SimPhase::ON_INNERSTEP_END]=firstTick[SimPhase::ON_INNERSTEP_BEGIN]+interval[SimPhase::ON_INNERSTEP_BEGIN];
76 interval[SimPhase::ON_INNERSTEP_END]=interval[SimPhase::ON_INNERSTEP_BEGIN];
77 interval[SimPhase::ON_STEP_END]=0;//meaningless but defined for same interface
78 interval[SimPhase::ON_EPISODE_END]=std::numeric_limits<std::uint64_t>::max();//meaningless but defined for same interface
79 interval[SimPhase::ON_GET_OBSERVATION_SPACE]=0;//meaningless but defined for same interface
80 interval[SimPhase::ON_GET_ACTION_SPACE]=0;//meaningless but defined for same interface
81 interval[SimPhase::ON_MAKE_OBS]=0;//meaningless but defined for same interface
82 interval[SimPhase::ON_DEPLOY_ACTION]=0;//meaningless but defined for same interface
83}
