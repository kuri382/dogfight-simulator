146 def run(self):
147 if(not self.isValid):
148 self.validate()
149 self.episodeCounter=0
150 for path in self.dataPaths:
151 self.episodeCounter += 1
152 self.currentFileName=os.path.splitext(os.path.basename(path))[0]
153 print("current: ",self.currentFileName)
154 with bz2.BZ2File(path,"rb") as fin:
155 unpickler=pickle.Unpickler(fin)
156 self.header=unpickler.load()
157 if(isinstance(self.header,nljson)):
158 self.header=self.header()
159 self.currentFrame=unpickler.load()
160 if(isinstance(self.currentFrame,nljson)):
161 self.currentFrame=self.currentFrame()
162 self.onEpisodeBegin()
163 while True:
164 try:
165 self.currentFrame=unpickler.load()
166 if(isinstance(self.currentFrame,nljson)):
167 self.currentFrame=self.currentFrame()
168 self.onInnerStepBegin()
169 self.onInnerStepEnd()
170 except EOFError:
171 break
172 self.onEpisodeEnd()
173 del unpickler
