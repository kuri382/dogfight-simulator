269//
270// call_funcを呼び出すためのマクロ
271// call_funcの第1引数は関数オブジェクトになるため、関数テンプレートを直接与えることができない。
272// このマクロは第1引数で与えた関数名をラップするジェネリックラムダを生成するので、
273// ユーザ目線では関数名を直接与えてcall_funcを呼び出すようなコードが書けるようになる。
274//
275#define ASRC_INTERNAL_FORWARD_VARIABLE(v) std::forward<decltype((v))>(v)
276#define ASRC_INTERNAL_GENERIC_LAMBDA(function_name) \
277[&](auto&& ... args){ \
278 return function_name(std::forward<decltype(args)>(args)...); \
279}

281#define ASRC_CALL_ARCHIVE_FUNC(function_name, archive_name, ...) \
282 ::asrc::core::util::call_func( \
283 ASRC_INTERNAL_GENERIC_LAMBDA(function_name) \
284 , ASRC_INTERNAL_FOREACH(ASRC_INTERNAL_FORWARD_VARIABLE, ASRC_INTERNAL_COMMA, archive_name __VA_OPT__(,) __VA_ARGS__) \
285 ) \

287//
288// 変数名と同じ名前をつけたNameValuePairを作りつつ、call_funcでserialize_with_type_infoに転送するマクロ
289// py::objectは型情報を付与し、それ以外はそのままの形でarchiveのoperator()に転送される。
290// ユーザ目線では多くの場合はこのマクロを使用すれば様々な型の変数を同じ記述方法で入出力できるようになる。
291//
292#define ASRC_SERIALIZE_FORWARD_AS_NVP(variable) ::cereal::make_nvp(#variable,std::forward<decltype((variable))>(variable))
293#define ASRC_SERIALIZE_NVP(archive_variable, ...) \
294 do { \
295 ::asrc::core::util::call_func( \
296 ASRC_INTERNAL_GENERIC_LAMBDA(::asrc::core::util::serialize_with_type_info) \
297 , archive_variable \
298 __VA_OPT__(, ASRC_INTERNAL_FOREACH(ASRC_SERIALIZE_FORWARD_AS_NVP, ASRC_INTERNAL_COMMA , __VA_ARGS__))\
299 ); \
300 } while (false);
