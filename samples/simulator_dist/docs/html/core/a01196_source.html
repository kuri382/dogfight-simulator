10class Draw2D:
11 '''3Dと2Dを切り替えるためのクラス。デフォルトが3Dとして、
12 Draw2D.begin()とDraw2D.end()で挟まれた範囲が、幅width,高さheightの矩形領域への2D描画として扱われる。
13 '''
14 @classmethod
15 def setShape(cls,width,height):
16 '''描画領域の指定
17 '''
18 cls.width=width
19 cls.height=height
20 @classmethod
21 def convertPos(cls,pos):
22 '''([0,width],[0,height])で与えた位置情報を、glVertex2Dで描画する際の範囲[-1,+1]に変換する。
23 '''
24 return (2.0*pos[0]/cls.width-1.0,2.0*pos[1]/cls.height-1.0)
25 @classmethod
26 def begin(cls):
27 '''2D描画の開始
28 '''
29 glDisable(GL_DEPTH_TEST)
30 cls.oldMatrixMode=glGetIntegerv(GL_MATRIX_MODE)
31 glMatrixMode(GL_PROJECTION)
32 glPushMatrix()
33 glLoadIdentity()
34 glMatrixMode(GL_MODELVIEW)
35 glPushMatrix()
36 glLoadIdentity()
37 @classmethod
38 def end(cls):
39 '''2D描画の終了
40 '''
41 glPopMatrix()
42 glMatrixMode(GL_PROJECTION)
43 glPopMatrix()
44 glMatrixMode(cls.oldMatrixMode)
45 glEnable(GL_DEPTH_TEST)

80def drawCircle3D(_pos, _x,_y,_r, _n):
81 '''3D空間に円(多角形)を描画する。
82
83 _x,_yに単位ベクトルでないものを与えれば楕円にもなる。また、直交判定は行わない。
84
85 Args:
86 _pos (3-dim array): 中心の座標
87 _x (3-dim array): 円のx軸方向ベクトル
88 _y (3-dim array): 円のy軸方向ベクトル
89 _r (float): 半径
90 _n (int): 分割数。
91 '''
92 y = np.cross(np.cross(_x,_y),_x)
93 glBegin(GL_LINE_LOOP)
94 for i in range(_n):
95 tmp = _pos + _x*(_r*cos(2 * pi*i / _n)) + _y*(_r*sin(2 * pi*i / _n))
96 glVertex3d(tmp[0], tmp[1], tmp[2])
97 glEnd()
98def drawCircle2D(_pos,_r,_n, _polar_ax=np.array([0.,0.])):
99 '''2D空間に円(多角形)を描画する。
100
101 Args:
102 _pos (3-dim array): 中心の座標
103 _r (float): 半径
104 _n (int): 分割数
105 _polar_ax (2-dim array): 始線のx,y方向成分
106 '''
107 glBegin(GL_LINE_LOOP)
108 base=atan2(_polar_ax[1],_polar_ax[0])
109 for i in range(_n):
110 tmp = Draw2D.convertPos(_pos + _r*np.array([cos(2 * pi*i / _n + base) , sin(2 * pi*i / _n + base)]))
111 glVertex2d(tmp[0], tmp[1])
112 glEnd()

247def drawPipe(_r0,_r1,_radius):
248 '''3D空間に円柱を描画する。
249
250 Args:
251 _r0 (3-dim array): 始点の座標
252 _r1 (3-dim array): 終点の座標
253 _radius (float): 半径
254 '''
255 r2=r1-r0
256 L=np.linalg.norm(r2)
257 ang=acos(r2[2]/L)/pi*180.0
258 arrow=gluNewQuadric()
259 glPushMatrix()
260 glTranslated(_r0[0], _r0[1], _r0[2])
261 glRotated(ang, -r2[1]*L, r2[0]*L, 0.0)
262 arrow = gluNewQuadric()
263 gluQuadricDrawStyle(arrow, GLU_FILL)
264 gluCylinder(arrow, _radius, _radius, len, 8, 1)
265 glPopMatrix()
266 gluDeleteQuadric(arrow)

268def polygonRegionCut(ps,reg,axis):
269 '''(仮実装)凸多角形をx軸またはy軸の指定範囲の境界で切断し、指定範囲内に収まる新たな多角形として返す。
270
271 pygameのウィンドウで複数のGLサーフェスを使えないため、
272 2種類以上の画面を並べる際に、隣の画面にはみ出さないようにするための処置。
273 例えばmodernGLを導入してFBOを使うなどすれば解決するので将来的には不要になる。
274
275 Args:
276 ps (list of 2-dim array): 多角形を構成する点の2D座標のリスト。
277 reg (2-dim array): 収めたい範囲。[最小値,最大値]
278 axis (int): 切断したい軸。x軸のときは0、y軸のときは1を入れる。
279 '''
280 #axis軸をreg[0],reg[1]に収まるように凸多角形を区切る
281 inRange=[reg[0]<=p[axis] and p[axis]<=reg[1] for p in ps]
282 ret=ps
283 if(inRange[0]):#1つめが内側
284 intercept=[-1]
285 mid=[]
286 for i in range(1,len(ps)):
287 if(len(intercept)%2==1 and not inRange[i]):
288 inner=ps[i-1]
289 outer=ps[i]
290 intercept[-1]=i
291 intercept.append(-1)
292 dst=reg[1] if outer[axis]>=reg[1] else reg[0]
293 mid.append(inner+(outer-inner)*((dst-inner[axis])/(outer[axis]-inner[axis])))
294 if(len(intercept)%2==0 and inRange[i]):
295 outer=ps[i-1]
296 inner=ps[i]
297 dst=reg[1] if outer[axis]>=reg[1] else reg[0]
298 mid.append(outer+(inner-outer)*((dst-outer[axis])/(inner[axis]-outer[axis])))
299 intercept[-1]=i
300 intercept.append(-1)
301 if(len(intercept)%2==0):
302 outer=ps[-1]
303 inner=ps[0]
304 dst=reg[1] if outer[axis]>=reg[1] else reg[0]
305 mid.append(outer+(inner-outer)*((dst-outer[axis])/(inner[axis]-outer[axis])))
306 intercept[-1]=0
307 ret=[]
308 bef=0
309 aft=-1
310 for i in range(int(len(mid)/2)):
311 aft=intercept[i*2]
312 ret=ret+ps[bef:aft]+mid[i*2:i*2+2]
313 bef=intercept[i*2+1]
314 if(len(mid)==0 or bef>0):
315 ret=ret+ps[bef:]
316 return ret
