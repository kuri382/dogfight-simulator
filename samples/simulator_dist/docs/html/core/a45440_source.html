14void Asset::initialize(){
15 BaseType::initialize();
16 removeWhenKilled=false;
17 std::shared_ptr<SimulationManagerAccessorBase> manager=std::dynamic_pointer_cast<SimulationManagerAccessorBase>(this->Entity::manager);
18 if(modelConfig.contains("interval")){
19 nl::json intervals=modelConfig.at("interval");
20 std::string intervalUnit="tick";
21 if(intervals.contains("unit")){
22 intervalUnit=intervals.at("unit");
23 }
24 if(intervalUnit=="time"){
25 double dt=manager->getBaseTimeStep();
26 interval[SimPhase::PERCEIVE]=std::max<std::uint64_t>(1,std::lround(getValueFromJsonKRD(intervals,"perceive",randomGen,dt)/dt));//reset at t=0 is not regarded as the "first time."
27 interval[SimPhase::CONTROL]=std::max<std::uint64_t>(1,std::lround(getValueFromJsonKRD(intervals,"control",randomGen,dt)/dt));
28 interval[SimPhase::BEHAVE]=std::max<std::uint64_t>(1,std::lround(getValueFromJsonKRD(intervals,"behave",randomGen,dt)/dt));
29 }else if(intervalUnit=="tick"){
30 interval[SimPhase::PERCEIVE]=getValueFromJsonKRD(intervals,"perceive",randomGen,1);//reset at t=0 is not regarded as the "first time."
31 interval[SimPhase::CONTROL]=getValueFromJsonKRD(intervals,"control",randomGen,1);
32 interval[SimPhase::BEHAVE]=getValueFromJsonKRD(intervals,"behave",randomGen,1);
33 }else{
34 throw std::runtime_error("Only \"tick\" or \"time\" can be designated as the unit of interval."+intervalUnit);
35 }
36 }else{
37 interval[SimPhase::PERCEIVE]=1;
38 interval[SimPhase::CONTROL]=1;
39 interval[SimPhase::BEHAVE]=1;
40 }
41 interval[SimPhase::VALIDATE]=std::numeric_limits<std::uint64_t>::max();
42 if(modelConfig.contains("firstTick")){
43 nl::json firstTicks=modelConfig.at("firstTick");
44 std::string firstTickUnit="tick";
45 if(firstTicks.contains("unit")){
46 firstTickUnit=firstTicks.at("unit");
47 }
48 if(firstTickUnit=="time"){
49 double dt=manager->getBaseTimeStep();
50 firstTick[SimPhase::PERCEIVE]=std::lround(getValueFromJsonKRD<double>(firstTicks,"perceive",randomGen,interval[SimPhase::PERCEIVE]*dt)/dt);//reset at t=0 is not regarded as the "first time."
51 firstTick[SimPhase::CONTROL]=std::lround(getValueFromJsonKRD(firstTicks,"control",randomGen,0.0)/dt);
52 firstTick[SimPhase::BEHAVE]=std::lround(getValueFromJsonKRD(firstTicks,"behave",randomGen,0.0)/dt);
53 }else if(firstTickUnit=="tick"){
54 firstTick[SimPhase::PERCEIVE]=getValueFromJsonKRD(firstTicks,"perceive",randomGen,interval[SimPhase::PERCEIVE]);//reset at t=0 is not regarded as the "first time."
55 firstTick[SimPhase::CONTROL]=getValueFromJsonKRD(firstTicks,"control",randomGen,0);
56 firstTick[SimPhase::BEHAVE]=getValueFromJsonKRD(firstTicks,"behave",randomGen,0);
57 }else{
58 throw std::runtime_error("Only \"tick\" or \"time\" can be designated as the unit of firstTick."+firstTickUnit);
59 }
60 }else{
61 firstTick[SimPhase::PERCEIVE]=interval[SimPhase::PERCEIVE];//reset at t=0 is not regarded as the "first time."
62 firstTick[SimPhase::CONTROL]=0;
63 firstTick[SimPhase::BEHAVE]=0;
64 }
65 firstTick[SimPhase::VALIDATE]=0;//meaningless but defined for same interface
66}
