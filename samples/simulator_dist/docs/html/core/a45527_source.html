35void GeodeticCRS::validateImpl(bool isFirstTime) const{
36 this->CoordinateReferenceSystem::validateImpl(isFirstTime);
37 //他のGeodeticCRSと対になるものとして生成される場合はそのdatumの情報を優先して用いる。
38 _autoGenerated=false;
39 if(isGeographic()){

42 if(instanceConfig.contains("ecef")){
43 j=instanceConfig.at("ecef");

45 }else if(modelConfig.contains("ecef")){

47 _autoGenerated=true;
48 }
49 if(_autoGenerated){
50 assert(j.is_object());
51 j["baseName"]="CoordinateReferenceSystem";
52 _ecef=createOrGetEntity<GeodeticCRS>(j);
53 if(_ecef){
54 setGeographicCRS(_ecef,non_const_this<GeographicCRS>());
55 _semiMajorAxis=_ecef->semiMajorAxis();
56 _inverseFlattening=_ecef->inverseFlattening();
57 }else{
58 throw std::runtime_error("The given json does not represent a valid ECEF. j="+j.dump());
59 }
60 return;
61 }
62 }else if(isEarthFixed()){
63 // 直交座標or極座標
64 if(instanceConfig.contains("isSpherical")){
65 _isSpherical=instanceConfig.at("isSpherical");
66 }else if(modelConfig.contains("isSpherical")){
67 _isSpherical=modelConfig.at("isSpherical");
68 }else{
69 _isSpherical=false;
70 }
71 if(!_isSpherical){
72 //Cartesian ECEFはGeographicから生成され得る。
73 nl::json j;
74 if(instanceConfig.contains("geog")){
75 j=instanceConfig.at("geog");
76 _autoGenerated=true;
77 }else if(modelConfig.contains("geog")){
78 j=modelConfig.at("geog");
79 _autoGenerated=true;
80 }
81 if(_autoGenerated){
82 assert(j.is_object());

84 auto shared=createOrGetEntity<GeographicCRS>(j);
85 if(shared){
86 setECEF(shared,non_const_this<GeodeticCRS>());
87 _semiMajorAxis=shared->semiMajorAxis();
88 _inverseFlattening=shared->inverseFlattening();
89 _geog=shared;
90 }else{
91 throw std::runtime_error("The given json does not represent a valid geographic CRS. j="+j.dump());
92 }
93 return;
94 }
95
96 //Cartesian ECEFはSpherical ECEFからも生成され得る。
97 if(instanceConfig.contains("spherical_ecef")){
98 j=instanceConfig.at("spherical_ecef");
99 _autoGenerated=true;
100 }else if(modelConfig.contains("spherical_ecef")){
101 j=modelConfig.at("spherical_ecef");
102 _autoGenerated=true;

104 if(_autoGenerated){
105 assert(j.is_object());

112 _spherical_ecef=shared;
113 }else{
114 throw std::runtime_error("The given json does not represent a valid geographic CRS. j="+j.dump());
115 }
116 return;
117 }
118 }else{
119 //Spherical ECEFはCartesian ECEFから生成され得る。
120 nl::json j;
121 if(instanceConfig.contains("ecef")){
122 j=instanceConfig.at("ecef");
123 _autoGenerated=true;
124 }else if(modelConfig.contains("ecef")){
125 j=modelConfig.at("ecef");
126 _autoGenerated=true;
127 }
128 if(_autoGenerated){
129 assert(j.is_object());
130 j["baseName"]="CoordinateReferenceSystem";
131 _ecef=createOrGetEntity<GeodeticCRS>(j);

133 setSphericalECEF(_ecef,non_const_this<GeodeticCRS>());
134 _semiMajorAxis=_ecef->semiMajorAxis();
135 _inverseFlattening=_ecef->inverseFlattening();

137 throw std::runtime_error("The given json does not represent a valid ECEF. j="+j.dump());

186 }else if(modelConfig.contains("isSpherical")){
187 _isSpherical=modelConfig.at("isSpherical");

733// 同じdatumでgeographic座標をECEF座標に変換する。
734Eigen::Vector3d GeodeticCRS::transformGeographicCoordinateToECEF(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const CoordinateType& coordinateType) const{
735 if(coordinateType==CoordinateType::POSITION_ABS){
736 // 絶対位置はEPSG:9602で変換する。
737 return transformGeographicCoordinateToECEF(value,CoordinateType::POSITION_ABS);
738 }else if(coordinateType==CoordinateType::POSITION_REL){
739 // 相対位置の変換は始点と終点の位置をそれぞれ変換して差をとる。
740 return (
741 transformGeographicCoordinateToECEF(location+value,CoordinateType::POSITION_ABS)-
742 transformGeographicCoordinateToECEF(location,CoordinateType::POSITION_ABS)
743 );
744 }else{
745 // 位置以外は局所的な座標軸の向きと縮尺を考慮してヤコビアンにより変換する。
746 // geocentric coordinate (X, Y, Z) in [m]
747 // geographic coordinate (φ, λ, h) latitude[deg],longitude[deg],elliptic altitude[m]
748
749 double lat=deg2rad(location(0));
750 double lon=deg2rad(location(1));
751 double alt=location(2);
752 double f = 1.0/inverseFlattening();
753 double e2 = f * (2 - f);
754 double e2sin2lat=e2*pow(sin(lat),2);
755 double nu=semiMajorAxis()/sqrt(1-e2sin2lat);
756
757 double coeff = (nu*(1-e2)/(1-e2sin2lat)+alt);
758 double dX_dphi = deg2rad(-coeff*sin(lat)*cos(lon));// [m/rad]→[m/deg]に換算しておく
759 double dY_dphi = deg2rad(-coeff*sin(lat)*sin(lon));// [m/rad]→[m/deg]に換算しておく
760 double dZ_dphi = deg2rad(coeff*cos(lat));// [m/rad]→[m/deg]に換算しておく
761
762 double dX_dlam = deg2rad(-(nu+alt)*cos(lat)*sin(lon));// [m/rad]→[m/deg]に換算しておく
763 double dY_dlam = deg2rad((nu+alt)*cos(lat)*cos(lon));// [m/rad]→[m/deg]に換算しておく
764
765 Eigen::Matrix3d Rcg({
766 {dX_dphi, dX_dlam, cos(lat)*cos(lon)},
767 {dY_dphi, dY_dlam, cos(lat)*sin(lon)},
768 {dZ_dphi, 0, sin(lat)}
769 });
770 if(coordinateType==CoordinateType::DIRECTION){
771 return (Rcg*value).normalized();
772 }else{
773 return Rcg*value;
774 }
775 }
776}
777Eigen::Vector3d GeodeticCRS::transformGeographicCoordinateToECEF(const Eigen::Vector3d& value, const CoordinateType& coordinateType) const{
778 if(coordinateType!=CoordinateType::POSITION_ABS){
779 //絶対位置以外は観測点の指定を強制する。
780 throw std::runtime_error("Geographic/ECEF conversion requires location of reference point, unless coordinateType==POSITION_ABS.");
781 }
782 // lat(rad),lon(rad),alt→XYZ
783 // EPSG:9602
784 double lat = deg2rad(value(0));
785 double lon = deg2rad(value(1));
786 double alt = value(2);
787 double f = 1.0/_inverseFlattening;
788 double e2 = f * (2 - f);
789 double nu = _semiMajorAxis / sqrt(1.0 - e2 * pow(sin(lat), 2));
790 return Eigen::Vector3d(
791 (nu + alt) * cos(lat) * cos(lon),
792 (nu + alt) * cos(lat) * sin(lon),
793 ((1 - e2) * nu + alt) * sin(lat)
794 );
795}
796// 同じdatumのECEF座標をgeographic座標に変換する。
797Eigen::Vector3d GeodeticCRS::transformECEFCoordinateToGeographicCRS(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const CoordinateType& coordinateType) const{
798 if(coordinateType==CoordinateType::POSITION_ABS){
799 // 絶対位置はEPSG:9602で変換する。
800 return transformECEFCoordinateToGeographicCRS(value,CoordinateType::POSITION_ABS);
801
802 }else if(coordinateType==CoordinateType::POSITION_REL){
803 // 相対位置の変換は始点と終点の位置をそれぞれ変換して差をとる。
804 return (
805 transformECEFCoordinateToGeographicCRS(location+value,CoordinateType::POSITION_ABS)-
806 transformECEFCoordinateToGeographicCRS(location,CoordinateType::POSITION_ABS)
807 );
808 }else{
809 // 位置以外は局所的な座標軸の向きと縮尺を考慮してヤコビアンにより変換する。
810 // geocentric coordinate (X, Y, Z) in [m]
811 // geographic coordinate (φ, λ, h) latitude[deg],longitude[deg],elliptic altitude[m]
812 Eigen::Vector3d lla=transformECEFCoordinateToGeographicCRS(location,CoordinateType::POSITION_ABS);
813 double lat=deg2rad(lla(0));
814 double lon=deg2rad(lla(1));
815 double alt=lla(2);
816 double f = 1.0/inverseFlattening();
817 double e2 = f * (2 - f);
818 double e2sin2lat=e2*pow(sin(lat),2);
819 double nu=semiMajorAxis()/sqrt(1-e2sin2lat);
820
821 double coeff = (nu*(1-e2)/(1-e2sin2lat)+alt);
822 double dX_dphi = deg2rad(-coeff*sin(lat)*cos(lon));// [m/rad]→[m/deg]に換算しておく
823 double dY_dphi = deg2rad(-coeff*sin(lat)*sin(lon));// [m/rad]→[m/deg]に換算しておく
824 double dZ_dphi = deg2rad(coeff*cos(lat));// [m/rad]→[m/deg]に換算しておく
825
826 double dX_dlam = deg2rad(-(nu+alt)*cos(lat)*sin(lon));// [m/rad]→[m/deg]に換算しておく
827 double dY_dlam = deg2rad((nu+alt)*cos(lat)*cos(lon));// [m/rad]→[m/deg]に換算しておく
828
829 Eigen::Matrix3d Rcg({
830 {dX_dphi, dX_dlam, cos(lat)*cos(lon)},
831 {dY_dphi, dY_dlam, cos(lat)*sin(lon)},
832 {dZ_dphi, 0, sin(lat)}
833 });
834 if(coordinateType==CoordinateType::DIRECTION){
835 return (Rcg.inverse()*value).normalized();
836 }else{
837 return Rcg.inverse()*value;
838 }
839 }

841Eigen::Vector3d GeodeticCRS::transformECEFCoordinateToGeographicCRS(const Eigen::Vector3d& value, const CoordinateType& coordinateType) const{
842 if(coordinateType!=CoordinateType::POSITION_ABS){
843 //絶対位置以外は観測点の指定を強制する。
844 throw std::runtime_error("Geographic/ECEF conversion requires location of reference point, unless coordinateType==POSITION_ABS.");
845 }
846 // XYZ→lat(rad),lon(rad),alt
847 // EPSG:9602
848 double X = value(0);
849 double Y = value(1);
850 double Z = value(2);
851 double f = 1.0/_inverseFlattening;
852 double e2 = f * (2 - f);
853 double eps = e2 / (1 - e2);
854 double b = _semiMajorAxis * (1 - f);
855 double p = sqrt(X * X + Y * Y);
856 double q = atan2(Z*_semiMajorAxis, p*b);
857 double lat = atan2(Z+eps*b*pow(sin(q),3), p-e2*_semiMajorAxis*pow(cos(q),3));
858 double lon = atan2(Y,X);
859 double nu = _semiMajorAxis / sqrt(1.0 - e2 * pow(sin(lat), 2));
860 return Eigen::Vector3d(
861 rad2deg(lat),
862 rad2deg(lon),
863 p/cos(lat)-nu
864 );
865}

1048Eigen::Vector3d GeodeticCRS::cartesianToSpherical(const Eigen::Vector3d& v) const{
1049 // ECEF X,Y,Z -> (geocentric lat, geocentric lon, geocentric radius)
1050 double xy=sqrt(v(0)*v(0)+v(1)*v(1));
1051 if(xy==0){
1052 return Eigen::Vector3d(
1053 v(2)>0 ? M_PI_2 : -M_PI_2,
1054 0,
1055 abs(v(2))
1056 );
1057 }else{
1058 double lat=atan2(v(2),xy);
1059 return Eigen::Vector3d(
1060 lat,
1061 atan2(v(1),v(0)),
1062 sqrt(xy*xy+v(2)*v(2))
1063 );
1064 }
1065}

1067 // (geocentric lat, geocentric lon, geocentric radius) -> ECEF X,Y,Z
1068 return Eigen::Vector3d(
1069 v(2)*cos(v(1))*cos(v(0)),
1070 v(2)*sin(v(1))*cos(v(0)),
1071 v(2)*sin(v(0))
1072 );

1244 [](unsigned char c){ return std::toupper(c);}
