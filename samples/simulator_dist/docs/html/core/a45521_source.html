133Coordinate ProjectedCRS::transformToIntermediateCRS(const Eigen::Vector3d& value, const Eigen::Vector3d& location, const Time& time, const CoordinateType& coordinateType) const{
134 //intermediateはECEF
135 if(!isValid()){validate();}
136 if(coordinateType==CoordinateType::POSITION_ABS){
137 return transformToIntermediateCRS(value,time,CoordinateType::POSITION_ABS);
138 }else if(coordinateType==CoordinateType::POSITION_REL){
139 //相対位置は観測点からの相対位置ベクトルと解釈して、始点と終点をそれぞれ変換して差をとる。
140 auto locationInECEF=transformToIntermediateCRS(location,time,CoordinateType::POSITION_ABS)();
141 return std::move(Coordinate(
142 transformToIntermediateCRS(location+value,time,CoordinateType::POSITION_ABS)() - locationInECEF,
143 locationInECEF,
144 getIntermediateCRS(),
145 time,
146 coordinateType
147 ));
148 }else{
149 //位置以外はlocationを原点とするTopocentricCRSを介して軸の向きと縮尺を考慮した変換を行う。
150 double gamma=getMeridianConversionAtProjectedCoordinate(location);
151 double h=getMeridionalScaleFactorAtProjectedCoordinate(location,false);
152 double k=getParallelScaleFactorAtProjectedCoordinate(location,false);
153 double theta=getIntersectionAngleAtProjectedCoordinate(location);
154
155 // projected coordinate (Ep, Np, U) easting[m], northing[m], elliptic altitude[m]
156 // topocentric coordinate (Et, Nt, U) easting[m], northing[m], elliptic altitude[m]
157 double dEp_dEt = k*cos(gamma-theta+M_PI_2);
158 double dEp_dNt = -h*sin(gamma);
159 double dNp_dEt = k*sin(gamma-theta+M_PI_2);
160 double dNp_dNt = h*cos(gamma);
161
162 Eigen::Matrix3d Rpt({
163 {dEp_dEt, dEp_dNt, 0},
164 {dNp_dEt, dNp_dNt, 0},
165 { 0, 0, 1}
166 }); // [projected(ENU) <- topocentric(ENU)]
167 Eigen::Vector3d valueInTopo = Rpt.inverse()*getQuaternionToENU().transformVector(value);
168 if(coordinateType==CoordinateType::DIRECTION){
169 valueInTopo.normalize();
170 }
171 auto ecef=getIntermediateCRS();
172 auto locationInECEF = transformToIntermediateCRS(location,time,CoordinateType::POSITION_ABS)();
173 auto valueInECEF=ecef->transformFromTopocentricCRS(
174 valueInTopo,
175 Eigen::Vector3d::Zero(),
176 time,
177 coordinateType,
178 locationInECEF,
179 "ENU",
180 false
181 );
182 return std::move(Coordinate(
183 valueInECEF,
184 locationInECEF,
185 ecef,
186 time,
187 coordinateType
188 ));
189 }

272Quaternion ProjectedCRS::getQuaternionTo(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS) const{
273 if(!isValid()){validate();}
274 if(!dstCRS){
275 throw std::runtime_error("dstCRS is nullptr!");
276 }
277 if(this==dstCRS.get()){
278 return Quaternion(1,0,0,0);
279 }
280 if(!isTransformableTo(dstCRS)){
281 throw std::runtime_error("can't be transformed to "+dstCRS->getFactoryClassName());
282 }
283 if(auto derived=std::dynamic_pointer_cast<DerivedCRS>(dstCRS)){
284 if(this==derived->getBaseProjectedCRS().get()){
285 // 自身からderiveされたものは子側の関数を呼ぶ
286 return derived->getQuaternionFrom(location,time,non_const_this());
287 }
288 }
289 // そうでなければlocationを原点とするTopocentricCRSと、baseに対応するECEFの二つを間に挟む
290 auto ecef=std::dynamic_pointer_cast<GeodeticCRS>(getIntermediateCRS());
291 auto locationInECEF=transformTo(location,location,time,ecef,CoordinateType::POSITION_ABS)();
292
293 auto qToECEFFromTopo=ecef->getEquivalentQuaternionToTopocentricCRS(locationInECEF,time,"ENU").conjugate();
294 auto qToTopoFromSelf=getEquivalentQuaternionToTopocentricCRS(location,time,"ENU");
295 auto qToECEFFromSelf=qToECEFFromTopo*qToTopoFromSelf;
296 if(dstCRS==ecef){
297 return qToECEFFromSelf;
298 }else{
299 return ecef->getQuaternionTo(locationInECEF,time,dstCRS)*qToECEFFromSelf;
300 }
301}
302Quaternion ProjectedCRS::getQuaternionFrom(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& srcCRS) const{
303 if(!isValid()){validate();}
304 if(!srcCRS){
305 throw std::runtime_error("srcCRS is nullptr!");
306 }
307 if(this==srcCRS.get()){
308 return Quaternion(1,0,0,0);
309 }
310 if(!isTransformableTo(srcCRS)){
311 throw std::runtime_error(getFactoryClassName()+" can't be transformed from "+srcCRS->getFactoryClassName());
312 }
313 if(auto derived=std::dynamic_pointer_cast<DerivedCRS>(srcCRS)){
314 if(this==derived->getBaseProjectedCRS().get()){
315 // 自身からderiveされたものは子側の関数を呼ぶ
316 return derived->getQuaternionTo(location,time,non_const_this());
317 }
318 }
319 // そうでなければlocationを原点とするTopocentricCRSと、baseに対応するECEFの二つを間に挟む
320 auto ecef=getIntermediateCRS();
321 auto locationInECEF = ecef->transformFrom(location,location,time,srcCRS,CoordinateType::POSITION_ABS)();
322 auto qToECEFFromSrc = ecef->getQuaternionFrom(location,time,srcCRS);
323
324 auto qToTopoFromECEF=ecef->getEquivalentQuaternionToTopocentricCRS(locationInECEF,time,"ENU");
325
326 auto locationInSelf=transformFrom(location,location,time,srcCRS,CoordinateType::POSITION_ABS)();
327 auto qToSelfFromTopo=getEquivalentQuaternionToTopocentricCRS(locationInSelf,time,"ENU").conjugate();
328 return qToSelfFromTopo*qToTopoFromECEF*qToECEFFromSrc;

330Quaternion ProjectedCRS::getEquivalentQuaternionToTopocentricCRS(const Eigen::Vector3d& origin, const Time& time, const std::string& axisOrder_) const{
331 // CRSとして生成せずにクォータニオンとして得る。
332 // 原点(origin)におけるTopocentricCRSとProjectedCRSの間で、「北」が一致するように鉛直方向を軸に回転するようなクォータニオンとする。
333 // 正角図法でない場合は子午線と緯線が直交しないが、子午線を合わせることを優先する。
334
335 // まずはENU(projected)からENU(topocentric)への回転クォータニオンを計算する。
336 // U軸まわりに子午線収差角(投影図上の北と真北の差)だけ回転すればよい。
337 double gamma=getMeridianConversionAtProjectedCoordinate(origin);
338 Eigen::Vector3d EpInTopo(cos(gamma),-sin(gamma),0); // 投影図の緯線(Easting軸)がTopocentricCRS(ENU)でどこを向いているか
339 Eigen::Vector3d NpInTopo(sin(gamma),cos(gamma),0); // 投影図の子午線(Northing軸)がTopocentricCRS(ENU)でどこを向いているか
340
341 Quaternion qTopoFromProj = Quaternion::fromBasis(EpInTopo,NpInTopo,Eigen::Vector3d(0,0,1)); // [ENU(topo) <- ENU(projected)]
342
343 //変換前後のaxisOrderに応じて成分を入れ替える
344 std::string dstAxisOrder=axisOrder_;
345 std::transform(dstAxisOrder.begin(),dstAxisOrder.end(),dstAxisOrder.begin(),
346 [](unsigned char c){ return std::toupper(c);}
347 );
348 return (
349 getAxisSwapQuaternion("ENU",dstAxisOrder)* // [Dst <- ENU(topo)]
350 qTopoFromProj* // [ENU(topo) <- ENU(projected)]
351 getQuaternionToENU() // [ENU(projected) <- Self]
352 );
353}

815 double E=ENU(0);
816 double N=ENU(1);
817 double xi=N/(scaleFactor*A);
818 double eta=E/(scaleFactor*A);
819 double xi_d=xi;
820 double eta_d=eta;
821 for(int j=1;j<=beta.size();++j){
822 double beta_sin_j=beta(j-1)*sin(2*j*xi);
823 double beta_cos_j=beta(j-1)*cos(2*j*xi);
