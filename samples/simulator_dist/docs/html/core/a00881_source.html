30template<class T>
31struct is_stl_array {
32 private:
33 using Type=get_true_value_type_t<T>;
34 template<class U, std::size_t N>
35 static std::true_type test(const std::array<U,N>*);
36 static std::false_type test(...);
37
38 public:
39 static constexpr bool value = decltype(test(std::declval<Type*>()))::value;
40};

42concept stl_array = is_stl_array<T>::value;
