44 template <class Scalar,class Archive>
45 inline void save_dtype_helper(Archive& ar){
46 if constexpr (
47 !traits::same_archive_as<Archive,::asrc::core::util::NLJSONOutputArchive>
48 ){
49 DTypes dtype;
50 if constexpr (std::same_as<Scalar,double>){
51 dtype=DTypes::DOUBLE_;
52 }else if constexpr (std::same_as<Scalar,float>){
53 dtype=DTypes::FLOAT_;
54 }else if constexpr (std::same_as<Scalar,std::uint64_t>){
55 dtype=DTypes::UINT64_;
56 }else if constexpr (std::same_as<Scalar,std::int64_t>){
57 dtype=DTypes::INT64_;
58 }else if constexpr (std::same_as<Scalar,std::uint32_t>){
59 dtype=DTypes::UINT32_;
60 }else if constexpr (std::same_as<Scalar,std::int32_t>){
61 dtype=DTypes::INT32_;
62 }else if constexpr (std::same_as<Scalar,std::uint16_t>){
63 dtype=DTypes::UINT16_;
64 }else if constexpr (std::same_as<Scalar,std::int16_t>){
65 dtype=DTypes::INT16_;
66 }else if constexpr (std::same_as<Scalar,std::uint8_t>){
67 dtype=DTypes::UINT8_;
68 }else if constexpr (std::same_as<Scalar,std::int8_t>){
69 dtype=DTypes::INT8_;
70 }else if constexpr (std::same_as<Scalar,bool>){
71 dtype=DTypes::BOOL_;
72 }
73 ar(::cereal::make_nvp("dtype",dtype));
74 }
75 }

77 inline DTypes extract_dtype_helper(const nl::ordered_json& j){
78 // 一つでも実数値があったらdouble、そうでなければsigned int64とする。
79 if(j.at(0).is_array()){
80 for(auto && v:j){
81 DTypes ret=extract_dtype_helper(v);
82 if(ret==DTypes::DOUBLE_){
83 return ret;
84 }
85 }
86 return DTypes::INT64_;
87 }else if(j.at(0).is_number()){
88 for(auto && v:j){
89 if(v.is_number_float()){
90 return DTypes::DOUBLE_;
91 }
92 }
93 return DTypes::INT64_;
94 }else{
95 throw cereal::Exception("json to be loaded as py::array must be an array of number.");
96 }
97 }

117 template <class Archive, traits::matrix MatrixType>
118 inline void save_without_swap( Archive & ar, MatrixType const & m )
119 {
120 using MatrixTraits=Eigen::internal::traits<MatrixType>;
121 constexpr bool isRowMajorType = MatrixType::IsRowMajor;
122 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
123
124 if constexpr (MatrixTraits::RowsAtCompileTime==1 || MatrixTraits::ColsAtCompileTime==1) {
125 ar( ::cereal::make_size_tag( static_cast<::cereal::size_type>(m.size()) ) );
126 if constexpr (
127 cereal::traits::is_output_serializable<cereal::BinaryData<const typename MatrixTraits::Scalar *>, Archive>::value
128 && (
129 std::derived_from<MatrixType,Eigen::PlainObjectBase<MatrixType>>
130 || std::derived_from<MatrixType,Eigen::MapBase<MatrixType>>
131 )
132 ) {
133 // Eigen::PlainObjectBase (Eigen::Matrix, Eigen::Vector, Eigen::Array等)
134 // Eigen::MapBase (Eigen::Map, Eigen::Ref, Eigen::Block等)
135 // m.data()でデータにアクセス可能なクラス全般
136 ar( ::cereal::binary_data( m.data(), m.size() * sizeof(typename MatrixTraits::Scalar) ) );
137 }else{
138 for(auto&& v : m){
139 ar(v);
140 }
141 }
142 }else{
143 ar( ::cereal::make_size_tag( static_cast<::cereal::size_type>(m.outerSize()) ) );
144 for(Eigen::Index outer=0; outer<m.outerSize(); ++outer){
145 // m.innerVector(outer)を使いたいところだが、
146 // MatrixTraits::ColsAtCompileTime==Eigen::Dynamicを要求されてしまうようなので、
147 // 固定サイズの行列には使用できない。
148 if constexpr (isRowMajorType) {
149 ar(m.row(outer));
150 }else{
151 ar(m.col(outer));
152 }
153 }
154 }
155 }
156 //
157 // loadはnon-constなPlainObjectBase又はMapBaseに限る。
158 //
159 template <class Archive, traits::matrix MatrixType>
160 requires (
161 (
162 std::derived_from<MatrixType,Eigen::PlainObjectBase<MatrixType>>
163 || std::derived_from<MatrixType,Eigen::MapBase<MatrixType>>
164 )
165 && !std::is_const_v<MatrixType>
166 )
167 inline void load_without_swap( Archive & ar, MatrixType & m )
168 {
169 using MatrixTraits=Eigen::internal::traits<MatrixType>;
170 constexpr bool isRowMajorType = MatrixType::IsRowMajor;
171 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
172
173 if constexpr (MatrixTraits::RowsAtCompileTime==1 || MatrixTraits::ColsAtCompileTime==1) {
174 cereal::size_type size;
175 ar( cereal::make_size_tag( size ) );
176 if constexpr (std::derived_from<MatrixType,Eigen::PlainObjectBase<MatrixType>>){
177 m.resize(size);
178 }else{
179 assert(m.size()==size);
180 }
181 if constexpr (cereal::traits::is_input_serializable<cereal::BinaryData<typename MatrixTraits::Scalar *>, Archive>::value) {
182 ar( cereal::binary_data( m.data(), static_cast<std::size_t>( size ) * sizeof(typename MatrixTraits::Scalar) ) );
183 }else{
184 for(Eigen::Index i=0;i<size;++i){
185 ar(m(i));
186 }
187 }
188 }else{
189 cereal::size_type outerSize;
190 ar( cereal::make_size_tag( outerSize ) );
191 for(Eigen::Index outer=0; outer<outerSize; ++outer){
192 if(outer==0){
193 if constexpr (isRowMajorType) {
194 Eigen::Matrix<
195 typename MatrixTraits::Scalar,
196 1,
197 MatrixTraits::ColsAtCompileTime,
198 Eigen::RowMajor,
199 1,
200 MatrixTraits::MaxColsAtCompileTime
201 > tmp;
202 ar(tmp);
203 if constexpr (std::derived_from<MatrixType,Eigen::PlainObjectBase<MatrixType>>){
204 m.resize(outerSize,tmp.cols());
205 }else{
206 assert(tmp.cols()==m.innerSize() && m.outerSize()==outerSize);
207 }
208 // m.innerVector(outer)を使いたいところだが、
209 // MatrixTraits::ColsAtCompileTime==Eigen::Dynamicを要求されてしまうようなので、
210 // 固定サイズの行列には使用できない。
211 m.row(outer)=std::move(tmp);
212 }else{
213 Eigen::Matrix<
214 typename MatrixTraits::Scalar,
215 MatrixTraits::RowsAtCompileTime,
216 1,
217 Eigen::ColMajor,
218 MatrixTraits::MaxRowsAtCompileTime,
219 1
220 > tmp;
221 ar(tmp);
222 if constexpr (std::derived_from<MatrixType,Eigen::PlainObjectBase<MatrixType>>){
223 m.resize(tmp.rows(),outerSize);
224 }else{
225 assert(tmp.rows()==m.innerSize() && m.outerSize()==outerSize);
226 }
227 m.col(outer)=std::move(tmp);
228 }
229 }else{
230 if constexpr (isRowMajorType) {
231 ar(m.row(outer));
232 }else{
233 ar(m.col(outer));
234 }
235 }
236 }
237 }

246 using Type = Eigen::Tensor<Scalar_, NumIndices_, Options_, IndexType_>;

249 constexpr bool is_text_archive = cereal::traits::is_text_archive<Archive>::value;
250
251 if constexpr (traits::NumDimensions==1) {

253 if constexpr (cereal::traits::is_output_serializable<cereal::BinaryData<const typename traits::Scalar *>, Archive>::value) {
254 ar( ::cereal::binary_data( m.data(), m.size() * sizeof(typename traits::Scalar) ) );
255 }else{
256 for(Eigen::Index i=0; i<m.size(); ++i){
257 ar(m(i));
258 }
259 }
260 }else{
261 // RowMajorは最後の軸に沿って連続、最初の軸が最外周

263 constexpr Eigen::Index outerDimIndex = (isRowMajorType ? 0 : traits::NumDimensions-1);

269 newDimensions[i]=m.dimensions()[i+dimIndexOffset];
270 }
271 Eigen::Index stride=m.size()/m.dimensions()[outerDimIndex];
272 for(Eigen::Index i=0; i<m.dimensions()[outerDimIndex]; ++i){
273 Eigen::TensorMap<const Eigen::Tensor<typename traits::Scalar, traits::NumDimensions-1, traits::Options, typename traits::Index>,Options_> mapped(
274 m.data() + stride*i,
275 newDimensions
276 );
277 ar(mapped);
278 }

282 inline void load_without_swap( Archive & ar, Eigen::Tensor<Scalar_, NumIndices_, Options_, IndexType_> & m )
283 {

294 ar( cereal::binary_data( m.data(), static_cast<std::size_t>( size ) * sizeof(typename traits::Scalar) ) );
295 }else{

297 ar(m(i));
298 }
299 }
300 }else{
301 // RowMajorは最後の軸に沿って連続、最初の軸が最外周

303 constexpr Eigen::Index outerDimIndex = (isRowMajorType ? 0 : traits::NumDimensions-1);
304 constexpr Eigen::Index dimIndexOffset = (isRowMajorType ? 1 : 0);
305

307 ar( ::cereal::make_size_tag( outerSize ) );

310 Eigen::DSizes<Eigen::Index, traits::NumDimensions> startIndices;
311 Eigen::DSizes<Eigen::Index, traits::NumDimensions> sliceSizes;
312 Eigen::Index stride;
313 for(Eigen::Index i=0; i<outerSize; ++i){

316 ar(sub);
317 subDimensions=sub.dimensions();
318 for(Eigen::Index i=0; i<traits::NumDimensions-1; ++i){
319 newDimensions[i+dimIndexOffset]=subDimensions[i];
320 startIndices[i+dimIndexOffset]=0;

324 m.resize(newDimensions);
325 startIndices[outerDimIndex]=0;
326 sliceSizes[outerDimIndex]=1;
327 m.slice(startIndices,sliceSizes)=std::move(sub.reshape(sliceSizes));
328 stride=m.size()/outerSize;
329 }else{
330 Eigen::TensorMap<Eigen::Tensor<typename traits::Scalar, traits::NumDimensions-1, traits::Options, typename traits::Index>,Options_> mapped(
331 m.data() + stride*i,
332 subDimensions
333 );
334 ar(mapped);
335 }

554 assert(size==m.dimensions()[0]);
555 auto evaluated=Eigen::TensorEvaluator(m,Eigen::DefaultDevice());
556 for(Eigen::Index i=0;i<size;++i){
557 ar(evaluated.coeffRef(i));
558 }

561 // ColMajorは最初の軸に沿って連続、最後の軸が最外周
562 constexpr Eigen::Index outerDimIndex = (isRowMajorType ? 0 : traits::NumDimensions-1);
563 constexpr Eigen::Index dimIndexOffset = (isRowMajorType ? 1 : 0);
564
565 cereal::size_type size;
566 ar( ::cereal::make_size_tag( size ) );
567 assert(size==m.dimensions()[outerDimIndex]);
568
569 Eigen::DSizes<Eigen::Index, traits::NumDimensions-1> newDimensions;
570 Eigen::DSizes<Eigen::Index, traits::NumDimensions> startIndices;
571 Eigen::DSizes<Eigen::Index, traits::NumDimensions> sliceSizes=m.dimensions();
572 for(Eigen::Index i=0; i<traits::NumDimensions-1; ++i){
573 newDimensions[i]=m.dimensions()[i+dimIndexOffset];
574 startIndices[i+dimIndexOffset]=0;
575 }
576 sliceSizes[outerDimIndex]=1;
577 for(Eigen::Index i=0; i<m.dimensions()[outerDimIndex]; ++i){
578 startIndices[outerDimIndex]=i;
579 ar(m.slice(startIndices,sliceSizes).reshape(newDimensions));
580 }
