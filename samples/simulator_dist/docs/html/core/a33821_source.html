14 //-π〜+π

21 //0〜+2π

35std::vector<std::pair<double,double>> mergeSectors(const std::pair<double,double>& s1,const std::pair<double,double>& s2){
36 std::vector<std::pair<double,double>> ret;
37 if(isInsideSector(s1,s2.first)){
38 double e1=asNonNegativeAngle(s1.second-s1.first);
39 double e2=asNonNegativeAngle(s2.second-s1.first);
40 ret.push_back(std::make_pair(s1.first,asAngle(s1.first+std::max(e1,e2))));
41 }else if(isInsideSector(s2,s1.first)){
42 double e1=asNonNegativeAngle(s1.second-s2.first);
43 double e2=asNonNegativeAngle(s2.second-s2.first);
44 ret.push_back(std::make_pair(s2.first,asAngle(s2.first+std::max(e1,e2))));
45 }else{
46 ret.push_back(s1);
47 ret.push_back(s2);
48 }

78std::pair<double,double> LinearSegment::calcUnacceptableRegion(const MotionState& motion, const Eigen::VectorXd& dstDir){
79 Eigen::Vector2d r(motion.pos(0),motion.pos(1));
80 double L=(p2-p1).norm();
81 double t = L>0 ? (r-p1).dot(p2-p1)/(L*L) : (r-p1).norm();
82 Eigen::Vector2d h=p1*(1-t)+p2*t;
83 double d,ah;
84 if(isOneSide){
85 d=normal.dot(r-h);
86 ah=getAZ(-normal);
87 }else{
88 d=(r-h).norm();
89 ah=getAZ(h-r);
90 }
91 double dAZ=M_PI/2.0-atan(adjustStrength*(d-limit));
92 double a1,a2;
93 if(t<0 && !infinite_p1){
94 bool inside = (r-p1).norm()<threshold;
95 double a0=getAZ(p1-r);
96 if(inside){
97 double dAZ1=asAngle(a0-ah);
98 if(dAZ1>=0){
99 if(dAZ1<dAZ){
100 a1=a0;
101 a2=asAngle(ah+dAZ);
102 }else{
103 a1=a0;
104 a2=a0;
105 }
106 }else{
107 if(-dAZ<dAZ1){
108 a1=asAngle(ah-dAZ);
109 a2=a0;
110 }else{
111 a1=a0;
112 a2=a0;
113 }
114 }
115 }else{
116 a1=a0;
117 a2=a0;
118 }
119 }else if(t>1 && !infinite_p2){
120 bool inside = (r-p2).norm()<threshold;
121 double a0=getAZ(p2-r);
122 if(inside){
123 double dAZ2=asAngle(a0-ah);
124 if(dAZ2>=0){
125 if(dAZ2<dAZ){
126 a1=a0;
127 a2=asAngle(ah+dAZ);
128 }else{
129 a1=a0;
130 a2=a0;
131 }
132 }else{
133 if(-dAZ<dAZ2){
134 a1=asAngle(ah-dAZ);
135 a2=a0;
136 }else{
137 a1=a0;
138 a2=a0;
139 }
140 }
141 }else{
142 a1=a0;
143 a2=a0;
144 }
145 }else{// if(t>=0 && t<=1){

148 a1=asAngle(ah-dAZ);
149 a2=asAngle(ah+dAZ);
150 }else{

161Eigen::VectorXd StaticCollisionAvoider2D::operator()(const MotionState& motion, const Eigen::VectorXd& dstDir){
162 std::vector<std::pair<double,double>> ret;
163 for(auto&& b:borders){
164 std::pair<double,double> s=b->calcUnacceptableRegion(motion,dstDir);
165 if(s.first==s.second){
166 continue;
167 }
168 if(ret.size()==0){
169 ret.push_back(s);
170 }else{
171 std::vector<std::pair<double,double>> merged;
172 for(auto&& r:ret){
173 std::vector<std::pair<double,double>> m=mergeSectors(r,s);
174 if(m.size()==1){//merged
175 s=m[0];
176 }//else{s=m[1];}
177 merged.push_back(m[0]);
178 }
179 ret=std::move(merged);
180 }
181 }
182 double dstAZ=getAZ(dstDir);
183 for(auto&& r:ret){
184 if(isInsideSector(r,dstAZ)){
185 double d1=asAngle(r.first-dstAZ);
186 double d2=asAngle(r.second-dstAZ);
187 if(abs(d1)<abs(d2)){
188 dstAZ=r.first;
189 }else{
190 dstAZ=r.second;
191 }
192 break;
193 }
194 }
195 if(dstDir.size()>2){
196 double factor=sqrt(1.0-dstDir(2)*dstDir(2));
197 return Eigen::Vector3d(cos(dstAZ)*factor,sin(dstAZ)*factor,dstDir(2));
198 }else{
199 return Eigen::Vector2d(cos(dstAZ),sin(dstAZ));
200 }
201}
