14// ある型Tから暗黙に変換可能な型のうち、一部の型への暗黙的な変換を禁止するためのラッパーの基底クラス。
15//
16struct ForwarderBase {};
17
18template<template<class> class Forwarder, class T,
19 std::enable_if_t<
20 std::is_base_of_v<ForwarderBase,Forwarder<T>>
21 ,std::nullptr_t> = nullptr
22>
23struct make_forwarder {
24 using type = Forwarder<std::remove_cv_t<std::remove_reference_t<T>>>;
25};
26template<template<class> class Forwarder, class T>
27using make_forwarder_t = typename make_forwarder<Forwarder,T>::type;

29//
30// ある型Tから「CV修飾子又は参照修飾子の異なるT」以外への暗黙的な変換を禁止するためのラッパー。
31// 大抵の型と変換できてしまう型(nl::json等)を経由して想定外のテンプレート実体化が成立してしまうのを抑制するためのSFINAEやconcept制約に用いる。
32//
33template<class T>
34struct NoConvertForwarder : ForwarderBase {
35 using RawType = std::remove_cv_t<std::remove_reference_t<T>>;
36 using StoredType = std::conditional_t<std::is_reference_v<T>,T,std::decay_t<T>>;
37
38 StoredType value;
39
40 explicit NoConvertForwarder(T value_) : value(value_){}
41
42 operator StoredType& () & {return value;}
43 operator const StoredType& () const & {return value;}
44 operator StoredType&& () && {return value;}
45 operator const StoredType&& () const && {return value;}
46
47 template<class Dest=StoredType, std::enable_if_t<!std::is_reference_v<Dest>, nullptr_t> = nullptr>
48 Dest* operator& (){return std::addressof(value);}
49 template<class Dest=StoredType, std::enable_if_t<!std::is_reference_v<Dest>, nullptr_t> = nullptr>
50 const Dest* operator& () const {return std::addressof(value);}
51};
