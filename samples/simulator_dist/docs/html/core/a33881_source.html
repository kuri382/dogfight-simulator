119void Reward::setupTarget(const std::string& query){
120 std::regex re;
121 if(query=="All"){
122 //まずAgent単位の報酬は常に保持する。
123 for(auto &&e:manager->getAgents()){
124 auto agent=getShared(e);
125 std::string agentFullName=agent->getFullName();
126 if(reward.find(agentFullName)==reward.end()){
127 if(!isTeamReward){
128 target.push_back(agentFullName);
129 }
130 reward[agentFullName]=0.0;
131 totalReward[agentFullName]=0.0;
132 }
133 }
134 //TeamRewardの場合は追加で陣営単位の報酬も保持する。
135 if(isTeamReward){
136 for(auto &&team:manager->getTeams()){
137 if(reward.find(team)==reward.end()){
138 target.push_back(team);
139 reward[team]=0.0;
140 totalReward[team]=0.0;
141 }
142 }
143 }
144 }else{
145 if(query.find("Team:")==0){
146 re=std::regex(query.substr(5));
147 //まずAgent単位の報酬は常に保持する。
148 for(auto &&e:manager->getAgents()){
149 auto agent=getShared(e);
150 std::string team=agent->getTeam();
151 std::string agentFullName=agent->getFullName();
152 if(std::regex_match(team,re) && reward.find(agentFullName)==reward.end()){
153 if(!isTeamReward){
154 target.push_back(agentFullName);
155 }
156 reward[agentFullName]=0.0;
157 totalReward[agentFullName]=0.0;
158 }
159 }
160 //TeamRewardの場合は追加で陣営単位の報酬も保持する。
161 if(isTeamReward){
162 for(auto &&team:manager->getTeams()){
163 if(std::regex_match(team,re) && reward.find(team)==reward.end()){
164 target.push_back(team);
165 reward[team]=0.0;
166 totalReward[team]=0.0;
167 }
168 }
169 }
170 }else if(query.find("Agent:")==0){
171 re=std::regex(query.substr(6));
172 //まずAgent単位の報酬は常に保持する。
173 for(auto &&e:manager->getAgents()){
174 auto agent=getShared(e);
175 std::string agentFullName=agent->getFullName();
176 if(std::regex_match(agentFullName,re) && reward.find(agentFullName)==reward.end()){
177 if(!isTeamReward){
178 target.push_back(agentFullName);
179 }
180 reward[agentFullName]=0.0;
181 totalReward[agentFullName]=0.0;
182 }
183 }
184 //TeamRewardの場合は追加で陣営単位の報酬も保持する。
185 if(isTeamReward){
186 for(auto &&e:manager->getAgents()){
187 auto agent=getShared(e);
188 std::string team=agent->getTeam();
189 std::string agentFullName=agent->getFullName();
190 if(std::regex_match(agentFullName,re) && reward.find(team)==reward.end()){
191 target.push_back(team);
192 reward[team]=0.0;
193 totalReward[team]=0.0;
194 }
195 }
196 }
197 }else{
198 throw std::runtime_error("Invalid reward target designation. Use either \"All\", \"Team:REGEX_PATTERN\", or \"Agent:REGEX_PATTERN\". The given query is \""+query+"\"");
199 }
200 }
