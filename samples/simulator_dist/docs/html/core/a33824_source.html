80void EkkerMissile::calcMotion(double tAftLaunch,double dt){
81 motion_prev=motion;
82 double V=motion.vel.norm();
83 auto propNav=getShared<PropNav>(controllers["Navigator"]);
84 Eigen::Vector3d accelCmd=propNav->commands["accel"];
85 Eigen::Vector3d omegaV=propNav->commands["omega"];
86 double omega=omegaV.norm();
87 double omegaMax=maxLoadG*gravity/V;
88 if(omega>omegaMax){
89 omegaV*=(omegaMax/omega);
90 omega=omegaMax;
91 accelCmd=omegaV.cross(motion.vel());
92 }
93 double alt=motion.getHeight();
94 bool pwroff=tAftLaunch>=tBurn;
95 double thrust_;
96 if(pwroff){
97 thrust_ = 0.0;
98 mass = massF;
99 lcg = lcgf;
100 }else{
101 thrust_ = thrust;
102 mass = massI+(massF-massI)*(tAftLaunch/tBurn-1);
103 lcg = lcgi+(lcgf-lcgi)*(tAftLaunch/tBurn-1);
104 }
105 std::map<std::string,double> atm=atmosphere(alt);
106 double M=V/atm["a"];
107 double qd=0.5*atm["rho"]*V*V;
108 Eigen::Vector3d ex=motion.vel()/V;
109 Eigen::Vector3d gravityAccelVector=motion.dirHtoP(Eigen::Vector3d(0,0,1),"NED")*gravity; //TODO 慣性力モデルに置換
110 Eigen::Vector3d ey=(accelCmd-gravityAccelVector).cross(ex)*mass;
111 double desiredSideForce=ey.norm();
112 Eigen::Vector3d sideAx;
113 double aoa,CL,CD;
114 if(desiredSideForce>0){
115 ey/=desiredSideForce;
116 sideAx=ex.cross(ey);
117 std::pair<double,double> tmp=EkkerMDT::cldcmd(d,l,ln,lcg,bt,St,M);
118 double cld=tmp.first;
119 double cmd=tmp.second;
120 auto func=[&](double aoa_){
121 std::pair<double,double> tmp=EkkerMDT::clacma(d,l,ln,lcg,bw,Sw,lw,bt,St,M,aoa_);
122 double cla=tmp.first;
123 double cma=tmp.second;
124 double delta=std::clamp<double>(-cma/cmd*aoa_,-maxD,maxD);
125 return (cla*aoa_+cld*delta)*(qd*Sref)+thrust_*sin(aoa_)-desiredSideForce;
126 };
127 if(func(0.0)>=0){
128 aoa=0.0;
129 }else if(func(maxA)<=0){
130 aoa=maxA;
131 }else{
132 try{
133 boost::math::tools::eps_tolerance<double> tol(16);
134 boost::uintmax_t max_iter=10;
135 auto result=boost::math::tools::toms748_solve(func,0.0,maxA,tol,max_iter);
136 aoa=(result.first+result.second)/2;
137 }catch(std::exception& e){
138 std::cout<<"exception at EkkerMissile::calcMotion"<<std::endl;
139 DEBUG_PRINT_EXPRESSION(func(0.0))
140 DEBUG_PRINT_EXPRESSION(func(maxA))
141 DEBUG_PRINT_EXPRESSION(alt)
142 DEBUG_PRINT_EXPRESSION(M)
143 DEBUG_PRINT_EXPRESSION(desiredSideForce)
144 DEBUG_PRINT_EXPRESSION(thrust_)
145 throw e;
146 }
147 }
148 tmp=EkkerMDT::clacma(d,l,ln,lcg,bw,Sw,lw,bt,St,M,aoa);
149 double cla=tmp.first;
150 double cma=tmp.second;
151 double delta=std::clamp<double>(-cma/cmd*aoa,-maxD,maxD);
152 //CL=(cla-cld*cma/cmd)*aoa;
153 CL=cla*aoa+cld*delta;
154 CD=EkkerMDT::cdtrim(d,l,ln,bw,Sw,thicknessRatio,bt,St,thicknessRatio,M,alt,aoa,delta,frictionFactor,pwroff);
155 }else{
156 sideAx<<0,0,0;
157 CL=0.0;
158 CD=EkkerMDT::cdtrim(d,l,ln,bw,Sw,thicknessRatio,bt,St,thicknessRatio,M,alt,0,0,frictionFactor,pwroff);
159 aoa=0.0;
160 }
161 Eigen::Vector3d sideAccel=sideAx*(CL*qd*Sref+thrust_*sin(aoa))/mass+(gravityAccelVector-ex*(gravityAccelVector.dot(ex)));
162 //attitude and deflection are assumed to be immidiately adjustable as desired.
163 accelScalar=(-CD*Sref*qd+thrust_*cos(aoa))/mass+gravityAccelVector.dot(ex);
164 Eigen::Vector3d accel=ex*accelScalar;
165 omegaV=motion.vel().cross(sideAccel)/(V*V);
166 omega=omegaV.norm();
167 if(omega*dt<1e-8){//ほぼ無回転
168 motion.setPos(motion.pos()+motion.vel()*dt+accel*dt*dt/2.);
169 motion.setVel(motion.vel()+accel*dt);
170 }else{
171 Eigen::Vector3d ex0=motion.vel()/V;
172 Eigen::Vector3d ey0=omegaV.cross(ex0).normalized();
173 double wt=omega*dt;
174 double acc=accelScalar;
175 Eigen::Vector3d vn=(ex0*cos(wt)+ey0*sin(wt)).normalized();
176 motion.setVel((V+acc*dt)*vn);
177 Eigen::Vector3d dr=V*dt*(ex0*sinc(wt)+ey0*oneMinusCos_x2(wt)*wt)+acc*dt*dt*(ex0*(sinc(wt)-oneMinusCos_x2(wt))+ey0*wt*(sincMinusOne_x2(wt)+oneMinusCos_x2(wt)));
178 motion.setPos(motion.pos()+dr);
179 accel=(motion.vel()-vel_prev())/dt;
180 }
181 motion.setOmega(omegaV);
182 calcQ();
183}
184bool EkkerMissile::hitCheck(const Eigen::Vector3d &tpos,const Eigen::Vector3d &tpos_prev){
185 Eigen::Vector3d r1=pos_prev();
186 Eigen::Vector3d r2=tpos_prev;
187 Eigen::Vector3d d1=pos()-r1;
188 Eigen::Vector3d d2=tpos-r2;
189 Eigen::Vector3d A=r1-r2;double a=A.norm();
190 Eigen::Vector3d B=d1-d2;double b=B.norm();
191 if(a<hitD){//初期位置であたり
192 return true;
193 }else if(b<1e-8){//相対速度がほぼ0}
194 return a<hitD;
195 }
196 double tMin=std::min(1.,std::max(0.,-2*A.dot(B)/(b*b)));
197 return (A+B*tMin).norm()<hitD;
198}
199bool EkkerMissile::checkDeactivateCondition(double tAftLaunch){
200 //最大飛翔時間超過or地面に激突or燃焼後に速度が規定値未満

203 (tAftLaunch>=tBurn && motion.vel.norm()<minV)
204 );
