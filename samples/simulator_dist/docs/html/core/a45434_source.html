39PyObjectTypeKind get_type_kind_of_type(const py::type& cls){
40 if(PyObject_IsSubclass(cls.ptr(), py::type::of<nl::json>().ptr())){
41 return PyObjectTypeKind::nl_json;
42 }else if(PyObject_IsSubclass(cls.ptr(), py::type::of<nl::ordered_json>().ptr())){
43 return PyObjectTypeKind::nl_ordered_json;
44 }else if(
45 py::hasattr(cls,"_allow_cereal_serialization_in_cpp")
46 && cls.attr("_allow_cereal_serialization_in_cpp").cast<bool>()
47 ){
48 return PyObjectTypeKind::cpp_type;
49 }else if(cls.ptr() == py::type::of(py::none()).ptr()){
50 return PyObjectTypeKind::none;
51 }else if(PyTypeInfo::isBoolean(cls)){
52 return PyObjectTypeKind::boolean;
53 }else if(PyTypeInfo::isIntegral(cls)){
54 return PyObjectTypeKind::integer;
55 }else if(PyTypeInfo::isReal(cls)){
56 return PyObjectTypeKind::float_;
57 }else if(PyObject_IsSubclass(cls.ptr(), (PyObject*)(&PyByteArray_Type))){
58 return PyObjectTypeKind::bytearray;
59 }else if(PyObject_IsSubclass(cls.ptr(), (PyObject*)(&PyBytes_Type))){
60 return PyObjectTypeKind::bytes;
61 }else if(PyObject_IsSubclass(cls.ptr(), (PyObject*)(&PyUnicode_Type))){
62 return PyObjectTypeKind::string;
63 }else if(PyTypeInfo::isMutableSet(cls)){
64 return PyObjectTypeKind::mutable_set;
65 }else if(PyTypeInfo::isSet(cls)){
66 return PyObjectTypeKind::set;
67 }else if(PyTypeInfo::isMutableSequence(cls)){
68 return PyObjectTypeKind::mutable_sequence;
69 }else if(PyTypeInfo::isSequence(cls)){
70 return PyObjectTypeKind::sequence;
71 }else if(PyTypeInfo::isMutableMapping(cls)){
72 return PyObjectTypeKind::mutable_mapping;
73 }else if(PyTypeInfo::isMapping(cls)){
74 return PyObjectTypeKind::mapping;
75 }else if(PyObject_IsSubclass(cls.ptr(), (PyObject*)(py::detail::npy_api::get().PyArray_Type_))){
76 return PyObjectTypeKind::numpy_array;
77 }else{
78 return PyObjectTypeKind::other;
79 }
80}
