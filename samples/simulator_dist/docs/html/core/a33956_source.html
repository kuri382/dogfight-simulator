530Eigen::Vector3d Fighter::toEulerAngle(){
531 Eigen::Vector3d ex=motion.relBtoH(Eigen::Vector3d(1,0,0),"NED");
532 Eigen::Vector3d ey=motion.relBtoH(Eigen::Vector3d(0,1,0),"NED");
533 Eigen::Vector3d horizontalY=Eigen::Vector3d(0,0,1).cross(ex).normalized();
534 double sinRoll=horizontalY.cross(ey).dot(ex);
535 double cosRoll=horizontalY.dot(ey);
536 double rollAtt=atan2(sinRoll,cosRoll);
537 return Eigen::Vector3d(rollAtt,motion.getEL(),motion.getAZ());
