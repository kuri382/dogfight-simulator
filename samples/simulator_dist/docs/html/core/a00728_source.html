22class NLJSONOutputArchive : public cereal::OutputArchive<NLJSONOutputArchive>, public cereal::traits::TextArchive{
23 // nl::ordered_json variant for cereal::JSONOutputArchive
24 enum class NodeType { StartObject, InObject, StartArray, InArray };
25 public:
26 NLJSONOutputArchive(nl::ordered_json& j_, bool discardRootNameValuePair_=false) :
27 cereal::OutputArchive<NLJSONOutputArchive>(this),
28 discardRootNameValuePair(discardRootNameValuePair_),
29 rootRef(j_),
30 itsNextName(nullptr)
31 {
32 rootRef=nl::ordered_json();
33 itsNodeRefStack.push(&rootRef);
34 itsNameCounter.push(0);
35 if(discardRootNameValuePair){
36 itsNodeStack.push(NodeType::InObject);
37 }else{
38 itsNodeStack.push(NodeType::StartObject);
39 }
40 }
41 ~NLJSONOutputArchive()=default;
42
43 void saveBinaryValue( const void * data, size_t size, const char * name = nullptr ){
44 setNextName( name );
45 writeName();
46
47 auto base64string = cereal::base64::encode( reinterpret_cast<const unsigned char *>( data ), size );
48 saveValue( base64string );
49 };
50
51 void startNode(){
52 writeName();
53 itsNodeStack.push(NodeType::StartObject);
54 itsNameCounter.push(0);
55 }
56 void finishNode(){
57 nl::ordered_json* currentNode = itsNodeRefStack.top();
58 if(discardRootNameValuePair && itsNodeStack.size()==1){
59 return;
60 }
61 switch(itsNodeStack.top()){
62 case NodeType::StartArray:
63 (*currentNode)=nl::ordered_json::array();
64 // fall through
65 case NodeType::InArray:
66 itsNodeRefStack.pop();
67 break;
68 case NodeType::StartObject:
69 (*currentNode)=nl::ordered_json::object();
70 // fall through
71 case NodeType::InObject:
72 itsNodeRefStack.pop();
73 break;
74 }
75 itsNodeStack.pop();
76 itsNameCounter.pop();
77 }
78 void setNextName( const char * name ){
79 itsNextName = name;
80 }
81 template<typename T>
82 void saveValue(const T& t){
83 nl::ordered_json* currentNode = itsNodeRefStack.top();
84 (*currentNode)=t;
85 if(discardRootNameValuePair && itsNodeRefStack.size()==1){
86 return;
87 }
88 itsNodeRefStack.pop();
89 }
90 void writeName(){
91 NodeType const & nodeType = itsNodeStack.top();
92 nl::ordered_json* currentNode = itsNodeRefStack.top();
93 if(nodeType == NodeType::StartArray){
94 (*currentNode)=nl::ordered_json::array();
95 itsNodeStack.top() = NodeType::InArray;
96 }else if(nodeType == NodeType::StartObject){
97 (*currentNode)=nl::ordered_json::object();
98 itsNodeStack.top() = NodeType::InObject;
99 }
100 if(discardRootNameValuePair && currentNode->is_null()){
101 return;
102 }
103 if(currentNode->is_object()){
104 if(itsNextName == nullptr){
105 std::string name = "value" + std::to_string( itsNameCounter.top()++ );
106 (*currentNode)[name]=nl::ordered_json();
107 itsNodeRefStack.push(&currentNode->at(name));
108 }else{
109 (*currentNode)[itsNextName]=nl::ordered_json();
110 itsNodeRefStack.push(&currentNode->at(itsNextName));
111 itsNextName = nullptr;
112 }
113 }else if(currentNode->is_array()){
114 currentNode->push_back(nl::ordered_json());
115 itsNodeRefStack.push(&currentNode->at(currentNode->size()-1));
116 }else{
117 throw cereal::Exception("Current node is neither array nor object");
118 }
119 }
120 void makeArray(){
121 itsNodeStack.top() = NodeType::StartArray;
122 }
123 const nl::ordered_json& getCurrentNode(){
124 // for debug purpose only
125 return *itsNodeRefStack.top();
126 }
127 private:
128 bool discardRootNameValuePair;
129 nl::ordered_json& rootRef; //root node reference
130
131 const char* itsNextName;
132 std::stack<std::uint32_t> itsNameCounter;
133 std::stack<nl::ordered_json*> itsNodeRefStack;
134 std::stack<NodeType> itsNodeStack;
135};
136class NLJSONInputArchive : public cereal::InputArchive<NLJSONInputArchive>, public cereal::traits::TextArchive{
137 // nl::ordered_json variant for cereal::JSONInputArchive
138 public:
139 NLJSONInputArchive(const nl::ordered_json& j, bool discardRootNameValuePair_=false) :
140 cereal::InputArchive<NLJSONInputArchive>(this),
141 discardRootNameValuePair(discardRootNameValuePair_),
142 rootRef(j),
143 itsNextName(nullptr)
144 {
145 itsNameCounter.push(0);
146 itsIndexCounter.push(0);
147 itsNodeRefStack.push(&rootRef);
148 }
149 ~NLJSONInputArchive()=default;
150
151 void loadBinaryValue( void * data, size_t size, const char * name = nullptr ){
152 itsNextName = name;
153
154 std::string encoded;
155 loadValue( encoded );
156 auto decoded = cereal::base64::decode( encoded );
157
158 if( size != decoded.size() )
159 throw cereal::Exception("Decoded binary data size does not match specified size");
160
161 std::memcpy( data, decoded.data(), decoded.size() );
162 itsNextName = nullptr;
163 };
164
165 void startNode(){
166 const nl::ordered_json* currentNode = itsNodeRefStack.top();
167 if(discardRootNameValuePair && itsNodeRefStack.size()==1){
168 itsNodeRefStack.push(currentNode);
169 itsNextName = nullptr;
170 itsNameCounter.push(0);
171 itsIndexCounter.push(0);
172 return;
173 }
174 if(currentNode->is_object()){
175 if(itsNextName == nullptr){
176 std::string name = "value" + std::to_string( itsNameCounter.top()++ );
177 itsNodeRefStack.push(&currentNode->at(name));
178 }else{
179 itsNodeRefStack.push(&currentNode->at(itsNextName));
180 itsNextName = nullptr;
181 }
182 }else{
183 itsNodeRefStack.push(&currentNode->at(itsIndexCounter.top()++));
184 }
185 itsNameCounter.push(0);
186 itsIndexCounter.push(0);
187 }
188
189 void finishNode(){
190 itsNodeRefStack.pop();
191 itsNameCounter.pop();
192 itsIndexCounter.pop();
193 }
194
195 void setNextName( const char * name ){
196 itsNextName = name;
197 }
198
199 template<typename T>
200 void loadValue(T& t){
201 const nl::ordered_json* currentNode = itsNodeRefStack.top();
202 if(discardRootNameValuePair && itsNodeRefStack.size()==1){
203 currentNode->get_to(t);
204 }else{
205 if(currentNode->is_object()){
206 if(itsNextName == nullptr){
207 std::string name = "value" + std::to_string( itsNameCounter.top()++ );
208 currentNode->at(name).get_to(t);
209 }else{
210 currentNode->at(itsNextName).get_to(t);
211 itsNextName = nullptr;
212 }
213 }else if(currentNode->is_array()){
214 currentNode->at(itsIndexCounter.top()++).get_to(t);
215 }else{
216 currentNode->get_to(t);
217 }
218 }
219 }
220 const nl::ordered_json& getCurrentNode(){
221 // 今いる階層のノードを返す。
222 return *itsNodeRefStack.top();
223 }
224 const nl::ordered_json& getNextNode(){
225 // 次に読み込まれるノードを返す。 (for internal use)
226 const nl::ordered_json* currentNode = itsNodeRefStack.top();
227 if(discardRootNameValuePair && itsNodeRefStack.size()==1){
228 return *itsNodeRefStack.top();
229 }else{
230 if(currentNode->is_object()){
231 if(itsNextName == nullptr){
232 std::string name = "value" + std::to_string( itsNameCounter.top()++ );
233 return currentNode->at(name);
234 }else{
235 return currentNode->at(itsNextName);
236 }
237 }else if(currentNode->is_array()){
238 return currentNode->at(itsIndexCounter.top());
239 }else{
240 return *currentNode;
241 }
242 }
243 }
244 bool contains(const typename nl::ordered_json::object_t::key_type& key){
245 const nl::ordered_json* currentNode = itsNodeRefStack.top();
246 if(currentNode->is_object()){
247 return currentNode->contains(key);
248 }else{
249 return false;
250 }
251 }
252
253 //! Loads the size for a SizeTag
254 void loadSize(cereal::size_type & size){
255 const nl::ordered_json* currentNode = itsNodeRefStack.top();
256 if(currentNode->is_array()){
257 size = currentNode->size();
258 }else{
259 throw std::runtime_error("cereal::SizeTag can only be loaded from a json array.");
260 }
261 }
262
263 void checkNull(){
264 // explicitなnullptr_tは値がnullかどうかをチェックする
265 const nl::ordered_json* currentNode = itsNodeRefStack.top();
266 const nl::ordered_json* nodeToBeChecked;
267 if(discardRootNameValuePair && itsNodeRefStack.size()==1){
268 nodeToBeChecked = currentNode;
269 }else{
270 if(currentNode->is_object()){
271 if(itsNextName == nullptr){
272 std::string name = "value" + std::to_string( itsNameCounter.top()++ );
273 nodeToBeChecked = &(currentNode->at(name));
274 }else{
275 nodeToBeChecked = &(currentNode->at(itsNextName));
276 itsNextName = nullptr;
277 }
278 }else if(currentNode->is_array()){
279 nodeToBeChecked = &(currentNode->at(itsIndexCounter.top()++));
280 }else{
281 nodeToBeChecked = currentNode;
282 }
283 }
284 if(!nodeToBeChecked->is_null()){
285 throw std::runtime_error("non-null value encountered at explicit std::nullptr_t loading.");
286 }
287 }
288
289 private:
290 bool discardRootNameValuePair;
291 const nl::ordered_json& rootRef; //root node reference
292
293 const char * itsNextName; //!< Next name set by NVP
294 std::stack<std::uint32_t> itsNameCounter;
295 std::stack<std::uint32_t> itsIndexCounter;
296 std::stack<const nl::ordered_json*> itsNodeRefStack;
297};

613// serializing string compatible char pointers
614// These functions disable C style array serialization in "cereal/types/commons.h".
615template <traits::string_literal T>
616inline void CEREAL_SAVE_FUNCTION_NAME(NLJSONOutputArchive & ar, const T& t)
617{
618 // 文字列リテラルはここに落ちる。nl::ordered_jsonを初期化子リストで初期化するときに文字列定数をリテラルで直接書いた場合等が該当する。
619 using CharT = traits::get_true_value_type_t<std::decay_t<T>>;
620 CEREAL_SAVE_FUNCTION_NAME(ar, std::basic_string(t));
621}
622template <traits::string_literal T>
623inline void CEREAL_LOAD_FUNCTION_NAME(NLJSONInputArchive & ar, T& t)
624{
625 // 非constのchar[N]型変数に読み込みたいという需要はあまり無いと思われるが、
626 // 一応用意しておく。
627 using CharT = traits::get_true_value_type_t<std::decay_t<T>>;
628 std::basic_string<CharT> str;
629 CEREAL_LOAD_FUNCTION_NAME(ar, str);
630 int i=0;
631 for(auto& v: t){
632 v=str[i];
633 ++i;
634 }
635}
