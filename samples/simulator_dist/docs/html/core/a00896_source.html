18// https://stackoverflow.com/questions/49904809/
19// ↑をC++20のconceptで書き直したもの
20template<typename T>
21concept dereferenceable = requires (T& t) {
22 *t;
23};
24template<typename T>
25struct is_dereferenceable : std::bool_constant<dereferenceable<T>> {};

27template<typename T>
28concept arrow_dereferenceable = requires (T& t) {
29 t.operator->();
30};
31template<typename T>
32struct is_arrow_dereferenceable : std::bool_constant<arrow_dereferenceable<T>> {};

34template<typename T>
35concept pointer = std::is_pointer<T>::value;

37template<typename T>
38struct is_pointer_like_dereferenceable : is_dereferenceable<T> {};
39template<typename T>
40concept pointer_like_dereferenceable = dereferenceable<T>;
41
42template<typename T>
43struct is_pointer_like_arrow_dereferenceable : std::disjunction<
44 std::is_pointer<T>,
45 is_arrow_dereferenceable<T>
46> {};
47template<typename T>
48concept pointer_like_arrow_dereferenceable = is_pointer_like_arrow_dereferenceable<T>::value;
49template<typename T>
50struct is_pointer_like : std::conjunction<
51 is_dereferenceable<T>,
52 is_arrow_dereferenceable<T>
53> {};
54template<typename T>
55concept pointer_like = pointer_like_dereferenceable<T> && pointer_like_arrow_dereferenceable<T>;

58// dereference<T>
59// pointer-like型の参照先の値を得る。
60// operator*による間接参照を行った後の型を返すものとする。
61// shared_ptr<T>等はTではなくT&を返すので、Tそのものを得たい場合は別途remove_referenceで囲む必要がある。
62template<typename T, bool = pointer_like<T>>
63struct dereference {};
64template<typename T>
65struct dereference<T,true> {
66 using type = decltype(*std::declval<T>());
67};
68template<typename T>
69struct dereference<T,false> {
70 using type = T;
71};
72template<typename T> using dereference_t = typename dereference<T>::type;

74//
75// get_true_value_type<T>
76// ポインタや参照型に対し、その参照先の値の型(CV修飾なし)を得る。
77// 再帰処理は行わないので入れ子になった型(ダブルポインタ等)は最外の参照のみ外される。
78template<typename T>
79struct get_true_value_type {
80 using type = std::remove_cv_t<std::remove_reference_t<dereference_t<T>>>;
81};
82template<typename T> using get_true_value_type_t = typename get_true_value_type<T>::type;
