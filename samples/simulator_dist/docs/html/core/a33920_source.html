60 if(!isValid()){validate();}

94Quaternion AffineCRS::getQuaternionTo(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& dstCRS) const{
95 if(!isValid()){validate();}
96 if(!dstCRS){
97 throw std::runtime_error("dstCRS is nullptr!");
98 }
99 if(this==std::dynamic_pointer_cast<AffineCRS>(dstCRS).get()){
100 return Quaternion(1,0,0,0);
101 }
102 if(!isTransformableTo(dstCRS)){
103 throw std::runtime_error(getFactoryClassName()+" can't be transformed to "+dstCRS->getFactoryClassName());
104 }
105 // MotionState::qは左からqをかけてderived座標をbase座標に移す
106 // v_base = q.transformVector(v_derived)
107 auto qToBaseFromThis=internalMotionState.q;
108 if(getBaseCRS()==dstCRS){
109 return qToBaseFromThis;
110 }else{
111 auto locationInBase=transformToBaseCRS(location,location,time,CoordinateType::POSITION_ABS)();
112 auto qToDstFromBase=getBaseCRS()->getQuaternionTo(locationInBase,time,dstCRS);
113 return qToDstFromBase*qToBaseFromThis;
114 }
115}
116Quaternion AffineCRS::getQuaternionFrom(const Eigen::Vector3d& location, const Time& time, const std::shared_ptr<CoordinateReferenceSystem>& srcCRS) const{
117 if(!isValid()){validate();}
118 if(!srcCRS){
119 throw std::runtime_error("srcCRS is nullptr!");
120 }
121 if(this==std::dynamic_pointer_cast<AffineCRS>(srcCRS).get()){
122 return Quaternion(1,0,0,0);
123 }
124 if(!isTransformableFrom(srcCRS)){
125 throw std::runtime_error(getFactoryClassName()+" can't be transformed from "+srcCRS->getFactoryClassName());
126 }
127 // MotionState::qは左からqをかけてderived座標をbase座標に移す
128 // v_base = q.transformVector(v_derived)
129 auto qToThisFromBase=internalMotionState.q.conjugate();
130 if(getBaseCRS()==srcCRS){
131 return qToThisFromBase;
132 }else{
133 auto qToBaseFromSrc=getBaseCRS()->getQuaternionFrom(location,time,srcCRS);
134 return qToThisFromBase*qToBaseFromSrc;
135 }

240Eigen::Vector3d AffineCRS::cartesianToSpherical(const Eigen::Vector3d& v) const{
241 // axisOrder -> FSD -> (Azimuth Elevation Range)
242 auto fsd=getAxisSwapQuaternion(getAxisOrder(),"FSD").transformVector(v);
243 double xy=sqrt(fsd(0)*fsd(0)+fsd(1)*fsd(1));
244 if(xy==0){
245 return Eigen::Vector3d(
246 0,
247 fsd(2)>0 ? -M_PI_2 : M_PI_2,
248 abs(fsd(2))
249 );
250 }else{
251 double el=atan2(-fsd(2),xy);
252 return Eigen::Vector3d(
253 atan2(fsd(1),fsd(0)),
254 el,
255 sqrt(xy*xy+fsd(2)*fsd(2))
256 );
257 }
258}
259Eigen::Vector3d AffineCRS::sphericalToCartesian(const Eigen::Vector3d& v) const{
260 // (Azimuth Elevation Range) -> FSD -> axisOrder
261 Eigen::Vector3d fsd(
262 v(2)*cos(v(0))*cos(v(1)),
263 v(2)*sin(v(0))*cos(v(1)),
264 -v(2)*sin(v(1))
265 );
266 return getAxisSwapQuaternion("FSD",getAxisOrder()).transformVector(fsd);
267}

275Quaternion AffineCRS::getEquivalentQuaternionToTopocentricCRS(const Eigen::Vector3d& origin, const Time& time, const std::string& axisOrder_) const{
276 //CRSとして生成せずにクォータニオンとして得る。
277 std::string dstAxisOrder=axisOrder_;
278 std::transform(dstAxisOrder.begin(),dstAxisOrder.end(),dstAxisOrder.begin(),
279 [](unsigned char c){ return std::toupper(c);}
280 );
281 auto intermediateCRS=getIntermediateCRS();
282 Coordinate cartesianOriginInIntermediateCRS = transformToIntermediateCRS(origin,time,CoordinateType::POSITION_ABS);
283 auto qToIntermediateCRSFromSelf=getQuaternionFrom(cartesianOriginInIntermediateCRS).conjugate();// [intermediateCRS <- this]
284 if(checkTopocentricCartesianAxisOrder(dstAxisOrder)){
285 return (
286 intermediateCRS->getEquivalentQuaternionToTopocentricCRS(cartesianOriginInIntermediateCRS(),time,dstAxisOrder) * //[new Topo <- intermediateCRS]
287 qToIntermediateCRSFromSelf // [intermediateCRS <- this]
288 );
289 }else if(checkBodyCartesianAxisOrder(dstAxisOrder)){
290 // baseのFWD方向のベクトルを水平面に射影した方向を「前方」に相当する基準軸Fとした局所水平座標系
291 // 「左方」に相当するP=U×Fの3軸を導出する必要がある。
292 auto qToENUFromIntermediateCRS = intermediateCRS->getEquivalentQuaternionToTopocentricCRS(cartesianOriginInIntermediateCRS(),time,"ENU");//[ENU <- intermediateCRS]
293 Eigen::Vector3d fwdInBody=getAxisSwapQuaternion("FPU",getAxisOrder()).transformVector(Eigen::Vector3d(1,0,0));
294 Eigen::Vector3d fwdInIntermediateCRS=transformTo(fwdInBody,time,intermediateCRS,CoordinateType::DIRECTION)();
295 Eigen::Vector3d fwdInENU=qToENUFromIntermediateCRS.transformVector(fwdInIntermediateCRS);
296 double xy=sqrt(fwdInENU(0)*fwdInENU(0)+fwdInENU(1)*fwdInENU(1));
297 double az=0.0;
298 if(xy>0){
299 az=atan2(fwdInENU(1),fwdInENU(0));
300 }
301 fwdInENU=Eigen::Vector3d(cos(az),sin(az),0);
302 Eigen::Vector3d portInENU(-sin(az),cos(az),0);
303 Eigen::Vector3d upInENU(0,0,1);
304 auto qToENUFromFPU=Quaternion::fromBasis(fwdInENU,portInENU,upInENU); // [ENU <- FPU]
305 return ( //[dstAxisOrder <- this] =
306 getAxisSwapQuaternion("FPU",dstAxisOrder) * //[dstAxisOrder <- FPU] *
307 qToENUFromFPU.conjugate() * //[FPU <- ENU] *
308 qToENUFromIntermediateCRS * //[ENU <- intermediateCRS] *
309 qToIntermediateCRSFromSelf //[intermediateCRS <- this]
310 );
311 }else{
312 throw std::runtime_error("'"+axisOrder_+"' is not a valid axisOrder of TopocentricCRS.");
313 }
