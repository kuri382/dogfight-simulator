231void BasicAACRuler01::onCrash(const nl::json& args){
232 std::lock_guard<std::mutex> lock(mtx);
233 std::shared_ptr<PhysicalAsset> asset=args;
234 std::string team=asset->getTeam();
235 std::string modelName=asset->getFactoryModelName();
236 auto beg=deadFighters.begin();
237 auto end=deadFighters.end();
238 if(std::find(beg,end,asset->getFullName())==end){
239 if(isToBeConsideredForElimination(team,modelName)){
240 getCrashCount(team,modelName)+=1;
241 lastDownPosition[team]=forwardAx[team].dot(asset->pos(getLocalCRS()).block<2,1>(0,0,2,1));
242 lastDownReason[team]=DownReason::CRASH;
243 }
244 deadFighters.push_back(asset->getFullName());

282void BasicAACRuler01::onInnerStepEnd(){
283 //生存数の監視
284 std::map<std::string,int> aliveCount;
285 for(auto& team:teams){
286 aliveCount[team]=0;
287 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
288 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset) && isToBeConsideredForElimination(team,asset->getFactoryModelName());
289 })){
290 aliveCount[team]++;
291 }
292 if(aliveCount[team]==0 && eliminatedTime[team]<0){
293 eliminatedTime[team]=manager->getElapsedTime();
294 }
295 }
296 //防衛ラインの突破タイミングを監視
297 std::vector<double> tmp;
298 for(auto& team:teams){
299 tmp.clear();
300 tmp.push_back(-dLine);
301 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
302 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset) && isToBeConsideredForBreak(team,asset->getFactoryModelName());
303 })){
304 auto f=getShared<Fighter>(e);
305 tmp.push_back(forwardAx[team].dot(f->pos(getLocalCRS()).block<2,1>(0,0,2,1)));
306 }
307 leadRange[team]=*std::max_element(tmp.begin(),tmp.end());
308 if(leadRange[team]>=dLine && breakTime[team]<0){
309 breakTime[team]=manager->getElapsedTime();
310 //得点計算 2：突破による加点(pBreak点)
311 debugPrint("2. Break",team,pBreak);
312 stepScore[team]+=pBreak;
313 }
314 }
315
316 for(auto& team:teams){
317 //得点計算 1：撃墜による加点(1機あたりpHit点)
318 for(auto&& c:hitCount[team]){
319 debugPrint("1. Hit("+c.first+")",team,c.second*getPHit(team,c.first));
320 stepScore[team]+=c.second*getPHit(team,c.first);
321 }
322 //得点計算 6-(a)：墜落による減点(1機あたりpCrash点)
323 for(auto&& c:crashCount[team]){
324 debugPrint("6-(a). Crash("+c.first+")",team,-c.second*getPCrash(team,c.first));
325 stepScore[team]-=c.second*getPCrash(team,c.first);
326 }
327 //得点計算 6-(b)：場外に対する減点(1秒、1kmにつきpOut点)
328 outDist[team]=0;
329 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
330 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset);
331 })){
332 auto f=getShared<Fighter>(e);
333 double d=abs(sideAx[team].dot(f->pos(getLocalCRS()).block<2,1>(0,0,2,1)))-dOut;
334 if(applyDOutBeyondLine){
335 d=std::max(d,abs(forwardAx[team].dot(f->pos(getLocalCRS()).block<2,1>(0,0,2,1)))-dLine);
336 }
337 outDist[team]+=std::max(0.0,d);
338 }
339 if(outDist[team]>0.0){
340 debugPrint("6-(b). Out",team,-(outDist[team]/1000.)*pOut*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep());
341 }
342 stepScore[team]-=(outDist[team]/1000.)*pOut*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep();
343 if(score[team]+stepScore[team]<=pDisq && disqTime[team]<0){
344 disqTime[team]=manager->getElapsedTime();
345 }
346 }
347 if(endReasonSub==EndReason::NOTYET){
348 if(terminalAtElimination &&
349 (eliminatedTime[westSider]>=0 || eliminatedTime[eastSider]>=0)
350 ){
351 //終了条件(1)が有効な場合
352 if(!(enableAdditionalTime && checkHitPossibility())){
353 //相討ちを考慮する場合、以後の撃墜の可能性がある限り継続
354 endReasonSub=EndReason::ELIMINATION;
355 }
356 }
357 if(endReasonSub==EndReason::NOTYET){
358 if(terminalAtBreak &&
359 (breakTime[westSider]>=0 || breakTime[eastSider]>=0)
360 ){
361 //終了条件(2)が有効な場合
362 if(!(enableAdditionalTime && checkHitPossibility())){
363 //相討ちを考慮する場合、以後の撃墜の可能性がある限り継続
364 endReasonSub=EndReason::BREAK;
365 }
366 }
367 if(endReasonSub==EndReason::NOTYET){
368 if(disqTime[westSider]>=0 || disqTime[eastSider]>=0){
369 //終了条件(5)の判定
370 endReasonSub=EndReason::PENALTY;
371 }else{
372 //終了条件(3)の判定
373 bool withdrawal=manager->getElapsedTime()>=minTime;
374 for(auto& team:teams){
375 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
376 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset);
377 })){
378 auto f=getShared<Fighter>(e);
379 if(forwardAx[team].dot(f->pos(getLocalCRS()).block<2,1>(0,0,2,1))>=-dLine){
380 withdrawal=false;
381 break;
382 }
383 }
384 if(!withdrawal){
385 break;
386 }
387 }
388 if(withdrawal){
389 endReasonSub=EndReason::WITHDRAWAL;
390 }
391 }
392 }
393 }
394 }

396void BasicAACRuler01::checkDone(){
397 //終了判定
398 dones.clear();
399 for(auto&& e:manager->getAgents()){
400 auto a=getShared(e);
401 dones[a->getFullName()]=!a->isAlive();
402 }
403 bool considerAdvantage=false;
404 //終了条件(1)：全機撃墜or墜落
405 if(endReasonSub==EndReason::ELIMINATION){
406 endReason=EndReason::ELIMINATION;
407 if(breakTime[westSider]<0 && breakTime[eastSider]<0){
408 //両者未突破の場合、得点計算3及び5の考慮が必要
409 if(eliminatedTime[westSider]>=0 && eliminatedTime[eastSider]>=0){
410 //両者全滅した場合、得点計算5に従い優勢度を考慮
411 considerAdvantage=true;
412 for(auto&& t:teams){
413 leadRange[t]=lastDownPosition[t];
414 }
415 }else{
416 //一方が生存していた場合
417 //得点計算 3：未突破の陣営に「そこから突破して更に帰還可能な」突破判定対象機が存在していれば+pBreak点
418 if(eliminatedTime[eastSider]<0){//東が生存
419 bool chk=false;
420 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
421 return asset->isAlive() && asset->getTeam()==eastSider && isinstance<Fighter>(asset) && isToBeConsideredForBreak(eastSider,asset->getFactoryModelName());
422 })){
423 if(isBreakableAndReturnableToBase(e.lock())){
424 chk=true;
425 break;
426 }
427 }
428 if(chk){
429 debugPrint("3. Break",eastSider,pBreak);
430 stepScore[eastSider]+=pBreak;
431 score[eastSider]+=pBreak;
432 }
433 }else{//西が生存
434 bool chk=false;
435 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
436 return asset->isAlive() && asset->getTeam()==westSider && isinstance<Fighter>(asset) && isToBeConsideredForBreak(westSider,asset->getFactoryModelName());
437 })){
438 if(isBreakableAndReturnableToBase(e.lock())){
439 chk=true;
440 break;
441 }
442 }
443 if(chk){
444 debugPrint("3. Break",westSider,pBreak);
445 stepScore[westSider]+=pBreak;
446 score[westSider]+=pBreak;
447 }
448 }
449 }
450 }
451 }
452 //終了条件(2)：防衛ラインの突破
453 else if(endReasonSub==EndReason::BREAK){
454 endReason=EndReason::BREAK;
455 //追加の得点増減はなし
456 }
457 //終了条件(5)：ペナルティによる敗北
458 else if(endReasonSub==EndReason::PENALTY){
459 endReason=EndReason::PENALTY;
460 if(breakTime[westSider]<0 && breakTime[eastSider]<0){
461 //両者未突破の場合、得点計算5に従い優勢度を計算
462 considerAdvantage=true;
463 }
464 }
465 //終了条件(3)：両者撤退による打ち切り
466 else if(endReasonSub==EndReason::WITHDRAWAL){
467 endReason=EndReason::WITHDRAWAL;
468 //追加の得点増減はなし(優勢度も計算しない)
469 }
470 //終了条件(4)：時間切れ
471 else if(manager->getElapsedTime()>=maxTime){
472 endReason=EndReason::TIMEUP;
473 if(breakTime[westSider]<0 && breakTime[eastSider]<0){
474 //両者未突破の場合、得点計算5に従い優勢度を計算
475 considerAdvantage=true;
476 }
477 }
478 //得点計算 4：生存点(1機あたりpAlive点)
479 //得点計算 6(a)：帰還不可による墜落ペナルティ(1機あたりpCrash点)
480 if(endReason!=EndReason::NOTYET){
481 for(auto& team:teams){
482 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
483 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset) && isToBeConsideredForElimination(team,asset->getFactoryModelName());
484 })){
485 auto asset=e.lock();
486 if(isReturnableToBase(asset)){
487 //帰還可能なら生存点
488 debugPrint("4. Alive("+asset->getFactoryModelName()+")",team,getPAlive(team,asset->getFactoryModelName()));
489 stepScore[team]+=getPAlive(team,asset->getFactoryModelName());
490 score[team]+=getPAlive(team,asset->getFactoryModelName());
491 }else{
492 //帰還不可能なら墜落ペナルティ
493 debugPrint("6(a). NoReturn("+asset->getFactoryModelName()+")",team,-getPCrash(team,asset->getFactoryModelName()));
494 stepScore[team]-=getPCrash(team,asset->getFactoryModelName());
495 score[team]-=getPCrash(team,asset->getFactoryModelName());
496 }
497 }
498 }
499 }
500 if(considerAdvantage){
501 //終了条件(4)、(5)または両者全滅の場合で、両者未突破だった場合の優勢度加点
502 //得点計算 5：進出度合いに対する加点(1kmにつきpAdv点)
503 if(leadRange[westSider]>leadRange[eastSider]){
504 double s=(leadRange[westSider]-leadRange[eastSider])/2./1000.*pAdv;
505 debugPrint("5. Adv",westSider,s);
506 stepScore[westSider]+=s;
507 score[westSider]+=s;
508 }else{
509 double s=(leadRange[eastSider]-leadRange[westSider])/2./1000.*pAdv;
510 debugPrint("5. Adv",eastSider,s);
511 stepScore[eastSider]+=s;
512 score[eastSider]+=s;
513 }
514 }
515 //打ち切り対策で常時勝者を仮設定しておく
516 if(score[westSider]>score[eastSider]){
517 winner=westSider;
518 }else if(score[westSider]<score[eastSider]){
519 winner=eastSider;
520 }else{

660 //この先撃墜の可能性があるかどうかを返す
661 if(eliminatedTime.at(westSider)<0){
662 //西側が生存している場合、東側の航空機か誘導弾が残っていれば可能性ありとする
663 for(auto&& asset:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
664 return asset->isAlive() && asset->getTeam()==eastSider && isinstance<Fighter>(asset);
665 })){
666 return true;
667 }
668 for(auto&& asset:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
669 return asset->isAlive() && asset->getTeam()==eastSider && isinstance<Missile>(asset);
670 })){
671 auto msl=getShared<Missile>(asset);
672 if(msl->isAlive() && msl->hasLaunched){
673 return true;
674 }
675 }
676 }
677 if(eliminatedTime.at(eastSider)<0){
678 //東側が生存している場合、西側の航空機か誘導弾が残っていれば可能性ありとする
679 for(auto&& asset:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
680 return asset->isAlive() && asset->getTeam()==westSider && isinstance<Fighter>(asset);
681 })){
682 return true;
683 }
684 for(auto&& asset:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
685 return asset->isAlive() && asset->getTeam()==westSider && isinstance<Missile>(asset);
686 })){
687 auto msl=getShared<Missile>(asset);
688 if(msl->isAlive() && msl->hasLaunched){
689 return true;
690 }
691 }
692 }
693 return false;

695bool BasicAACRuler01::isReturnableToBase(const std::shared_ptr<PhysicalAsset>& asset) const{
696 //現在位置から帰還可能かどうか
697 if(!considerFuelConsumption){
698 return true;
699 }
700 auto fgtr=getShared<Fighter>(asset);
701 std::string team=fgtr->getTeam();
702 double range=fgtr->getMaxReachableRange()/(1+fuelMargin);//今の残燃料で到達できる距離
703 double distanceFromLine=forwardAx.at(team).dot(fgtr->pos(getLocalCRS()).block<2,1>(0,0,2,1))+dLine;
704 return distanceFromBase.at(team)+distanceFromLine<=range;
705}
706bool BasicAACRuler01::isBreakableAndReturnableToBase(const std::shared_ptr<PhysicalAsset>& asset) const{
707 //現在位置から防衛ラインを突破して更に帰還可能かどうか

715 return distanceFromBase.at(team)+4*dLine-distanceFromLine<=range;
