305void BasicAACRuleBasedAgent01::updateTrackInfo(TrackInfo& ti,bool isFull){
306 /*TrackInfoの付帯情報の更新を行う。
307 * 射程情報: 双方の距離,RmaxHead,Rmax1,Rmax2
308 * 追尾状況: 自機が追尾中か否か,誰かが追尾中か否か,追尾機数(IN,LIMIT,OUT) ただし、stateがEVADEまたはWITHDRAWの機体はINでもLIMIT扱いとする。
309 * isFullにTrueを入れたときはdeploy用に射程計算等を含む全てを計算し、Falseを入れたときは位置と速度だけ更新する。
310 */
311 Eigen::Vector3d tpos=ti.pos(getLocalCRS());
312 Eigen::Vector3d tvel=ti.vel(getLocalCRS());
313 Eigen::Vector3d dpos=pos-tpos;
314 double L=dpos.norm();
315 ti.distance=L;
316 if(isFull && ti.state!=TrackInfo::UpdateState::LOST){
317 ti.myRHead=calcRHead(pos,vel,tpos,tvel);
318 ti.myRTail=calcRTail(pos,vel,tpos,tvel);
319 ti.hisRHead=calcRHead(tpos,tvel,pos,vel);
320 ti.hisRTail=calcRTail(tpos,tvel,pos,vel);
321 Eigen::Vector3d tEx=tvel.normalized();
322 if(ti.state==TrackInfo::UpdateState::TRACK){
323 ti.trackers.clear();
324 for(auto& src:parent->observables.at("/sensor/trackSource"_json_pointer)[ti.idx]){
325 if(std::find(ti.trackers.begin(),ti.trackers.end(),src)==ti.trackers.end()){
326 ti.trackers.push_back(src);
327 }
328 }
329 ti.numTracker=ti.trackers.size();
330 ti.nonLimitTrackers.clear();
331 ti.limitTrackers.clear();
332 ti.numTrackerLimit=0;
333 ti.inOurSensor=TrackInfo::SensingState::OUTSIDE;
334 ti.inMySensor=TrackInfo::SensingState::OUTSIDE;
335 for(auto& e:ti.trackers){
336 const nl::json& fObs=e==parent->getFullName() ?
337 parent->observables :
338 parent->observables.at("/shared/fighter"_json_pointer).at(e);
339 const nl::json& aObs=parent->observables.at("/shared/agent"_json_pointer).at(e);
340 State aState=State::NONE;
341 try{
342 aObs.at("state").get_to(aState);
343 }catch(...){
344 }
345 Eigen::Vector3d tposS=MotionState(fObs.at("motion")).absAtoB(tpos,getLocalCRS());
346 double L=tposS.norm();
347 double angle=acos(std::max(-1.0,std::min(1.0,tposS(0)/L)));
348 double fLref,fAngle;
349 if(fObs.contains("/spec/sensor/radar"_json_pointer)){
350 fLref=fObs.at("/spec/sensor/radar/Lref"_json_pointer);
351 fAngle=fObs.at("/spec/sensor/radar/thetaFOR"_json_pointer);
352 }else{
353 fLref=0;
354 fAngle=0;
355 }
356 if(
357 L<=fLref-sensorInRangeLimit &&
358 angle<=fAngle-sensorInCoverageLimit &&
359 (!considerStateForTrackingCondition || (
360 aState!=State::EVADE &&
361 aState!=State::WITHDRAW &&
362 aState!=State::NONE
363 ))
364 ){
365 ti.inOurSensor=TrackInfo::SensingState::INSIDE;
366 ti.nonLimitTrackers.push_back(e);
367 if(e==parent->getFullName()){
368 ti.inMySensor=TrackInfo::SensingState::INSIDE;
369 }
370 }else if(L<=fLref && angle<=fAngle){
371 ti.limitTrackers.push_back(e);
372 ti.numTrackerLimit++;
373 if(ti.inOurSensor!=TrackInfo::SensingState::INSIDE){
374 ti.inOurSensor=TrackInfo::SensingState::LIMIT;
375 }
376 if(e==parent->getFullName()){
377 if(ti.inMySensor!=TrackInfo::SensingState::INSIDE){
378 ti.inMySensor=TrackInfo::SensingState::LIMIT;
379 }
380 }
381 }
382 }
383 }else{
384 ti.trackers.clear();
385 ti.nonLimitTrackers.clear();
386 ti.limitTrackers.clear();
387 ti.numTracker=0;
388 ti.numTrackerLimit=0;
389 ti.inOurSensor=TrackInfo::SensingState::OUTSIDE;
390 ti.inMySensor=TrackInfo::SensingState::OUTSIDE;
391 }
392 }

394void BasicAACRuleBasedAgent01::updateTargetStatus(bool isFull){
395 /*主目標と副目標を更新する。副目標は、前回の各機の主目標と、まだ誘導が必要な全ての誘導弾目標とし、副目標内でのtruthの重複はないようにする。
396 「誘導が必要」とは、誘導弾自身のセンサで捉えていない状態(mode!=SELF)を指すものとする。
397 なお、副目標には自機の主目標も含む(敢えて除外していない)
398 isFullにTrueを入れたときは副目標の再抽出も行い全て更新するが、Falseを入れたときはその時点で抽出されているものの位置と速度だけ更新する。
399 */
400 //主目標の更新
401 updateTargetStatusSub(target,isFull);
402 updateTargetStatusSub(launchedTarget,isFull);
403 //副目標の確定(isFullの時のみ)
404 if(isFull){
405 std::vector<bool> flag;//いらないものを削除するためのフラグ
406 for(auto& t:additionalTargets){
407 flag.push_back(false);
408 }
409 for(auto&& fr:parent->observables.at("/shared/fighter"_json_pointer).items()){
410 if(fr.value().at("isAlive")){
411 const nl::json& ag=parent->observables.at("/shared/agent"_json_pointer).at(fr.key());
412 TrackInfo tgt;
413 //主目標
414 if(ag.contains("target")){
415 tgt=TrackInfo(ag.at("target"),-1);
416 if(!tgt.is_none()){
417 bool ret=true;
418 int idx=0;
419 for(auto& aTgt:additionalTargets){
420 if(aTgt.isSame(tgt)){
421 ret=false;
422 flag[idx]=true;
423 break;
424 }
425 idx++;
426 }
427 if(ret){
428 additionalTargets.push_back(tgt);
429 flag.push_back(true);
430 }
431 }
432 }
433 //直近の射撃目標
434 if(ag.contains("launchedTarget")){
435 tgt=TrackInfo(ag.at("launchedTarget"),-1);
436 if(!tgt.is_none()){
437 int nextMsl=fr.value().at("/weapon/nextMsl"_json_pointer);
438 if(nextMsl>0){
439 const nl::json& msl=fr.value().at("/weapon/missiles"_json_pointer)[nextMsl-1];
440 if(msl.at("isAlive").get<bool>() && msl.at("mode").get<Missile::Mode>()!=Missile::Mode::SELF){//まだ誘導が必要
441 bool ret=true;
442 int idx=0;
443 for(auto& aTgt:additionalTargets){
444 if(aTgt.isSame(tgt)){
445 ret=false;
446 flag[idx]=true;
447 break;
448 }
449 idx++;
450 }
451 if(ret){
452 additionalTargets.push_back(tgt);
453 flag.push_back(true);
454 }
455 }
456 }
457 }
458 }
459 }
460 }
461 //まだ誘導が必要な誘導弾の目標は直近のでなくても残す
462 for(auto&& msl:parent->observables.at("/weapon/missiles"_json_pointer)){
463 if(msl.at("isAlive").get<bool>() && msl.at("hasLaunched").get<bool>() && msl.at("mode").get<Missile::Mode>()!=Missile::Mode::SELF){
464 int idx=0;
465 for(auto& aTgt:additionalTargets){
466 if(msl.at("target").get<Track3D>().isSame(aTgt)){
467 flag[idx]=true;
468 idx++;
469 }
470 }
471 }
472 }
473 for(auto&& fr:parent->observables.at("/shared/fighter"_json_pointer).items()){
474 if(fr.key()!=parent->getFullName()){
475 for(auto&& msl:fr.value().at("/weapon/missiles"_json_pointer)){
476 if(msl.at("isAlive").get<bool>() && msl.at("hasLaunched").get<bool>() && msl.at("mode").get<Missile::Mode>()!=Missile::Mode::SELF){
477 int idx=0;
478 for(auto& aTgt:additionalTargets){
479 if(msl.at("target").get<Track3D>().isSame(aTgt)){
480 flag[idx]=true;
481 idx++;
482 }
483 }
484 }
485 }
486 }
487 }
488 auto ait=additionalTargets.begin();
489 for(auto fit=flag.begin();fit!=flag.end();){
490 if((*fit) && (*ait).state!=TrackInfo::UpdateState::LOST){
491 ++fit;
492 ++ait;
493 }else{
494 fit=flag.erase(fit);
495 ait=additionalTargets.erase(ait);
496 }
497 }
498 }
499 //副目標の更新
500 for(auto& aTgt:additionalTargets){
501 updateTargetStatusSub(aTgt,isFull);
502 }
503 //行動判断の考慮対象(航跡＋メモリトラック中の主or副目標)の抽出
504 allEnemies.clear();
505 for(auto& t:track){
506 allEnemies.push_back(t);
507 }
508 for(auto& t:additionalTargets){
509 if(t.state==TrackInfo::UpdateState::MEMORY){
510 allEnemies.push_back(t);
511 }
512 }

514void BasicAACRuleBasedAgent01::updateTargetStatusSub(TrackInfo& tgt,bool isFull){
515 /*指定した目標情報を更新する。
516 */
517 if(!tgt.is_none()){
518 bool updated=false;
519 for(auto& t:track){//航跡情報から更新対象を検索
520 if(tgt.isSame(t)){
521 tgt.update(t,isFull);
522 updated=true;
523 break;
524 }
525 }
526 if(!updated){//見つからなかった場合、メモリトラック
527 if(tgt.state==TrackInfo::UpdateState::TRACK){
528 tgt.memoryStartTime=manager->getTime();
529 tgt.state=TrackInfo::UpdateState::MEMORY;
530 }else if(tgt.state==TrackInfo::UpdateState::MEMORY){
531 if(manager->getTime()>=tMaxMemory+tgt.memoryStartTime){
532 tgt.state=TrackInfo::UpdateState::LOST;
533 }
534 }
535 }
536 if(tgt.state!=TrackInfo::UpdateState::TRACK){//外挿
537 tgt.extrapolateTo(manager->getTime());
538 updateTrackInfo(tgt,isFull);
539 }

543 //余剰燃料のチェック

550 double fuelMargin=rulerObs.at("fuelMargin");

552 return maxReachableRange/(1+fuelMargin)-(distanceFromLine+distanceFromBase) >= RTBExitMargin;
553 }else{
554 return maxReachableRange/(1+fuelMargin)-(distanceFromLine+distanceFromBase) >= RTBEnterMargin;

568std::pair<bool,std::pair<TrackInfo,double>> BasicAACRuleBasedAgent01::chkConditionForShoot(TrackInfo& tgt){
569 /*(c1)射撃条件の判定
570 1. 射程による判定・・・その航跡に対してRtailを0、Rheadを1としたときに現在の距離がkShoot以下であること
571 2. 探知状況による判定・・・その航跡を自身か味方の少なくとも１機が余裕をもって捉えていること
572 3. 射撃状況による判定・・・その航跡に対して自身が発射した飛翔中の誘導弾がnMslSimul発未満であること
573 */
574 //条件0：機体側の制限も確認しておく
575 bool ret0=getShared<FighterAccessor>(parent)->isLaunchableAt(tgt);
576 //条件1：射程による判定
577 bool ret1=!tgt.is_none();
578 double r;
579 if(ret1){
580 r=tgt.distance-tgt.myRTail;
581 double delta=tgt.myRHead-tgt.myRTail;
582 if(delta==0){
583 if(r<0){
584 r=-std::numeric_limits<double>::infinity();
585 }else if(r>0){
586 r=std::numeric_limits<double>::infinity();
587 }else{
588 r=0;
589 }
590 }else{
591 r/=delta;
592 }
593 ret1=r<=kShoot;
594 }else{
595 r=0;
596 }
597 //条件2：探知状況による判定
598 bool ret2=false;
599 for(auto& f:tgt.nonLimitTrackers){
600 State aState=State::NONE;
601 try{
602 parent->observables.at("/shared/agent"_json_pointer).at(f).at("state").get_to(aState);
603 }catch(...){
604 }
605 if(!considerStateForTrackingCondition || (aState!=State::EVADE && aState!=State::WITHDRAW&& aState!=State::NONE)){
606 ret2=true;
607 break;
608 }
609 }
610 //条件3:射撃状況による判定
611 bool ret3=true;
612 int count=0;
613 for(auto&& msl:parent->observables.at("/weapon/missiles"_json_pointer)){
614 if(msl.at("isAlive").get<bool>() && msl.at("hasLaunched").get<bool>()){
615 if(msl.at("target").get<Track3D>().isSame(tgt)){
616 count++;
617 if(count>=nMslSimul){
618 ret3=false;
619 break;
620 }
621 }
622 }
623 }
624 return std::make_pair(ret0 && ret1 && ret2 && ret3,std::make_pair(tgt,r));
625}

672void BasicAACRuleBasedAgent01::selectTarget(){
673 /*主目標の選択を行う。候補は全航跡に加え、メモリトラック中の副目標(主目標も含む)とする
674 1. 自陣に十分近いか、敵陣側扇範囲内に捉えている味方がいないのいずれを満たす敵がいれば最も近い味方を割り振る。
675 パラメータはdPrioritizedAimLeaker,thetaPrioritizedAimLeakerの二つ
676 2. 距離の近い順に割り振る。
677 */
678 //まず、lostしてしまったものを削除
679 if(!target.is_none() && target.state==TrackInfo::UpdateState::LOST){
680 target=TrackInfo();
681 }
682 if(!launchedTarget.is_none() && launchedTarget.state==TrackInfo::UpdateState::LOST){
683 launchedTarget=TrackInfo();
684 }
685 std::vector<Eigen::Vector3d> fpos,fvel;
686 std::vector<bool> assigned;
687 assigned.push_back(false);
688 fpos.push_back(pos);
689 fvel.push_back(vel);
690 for(auto&& f:parent->observables.at("/shared/fighter"_json_pointer).items()){
691 if(f.value().at("isAlive").get<bool>()){
692 assigned.push_back(false);
693 MotionState mo(f.value().at("motion"));
694 fpos.push_back(mo.pos(getLocalCRS()));
695 fvel.push_back(mo.vel(getLocalCRS()));
696 }
697 }
698 std::size_t numF=fpos.size();
699 std::vector<TrackInfo*> candidates;
700 std::vector<Eigen::Vector3d> tpos;
701 for(auto& t:allEnemies){
702 if(t.state!=TrackInfo::UpdateState::LOST){
703 candidates.push_back(&t);
704 tpos.push_back(t.pos(getLocalCRS()));
705 }
706 }
707 isAssignedAsInterceptor=false;
708 if(candidates.size()==0){//何も見えていない場合
709 target=TrackInfo();
710 return;
711 }
712 //(1)突破阻止(複数該当した場合は、とりあえずラインに近い敵から順に、最も近い未割当の味方を割り当てる)
713 std::vector<std::pair<double,int>> leakers,nearestF;
714 Eigen::MatrixXd dist=Eigen::MatrixXd::Zero(fpos.size(),tpos.size());
715 int i=0;int j=0;
716 for(auto& tp:tpos){
717 double distFromLine=dLine-tp.dot(-forward);
718 bool isLeaker=distFromLine<dPrioritizedAimLeaker*dLine;
719 double angle=M_PI;
720 i=0;
721 for(auto& fp:fpos){
722 dist(i,j)=(fp-tp).norm();
723 angle=std::min(angle,
724 acos(std::max(-1.0,std::min(1.0,
725 (tp-fp).block<2,1>(0,0,2,1).normalized().dot(forward.block<2,1>(0,0,2,1))
726 )))
727 );
728 i++;
729 }
730 if(angle>thetaPrioritizedAimLeaker){
731 isLeaker=true;
732 }
733 if(isLeaker){
734 leakers.push_back(std::make_pair(distFromLine,j));
735 }
736 j++;
737 }
738 std::sort(leakers.begin(),leakers.end(),
739 [](const std::pair<double,int> &lhs,const std::pair<double,int> &rhs){
740 return lhs.first<rhs.first;
741 });
742 int numAssigned=0;
743 //typedef std::pair<double,int> viType;
744 for(auto& li:leakers){
745 nearestF.clear();
746 for(int i=0;i<fpos.size();++i){
747 nearestF.push_back(std::make_pair(dist(i,li.second),i));
748 }
749 std::sort(nearestF.begin(),nearestF.end(),
750 [](const std::pair<double,int> &lhs,const std::pair<double,int> &rhs){
751 return lhs.first<rhs.first;
752 });
753 for(auto& nf:nearestF){
754 if(!assigned[nf.second]){
755 numAssigned++;
756 if(nf.second==0){
757 target=*candidates[li.second];
758 isAssignedAsInterceptor=true;
759 }
760 break;
761 }
762 }
763 if(numAssigned==fpos.size()){
764 break;
765 }
766 }
767 //(2)最も近い敵(重複可)
768 std::vector<TrackInfo*> nearestE;
769 for(int i=0;i<fpos.size();++i){
770 int idx;
771 Eigen::VectorXd tmp=dist.block(i,0,1,tpos.size()).transpose();
772 tmp.minCoeff(&idx);
773 nearestE.push_back(candidates[idx]);
774 }
775 target=*nearestE[0];
776 return;
777}

782bool BasicAACRuleBasedAgent01::chooseTransitionNormal(){
783 /*通常時の状態遷移(前進、接近、横行、後退)の選択を行う。状況に応じて確率で決定
784 1. 帰還できなくなりそうなとき、確定で後退
785 2. 見えている敵がいない場合、確定で前進
786 3. 敵陣に十分近いか、敵陣側扇範囲内に敵がいないのいずれかであればライン突破を優先し前進、
787 　主目標が突破阻止を目的として割り当てられているとき、突破阻止を優先し接近を選択する。
788 　両方満たす場合、自身が先に突破しそうであれば突破を優先し前進を選択、相手が先に突破しそうであれば突破阻止を優先し接近を選択
789 4. 主目標を味方が余裕をもって捉えていれば前進、接近、横行から確率で選択
790 5. 主目標を味方が余裕をもって捉えておらず、自身で余裕をもって捉えていなければ確定で接近
791 6. 主目標を味方が余裕をもって捉えておらず、自身で余裕をもって捉えていれば横行または接近から確率で選択
792 Returns:
793 bool: Trueのとき、もう一度回す。
794 */
795 const nl::json& fObs=parent->observables;
796 bool reconsider=manager->getTime()-lastNormalTransitionTime>=reconsiderationCycle;
797 //1. 帰還できなくなりそう
798 if(!chkFuelSufficiency()){
799 if(transitionCause!=TransitionCause::FUEL_SHORTAGE || reconsider){
800 transitionCause=TransitionCause::FUEL_SHORTAGE;
801 immidiateTransition(State::RTB);
802 }
803 return false;
804 }
805 //2. 見えている相手がいない
806 if(allEnemies.size()==0){
807 if(transitionCause!=TransitionCause::NO_ENEMY || reconsider){
808 transitionCause=TransitionCause::NO_ENEMY;
809 immidiateTransition(State::ADVANCE);
810 }
811 return false;
812 }
813 //3. ライン突破を優先 or 突破阻止を優先
814 double dist=dLine-pos.dot(forward);
815 double angle=M_PI;
816 bool tryBreak=false;
817 for(auto& t:allEnemies){
818 Eigen::Vector3d tpos=t.pos(getLocalCRS());
819 angle=std::min(angle,acos(std::max(-1.0,std::min(1.0,(tpos-pos).block<2,1>(0,0,2,1).normalized().dot(forward.block<2,1>(0,0,2,1))))));
820 }
821 tryBreak=dist<dPrioritizedAdvance*dLine || angle>thetaPrioritizedAdvance;
822 if(isAssignedAsInterceptor){
823 if(tryBreak){
824 double eDist=dLine-target.pos(getLocalCRS()).dot(-forward);
825 if(eDist<dist){
826 //相手の方が先に突破する可能性が高い場合、突破阻止を優先
827 if(transitionCause!=TransitionCause::INTERCEPT_LEAKER || reconsider){
828 transitionCause=TransitionCause::INTERCEPT_LEAKER;
829 immidiateTransition(State::APPROACH_TARGET);
830 }
831 return false;
832 }
833 }
834 }
835 if(tryBreak){
836 //自分の方が先に突破する可能性が高い場合、突破を優先
837 if(transitionCause!=TransitionCause::TRY_BREAK || reconsider){
838 transitionCause=TransitionCause::TRY_BREAK;
839 immidiateTransition(State::ADVANCE);
840 }
841 return false;
842 }
843 //4〜6
844 if(target.nonLimitTrackers.size()>=2){
845 //4. 味方が余裕をもって捉えている
846 if(transitionCause!=TransitionCause::TRACKING_BY_ALLY || reconsider){
847 std::vector<nl::json> candidates={State::ADVANCE,State::APPROACH_TARGET,State::KEEP_SENSING};
848 Eigen::VectorXd probs(3);
849 probs<<pAdvanceAlly,pApproachAlly,pKeepSensingAlly;
850 probs/=probs.sum();
851 nl::json config={
852 {"type","choice"},
853 {"weights",probs},
854 {"candidates",candidates}
855 };
856 auto nextState=getValueFromJsonR(config,randomGen).get<State>();
857 transitionCause=TransitionCause::TRACKING_BY_ALLY;
858 immidiateTransition(nextState);
859 }
860 return false;
861 }else if(target.nonLimitTrackers.size()==1){
862 if(target.nonLimitTrackers[0]!=parent->getFullName()){
863 //4. 味方が余裕をもって捉えている
864 if(transitionCause!=TransitionCause::TRACKING_BY_ALLY || reconsider){
865 std::vector<nl::json> candidates={State::ADVANCE,State::APPROACH_TARGET,State::KEEP_SENSING};
866 Eigen::VectorXd probs(3);
867 probs<<pAdvanceAlly,pApproachAlly,pKeepSensingAlly;
868 probs/=probs.sum();
869 nl::json config={
870 {"type","choice"},
871 {"weights",probs},
872 {"candidates",candidates}
873 };
874 auto nextState=getValueFromJsonR(config,randomGen).get<State>();
875 transitionCause=TransitionCause::TRACKING_BY_ALLY;
876 immidiateTransition(nextState);
877 }
878 return false;
879 }else{
880 //5. 自分だけが余裕をもって捉えている
881 if(transitionCause!=TransitionCause::TRACKING_BY_MYSELF || reconsider){
882 std::vector<nl::json> candidates={State::APPROACH_TARGET,State::KEEP_SENSING};
883 Eigen::VectorXd probs(2);
884 probs<<pApproachMyself,pKeepSensingMyself;
885 probs/=probs.sum();
886 nl::json config={
887 {"type","choice"},
888 {"weights",probs},
889 {"candidates",candidates}
890 };
891 auto nextState=getValueFromJsonR(config,randomGen).get<State>();
892 transitionCause=TransitionCause::TRACKING_BY_MYSELF;
893 immidiateTransition(nextState);
894 }
895 return false;
896 }
897 }else{
898 //6. 誰も余裕をもって捉えていない
899 if(transitionCause!=TransitionCause::TRACKING_BY_NOBODY || reconsider){
900 transitionCause=TransitionCause::TRACKING_BY_NOBODY;
901 immidiateTransition(State::APPROACH_TARGET);
902 }
903 return false;
904 }

919bool BasicAACRuleBasedAgent01::deploySub(){
920 /*
921 遷移待機の終了判定⇛回避判定⇛離脱判定⇛射撃判定⇛通常遷移の順に判定
922 特に遅延遷移の問題で、一度のdeployで二周以上回すこともある。
923 Returns:
924 bool: Trueのとき、もう一度回す。
925 */
926 //遷移待ち中
927 if(waitingTransition){
928 if(chkTransitionCount()){//遷移待ち時間の経過
929 //一旦遷移を完了させてもう一度回す。
930 completeTransition();
931 return true;
932 }
933 }
934 //回避の判定(回避は最優先)
935 if(chkMWS()){
936 if(state!=State::EVADE){
937 immidiateTransition(State::EVADE);
938 }
939 return false;
940 }

942 bool needToWithdraw=false;
943 bool isWithdrawing= state==State::WITHDRAW || (waitingTransition && transitionAfter==State::WITHDRAW);
944 if(isWithdrawing){
945 needToWithdraw=true;
946 Time t=manager->getTime();
947 if(t>=withdrawnTime+tWithdraw){
948 needToWithdraw=false;
949 }
950 }
951 std::vector<bool> bs=chkBreak();
952 std::vector<bool> breakFlag;
953 bool bAny=false;
954 for(auto&& b:bs){
955 breakFlag.push_back(b);
956 bAny=bAny||b;
957 }
958 if(bAny){
959 if(state!=State::WITHDRAW){
960 withdrawnTrigger.clear();
961 for(int i=0;i<allEnemies.size();++i){
962 if(breakFlag[i]){
963 withdrawnTrigger.push_back(allEnemies[i].copy());
964 }
965 }
966 immidiateTransition(State::WITHDRAW);
967 }
968 withdrawnTime=manager->getTime(); //時間経過による離脱終了の起点の時刻だけは更新
969 needToWithdraw=true;
970 }
971 //射撃の判定
972 if(chkLaunchable()){//自機が射撃可能状態
973 std::vector<std::pair<bool,std::pair<TrackInfo,double>>> res;
974 for(auto&& t:track){
975 res.push_back(chkConditionForShoot(t));
976 }
977 std::sort(res.begin(),res.end(),
978 [](const std::pair<bool,std::pair<TrackInfo,double>>& lhs,const std::pair<bool,std::pair<TrackInfo,double>>& rhs){
979 if(lhs.first){
980 if(rhs.first){
981 return lhs.second.second<rhs.second.second;
982 }else{
983 return true;
984 }
985 }else{
986 if(rhs.first){
987 return false;
988 }else{
989 return lhs.second.second<rhs.second.second;
990 }
991 }
992 });
993 if(res.size()>0 && res[0].first){
994 decideShoot(res[0].second.first);
995 }
996 }
997 if(needToWithdraw){
998 return false;
999 }
1000 //離脱も回避もしなかった場合、通常時の行動選択を実施
1001 return chooseTransitionNormal();

1004void BasicAACRuleBasedAgent01::control(){
1005 /*stateに応じた制御出力を決定する。
1006 * stateに応じて進みたい方向dstDirと進みたい速さdstVを計算する。
1007 * dstDirに対しては、現時点の進行方向から導出される回転方向に回転するような角速度を発生させる。
1008 * dstVに対しては、現時点の速度から導出される加速度を発生させる。
1009 * 高度維持、マルチセンシングもこの関数内でdstDirに補正をかけることで行う。
1010 * 目標への接近はその瞬間に目標がいる方向へ最大速度とする。
1011 */
1012 if(!isAfterDeploy){
1013 updateMyInfo();//自機情報の更新
1014 updateTracks(false);//航跡情報の更新
1015 updateTargetStatus(false);//目標情報の更新
1016 }
1017 isAfterDeploy=false;
1018 MotionState myMotion=MotionState(parent->observables.at("motion")).transformTo(getLocalCRS());
1019 Eigen::Vector3d ex=myMotion.dirBtoP(Eigen::Vector3d(1,0,0));
1020 Eigen::Vector3d dstDirH;
1021 //水平方向の決定
1022 if(state==State::ADVANCE){
1023 //前進
1024 dstDirH=forward;
1025 }else if(state==State::APPROACH_TARGET){
1026 //その瞬間に目標がいる方向
1027 Eigen::Vector3d tpos=target.pos(getLocalCRS());
1028 dstDirH=tpos-pos;
1029 dstDirH(2)=0;
1030 dstDirH.normalize();
1031 }else if(state==State::KEEP_SENSING){
1032 //横行ははじめから複数捉えることを考慮する
1033 //主目標方向を正面として、左右どちらに進むかを決める
1034 //(1)場外に出そうならば内側向き(dOutLimitKeepSensing)
1035 //(2)場外に出ないならば副目標を多数捉えられる向きを優先
1036 //(3)副目標が同数ならば前進優先
1037 Eigen::Vector3d tDir=target.pos(getLocalCRS())-pos;
1038 double az=atan2(tDir(1),tDir(0));
1039 Eigen::Vector3d right=Eigen::Vector3d(cos(az+M_PI/2.-thetaKeepSensing),sin(az+M_PI/2.-thetaKeepSensing),0.);
1040 Eigen::Vector3d left=Eigen::Vector3d(cos(az-M_PI/2.+thetaKeepSensing),sin(az-M_PI/2.+thetaKeepSensing),0.);
1041 if(abs(pos(0))>=dOut-dOutLimitKeepSensing){
1042 //(1)
1043 Eigen::Vector3d inner;
1044 if(pos(0)>0){
1045 inner<<-1.,0.,0.;
1046 }else{
1047 inner<<1.,0.,0.;
1048 }
1049 double r=right.dot(inner);
1050 double l=left.dot(inner);
1051 if(r>=l){
1052 dstDirH=right;
1053 }else{
1054 dstDirH=left;
1055 }
1056 }else{
1057 //(2)
1058 int leftCount=0;
1059 int rightCount=0;
1060 if(parent->observables.contains("/spec/sensor/radar"_json_pointer)){
1061 for(auto& t:additionalTargets){
1062 Eigen::Vector3d tpos=t.pos(getLocalCRS());
1063 tDir=(tpos-pos).array()*Eigen::Array3d(1,1,0);
1064 double sAngle=parent->observables.at("/spec/sensor/radar/thetaFOR"_json_pointer);
1065 if(tDir.dot(right)>=cos(sAngle)){
1066 rightCount++;
1067 }
1068 if(tDir.dot(left)>=cos(sAngle)){
1069 leftCount++;
1070 }
1071 }
1072 }
1073 if(rightCount>leftCount){
1074 dstDirH=right;
1075 }else if(leftCount>rightCount){
1076 dstDirH=left;
1077 }else{
1078 //(3)
1079 double r=right.dot(forward);
1080 double l=left.dot(forward);
1081 if(r>=l){
1082 dstDirH=right;
1083 }else{
1084 dstDirH=left;
1085 }
1086 }
1087 }
1088 }else if(state==State::RTB){
1089 //後退
1090 dstDirH=-forward;
1091 }else if(state==State::WITHDRAW){
1092 //基本的には離脱のきっかけとなった敵から最も離れる方位に向かう
1093 std::vector<Eigen::Vector3d> tpos;
1094 for(auto& t:allEnemies){
1095 for(auto& trig:withdrawnTrigger){
1096 if(t.isSame(trig)){
1097 tpos.push_back(t.pos(getLocalCRS())-pos);
1098 break;
1099 }
1100 }
1101 }
1102 if(tpos.size()==0){
1103 //対象の敵が見えなくなったら真後ろへ
1104 dstDirH=-forward;
1105 }else{
1106 std::vector<double> az;
1107 for(auto& p:tpos){
1108 az.push_back(atan2(p(1),p(0)));
