75void BasicAACReward01::onEpisodeBegin(){
76 ruler=getShared<BasicAACRuler01>(manager->getRuler());
77 assert(ruler);
78 this->TeamReward::onEpisodeBegin();
79 assert(target.size()<=2);
80 for(auto&& team:target){
81 assert(team==ruler->westSider || team==ruler->eastSider);
82 }
83 opponentName.clear();
84 opponentName[ruler->westSider]=ruler->eastSider;
85 opponentName[ruler->eastSider]=ruler->westSider;
86 advPrev.clear();
87 advOffset.clear();
88 eliminatedTime.clear();
89 breakTime.clear();
90 disqTime.clear();
91 ruler->_setupPDownConfig(rHit,modelConfig,"rHit",0.0);
92 ruler->_setupPDownConfig(rHitE,modelConfig,"rHitE",0.0);
93 ruler->_setupPDownConfig(rCrash,modelConfig,"rCrash",0.0);
94 ruler->_setupPDownConfig(rCrashE,modelConfig,"rCrashE",0.0);
95 ruler->_setupPDownConfig(rAlive,modelConfig,"rAlive",0.0);
96 ruler->_setupPDownConfig(rAliveE,modelConfig,"rAliveE",0.0);
97 ruler->_setupPDownScale(rHitScale,rHit,rHitPerAircraft);
98 ruler->_setupPDownScale(rHitEScale,rHitE,rHitEPerAircraft);
99 ruler->_setupPDownScale(rCrashScale,rCrash,rCrashPerAircraft);
100 ruler->_setupPDownScale(rCrashEScale,rCrashE,rCrashEPerAircraft);
101 ruler->_setupPDownScale(rAliveScale,rAlive,rAlivePerAircraft);
102 ruler->_setupPDownScale(rAliveEScale,rAliveE,rAliveEPerAircraft);
103 for(auto& t:ruler->teams){
104 eliminatedTime[t]=-1;
105 breakTime[t]=-1;
106 disqTime[t]=-1;
107 advPrev[t]=(ruler->leadRange[t]-ruler->leadRange[opponentName[t]])/2;
108 if(!acceptNegativeAdv){
109 advPrev[t]=std::max(0.0,advPrev[t]);
110 }else if(adjustZerosum){
111 advPrev[t]/=2;
112 }
113 advOffset[t]=advPrev[t];
114 }

116void BasicAACReward01::onInnerStepEnd(){
117 for(auto& team:target){
118 //得点計算 1：撃墜による加点(1機あたりpHit点)
119 for(auto&& c:ruler->hitCount[team]){
120 debugPrint("1-1. Hit("+c.first+")",team,c.second*ruler->getPHit(team,c.first)*(1+rHitRatio));
121 reward[team]+=c.second*ruler->getPHit(team,c.first)*(1+rHitRatio);
122 debugPrint("1-2. Hit("+c.first+")",team,c.second*getRHit(team,c.first));
123 reward[team]+=c.second*getRHit(team,c.first);
124 debugPrint("1-3. Hit("+c.first+")",opponentName[team],c.second*getRHitE(opponentName[team],c.first));
125 reward[opponentName[team]]+=c.second*getRHitE(opponentName[team],c.first);
126 }
127 //得点計算 6-(a)：墜落による減点(1機あたりpCrash点)
128 for(auto&& c:ruler->crashCount[team]){
129 debugPrint("6-(a)-1. Crash("+c.first+")",team,-c.second*ruler->getPCrash(team,c.first)*(1+rCrashRatio));
130 reward[team]-=c.second*ruler->getPCrash(team,c.first)*(1+rCrashRatio);
131 debugPrint("6-(a)-2. Crash("+c.first+")",team,c.second*getRCrash(team,c.first));
132 reward[team]+=c.second*getRCrash(team,c.first);
133 debugPrint("6-(a)-3. Crash("+c.first+")",opponentName[team],c.second*getRCrashE(opponentName[team],c.first));
134 reward[opponentName[team]]+=c.second*getRCrashE(opponentName[team],c.first);
135 }
136 //得点計算 6-(b)：場外に対する減点(1秒、1kmにつきpOut点)
137 if(ruler->outDist[team]>0.0){
138 debugPrint("6-(b). Out",team,-(ruler->outDist[team]/1000.)*ruler->pOut*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep()*(1+rOutRatio));
139 }
140 reward[team]-=(ruler->outDist[team]/1000.)*ruler->pOut*interval[SimPhase::ON_INNERSTEP_END]*manager->getBaseTimeStep()*(1+rOutRatio);
141 }
142 for(auto& team:ruler->teams){
143 //全滅に対する追加報酬
144 if(ruler->eliminatedTime[team]>=0 && eliminatedTime[team]<0){
145 if(reward.count(team)>0){
146 debugPrint("7-1. Elim",team,rElimE);
147 reward[team]+=rElimE;
148 }
149 if(reward.count(opponentName[team])>0){
150 debugPrint("7-2. Elim",opponentName[team],rElim);
151 reward[opponentName[team]]+=rElim;
152 }
153 eliminatedTime[team]=manager->getElapsedTime();
154 }
155 //突破に対する追加報酬
156 if(ruler->breakTime[team]>=0 && breakTime[team]<0){
157 if(reward.count(team)>0){
158 debugPrint("2-1. Break",team,rBreak);
159 reward[team]+=rBreak;
160 //得点計算 2：突破による加点(pBreak点)
161 debugPrint("2-2. Break",team,ruler->pBreak*(1+rBreakRatio));
162 reward[team]+=ruler->pBreak*(1+rBreakRatio);
163 }
164 if(reward.count(opponentName[team])>0){
165 debugPrint("2-3. Break",opponentName[team],rBreakE);
166 reward[opponentName[team]]+=rBreakE;
167 }
168 breakTime[team]=manager->getElapsedTime();
169 }
170 //失格に対する追加報酬
171 if(ruler->disqTime[team]>=0 && disqTime[team]<0){
172 if(reward.count(team)>0){
173 debugPrint("8-1. Disq",team,rDisq);
174 reward[team]+=rDisq;
175 }
176 if(reward.count(opponentName[team])>0){
177 debugPrint("8-1. Disq",opponentName[team],rDisqE);
178 reward[opponentName[team]]+=rDisqE;
179 }
180 disqTime[team]=manager->getElapsedTime();
181 }
182 }

184void BasicAACReward01::onStepEnd(){
185 //得点計算 5：進出度合いに対する加点(1kmにつきpAdv点)
186 double adv=(ruler->leadRange[ruler->westSider]-ruler->leadRange[ruler->eastSider])/2;
187 if(!acceptNegativeAdv){
188 adv=std::max(0.0,adv);
189 }else if(adjustZerosum){
190 adv/=2;
191 }
192 if(reward.count(ruler->westSider)>0){
193 if(ruler->pAdv*(1+rAdvRatio)!=0.0){
194 debugPrint("5. Adv",ruler->westSider,(adv-advPrev[ruler->westSider])/1000.*ruler->pAdv*(1+rAdvRatio));
195 }
196 reward[ruler->westSider]+=(adv-advPrev[ruler->westSider])/1000.*ruler->pAdv*(1+rAdvRatio);
197 }
198 advPrev[ruler->westSider]=adv;
199 adv=(ruler->leadRange[ruler->eastSider]-ruler->leadRange[ruler->westSider])/2;
200 if(!acceptNegativeAdv){
201 adv=std::max(0.0,adv);
202 }else if(adjustZerosum){
203 adv/=2;
204 }
205 if(reward.count(ruler->eastSider)>0){
206 if(ruler->pAdv*(1+rAdvRatio)!=0.0){
207 debugPrint("5. Adv",ruler->eastSider,(adv-advPrev[ruler->eastSider])/1000.*ruler->pAdv*(1+rAdvRatio));
208 }
209 reward[ruler->eastSider]+=(adv-advPrev[ruler->eastSider])/1000.*ruler->pAdv*(1+rAdvRatio);
210 }
211 advPrev[ruler->eastSider]=adv;
212 //終了時の帳尻合わせ
213 if(ruler->endReason!=BasicAACRuler01::EndReason::NOTYET){
214 bool considerAdvantage=false;
215 //終了条件(1)：全機撃墜or墜落
216 if(ruler->endReason==BasicAACRuler01::EndReason::ELIMINATION){
217 if(breakTime[ruler->westSider]<0 && breakTime[ruler->eastSider]<0){
218 //両者未突破の場合、得点計算3及び5の考慮が必要
219 if(eliminatedTime[ruler->westSider]>=0 && eliminatedTime[ruler->eastSider]>=0){
220 //両者全滅した場合、得点計算5に従い優勢度を考慮
221 considerAdvantage=true;
222 }else{
223 //一方が生存していた場合
224 //得点計算 3：未突破の陣営に「そこから突破して更に帰還可能な」突破判定対象機が存在していれば+pBreak点
225 if(eliminatedTime[ruler->eastSider]<0){//東が生存
226 bool chk=false;
227 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
228 return asset->isAlive() && asset->getTeam()==ruler->eastSider && isinstance<Fighter>(asset) && ruler->isToBeConsideredForBreak(ruler->eastSider,asset->getFactoryModelName());
229 })){
230 if(ruler->isBreakableAndReturnableToBase(e.lock())){
231 chk=true;
232 break;
233 }
234 }
235 if(chk){
236 if(reward.count(ruler->eastSider)>0){
237 debugPrint("3-1. Break",ruler->eastSider,rBreak);
238 reward[ruler->eastSider]+=rBreak;
239 debugPrint("3-2. Break",ruler->eastSider,ruler->pBreak*(1+rBreakRatio));
240 reward[ruler->eastSider]+=ruler->pBreak*(1+rBreakRatio);
241 }
242 if(reward.count(ruler->westSider)>0){
243 debugPrint("3-3. Break",ruler->westSider,rBreakE);
244 reward[ruler->westSider]+=rBreakE;
245 }
246 }
247 }else{//西が生存
248 bool chk=false;
249 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
250 return asset->isAlive() && asset->getTeam()==ruler->westSider && isinstance<Fighter>(asset) && ruler->isToBeConsideredForBreak(ruler->westSider,asset->getFactoryModelName());
251 })){
252 if(ruler->isBreakableAndReturnableToBase(e.lock())){
253 chk=true;
254 break;
255 }
256 }
257 if(chk){
258 if(reward.count(ruler->westSider)>0){
259 debugPrint("3-1. Break",ruler->westSider,rBreak);
260 reward[ruler->westSider]+=rBreak;
261 debugPrint("3-2. Break",ruler->westSider,ruler->pBreak*(1+rBreakRatio));
262 reward[ruler->westSider]+=ruler->pBreak*(1+rBreakRatio);
263 }
264 if(reward.count(ruler->eastSider)>0){
265 debugPrint("3-3. Break",ruler->eastSider,rBreakE);
266 reward[ruler->eastSider]+=rBreakE;
267 }
268 }
269 }
270 }
271 }
272 }
273 //終了条件(2)：防衛ラインの突破
274 else if(ruler->endReason==BasicAACRuler01::EndReason::BREAK){
275 //追加の得点増減はなし
276 }
277 //終了条件(5)：ペナルティによる敗北
278 else if(ruler->endReasonSub==BasicAACRuler01::EndReason::PENALTY){
279 if(breakTime[ruler->westSider]<0 && breakTime[ruler->eastSider]<0){
280 //両者未突破の場合、得点計算5に従い優勢度を計算
281 considerAdvantage=true;
282 }
283 }
284 //終了条件(3)：両者撤退による打ち切り
285 else if(ruler->endReason==BasicAACRuler01::EndReason::WITHDRAWAL){
286 //追加の得点増減はなし(優勢度も計算しない)
287 }
288 //終了条件(4)：時間切れ
289 else if(manager->getElapsedTime()>=ruler->maxTime){
290 if(breakTime[ruler->westSider]<0 && breakTime[ruler->eastSider]<0){
291 //両者未突破の場合、得点計算5に従い優勢度を計算
292 considerAdvantage=true;
293 }
294 for(auto&& team:target){
295 debugPrint("9. TimeUp",team,rTimeup);
296 reward[team]+=rTimeup;
297 }
298 }
299 //得点計算 4：生存点(1機あたりpAlive点)
300 //得点計算 6(a)：帰還不可による墜落ペナルティ(1機あたりpCrash点)
301 for(auto& team:ruler->teams){
302 for(auto&& e:manager->getAssets([&](const std::shared_ptr<const Asset>& asset)->bool {
303 return asset->isAlive() && asset->getTeam()==team && isinstance<Fighter>(asset) && ruler->isToBeConsideredForElimination(team,asset->getFactoryModelName());
304 })){
305 auto asset=e.lock();
306 if(ruler->isReturnableToBase(asset)){
307 //帰還可能なら生存点
308 if(reward.count(team)>0){
309 debugPrint("4-1. Alive("+asset->getFactoryModelName()+")",team,ruler->getPAlive(team,asset->getFactoryModelName())*(1+rAliveRatio));
310 reward[team]+=ruler->getPAlive(team,asset->getFactoryModelName())*(1+rAliveRatio);
311 debugPrint("4-2. Alive("+asset->getFactoryModelName()+")",team,getRAlive(team,asset->getFactoryModelName()));
312 reward[team]+=getRAlive(team,asset->getFactoryModelName());
313 }
314 if(reward.count(opponentName[team])>0){
315 debugPrint("4-3. Alive("+asset->getFactoryModelName()+")",opponentName[team],getRAliveE(opponentName[team],asset->getFactoryModelName()));
316 reward[opponentName[team]]+=getRAliveE(opponentName[team],asset->getFactoryModelName());
317 }
318 }else{
319 //帰還不可能なら墜落ペナルティ
320 if(reward.count(team)>0){
321 debugPrint("6-(a)-1. NoReturn("+asset->getFactoryModelName()+")",team,-ruler->getPCrash(team,asset->getFactoryModelName())*(1+rCrashRatio));
322 reward[team]-=ruler->getPCrash(team,asset->getFactoryModelName())*(1+rCrashRatio);
323 debugPrint("6-(a)-2. NoReturn("+asset->getFactoryModelName()+")",team,getRCrash(team,asset->getFactoryModelName()));
324 reward[team]+=getRCrash(team,asset->getFactoryModelName());
325 }
326 if(reward.count(opponentName[team])>0){
327 debugPrint("6-(a)-3. NoReturn("+asset->getFactoryModelName()+")",opponentName[team],getRCrashE(opponentName[team],asset->getFactoryModelName()));
328 reward[opponentName[team]]+=getRCrashE(opponentName[team],asset->getFactoryModelName());
329 }
330 }
331 }
332 }
333 if(adjustBreakEnd && !considerAdvantage){
334 //Rulerが進出度合いによる得点を使用しない場合に、即時報酬も無効化する場合
335 if(reward.count(ruler->westSider)>0){
336 debugPrint("5. Adv(cancel)",ruler->westSider,-(advPrev[ruler->westSider]-advOffset[ruler->westSider])/1000.*ruler->pAdv*(1+rAdvRatio));
337 reward[ruler->westSider]-=(advPrev[ruler->westSider]-advOffset[ruler->westSider])/1000.*ruler->pAdv*(1+rAdvRatio);
338 }
339 if(reward.count(ruler->eastSider)>0){
340 debugPrint("5. Adv(cancel)",ruler->eastSider,-(advPrev[ruler->eastSider]-advOffset[ruler->eastSider])/1000.*ruler->pAdv*(1+rAdvRatio));
341 reward[ruler->eastSider]-=(advPrev[ruler->eastSider]-advOffset[ruler->eastSider])/1000.*ruler->pAdv*(1+rAdvRatio);
342 }
343 }
344 }
345 if(adjustZerosum){
346 //ゼロサム化する場合
347 std::map<std::string,double> buffer;
348 for(auto&& t:target){
349 buffer[t]=reward[t];
350 }
351 for(auto&& t:target){
352 if(buffer[opponentName[t]]!=0.0){
353 debugPrint("10. ZeroSum",t,-buffer[opponentName[t]]);
354 }
355 reward[t]-=buffer[opponentName[t]];
356 }
357 }
358 this->TeamReward::onStepEnd();
359 for(auto&& t:target){
360 if(reward[t]!=0.0){
361 debugPrint("reward",t,reward[t]);
362 debugPrint("totalReward",t,totalReward[t]);
363 }
364 }
