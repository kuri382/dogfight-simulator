25class TruncatedStandardNormal(Distribution):
26 """
27 Truncated Standard Normal distribution
28 https://people.sc.fsu.edu/~jburkardt/presentations/truncated_normal.pdf
29 """
30
31 arg_constraints = {
32 'a': constraints.real,
33 'b': constraints.real,
34 }
35 has_rsample = True
36
37 def __init__(self, a, b, validate_args=None):
38 self.a, self.b = broadcast_all(a, b)
39 if isinstance(a, Number) and isinstance(b, Number):
40 batch_shape = torch.Size()
41 else:
42 batch_shape = self.a.size()
43 super(TruncatedStandardNormal, self).__init__(batch_shape, validate_args=validate_args)
44 if self.a.dtype != self.b.dtype:
45 raise ValueError('Truncation bounds types are different')
46 if any((self.a >= self.b).view(-1,).tolist()):
47 raise ValueError('Incorrect truncation range')
48 eps = torch.finfo(self.a.dtype).eps
49 self._dtype_min_gt_0 = eps
50 self._dtype_max_lt_1 = 1 - eps
51 self._little_phi_a = self._little_phi(self.a)
52 self._little_phi_b = self._little_phi(self.b)
53 self._big_phi_a = self._big_phi(self.a)
54 self._big_phi_b = self._big_phi(self.b)
55 self._Z = (self._big_phi_b - self._big_phi_a).clamp_min(eps)
56 self._log_Z = self._Z.log()
57 little_phi_coeff_a = torch.nan_to_num(self.a, nan=math.nan)
58 little_phi_coeff_b = torch.nan_to_num(self.b, nan=math.nan)
59 self._lpbb_m_lpaa_d_Z = (self._little_phi_b * little_phi_coeff_b - self._little_phi_a * little_phi_coeff_a) / self._Z
60 self._mean = -(self._little_phi_b - self._little_phi_a) / self._Z
61 self._variance = 1 - self._lpbb_m_lpaa_d_Z - ((self._little_phi_b - self._little_phi_a) / self._Z) ** 2
62 self._entropy = CONST_LOG_SQRT_2PI_E + self._log_Z - 0.5 * self._lpbb_m_lpaa_d_Z
63
64 @constraints.dependent_property
65 def support(self):
66 return constraints.interval(self.a, self.b)
67
68 @property
69 def mean(self):
70 return self._mean
71
72 @property
73 def variance(self):
74 return self._variance
75
76 def entropy(self):
77 return self._entropy
78
79 @property
80 def auc(self):
81 return self._Z
82
83 @staticmethod
84 def _little_phi(x):
85 return (-(x ** 2) * 0.5).exp() * CONST_INV_SQRT_2PI
86
87 @staticmethod
88 def _big_phi(x):
89 return 0.5 * (1 + (x * CONST_INV_SQRT_2).erf())
90
91 @staticmethod
92 def _inv_big_phi(x):
93 return CONST_SQRT_2 * (2 * x - 1).erfinv()
94
95 def cdf(self, value):
96 if self._validate_args:
97 self._validate_sample(value)
98 return ((self._big_phi(value) - self._big_phi_a) / self._Z).clamp(0, 1)
99
100 def icdf(self, value):
101 return self._inv_big_phi(self._big_phi_a + value * self._Z)
102
103 def log_prob(self, value):
104 if self._validate_args:
105 self._validate_sample(value)
106 return CONST_LOG_INV_SQRT_2PI - self._log_Z - (value ** 2) * 0.5
107
108 def rsample(self, sample_shape=torch.Size()):
109 shape = self._extended_shape(sample_shape)
110 p = torch.empty(shape, device=self.a.device).uniform_(self._dtype_min_gt_0, self._dtype_max_lt_1)
111 return self.icdf(p)

114class TruncatedNormal(TruncatedStandardNormal):
115 """
116 Truncated Normal distribution
117 https://people.sc.fsu.edu/~jburkardt/presentations/truncated_normal.pdf
118 """
119
120 has_rsample = True
121
122 def __init__(self, loc, scale, a, b, validate_args=None):
123 self.loc, self.scale, a, b = broadcast_all(loc, scale, a, b)
124 a = (a - self.loc) / self.scale
125 b = (b - self.loc) / self.scale
126 super(TruncatedNormal, self).__init__(a, b, validate_args=validate_args)
127 self._log_scale = self.scale.log()
128 self._mean = self._mean * self.scale + self.loc
129 self._variance = self._variance * self.scale ** 2
130 self._entropy += self._log_scale
131
132 def _to_std_rv(self, value):
133 return (value - self.loc) / self.scale
134
135 def _from_std_rv(self, value):
136 return value * self.scale + self.loc
137
138 def cdf(self, value):
139 return super(TruncatedNormal, self).cdf(self._to_std_rv(value))
140
141 def icdf(self, value):
142 return self._from_std_rv(super(TruncatedNormal, self).icdf(value))
143
144 def log_prob(self, value):
145 return super(TruncatedNormal, self).log_prob(self._to_std_rv(value)) - self._log_scale
