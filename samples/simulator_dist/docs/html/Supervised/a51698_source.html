51bool UnsupervisedTaskMixer::isTarget(const std::shared_ptr<Agent>& agent){
52 std::string team=agent->getTeam();
53 auto found=targetAgentModels.find(team);
54 if(found==targetAgentModels.end()){
55 return false;
56 }
57 for(auto&& query:found->second){
58 std::regex re=std::regex(query);
59 if(std::regex_match(agent->getFactoryModelName(),re)){
60 return true;
61 }
62 }
63 return false;
64}

80void SupervisedTaskMixer::onEpisodeBegin(){
81 truths.clear();
82 truth_spaces.clear();
83 std::map<std::string,py::object> truth;
84 py::dict& observation=manager->observation();//makeObsは既に呼ばれた後である
85 for(auto&& e:observation){
86 std::string key=py::cast<std::string>(e.first);
87 auto agent=getShared(manager->getAgent(key));
88 if(isTarget(agent)){
89 std::pair<bool,py::object> ret=calcTaskTruth(agent);
90 if(ret.first){//即時計算出来た場合
91 truth[agent->getFullName()]=ret.second;
92 }else{//即時計算出来ない場合は一旦Noneを入れておき、派生クラスにおいて適切な場所で計算して格納するものとする
93 truth[agent->getFullName()]=py::none();
94 }
95 truth_spaces[agent->getFullName()]=calcTaskTruthSpace(agent);
96 }
97 }
98 truths.push_back(truth);
99}
