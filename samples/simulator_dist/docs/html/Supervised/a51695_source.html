87py::object SimpleSupervisedTaskSample02::calcTaskTruthSpace(const std::shared_ptr<Agent>& agent){
88 py::module_ spaces=py::module_::import("gymnasium.spaces");
89 float vMin,vMax;
90 if(lower*upper<=0.0){//入力の変域に0を含む
91 vMin=lower*upper;
92 vMax=std::max<float>(abs(lower*lower),abs(upper*upper));
93 }else{//入力の変域に0を含まない
94 vMin=std::min<float>(abs(lower*lower),abs(upper*upper));
95 vMax=std::max<float>(abs(lower*lower),abs(upper*upper));
96 }
97 Eigen::VectorXi shape(1);
98 shape<<1;
99 return spaces.attr("Box")(vMin,vMax,shape,py::dtype::of<float>());
100}
